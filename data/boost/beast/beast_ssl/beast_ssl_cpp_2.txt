// Пример 2: HTTPS-сервер на Boost.Beast + SSL (минимальный)

#include <boost/asio.hpp>
#include <boost/asio/ssl.hpp>
#include <boost/beast.hpp>
#include <iostream>
#include <memory>

namespace asio  = boost::asio;
namespace ssl   = asio::ssl;
namespace beast = boost::beast;
namespace http  = beast::http;
using tcp       = asio::ip::tcp;

class https_session : public std::enable_shared_from_this<https_session>
{
public:
    https_session(tcp::socket socket, ssl::context& ctx)
        : stream_(std::move(socket), ctx)
        , strand_(stream_.get_executor())
    {}

    void run()
    {
        auto self = shared_from_this();
        stream_.async_handshake(
            ssl::stream_base::server,
            asio::bind_executor(
                strand_,
                [self](beast::error_code ec)
                {
                    if (!ec) self->do_read();
                }));
    }

private:
    void do_read()
    {
        req_ = {};

        http::async_read(
            stream_,
            buffer_,
            req_,
            asio::bind_executor(
                strand_,
                std::bind(
                    &https_session::on_read,
                    shared_from_this(),
                    std::placeholders::_1,
                    std::placeholders::_2)));
    }

    void on_read(beast::error_code ec, std::size_t)
    {
        if (ec == http::error::end_of_stream)
            return do_shutdown();

        if (ec)
        {
            std::cerr << "read: " << ec.message() << "\n";
            return;
        }

        http::response<http::string_body> res{
            http::status::ok, req_.version()};

        res.set(http::field::server, "beast-https");
        res.set(http::field::content_type, "text/plain");
        res.keep_alive(req_.keep_alive());
        res.body() = "Hello from HTTPS";
        res.prepare_payload();

        auto self = shared_from_this();
        http::async_write(
            stream_,
            res,
            asio::bind_executor(
                strand_,
                [self](beast::error_code ec, std::size_t)
                {
                    if (ec)
                    {
                        std::cerr << "write: " << ec.message() << "\n";
                        return;
                    }

                    if (!self->req_.keep_alive())
                        self->do_shutdown();
                    else
                        self->do_read();
                }));
    }

    void do_shutdown()
    {
        auto self = shared_from_this();
        stream_.async_shutdown(
            asio::bind_executor(
                strand_,
                [self](beast::error_code ec)
                {
                    if (ec && ec != asio::error::eof)
                        std::cerr << "shutdown: " << ec.message() << "\n";
                }));
    }

    beast::ssl_stream<tcp::socket>       stream_;
    asio::strand<asio::any_io_executor>  strand_;
    beast::flat_buffer                   buffer_;
    http::request<http::string_body>     req_;
};

class https_listener : public std::enable_shared_from_this<https_listener>
{
public:
    https_listener(asio::io_context& ioc, ssl::context& ctx, tcp::endpoint ep)
        : ioc_(ioc)
        , ctx_(ctx)
        , acceptor_(ioc)
    {
        beast::error_code ec;

        acceptor_.open(ep.protocol(), ec);
        if (ec) throw beast::system_error(ec);

        acceptor_.set_option(asio::socket_base::reuse_address(true), ec);
        if (ec) throw beast::system_error(ec);

        acceptor_.bind(ep, ec);
        if (ec) throw beast::system_error(ec);

        acceptor_.listen(asio::socket_base::max_listen_connections, ec);
        if (ec) throw beast::system_error(ec);
    }

    void run()
    {
        do_accept();
    }

private:
    void do_accept()
    {
        acceptor_.async_accept(
            asio::make_strand(ioc_),
            std::bind(
                &https_listener::on_accept,
                shared_from_this(),
                std::placeholders::_1,
                std::placeholders::_2));
    }

    void on_accept(beast::error_code ec, tcp::socket socket)
    {
        if (!ec)
            std::make_shared<https_session>(std::move(socket), ctx_)->run();
        else
            std::cerr << "accept: " << ec.message() << "\n";

        do_accept();
    }

    asio::io_context& ioc_;
    ssl::context&     ctx_;
    tcp::acceptor     acceptor_;
};

int main()
{
    try
    {
        asio::io_context ioc;
        ssl::context     ctx{ssl::context::tls_server};

        // Загрузка сертификата/ключа (заглушка — сюда твой код)
        // ctx.use_certificate_chain_file("server.crt");
        // ctx.use_private_key_file("server.key", ssl::context::pem);

        auto ep = tcp::endpoint(tcp::v4(), 8443);
        std::make_shared<https_listener>(ioc, ctx, ep)->run();
        ioc.run();
    }
    catch (std::exception& e)
    {
        std::cerr << "fatal: " << e.what() << "\n";
    }
}
