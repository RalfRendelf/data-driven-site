// http_body_cpp_1.txt
// Пример 1: string_body и dynamic_body в HTTP-клиенте

#include <boost/asio.hpp>
#include <boost/beast.hpp>
#include <iostream>

namespace asio  = boost::asio;
namespace beast = boost::beast;
namespace http  = beast::http;
using tcp       = asio::ip::tcp;

int main()
{
    try
    {
        asio::io_context ioc;

        std::string host   = "example.org";
        std::string port   = "80";
        std::string target = "/";

        tcp::resolver resolver{ioc};
        auto results = resolver.resolve(host, port);

        beast::tcp_stream stream{ioc};
        stream.connect(results);

        // Запрос без тела (empty_body)
        http::request<http::empty_body> req{http::verb::get, target, 11};
        req.set(http::field::host, host);
        req.set(http::field::user_agent, "http-body-demo");

        http::write(stream, req);

        // Ответ в string_body (полностью в память)
        {
            beast::flat_buffer buffer;
            http::response<http::string_body> res;

            http::read(stream, buffer, res);

            std::cout << "=== string_body response ===\n";
            std::cout << "Status: " << res.result_int() << "\n";
            std::cout << "Body size: " << res.body().size() << "\n";
        }

        // Повторный запрос, но читаем в dynamic_body
        stream.socket().shutdown(tcp::socket::shutdown_both);
        stream = beast::tcp_stream{ioc};
        stream.connect(results);

        http::write(stream, req);

        {
            beast::flat_buffer buffer;
            http::response<http::dynamic_body> res;

            http::read(stream, buffer, res);

            std::cout << "\n=== dynamic_body response ===\n";
            auto size = res.body().size();
            std::cout << "Body size (bytes): " << size << "\n";

            // Преобразуем часть тела в строку
            auto data = beast::buffers_to_string(res.body().data());
            std::cout << "First bytes:\n"
                      << data.substr(0, 200) << "\n";
        }

        beast::error_code ec;
        stream.socket().shutdown(tcp::socket::shutdown_both, ec);
    }
    catch (std::exception& e)
    {
        std::cerr << "fatal: " << e.what() << "\n";
    }
}
