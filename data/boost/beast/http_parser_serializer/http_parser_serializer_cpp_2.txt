// http_parser_serializer_cpp_2.txt
// Пример 2: Потоковая отправка ответа через serializer (chunked-like)

#include <boost/asio.hpp>
#include <boost/beast.hpp>
#include <iostream>
#include <memory>

namespace asio  = boost::asio;
namespace beast = boost::beast;
namespace http  = beast::http;
using tcp       = asio::ip::tcp;

class streaming_session : public std::enable_shared_from_this<streaming_session>
{
public:
    explicit streaming_session(tcp::socket socket)
        : socket_(std::move(socket))
        , strand_(socket_.get_executor())
    {}

    void run()
    {
        do_read();
    }

private:
    void do_read()
    {
        req_ = {};

        http::async_read(
            socket_,
            buffer_,
            req_,
            asio::bind_executor(
                strand_,
                std::bind(
                    &streaming_session::on_read,
                    shared_from_this(),
                    std::placeholders::_1,
                    std::placeholders::_2)));
    }

    void on_read(beast::error_code ec, std::size_t)
    {
        if (ec == http::error::end_of_stream)
            return do_close();

        if (ec)
        {
            std::cerr << "read: " << ec.message() << "\n";
            return;
        }

        // Готовим ответ с dynamic_body
        res_.emplace(http::status::ok, req_.version());
        res_->set(http::field::server, "serializer-demo");
        res_->set(http::field::content_type, "text/plain");
        res_->keep_alive(req_.keep_alive());

        // Начальное тело
        beast::ostream(res_->body()) << "Streaming response start...\n";

        // Создаём сериализатор
        serializer_.emplace(*res_);

        do_write();
    }

    void do_write()
    {
        auto self = shared_from_this();

        http::async_write(
            socket_,
            *serializer_,
            asio::bind_executor(
                strand_,
                [self](beast::error_code ec, std::size_t bytes)
                {
                    if (ec)
                    {
                        std::cerr << "write: " << ec.message() << "\n";
                        return;
                    }

                    // Если сериализатор ещё не закончил — можно дописать данные
                    if (!self->serializer_->is_done())
                    {
                        // Для демонстрации: добавим ещё немного данных
                        beast::ostream(self->res_->body())
                            << "More data chunk, bytes written: " << bytes << "\n";

                        self->do_write();
                    }
                    else
                    {
                        if (!self->res_->keep_alive())
                            self->do_close();
                        else
                            self->do_read();
                    }
                }));
    }

    void do_close()
    {
        beast::error_code ec;
        socket_.shutdown(tcp::socket::shutdown_send, ec);
    }

    tcp::socket                         socket_;
    asio::strand<asio::any_io_executor> strand_;
    beast::flat_buffer                  buffer_;
    http::request<http::string_body>    req_;

    std::optional<http::response<http::dynamic_body>>          res_;
    std::optional<http::response_serializer<http::dynamic_body>> serializer_;
};

class streaming_listener : public std::enable_shared_from_this<streaming_listener>
{
public:
    streaming_listener(asio::io_context& ioc, tcp::endpoint ep)
        : ioc_(ioc)
        , acceptor_(ioc)
    {
        beast::error_code ec;

        acceptor_.open(ep.protocol(), ec);
        if (ec) throw beast::system_error(ec);

        acceptor_.set_option(asio::socket_base::reuse_address(true), ec);
        if (ec) throw beast::system_error(ec);

        acceptor_.bind(ep, ec);
        if (ec) throw beast::system_error(ec);

        acceptor_.listen(asio::socket_base::max_listen_connections, ec);
        if (ec) throw beast::system_error(ec);
    }

    void run()
    {
        do_accept();
    }

private:
    void do_accept()
    {
        acceptor_.async_accept(
            asio::make_strand(ioc_),
            std::bind(
                &streaming_listener::on_accept,
                shared_from_this(),
                std::placeholders::_1,
                std::placeholders::_2));
    }

    void on_accept(beast::error_code ec, tcp::socket socket)
    {
        if (!ec)
            std::make_shared<streaming_session>(std::move(socket))->run();
        else
            std::cerr << "accept: " << ec.message() << "\n";

        do_accept();
    }

    asio::io_context& ioc_;
    tcp::acceptor     acceptor_;
};

int main()
{
    try
    {
        asio::io_context ioc;
        auto ep = tcp::endpoint(tcp::v4(), 8083);

        std::make_shared<streaming_listener>(ioc, ep)->run();
        ioc.run();
    }
    catch (std::exception& e)
    {
        std::cerr << "fatal: " << e.what() << "\n";
    }
}
