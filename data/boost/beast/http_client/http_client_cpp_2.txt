// Пример 2: Продвинутый HTTP-клиент (POST + JSON + таймауты)

#include <boost/asio.hpp>
#include <boost/beast.hpp>
#include <iostream>

namespace asio  = boost::asio;
namespace beast = boost::beast;
namespace http  = beast::http;
using tcp       = asio::ip::tcp;

int main()
{
    try
    {
        asio::io_context ioc;

        std::string host   = "httpbin.org";
        std::string port   = "80";
        std::string target = "/post";

        tcp::resolver resolver{ioc};
        auto results = resolver.resolve(host, port);

        beast::tcp_stream stream{ioc};

        // Таймауты
        stream.expires_after(std::chrono::seconds(5));

        stream.connect(results);

        // JSON тело
        std::string json = R"({"name":"Vladislav","value":42})";

        http::request<http::string_body> req{http::verb::post, target, 11};
        req.set(http::field::host, host);
        req.set(http::field::content_type, "application/json");
        req.set(http::field::user_agent, "beast-http-client");
        req.body() = json;
        req.prepare_payload();

        http::write(stream, req);

        beast::flat_buffer buffer;
        http::response<http::dynamic_body> res;

        http::read(stream, buffer, res);

        std::cout << "Status: " << res.result_int() << "\n";
        std::cout << "Body:\n"
                  << beast::buffers_to_string(res.body().data()) << "\n";

        beast::error_code ec;
        stream.socket().shutdown(tcp::socket::shutdown_both, ec);
    }
    catch (std::exception& e)
    {
        std::cerr << "fatal: " << e.what() << "\n";
    }
}
