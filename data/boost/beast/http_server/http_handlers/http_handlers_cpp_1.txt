// Пример 1: Базовые обработчики (GET и POST)

#include <boost/beast.hpp>
#include <boost/asio.hpp>
#include <iostream>

namespace beast = boost::beast;
namespace http  = beast::http;

// Обработчик GET /hello
void handle_hello(const http::request<http::string_body>& req,
                  http::response<http::string_body>& res)
{
    res.result(http::status::ok);
    res.set(http::field::content_type, "text/plain");
    res.body() = "Hello from Beast!";
    res.prepare_payload();
}

// Обработчик POST /echo
void handle_echo(const http::request<http::string_body>& req,
                 http::response<http::string_body>& res)
{
    res.result(http::status::ok);
    res.set(http::field::content_type, "text/plain");
    res.body() = "You sent: " + req.body();
    res.prepare_payload();
}
