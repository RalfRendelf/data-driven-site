// Пример 1: Базовый thread_pool

#include <boost/asio.hpp>
#include <iostream>
#include <thread>

int main()
{
    boost::asio::thread_pool pool(4);

    for (int i = 0; i < 8; ++i)
    {
        boost::asio::post(pool, [i]{
            std::cout << "Task " << i
                      << " executed in thread "
                      << std::this_thread::get_id() << "\n";
        });
    }

    pool.join();
}
