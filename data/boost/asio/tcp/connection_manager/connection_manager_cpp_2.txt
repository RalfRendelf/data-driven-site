// Пример 2: Connection Manager в полноценном TCP-сервере

#include <boost/asio.hpp>
#include <iostream>
#include <set>
#include <memory>

using boost::asio::ip::tcp;

class session;
class connection_manager;

class session : public std::enable_shared_from_this<session>
{
public:
    session(tcp::socket socket, connection_manager& manager)
        : socket_(std::move(socket)),
          manager_(manager)
    {}

    void start();
    void stop();

private:
    void do_read();
    void do_write(std::size_t n);

    tcp::socket socket_;
    connection_manager& manager_;
    std::array<char, 1024> data_;
};

class connection_manager
{
public:
    void start(std::shared_ptr<session> s)
    {
        sessions_.insert(s);
        s->start();
    }

    void stop(std::shared_ptr<session> s)
    {
        sessions_.erase(s);
        s->stop();
    }

    void stop_all()
    {
        for (auto& s : sessions_)
            s->stop();
        sessions_.clear();
    }

private:
    std::set<std::shared_ptr<session>> sessions_;
};

void session::start()
{
    do_read();
}

void session::stop()
{
    socket_.close();
}

void session::do_read()
{
    auto self = shared_from_this();

    socket_.async_read_some(
        boost::asio::buffer(data_),
        [this, self](auto ec, std::size_t n)
        {
            if (!ec)
                do_write(n);
            else
                manager_.stop(self);
        }
    );
}

void session::do_write(std::size_t n)
{
    auto self = shared_from_this();

    boost::asio::async_write(
        socket_,
        boost::asio::buffer(data_, n),
        [this, self](auto ec, std::size_t)
        {
            if (!ec)
                do_read();
            else
                manager_.stop(self);
        }
    );
}

int main()
{
    boost::asio::io_context io;
    connection_manager manager;

    tcp::acceptor acc(io, tcp::endpoint(tcp::v4(), 9000));

    std::function<void()> do_accept;
    do_accept = [&]{
        acc.async_accept([&](auto ec, tcp::socket socket){
            if (!ec)
            {
                auto s = std::make_shared<session>(std::move(socket), manager);
                manager.start(s);
            }
            do_accept();
        });
    };

    do_accept();
    io.run();
}
