#include <sys/wait.h>
#include <unistd.h>
#include <iostream>

int main() {
    pid_t pid = fork();

    if (pid == 0) {
        sleep(2);
        return 42;
    }

    int status;
    waitpid(pid, &status, 0);

    if (WIFEXITED(status)) {
        std::cout << "Child exited with code " << WEXITSTATUS(status) << std::endl;
    }
}
