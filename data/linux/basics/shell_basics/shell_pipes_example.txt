# Пайпы (|) позволяют передавать stdout одной команды на stdin другой.
# Это основа потоковой обработки данных в Linux.

# 1. Вывести список файлов и отфильтровать только .txt
ls -l | grep ".txt"
# ls -l  → выводит подробный список файлов
# grep ".txt" → оставляет строки, содержащие ".txt"

# 2. Найти процессы, связанные с nginx
ps aux | grep nginx
# ps aux → выводит все процессы
# grep nginx → фильтрует строки, где встречается "nginx"

# 3. Просматривать вывод dmesg постранично
dmesg | less
# dmesg → выводит сообщения ядра
# less → позволяет листать вывод вверх/вниз
