#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/socket.h>
#include <sys/select.h>
#include <netinet/in.h>
#include <arpa/inet.h>

#define PORT 8080
#define BUFFER_SIZE 1024
#define MAX_CLIENTS 10

int main() {
    int server_fd, client_fds[MAX_CLIENTS];
    struct sockaddr_in address;
    fd_set read_fds, master_fds;
    int max_fd = 0;
    char buffer[BUFFER_SIZE];

    // Инициализация массива клиентских дескрипторов
    for (int i = 0; i < MAX_CLIENTS; i++) {
        client_fds[i] = -1;
    }

    // Создание сокета
    if ((server_fd = socket(AF_INET, SOCK_STREAM, 0)) == 0) {
        perror("socket failed");
        exit(EXIT_FAILURE);
    }

    // Настройка адреса сервера
    address.sin_family = AF_INET;
    address.sin_addr.s_addr = INADDR_ANY;
    address.sin_port = htons(PORT);

    // Связывание сокета с адресом
    if (bind(server_fd, (struct sockaddr *)&address, sizeof(address)) < 0) {
        perror("bind failed");
        exit(EXIT_FAILURE);
    }

    // Прослушивание входящих соединений
    if (listen(server_fd, MAX_CLIENTS) < 0) {
        perror("listen");
        exit(EXIT_FAILURE);
    }

    // Инициализация master_fds
    FD_ZERO(&master_fds);
    FD_SET(server_fd, &master_fds);
    max_fd = server_fd;

    printf("Server listening on port %d...\n", PORT);

    while (1) {
        read_fds = master_fds;
        if (select(max_fd + 1, &read_fds, NULL, NULL, NULL) == -1) {
            perror("select");
            break;
        }

        // Проверка серверного сокета на новые соединения
        if (FD_ISSET(server_fd, &read_fds)) {
            struct sockaddr_in client_addr;
            socklen_t client_len = sizeof(client_addr);
            int client_fd = accept(server_fd, (struct sockaddr *)&client_addr, &client_len);
            if (client_fd == -1) {
                perror("accept");
            } else {
                // Добавление нового клиента в master_fds
                for (int i = 0; i < MAX_CLIENTS; i++) {
                    if (client_fds[i] == -1) {
                        client_fds[i] = client_fd;
                        FD_SET(client_fd, &master_fds);
                        if (client_fd > max_fd) {
                            max_fd = client_fd;
                        }
                        printf("New client connected (fd=%d)\n", client_fd);
                        break;
                    }
                }
            }
        }

        // Обработка данных от клиентов
        for (int i = 0; i < MAX_CLIENTS; i++) {
            if (client_fds[i] != -1 && FD_ISSET(client_fds[i], &read_fds)) {
                ssize_t bytes_read = recv(client_fds[i], buffer, BUFFER_SIZE, 0);
                if (bytes_read <= 0) {
                    close(client_fds[i]);
                    FD_CLR(client_fds[i], &master_fds);
                    client_fds[i] = -1;
                    printf("Client disconnected (fd=%d)\n", client_fds[i]);
                } else {
                    buffer[bytes_read] = '\0';
                    printf("Received from client %d: %s\n", client_fds[i], buffer);
                    send(client_fds[i], buffer, bytes_read, 0); // Эхо-ответ
                }
            }
        }
    }

    close(server_fd);
    return 0;
}
