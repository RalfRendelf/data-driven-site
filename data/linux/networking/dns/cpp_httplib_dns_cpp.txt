#include <iostream>
#include <string>
#include <httplib.h>

int main() {
    httplib::Client cli("https://dns.google");

    // Выполняем GET-запрос к API DNS
    auto res = cli.Get("/resolve?name=example.com&type=A");

    if (res && res->status == 200) {
        std::cout << "DNS response:\n" << res->body << std::endl;
    } else {
        std::cerr << "Error: " << (res ? res->status : -1) << std::endl;
    }

    return 0;
}
