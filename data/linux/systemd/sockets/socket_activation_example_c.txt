#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <systemd/sd-daemon.h>

int main() {
    int n_sockets;
    int sockets[SD_LISTEN_FDS_START + 1];
    int i;

    // Получаем количество сокетов из переменной окружения LISTEN_FDS
    n_sockets = sd_listen_fds(0);
    if (n_sockets < 0) {
        fprintf(stderr, "Failed to get listen fds: %s\n", strerror(-n_sockets));
        return 1;
    }

    // Получаем файловые дескрипторы сокетов
    for (i = 0; i < n_sockets; i++) {
        sockets[i] = SD_LISTEN_FDS_START + i;
    }

    // Обрабатываем каждый сокет
    for (i = 0; i < n_sockets; i++) {
        struct sockaddr_in addr;
        socklen_t addr_len = sizeof(addr);

        // Получаем информацию о сокете
        if (getsockname(sockets[i], (struct sockaddr *)&addr, &addr_len) < 0) {
            perror("getsockname");
            continue;
        }

        printf("Socket %d: listening on %s:%d\n",
               sockets[i],
               inet_ntoa(addr.sin_addr),
               ntohs(addr.sin_port));

        // Здесь можно добавить логику обработки соединений
        // Например, для TCP-сокета:
        // accept(sockets[i], ...);
    }

    // Основной цикл сервиса
    while (1) {
        // Обработка соединений
        sleep(1);
    }

    return 0;
}
