<h2>Singleton (Одиночка)</h2>

<p><strong>Категория:</strong> Порождающий</p>

<h3>Описание</h3>
<p>Singleton - это порождающий паттерн проектирования, который гарантирует, что у класса есть только один экземпляр, и предоставляет глобальную точку доступа к этому экземпляру. Он используется для управления общей ресурсами, такими как базы данных или сетевые соединения.</p>

<h3>Принцип работы</h3>
<ol>
<li>Приватный конструктор предотвращает создание экземпляров класса извне</li>
<li>Статическое поле хранит уникальный экземпляр класса</li>
<li>Статический метод предоставляет глобальную точку доступа к экземпляру</li>
</ol>

<h3>Преимущества и недостатки</h3>
<p>Преимущества:</p>
<ul>
<li>Гарантирует наличие только одного экземпляра класса</li>
<li>Предоставляет глобальную точку доступа</li>
<li>Ленивая инициализация возможна</li>
<li>Контролирует доступ к глобальному ресурсу</li>
</ul>

<p>Недостатки:</p>
<ul>
<li>Нарушает принцип единственной ответственности</li>
<li>Может создать проблемы с многопоточностью</li>
<li>Сложно использовать в тестировании</li>
<li>Создает жесткую зависимость в коде</li>
</ul>

class SingletonMeta(type):
    """
    В Python паттерн Singleton можно реализовать по-разному.
    Возможные способы включают в себя: базовый класс, декоратор, метакласс.
    Ниже представлен метакласс.
    """
    _instances = {}

    def __call__(cls, *args, **kwargs):
        if cls not in cls._instances:
            instance = super().__call__(*args, **kwargs)
            cls._instances[cls] = instance
        return cls._instances[cls]


class Singleton(metaclass=SingletonMeta):
    def some_business_logic(self):
        """
        Наконец, любой одиночка должен содержать некоторую бизнес-логику,
        которая может быть выполнена на его экземпляре.
        """
        pass


if __name__ == "__main__":
    s1 = Singleton()
    s2 = Singleton()

    if id(s1) == id(s2):
        print("Singleton works, both variables contain the same instance.")
    else:
        print("Singleton failed, variables contain different instances.")

// Классическая реализация Singleton
class Singleton {
    constructor() {
        if (Singleton.instance) {
            return Singleton.instance;
        }
        
        // Здесь инициализация объекта
        this.data = [];
        
        Singleton.instance = this;
        return this;
    }
    
    addData(item) {
        this.data.push(item);
    }
    
    getData() {
        return this.data;
    }
}

// Пример использования
const singleton1 = new Singleton();
const singleton2 = new Singleton();

console.log(singleton1 === singleton2); // true

singleton1.addData("First item");
console.log(singleton2.getData()); // ["First item"]

// Альтернативная реализация с замыканием
const SingletonClosure = (function() {
    let instance;
    
    function Singleton() {
        if (instance) {
            return instance;
        }
        
        this.data = [];
        instance = this;
        return this;
    }
    
    return Singleton;
})();

// ES6 модульный способ (используя экспорт переменной)
let instance = null;

class ModuleSingleton {
    constructor() {
        if (instance) {
            return instance;
        }
        
        this.data = [];
        instance = this;
        return this;
    }
    
    addData(item) {
        this.data.push(item);
    }
    
    getData() {
        return this.data;
    }
}

export default new ModuleSingleton();

<code>
#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;

// Структура для хранения данных синглтона
typedef struct {
    int counter;
} Singleton;

// Статическая переменная для хранения экземпляра
static Singleton* instance = NULL;

// Функция для получения экземпляра синглтона
Singleton* getSingletonInstance() {
    if (instance == NULL) {
        instance = malloc(sizeof(Singleton));
        instance->counter = 0;
        printf("Создан новый экземпляр синглтона\n");
    } else {
        printf("Возвращен существующий экземпляр синглтона\n");
    }
    return instance;
}

// Функция для освобождения экземпляра
void destroySingleton() {
    if (instance != NULL) {
        free(instance);
        instance = NULL;
    }
}

// Функция для увеличения счетчика
void incrementCounter() {
    Singleton* s = getSingletonInstance();
    s->counter++;
}

// Функция для получения значения счетчика
int getCounter() {
    Singleton* s = getSingletonInstance();
    return s->counter;
}

// Пример использования
int main() {
    Singleton* s1 = getSingletonInstance();
    Singleton* s2 = getSingletonInstance();
    
    printf("s1 адрес: %p\n", (void*)s1);
    printf("s2 адрес: %p\n", (void*)s2);
    
    incrementCounter();
    incrementCounter();
    
    printf("Значение счетчика: %d\n", getCounter());
    
    destroySingleton();
    
    return 0;
}

#include &lt;iostream&gt;
#include &lt;thread&gt;
#include &lt;mutex&gt;

/**
 * Реализация потокобезопасного синглтона
 */
class Singleton
{
private:
    static Singleton* instance;
    static std::mutex mutex_;

    // Конструкторы должны быть объявлены приватными, чтобы 
    // предотвратить создание экземпляров синглтона.
    Singleton() {}
    ~Singleton() {}

public:
    /**
     * Удаление оператора копирования.
     */
    Singleton(Singleton &other) = delete;
    /**
     * Удаление оператора присваивания.
     */
    void operator=(const Singleton &) = delete;
    
    /**
     * Глобальная точка доступа к экземпляру синглтона.
     */
    static Singleton *GetInstance();
    
    void someBusinessLogic() {
        std::cout << "Выполнение бизнес-логики синглтона..." << std::endl;
    }
};

Singleton* Singleton::instance = nullptr;
std::mutex Singleton::mutex_;

Singleton *Singleton::GetInstance()
{
    std::lock_guard&lt;std::mutex&gt; lock(mutex_);
    if (instance == nullptr) {
        instance = new Singleton();
    }
    return instance;
}

int main()
{
    std::thread t1([] {
        Singleton* singleton = Singleton::GetInstance();
        std::cout << "Thread 1: " << singleton << std::endl;
    });
    
    std::thread t2([] {
        Singleton* singleton = Singleton::GetInstance();
        std::cout << "Thread 2: " << singleton << std::endl;
    });
    
    t1.join();
    t2.join();
    
    return 0;
}

// Потокобезопасная реализация синглтона с ленивой инициализацией
public class Singleton {
    private static volatile Singleton instance;

    private Singleton() {
        // Запрещаем создание экземпляров
    }

    public static Singleton getInstance() {
        if (instance == null) {
            synchronized (Singleton.class) {
                if (instance == null) {
                    instance = new Singleton();
                }
            }
        }
        return instance;
    }

    public void someBusinessLogic() {
        System.out.println("Выполнение бизнес-логики синглтона...");
    }
}

// Альтернативная реализация с использованием статического блока инициализации
class StaticBlockSingleton {
    private static final Singleton INSTANCE;

    static {
        try {
            INSTANCE = new Singleton();
        } catch (Exception e) {
            throw new RuntimeException("Ошибка при создании синглтона", e);
        }
    }

    private StaticBlockSingleton() {}

    public static Singleton getInstance() {
        return INSTANCE;
    }
}

// Реализация с использованием Enum (рекомендуемый способ в Effective Java)
enum EnumSingleton {
    INSTANCE;

    public void someBusinessLogic() {
        System.out.println("Выполнение бизнес-логики синглтона через Enum...");
    }
}

// Пример использования
public class Main {
    public static void main(String[] args) {
        Singleton s1 = Singleton.getInstance();
        Singleton s2 = Singleton.getInstance();

        System.out.println(s1 == s2); // true - это один и тот же объект
        s1.someBusinessLogic();
    }
}
