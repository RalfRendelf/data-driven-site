<h2>Monotonic Stack (Монотонный стек)</h2>

<p><strong>Категория:</strong> Структуры данных</p>

<h3>Описание</h3>
<p>Монотонный стек - это стек, в котором элементы хранятся в отсортированном порядке (либо по возрастанию, либо по убыванию). Он используется для решения задач, где нужно эффективно находить следующий больший/меньший элемент для каждого элемента массива.</p>

<h3>Принцип работы</h3>
<ol>
<li>Поддерживать монотонный порядок в стеке при добавлении новых элементов</li>
<li>При добавлении нового элемента, удалять элементы из стека, нарушающие монотонность</li>
<li>Использовать стек для эффективного поиска следующего/предыдущего большего/меньшего элемента</li>
</ol>

<h3>Сложность алгоритма</h3>
<ul>
<li>Временная сложность: O(n) - каждый элемент добавляется и удаляется из стека максимум один раз</li>
<li>Пространственная сложность: O(n) - для хранения стека</li>
</ul>

<h3>Преимущества и недостатки</h3>
<p>Преимущества:</p>
<ul>
<li>Эффективное решение задач на поиск следующего/предыдущего элемента</li>
<li>Линейное время выполнения</li>
<li>Применимость к широкому кругу задач</li>
</ul>

<p>Недостатки:</p>
<ul>
<li>Требует дополнительной памяти</li>
<li>Сложнее для понимания и реализации</li>
</ul>

<h3>Пример на C++: Найти следующий больший элемент</h3>
<pre>
<code>
#include &lt;iostream&gt;
#include &lt;vector&gt;
#include &lt;stack&gt;

std::vector&lt;int&gt; nextGreaterElement(std::vector&lt;int&gt;&amp; nums) {
    int n = nums.size();
    std::vector&lt;int&gt; result(n, -1);
    std::stack&lt;int&gt; st;
    
    for (int i = 0; i < n; i++) {
        // Пока стек не пуст и текущий элемент больше элемента на вершине стека
        while (!st.empty() && nums[i] > nums[st.top()]) {
            int index = st.top();
            st.pop();
            result[index] = nums[i];  // nums[i] - следующий больший элемент для nums[index]
        }
        st.push(i);  // Добавляем текущий индекс в стек
    }
    
    return result;
}

// Найти наибольший прямоугольник в гистограмме
int largestRectangleArea(std::vector&lt;int&gt;&amp; heights) {
    std::stack&lt;int&gt; st;
    int maxArea = 0;
    int n = heights.size();
    
    for (int i = 0; i <= n; i++) {
        int currentHeight = (i == n) ? 0 : heights[i];
        
        while (!st.empty() && currentHeight < heights[st.top()]) {
            int height = heights[st.top()];
            st.pop();
            int width = st.empty() ? i : i - st.top() - 1;
            maxArea = std::max(maxArea, height * width);
        }
        st.push(i);
    }
    
    return maxArea;
}

int main() {
    std::vector&lt;int&gt; nums = {4, 5, 2, 25};
    std::vector&lt;int&gt; result = nextGreaterElement(nums);
    
    std::cout << "Next greater elements: ";
    for (int num : result) {
        std::cout << num << " ";
    }
    std::cout << std::endl;
    
    std::vector&lt;int&gt; heights = {2, 1, 5, 6, 2, 3};
    int maxRect = largestRectangleArea(heights);
    std::cout << "Largest rectangle area: " << maxRect << std::endl;
    
    return 0;
}

<h3>Пример на C: Найти предыдущий меньший элемент</h3>
<pre>
<code>
#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;

// Структура стека
struct Stack {
    int* data;
    int top;
    int capacity;
};

struct Stack* createStack(int capacity) {
    struct Stack* stack = malloc(sizeof(struct Stack));
    stack->capacity = capacity;
    stack->top = -1;
    stack->data = malloc(capacity * sizeof(int));
    return stack;
}

int isEmpty(struct Stack* stack) {
    return stack->top == -1;
}

void push(struct Stack* stack, int item) {
    stack->data[++stack->top] = item;
}

int pop(struct Stack* stack) {
    return stack->data[stack->top--];
}

int peek(struct Stack* stack) {
    return stack->data[stack->top];
}

void previousSmaller(int arr[], int n, int result[]) {
    struct Stack* stack = createStack(n);
    
    for (int i = 0; i < n; i++) {
        // Удаляем элементы из стека, которые >= текущему
        while (!isEmpty(stack) && peek(stack) >= arr[i]) {
            pop(stack);
        }
        
        // Если стек пуст, нет меньшего элемента слева
        if (isEmpty(stack)) {
            result[i] = -1;
        } else {
            result[i] = peek(stack);
        }
        
        push(stack, arr[i]);
    }
    
    free(stack->data);
    free(stack);
}

int main() {
    int arr[] = {1, 6, 4, 10, 2, 5};
    int n = sizeof(arr) / sizeof(arr[0]);
    
    int result[n];
    previousSmaller(arr, n, result);
    
    printf("Previous smaller elements: ");
    for (int i = 0; i < n; i++) {
        if (result[i] == -1) {
            printf("_, ");
        } else {
            printf("%d, ", result[i]);
        }
    }
    printf("\n");
    
    return 0;
}
