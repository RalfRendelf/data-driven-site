<h2>Z-function / Prefix-function / Manacher (Z-функция / Префикс-функция / Алгоритм Манакера)</h2>

<p><strong>Категория:</strong> Строки</p>

<h3>Описание</h3>
<p>Это группа алгоритмов, используемых для решения задач на строки:</p>
<ul>
<li>Z-функция: для каждой позиции строки вычисляет длину самого длинного подотрезка, начинающегося в этой позиции и совпадающего с префиксом строки</li>
<li>Префикс-функция: для каждой позиции строки вычисляет длину самого длинного префикса строки, являющегося суффиксом подстроки до этой позиции</li>
<li>Алгоритм Манакера: находит все палиндромы в строке за линейное время</li>
</ul>

<h3>Принцип работы</h3>
<ol>
<li>Z-функция: использует уже вычисленные значения для ускорения вычислений</li>
<li>Префикс-функция: также использует предыдущие значения для эффективного вычисления текущего</li>
<li>Манакер: использует симметрию палиндромов для ускорения вычислений</li>
</ol>

<h3>Сложность алгоритма</h3>
<ul>
<li>Временная сложность: O(n) для всех алгоритмов</li>
<li>Пространственная сложность: O(n) для всех алгоритмов</li>
</ul>

<h3>Преимущества и недостатки</h3>
<p>Преимущества:</p>
<ul>
<li>Линейное время выполнения</li>
<li>Эффективное решение задач на строки</li>
<li>Могут быть использованы для решения задач на поиск подстрок</li>
</ul>

<p>Недостатки:</p>
<ul>
<li>Сложность понимания и реализации</li>
<li>Избыточность для простых задач</li>
</ul>

<h3>Пример на C++: Реализация Z-функции и алгоритма Манакера</h3>
<pre>
<code>
#include &lt;iostream&gt;
#include &lt;vector&gt;
#include &lt;string&gt;
#include &lt;algorithm&gt;

// Вычисление Z-функции
std::vector&lt;int&gt; z_function(const std::string&amp; s) {
    int n = s.length();
    std::vector&lt;int&gt; z(n);
    int l = 0, r = 0;
    
    for (int i = 1; i < n; i++) {
        if (i <= r) {
            z[i] = std::min(r - i + 1, z[i - l]);
        }
        
        while (i + z[i] < n && s[z[i]] == s[i + z[i]]) {
            z[i]++;
        }
        
        if (i + z[i] - 1 > r) {
            l = i;
            r = i + z[i] - 1;
        }
    }
    
    return z;
}

// Вычисление префикс-функции
std::vector&lt;int&gt; prefix_function(const std::string&amp; s) {
    int n = s.length();
    std::vector&lt;int&gt; pi(n);
    
    for (int i = 1; i < n; i++) {
        int j = pi[i - 1];
        
        while (j > 0 && s[i] != s[j]) {
            j = pi[j - 1];
        }
        
        if (s[i] == s[j]) {
            j++;
        }
        
        pi[i] = j;
    }
    
    return pi;
}

// Алгоритм Манакера для поиска всех палиндромов
std::vector&lt;int&gt; manacher_odd(const std::string&amp; s) {
    int n = s.length();
    std::vector&lt;int&gt; d(n);
    int l = 0, r = -1;
    
    for (int i = 0; i < n; i++) {
        int k = (i > r) ? 1 : std::min(d[l + r - i], r - i + 1);
        
        while (0 <= i - k && i + k < n && s[i - k] == s[i + k]) {
            k++;
        }
        
        d[i] = k--;
        
        if (i + k > r) {
            l = i - k;
            r = i + k;
        }
    }
    
    return d;
}

// Поиск всех палиндромов (чётной и нечётной длины)
std::vector&lt;std::vector&lt;bool&gt;&gt; manacher_palindromes(const std::string&amp; s) {
    int n = s.length();
    std::vector&lt;std::vector&lt;bool&gt;&gt; is_pal(n, std::vector&lt;bool&gt;(n, false));
    
    // Нечётные палиндромы
    for (int center = 0; center < n; center++) {
        for (int len = 0; center - len >= 0 && center + len < n && s[center - len] == s[center + len]; len++) {
            is_pal[center - len][center + len] = true;
        }
    }
    
    // Чётные палиндромы
    for (int center = 0; center < n - 1; center++) {
        for (int len = 0; center - len >= 0 && center + 1 + len < n && s[center - len] == s[center + 1 + len]; len++) {
            is_pal[center - len][center + 1 + len] = true;
        }
    }
    
    return is_pal;
}

int main() {
    std::string s = "abacabad";
    
    std::vector&lt;int&gt; z = z_function(s);
    std::cout << "Z-function: ";
    for (int val : z) {
        std::cout << val << " ";
    }
    std::cout << std::endl;
    
    std::vector&lt;int&gt; pi = prefix_function(s);
    std::cout << "Prefix function: ";
    for (int val : pi) {
        std::cout << val << " ";
    }
    std::cout << std::endl;
    
    std::vector&lt;int&gt; man_odd = manacher_odd(s);
    std::cout << "Manacher odd: ";
    for (int val : man_odd) {
        std::cout << val << " ";
    }
    std::cout << std::endl;
    
    return 0;
}

<h3>Пример на C: Реализация префикс-функции</h3>
<pre>
<code>
#include &lt;stdio.h&gt;
#include &lt;string.h&gt;

// Вычисление префикс-функции
void prefix_function(char* s, int* pi, int n) {
    for (int i = 1; i < n; i++) {
        int j = pi[i - 1];
        
        while (j > 0 && s[i] != s[j]) {
            j = pi[j - 1];
        }
        
        if (s[i] == s[j]) {
            j++;
        }
        
        pi[i] = j;
    }
}

int main() {
    char s[] = "abacabad";
    int n = strlen(s);
    int pi[n];
    
    // Инициализация
    pi[0] = 0;
    prefix_function(s, pi, n);
    
    printf("Prefix function: ");
    for (int i = 0; i < n; i++) {
        printf("%d ", pi[i]);
    }
    printf("\n");
    
    return 0;
}
