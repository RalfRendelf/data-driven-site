<h2>Merge Sort (Сортировка слиянием)</h2>

<p><strong>Категория:</strong> Сортировка</p>

<h3>Описание</h3>
<p>Merge Sort - это эффективный алгоритм сортировки методом "разделяй и властвуй", который делит массив на две половины, рекурсивно сортирует каждую, а затем объединяет отсортированные половины в окончательно отсортированный массив.</p>

<h3>Принцип работы</h3>
<ol>
<li>Разделить массив пополам до тех пор, пока каждый подмассив не будет содержать один элемент</li>
<li>Объединить пары отсортированных подмассивов в более крупные отсортированные массивы</li>
<li>Повторять процесс до получения одного полностью отсортированного массива</li>
</ol>

<h3>Сложность алгоритма</h3>
<ul>
<li>Лучший случай: O(n log n)</li>
<li>Средний случай: O(n log n)</li>
<li>Худший случай: O(n log n)</li>
<li>Пространственная сложность: O(n)</li>
</ul>

<h3>Преимущества и недостатки</h3>
<p>Преимущества:</p>
<ul>
<li>Гарантированная эффективность: O(n log n) во всех случаях</li>
<li>Стабильная сортировка (сохраняет относительный порядок одинаковых элементов)</li>
<li>Хорошо работает с большими объемами данных</li>
<li>Поддается параллелизации</li>
</ul>

<p>Недостатки:</p>
<ul>
<li>Требует дополнительной памяти размером n</li>
<li>Медленнее Quick Sort в среднем случае</li>
<li>Не является in-place алгоритмом</li>
</ul>

def merge_sort(arr):
    if len(arr) <= 1:
        return arr
    
    mid = len(arr) // 2
    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])
    
    return merge(left, right)


def merge(left, right):
    result = []
    left_idx = right_idx = 0
    
    while left_idx < len(left) and right_idx < len(right):
        if left[left_idx] < right[right_idx]:
            result.append(left[left_idx])
            left_idx += 1
        else:
            result.append(right[right_idx])
            right_idx += 1
    
    # Добавляем оставшиеся элементы
    result.extend(left[left_idx:])
    result.extend(right[right_idx:])
    
    return result


# Пример использования
if __name__ == "__main__":
    unsorted_list = [38, 27, 43, 3, 9, 82, 10]
    sorted_list = merge_sort(unsorted_list)
    print(f"Исходный список: {unsorted_list}")
    print(f"Отсортированный список: {sorted_list}")

function mergeSort(arr) {
  if (arr.length <= 1) {
    return arr;
  }
  
  const mid = Math.floor(arr.length / 2);
  const left = mergeSort(arr.slice(0, mid));
  const right = mergeSort(arr.slice(mid));
  
  return merge(left, right);
}

function merge(left, right) {
  const result = [];
  let leftIdx = 0;
  let rightIdx = 0;
  
  while (leftIdx < left.length && rightIdx < right.length) {
    if (left[leftIdx] < right[rightIdx]) {
      result.push(left[leftIdx]);
      leftIdx++;
    } else {
      result.push(right[rightIdx]);
      rightIdx++;
    }
  }
  
  // Добавляем оставшиеся элементы
  return result.concat(left.slice(leftIdx)).concat(right.slice(rightIdx));
}

// Пример использования
const unsortedArray = [38, 27, 43, 3, 9, 82, 10];
const sortedArray = mergeSort(unsortedArray);
console.log("Исходный массив:", unsortedArray);
console.log("Отсортированный массив:", sortedArray);

<code>
#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;

void merge(int arr[], int left, int mid, int right) {
    int n1 = mid - left + 1;
    int n2 = right - mid;
    
    // Создаем временные массивы
    int *leftArr = malloc(n1 * sizeof(int));
    int *rightArr = malloc(n2 * sizeof(int));
    
    // Копируем данные во временные массивы
    for (int i = 0; i < n1; i++)
        leftArr[i] = arr[left + i];
    for (int j = 0; j < n2; j++)
        rightArr[j] = arr[mid + 1 + j];
    
    // Объединяем временные массивы обратно в arr[left..right]
    int i = 0, j = 0, k = left;
    
    while (i < n1 && j < n2) {
        if (leftArr[i] <= rightArr[j]) {
            arr[k] = leftArr[i];
            i++;
        } else {
            arr[k] = rightArr[j];
            j++;
        }
        k++;
    }
    
    // Копируем оставшиеся элементы leftArr[], если они есть
    while (i < n1) {
        arr[k] = leftArr[i];
        i++;
        k++;
    }
    
    // Копируем оставшиеся элементы rightArr[], если они есть
    while (j < n2) {
        arr[k] = rightArr[j];
        j++;
        k++;
    }
    
    // Освобождаем память
    free(leftArr);
    free(rightArr);
}

void mergeSort(int arr[], int left, int right) {
    if (left < right) {
        int mid = left + (right - left) / 2;
        
        mergeSort(arr, left, mid);
        mergeSort(arr, mid + 1, right);
        
        merge(arr, left, mid, right);
    }
}

// Пример использования
int main() {
    int arr[] = {38, 27, 43, 3, 9, 82, 10};
    int arrSize = sizeof(arr) / sizeof(arr[0]);
    
    printf("Исходный массив: ");
    for (int i = 0; i < arrSize; i++)
        printf("%d ", arr[i]);
    printf("\n");
    
    mergeSort(arr, 0, arrSize - 1);
    
    printf("Отсортированный массив: ");
    for (int i = 0; i < arrSize; i++)
        printf("%d ", arr[i]);
    printf("\n");
    
    return 0;
}
