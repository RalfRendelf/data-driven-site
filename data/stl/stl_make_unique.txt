<h2>std::make_unique</h2>

<p><strong>Категория:</strong> Умные указатели</p>

<h3>Описание</h3>
<p>std::make_unique - это функция, добавленная в C++14, которая создает std::unique_ptr для объекта, размещенного в динамической памяти. Она автоматически выделяет память и создает объект в этой памяти, возвращая unique_ptr, который будет автоматически управлять этой памятью.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++14</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::make_unique и их реализация на C (для демонстрации аналогов):</p>

<h4>C++</h4>
<pre>
<code>
// =======================
// C++ пример std::make_unique
// =======================

#include &lt;memory&gt;
#include &lt;iostream&gt;
#include &lt;vector&gt;

struct Person {
int id;
std::string name;
};

int main() {
auto int_ptr = std::make_unique&lt;int&gt;(42);
std::cout &lt;&lt; *int_ptr &lt;&lt; "\n";

auto dbl_ptr = std::make_unique&lt;double&gt;(3.14);
std::cout &lt;&lt; *dbl_ptr &lt;&lt; "\n";

auto str_ptr = std::make_unique&lt;std::string&gt;("Hello");
std::cout &lt;&lt; *str_ptr &lt;&lt; "\n";

auto person_ptr = std::make_unique&lt;Person&gt;(Person{1, "Alice"});
std::cout &lt;&lt; person_ptr-&gt;id &lt;&lt; " " &lt;&lt; person_ptr-&gt;name &lt;&lt; "\n";

auto arr_ptr = std::make_unique&lt;int[]&gt;(5);
for (int i = 0; i &lt; 5; i++) arr_ptr[i] = i + 1;
for (int i = 0; i &lt; 5; i++) std::cout &lt;&lt; arr_ptr[i] &lt;&lt; " ";
std::cout &lt;&lt; "\n";

return 0;
}

// Пример реализации аналога std::make_unique на C

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;string.h&gt;

typedef struct {
void* ptr;
void (deleter)(void);
} UniquePtr;

#define MAKE_UNIQUE(type, ...) make_unique_##type(VA_ARGS)

UniquePtr* make_unique_int(int value) {
int* ptr = malloc(sizeof(int));
ptr = value;
UniquePtr uptr = malloc(sizeof(UniquePtr));
uptr->ptr = ptr;
uptr->deleter = &free;
return uptr;
}

UniquePtr* make_unique_double(double value) {
double* ptr = malloc(sizeof(double));
ptr = value;
UniquePtr uptr = malloc(sizeof(UniquePtr));
uptr->ptr = ptr;
uptr->deleter = &free;
return uptr;
}

UniquePtr* make_unique_string(const char* str) {
char* ptr = malloc(strlen(str) + 1);
strcpy(ptr, str);
UniquePtr* uptr = malloc(sizeof(UniquePtr));
uptr->ptr = ptr;
uptr->deleter = &free;
return uptr;
}

UniquePtr* make_unique_array(size_t count, size_t element_size) {
void* ptr = calloc(count, element_size);
UniquePtr* uptr = malloc(sizeof(UniquePtr));
uptr->ptr = ptr;
uptr->deleter = &free;
return uptr;
}

#define MAKE_UNIQUE_ARRAY(type, count) make_unique_array(count, sizeof(type))

void* unique_ptr_get(UniquePtr* uptr) {
return uptr ? uptr->ptr : NULL;
}

void unique_ptr_destroy(UniquePtr* uptr) {
if (uptr) {
if (uptr->ptr && uptr->deleter) {
uptr->deleter(uptr->ptr);
}
free(uptr);
}
}

typedef struct {
int id;
char name[50];
} Person;

UniquePtr* make_unique_person(int id, const char* name) {
Person* ptr = malloc(sizeof(Person));
ptr->id = id;
strcpy(ptr->name, name);
UniquePtr* uptr = malloc(sizeof(UniquePtr));
uptr->ptr = ptr;
uptr->deleter = &free;
return uptr;
}

int main() {
UniquePtr* int_ptr = make_unique_int(42);
printf("%d\n", (int)unique_ptr_get(int_ptr));

UniquePtr* dbl_ptr = make_unique_double(3.14);
printf("%.2f\n", (double)unique_ptr_get(dbl_ptr));

UniquePtr* str_ptr = make_unique_string("Hello");
printf("%s\n", (char*)unique_ptr_get(str_ptr));

UniquePtr* arr_ptr = make_unique_array(5, sizeof(int));
int* arr = (int*)unique_ptr_get(arr_ptr);
for (int i = 0; i < 5; i++) arr[i] = i + 1;
for (int i = 0; i < 5; i++) printf("%d ", arr[i]);
printf("\n");

UniquePtr* person_ptr = make_unique_person(1, "Alice");
Person* p = (Person*)unique_ptr_get(person_ptr);
printf("%d %s\n", p->id, p->name);

unique_ptr_destroy(int_ptr);
unique_ptr_destroy(dbl_ptr);
unique_ptr_destroy(str_ptr);
unique_ptr_destroy(arr_ptr);
unique_ptr_destroy(person_ptr);

return 0;
}
