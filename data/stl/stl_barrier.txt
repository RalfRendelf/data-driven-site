<h2>std::barrier</h2>

<p><strong>Категория:</strong> Многопоточность</p>

<h3>Описание</h3>
<p>std::barrier - это примитив синхронизации, который позволяет группе потоков ожидать друг друга в определенной точке выполнения. Все потоки в группе должны достичь барьера, прежде чем они смогут продолжить выполнение. В отличие от std::latch, барьер может использоваться многократно. Введен в C++20.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++20</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::barrier:</p>

<pre>
<code>
// Пример использования std::barrier на C++

#include &lt;iostream&gt;
#include &lt;thread&gt;
#include &lt;vector&gt;
#include &lt;barrier&gt;
#include &lt;chrono&gt;

int main() {
    const size_t num_threads = 5;
    std::vector&lt;std::thread&gt; threads;
    
    // Создаем barrier для ожидания num_threads потоков
    std::barrier barrier(num_threads, []() {
        // Этот completion-функциональный объект вызывается после
        // того, как все потоки достигли барьера
        std::cout << "Все потоки достигли барьера!" << std::endl;
    });
    
    // Запускаем потоки
    for (size_t i = 0; i < num_threads; ++i) {
        threads.emplace_back([&barrier, i]() {
            std::cout << "Поток " << i << " начал работу" << std::endl;
            
            // Имитация работы
            std::this_thread::sleep_for(std::chrono::milliseconds(100 * (i + 1)));
            
            std::cout << "Поток " << i << " ожидает у барьера" << std::endl;
            
            // Ожидание у барьера
            barrier.arrive_and_wait();
            
            std::cout << "Поток " << i << " продолжает работу после барьера" << std::endl;
            
            // Имитация дополнительной работы
            std::this_thread::sleep_for(std::chrono::milliseconds(50 * (i + 1)));
            
            std::cout << "Поток " << i << " завершает работу" << std::endl;
        });
    }
    
    // Ожидаем завершения всех потоков
    for (auto&amp; t : threads) {
        t.join();
    }
    
    std::cout << "Все потоки завершены" << std::endl;
    
    return 0;
}
</code>

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка POSIX threads
 * Компиляция: gcc -o barrier_example barrier_example.c -lpthread
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;pthread.h&gt;
#include &lt;unistd.h&gt;

// Структура для имитации std::barrier
typedef struct {
    pthread_mutex_t mutex;
    pthread_cond_t cond;
    int count;
    int total;
    int cycle;  // для многократного использования
} Barrier;

// Инициализация барьера
int barrier_init(Barrier* barrier, int total) {
    barrier->total = total;
    barrier->count = 0;
    barrier->cycle = 0;
    return pthread_mutex_init(&barrier->mutex, NULL) ||
           pthread_cond_init(&barrier->cond, NULL);
}

// Уничтожение барьера
int barrier_destroy(Barrier* barrier) {
    int result = pthread_mutex_destroy(&barrier->mutex);
    result |= pthread_cond_destroy(&barrier->cond);
    return result;
}

// Ожидание у барьера
void barrier_wait(Barrier* barrier) {
    pthread_mutex_lock(&barrier->mutex);
    
    barrier->count++;
    int current_cycle = barrier->cycle;
    
    if (barrier->count == barrier->total) {
        // Последний поток достиг барьера
        barrier->count = 0;
        barrier->cycle++;
        pthread_cond_broadcast(&barrier->cond);
    } else {
        // Ожидание, пока не соберутся все потоки
        do {
            pthread_cond_wait(&barrier->cond, &barrier->mutex);
        } while (current_cycle == barrier->cycle);
    }
    
    pthread_mutex_unlock(&barrier->mutex);
}

// Структура для передачи данных в поток
typedef struct {
    Barrier* barrier;
    int id;
} ThreadData;

// Функция потока
void* thread_function(void* arg) {
    ThreadData* data = (ThreadData*)arg;
    
    printf("Поток %d начал работу\n", data->id);
    
    // Имитация работы
    usleep(100000 * (data->id + 1));  // спим разное время для каждого потока
    
    printf("Поток %d ожидает у барьера\n", data->id);
    
    // Ожидание у барьера
    barrier_wait(data->barrier);
    
    printf("Поток %d продолжает работу после барьера\n", data->id);
    
    // Имитация дополнительной работы
    usleep(50000 * (data->id + 1));
    
    printf("Поток %d завершает работу\n", data->id);
    
    return NULL;
}

// Пример использования
int main() {
    const int num_threads = 5;
    pthread_t threads[num_threads];
    Barrier barrier;
    
    // Инициализация барьера для num_threads потоков
    if (barrier_init(&barrier, num_threads) != 0) {
        fprintf(stderr, "Ошибка инициализации барьера\n");
        return 1;
    }
    
    // Подготовка данных для потоков
    ThreadData thread_data[num_threads];
    for (int i = 0; i < num_threads; i++) {
        thread_data[i].barrier = &barrier;
        thread_data[i].id = i;
    }
    
    // Создание потоков
    for (int i = 0; i < num_threads; i++) {
        if (pthread_create(&threads[i], NULL, thread_function, &thread_data[i]) != 0) {
            fprintf(stderr, "Ошибка создания потока %d\n", i);
            return 1;
        }
    }
    
    // Ожидание завершения всех потоков
    for (int i = 0; i < num_threads; i++) {
        pthread_join(threads[i], NULL);
    }
    
    // Уничтожение барьера
    if (barrier_destroy(&barrier) != 0) {
        fprintf(stderr, "Ошибка уничтожения барьера\n");
    }
    
    printf("Все потоки завершены\n");
    
    return 0;
}
