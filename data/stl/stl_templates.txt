<h2>Шаблонные функции</h2>

<p><strong>Категория:</strong> Шаблоны</p>

<h3>Описание</h3>
<p>Шаблонные функции позволяют создавать обобщённые функции, которые могут работать с различными типами данных. Компилятор генерирует специализированные версии функции для каждого типа, с которым используется шаблон. Это позволяет писать универсальный и повторно используемый код.</p>

<h3>Добавлен в стандарт:</h3>
<p>С самого начала C++ (до стандарта C++98)</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные концепции шаблонных функций и их реализация на C (для демонстрации аналогов):</p>

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;string.h&gt;

#define DEFINE_TEMPLATE_FUNCTION(name, type) \
    type##_##name

#define DEFINE_MAX_FUNC(type) \
    type DEFINE_TEMPLATE_FUNCTION(max, type)(type a, type b) { \
        return (a > b) ? a : b; \
    }

DEFINE_MAX_FUNC(int)
DEFINE_MAX_FUNC(double)
DEFINE_MAX_FUNC(float)

#define DEFINE_SWAP_FUNC(type) \
    void DEFINE_TEMPLATE_FUNCTION(swap, type)(type* a, type* b) { \
        type temp = *a; \
        *a = *b; \
        *b = temp; \
    }

DEFINE_SWAP_FUNC(int)
DEFINE_SWAP_FUNC(double)
DEFINE_SWAP_FUNC(char)

#define DEFINE_EQUAL_FUNC(type) \
    int DEFINE_TEMPLATE_FUNCTION(equal, type)(type a, type b) { \
        return a == b; \
    }

DEFINE_EQUAL_FUNC(int)
DEFINE_EQUAL_FUNC(double)
DEFINE_EQUAL_FUNC(char)

#define DEFINE_COPY_ARRAY_FUNC(type) \
    void DEFINE_TEMPLATE_FUNCTION(copy_array, type)(type* source, type* destination, int size) { \
        for (int i = 0; i < size; i++) { \
            destination[i] = source[i]; \
        } \
    }

DEFINE_COPY_ARRAY_FUNC(int)
DEFINE_COPY_ARRAY_FUNC(double)
DEFINE_COPY_ARRAY_FUNC(char)

#define DEFINE_FIND_FUNC(type) \
    int DEFINE_TEMPLATE_FUNCTION(find, type)(type* array, int size, type value) { \
        for (int i = 0; i < size; i++) { \
            if (DEFINE_TEMPLATE_FUNCTION(equal, type)(array[i], value)) { \
                return i; \
            } \
        } \
        return -1; \
    }

DEFINE_FIND_FUNC(int)
DEFINE_FIND_FUNC(double)
DEFINE_FIND_FUNC(char)

typedef struct {
    void* data;
    size_t size;
    size_t element_size;
    char type_name[20];
} GenericArray;

GenericArray* create_generic_array(size_t count, size_t element_size, const char* type_name) {
    GenericArray* arr = malloc(sizeof(GenericArray));
    arr->data = calloc(count, element_size);
    arr->size = count;
    arr->element_size = element_size;
    strncpy(arr->type_name, type_name, sizeof(arr->type_name) - 1);
    arr->type_name[sizeof(arr->type_name) - 1] = '\0';
    return arr;
}

void* get_element(GenericArray* arr, size_t index) {
    if (index >= arr->size)
        return NULL;
    return (char*)arr->data + (index * arr->element_size);
}

void set_element(GenericArray* arr, size_t index, void* value) {
    if (index >= arr->size)
        return;
    void* target = get_element(arr, index);
    memcpy(target, value, arr->element_size);
}

void destroy_generic_array(GenericArray* arr) {
    if (arr) {
        if (arr->data)
            free(arr->data);
        free(arr);
    }
}

int main() {
    printf("%d\n", int_max(10, 20));
    printf("%.2f\n", double_max(3.14, 2.71));

    int a = 10, b = 20;
    int_swap(&a, &b);

    int numbers[] = {1, 5, 3, 9, 7, 2};
    int idx = int_find(numbers, 6, 9);
    printf("%d\n", idx);

    GenericArray* gen_arr = create_generic_array(5, sizeof(int), "int");

    for (int i = 0; i < 5; i++) {
        int value = (i + 1) * 10;
        set_element(gen_arr, i, &value);
    }

    for (int i = 0; i < 5; i++) {
        int* value = (int*)get_element(gen_arr, i);
        printf("%d ", *value);
    }
    printf("\n");

    destroy_generic_array(gen_arr);

    return 0;
}
