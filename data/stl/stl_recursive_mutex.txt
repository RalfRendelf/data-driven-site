<h2>std::recursive_mutex</h2>

<p><strong>Категория:</strong> Многопоточность</p>

<h3>Описание</h3>
<p>std::recursive_mutex - это примитив синхронизации, который позволяет одному и тому же потоку захватывать мьютекс несколько раз подряд. В отличие от std::mutex, который может быть захвачен только один раз (иначе произойдет deadlock), recursive_mutex позволяет захватывать его рекурсивно, при этом требуется соответствующее количество вызовов unlock для полного освобождения.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++11</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::recursive_mutex и их реализация на C (для демонстрации аналогов):</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка POSIX threads
 * Компиляция: gcc -o recursive_mutex_example recursive_mutex_example.c -lpthread
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;pthread.h&gt;
#include &lt;unistd.h&gt;

// Структура для хранения информации о рекурсивном мьютексе
typedef struct {
    pthread_mutex_t mutex;
    pthread_t owner;      // ID потока-владельца
    int lock_count;       // количество захватов мьютекса
    int is_locked;        // состояние блокировки
} RecursiveMutex;

// Инициализация рекурсивного мьютекса
RecursiveMutex* recursive_mutex_create() {
    RecursiveMutex* rm = malloc(sizeof(RecursiveMutex));
    
    // Настройка атрибутов мьютекса для рекурсивного использования
    pthread_mutexattr_t attr;
    pthread_mutexattr_init(&attr);
    pthread_mutexattr_settype(&attr, PTHREAD_MUTEX_RECURSIVE);
    
    pthread_mutex_init(&rm->mutex, &attr);
    rm->owner = 0;
    rm->lock_count = 0;
    rm->is_locked = 0;
    
    pthread_mutexattr_destroy(&attr);
    return rm;
}

// Блокировка рекурсивного мьютекса (аналог lock())
void recursive_mutex_lock(RecursiveMutex* rm) {
    if (rm != NULL) {
        pthread_mutex_lock(&rm->mutex);
        
        // Устанавливаем владельца при первом захвате
        if (rm->lock_count == 0) {
            rm->owner = pthread_self();
        }
        rm->lock_count++;
        rm->is_locked = 1;
    }
}

// Освобождение рекурсивного мьютекса (аналог unlock())
void recursive_mutex_unlock(RecursiveMutex* rm) {
    if (rm != NULL && rm->lock_count > 0) {
        rm->lock_count--;
        if (rm->lock_count == 0) {
            rm->is_locked = 0;
        }
        pthread_mutex_unlock(&rm->mutex);
    }
}

// Попытка заблокировать рекурсивный мьютекс (аналог try_lock())
int recursive_mutex_try_lock(RecursiveMutex* rm) {
    if (rm != NULL) {
        int result = pthread_mutex_trylock(&rm->mutex);
        if (result == 0) {
            // Устанавливаем владельца при первом захвате
            if (rm->lock_count == 0) {
                rm->owner = pthread_self();
            }
            rm->lock_count++;
            rm->is_locked = 1;
            return 1;  // Успешно заблокирован
        }
    }
    return 0;  // Не удалось заблокировать
}

// Проверка, заблокирован ли мьютекс
int recursive_mutex_is_locked(RecursiveMutex* rm) {
    if (rm != NULL) {
        return rm->is_locked;
    }
    return 0;
}

// Получить количество захватов
int recursive_mutex_get_lock_count(RecursiveMutex* rm) {
    if (rm != NULL) {
        return rm->lock_count;
    }
    return 0;
}

// Уничтожение рекурсивного мьютекса
void recursive_mutex_destroy(RecursiveMutex* rm) {
    if (rm != NULL) {
        pthread_mutex_destroy(&rm->mutex);
        free(rm);
    }
}

// Общие данные для примера
typedef struct {
    int counter;
    RecursiveMutex* mutex;
} SharedData;

// Рекурсивная функция для демонстрации использования
void recursive_increment(SharedData* data, int level) {
    if (level <= 0) return;
    
    recursive_mutex_lock(data->mutex);
    printf("Уровень %d: захват мьютекса, количество захватов: %d\n", 
           level, recursive_mutex_get_lock_count(data->mutex));
    
    // Изменяем данные
    data->counter++;
    
    // Рекурсивный вызов
    recursive_increment(data, level - 1);
    
    printf("Уровень %d: освобождение мьютекса\n", level);
    recursive_mutex_unlock(data->mutex);
}

// Функция для выполнения в потоке
void* worker_function(void* arg) {
    SharedData* data = (SharedData*)arg;
    
    printf("Поток %lu начинает работу\n", pthread_self());
    
    // Вызов рекурсивной функции с 3 уровнями
    recursive_increment(data, 3);
    
    printf("Поток %lu завершил работу\n", pthread_self());
    return NULL;
}

// Пример использования
int main() {
    SharedData data;
    data.counter = 0;
    data.mutex = recursive_mutex_create();
    
    pthread_t thread1, thread2;
    
    // Создание двух потоков
    pthread_create(&thread1, NULL, worker_function, &data);
    pthread_create(&thread2, NULL, worker_function, &data);
    
    // Ожидание завершения потоков
    pthread_join(thread1, NULL);
    pthread_join(thread2, NULL);
    
    printf("Финальное значение счетчика: %d\n", data.counter);
    
    recursive_mutex_destroy(data.mutex);
    
    return 0;
}
