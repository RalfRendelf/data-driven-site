<h2>std::memory_resource</h2>

<p><strong>Категория:</strong> Умные указатели</p>

<h3>Описание</h3>
<p>std::memory_resource - это абстрактный класс, введенный в C++17 в рамках библиотеки pmr (polymorphic memory resources), позволяющий определять различные стратегии выделения и освобождения памяти. Он предоставляет интерфейс для пользовательских аллокаторов, что позволяет эффективно управлять памятью в соответствии с конкретными требованиями производительности.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++17</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::memory_resource:</p>

<pre>
<code>
// Пример использования std::memory_resource на C++

#include &lt;iostream&gt;
#include &lt;memory_resource&gt;
#include &lt;vector&gt;
#include &lt;string&gt;

// Пользовательский memory resource
class DebugMemoryResource : public std::pmr::memory_resource {
private:
    std::pmr::memory_resource* upstream_resource;
    
public:
    DebugMemoryResource(std::pmr::memory_resource* upstream = std::pmr::get_default_resource())
        : upstream_resource(upstream) {}
    
protected:
    void* do_allocate(std::size_t bytes, std::size_t alignment) override {
        std::cout << "Allocating " << bytes << " bytes with alignment " << alignment << std::endl;
        return upstream_resource->allocate(bytes, alignment);
    }
    
    void do_deallocate(void* ptr, std::size_t bytes, std::size_t alignment) override {
        std::cout << "Deallocating " << bytes << " bytes at " << ptr << std::endl;
        upstream_resource->deallocate(ptr, bytes, alignment);
    }
    
    bool do_is_equal(const std::pmr::memory_resource& other) const noexcept override {
        return this == &other;
    }
};

int main() {
    // Создаем пользовательский memory resource
    DebugMemoryResource debug_mr;
    
    // Используем его для создания polymorphic allocator
    std::pmr::polymorphic_allocator<int> alloc(&debug_mr);
    
    // Создаем вектор с использованием нашего аллокатора
    std::pmr::vector<int> vec(alloc);
    
    // Заполняем вектор
    for (int i = 0; i < 5; ++i) {
        vec.push_back(i * 10);
    }
    
    std::cout << "Vector contents: ";
    for (const auto& item : vec) {
        std::cout << item << " ";
    }
    std::cout << std::endl;
    
    // Пример использования pool_resource (C++17)
    std::pmr::synchronized_pool_resource pool_res;
    auto* pool_alloc = pool_res.allocate(100, alignof(std::max_align_t));
    std::cout << "Allocated from pool at: " << pool_alloc << std::endl;
    pool_res.deallocate(pool_alloc, 100, alignof(std::max_align_t));
    
    // Пример с monotonic_buffer_resource
    alignas(std::max_align_t) char buffer[1024];
    std::pmr::monotonic_buffer_resource mono_res(buffer, sizeof(buffer));
    
    std::pmr::vector<int> mono_vec(&mono_res);
    for (int i = 0; i < 10; ++i) {
        mono_vec.push_back(i);
    }
    
    std::cout << "Monotonic buffer vector: ";
    for (const auto& item : mono_vec) {
        std::cout << item << " ";
    }
    std::cout << std::endl;
    
    return 0;
}

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o memory_resource_example memory_resource_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;stdint.h&gt;
#include &lt;string.h&gt;

// Тип функции для выделения памяти
typedef void* (*allocate_func)(size_t bytes, size_t alignment);
// Тип функции для освобождения памяти
typedef void (*deallocate_func)(void* ptr, size_t bytes, size_t alignment);
// Тип функции для проверки равенства
typedef int (*is_equal_func)(void* resource1, void* resource2);

// Структура для memory_resource
typedef struct {
    allocate_func allocate;
    deallocate_func deallocate;
    is_equal_func is_equal;
    void* upstream_resource;
} MemoryResource;

// Реализация выравнивания
static inline size_t align_to(size_t value, size_t alignment) {
    return (value + alignment - 1) & ~(alignment - 1);
}

// Простая реализация выделения памяти с выравниванием
void* default_allocate(size_t bytes, size_t alignment) {
    // Выделяем дополнительную память для сохранения смещения
    size_t offset = sizeof(void*) + alignment - 1;
    void* raw_memory = malloc(bytes + offset);
    if (!raw_memory) return NULL;
    
    // Выравниваем указатель
    void* aligned_ptr = (void*)align_to(((uintptr_t)raw_memory + sizeof(void*)), alignment);
    void** header = (void**)aligned_ptr - 1;
    *header = raw_memory;  // Сохраняем указатель на исходную память
    
    return aligned_ptr;
}

// Реализация освобождения памяти
void default_deallocate(void* ptr, size_t bytes, size_t alignment) {
    if (!ptr) return;
    void** header = (void**)ptr - 1;
    free(*header);
}

// Реализация проверки равенства
int default_is_equal(void* resource1, void* resource2) {
    return resource1 == resource2;
}

// Создание стандартного ресурса памяти
MemoryResource* create_default_resource() {
    MemoryResource* mr = malloc(sizeof(MemoryResource));
    mr->allocate = default_allocate;
    mr->deallocate = default_deallocate;
    mr->is_equal = default_is_equal;
    mr->upstream_resource = NULL;
    return mr;
}

// Отладочный ресурс памяти
MemoryResource* create_debug_resource(MemoryResource* upstream) {
    MemoryResource* debug_mr = malloc(sizeof(MemoryResource));
    debug_mr->allocate = default_allocate;
    debug_mr->deallocate = default_deallocate;
    debug_mr->is_equal = default_is_equal;
    debug_mr->upstream_resource = upstream;
    return debug_mr;
}

// Функция для выделения с отладочной информацией
void* debug_allocate(MemoryResource* mr, size_t bytes, size_t alignment) {
    printf("Debug allocating %zu bytes with alignment %zu\n", bytes, alignment);
    return mr->allocate(bytes, alignment);
}

// Функция для освобождения с отладочной информацией
void debug_deallocate(MemoryResource* mr, void* ptr, size_t bytes, size_t alignment) {
    printf("Debug deallocating %zu bytes at %p\n", bytes, ptr);
    mr->deallocate(ptr, bytes, alignment);
}

// Простой вектор для демонстрации
typedef struct {
    MemoryResource* resource;
    int* data;
    size_t size;
    size_t capacity;
} IntVector;

// Инициализация вектора с указанным ресурсом
IntVector* create_vector(MemoryResource* mr) {
    IntVector* vec = mr->allocate(sizeof(IntVector), _Alignof(IntVector));
    vec->resource = mr;
    vec->size = 0;
    vec->capacity = 1;
    vec->data = mr->allocate(sizeof(int) * vec->capacity, _Alignof(int));
    return vec;
}

// Добавление элемента в вектор
void push_back(IntVector* vec, int value) {
    if (vec->size >= vec->capacity) {
        vec->capacity *= 2;
        int* new_data = vec->resource->allocate(sizeof(int) * vec->capacity, _Alignof(int));
        for (size_t i = 0; i < vec->size; i++) {
            new_data[i] = vec->data[i];
        }
        vec->resource->deallocate(vec->data, sizeof(int) * vec->capacity / 2, _Alignof(int));
        vec->data = new_data;
    }
    
    vec->data[vec->size] = value;
    vec->size++;
}

// Печать вектора
void print_vector(IntVector* vec, const char* label) {
    printf("%s: ", label);
    for (size_t i = 0; i < vec->size; i++) {
        printf("%d ", vec->data[i]);
    }
    printf("\n");
}

// Освобождение вектора
void destroy_vector(IntVector* vec) {
    if (vec) {
        vec->resource->deallocate(vec->data, sizeof(int) * vec->capacity, _Alignof(int));
        vec->resource->deallocate(vec, sizeof(IntVector), _Alignof(IntVector));
    }
}

// Пример использования
int main() {
    // Создаем ресурс памяти
    MemoryResource* default_mr = create_default_resource();
    MemoryResource* debug_mr = create_debug_resource(default_mr);
    
    // Создаем вектор с использованием отладочного ресурса
    IntVector* vec = create_vector(debug_mr);
    
    // Заполняем вектор
    for (int i = 0; i < 5; i++) {
        debug_deallocate(debug_mr, NULL, 0, 1); // просто для вызова debug функций
        push_back(vec, i * 10);
    }
    
    print_vector(vec, "Vector contents");
    
    // Освобождаем ресурсы
    destroy_vector(vec);
    free(debug_mr);
    free(default_mr);
    
    return 0;
}
