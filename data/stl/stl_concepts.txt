<h2>Concepts (C++20)</h2>

<p><strong>Категория:</strong> Язык</p>

<h3>Описание</h3>
<p>Concepts - это новая возможность в C++20, которая позволяет определять требования к типам данных в шаблонах. Concepts позволяют создавать более понятные, читаемые и отладочные шаблоны, а также улучшают сообщения об ошибках при несоответствии типов требованиям шаблона.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++20</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные концепции Concepts и их реализация на C (для демонстрации аналогов):</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o concepts_example concepts_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;string.h&gt;

// Макросы для определения "концептов" (ограничений на типы) в C
#define INTEGRAL_TYPES char, short, int, long, long long, unsigned char, unsigned short, unsigned int, unsigned long, unsigned long long
#define FLOATING_POINT_TYPES float, double, long double
#define ARITHMETIC_TYPES INTEGRAL_TYPES, FLOATING_POINT_TYPES

// Макрос для проверки, является ли тип арифметическим
#define IS_ARITHMETIC(type) _Generic((type){0}, \
    ARITHMETIC_TYPES: 1, \
    default: 0 \
)

// Макрос для проверки, является ли тип целочисленным
#define IS_INTEGRAL(type) _Generic((type){0}, \
    INTEGRAL_TYPES: 1, \
    default: 0 \
)

// Макрос для проверки, является ли тип числом с плавающей точкой
#define IS_FLOATING_POINT(type) _Generic((type){0}, \
    FLOATING_POINT_TYPES: 1, \
    default: 0 \
)

// Структура для хранения информации о типе
typedef struct {
    char type_name[50];
    int satisfies_arithmetic;
    int satisfies_integral;
    int satisfies_floating_point;
} ConceptCheckResult;

// Макрос для создания ConceptCheckResult
#define CHECK_CONCEPTS(value, result_name) \
    ConceptCheckResult result_name = {0}; \
    snprintf(result_name.type_name, sizeof(result_name.type_name), "%s", #value); \
    result_name.satisfies_arithmetic = IS_ARITHMETIC(value); \
    result_name.satisfies_integral = IS_INTEGRAL(value); \
    result_name.satisfies_floating_point = IS_FLOATING_POINT(value);

// Макрос для условного выполнения в зависимости от "удовлетворения концепту"
#define IF_SATISFIES_ARITHMETIC(value, arithmetic_action, non_arithmetic_action) \
    (IS_ARITHMETIC(value) ? arithmetic_action : non_arithmetic_action)

#define IF_SATISFIES_INTEGRAL(value, integral_action, non_integral_action) \
    (IS_INTEGRAL(value) ? integral_action : non_integral_action)

#define IF_SATISFIES_FLOATING_POINT(value, floating_action, non_floating_action) \
    (IS_FLOATING_POINT(value) ? floating_action : non_floating_action)

// Макрос для определения "функции с ограничениями" (аналог requires)
#define REQUIRE_ARITHMETIC_OPERATION(op, a, b) \
    _Generic((a), \
        ARITHMETIC_TYPES: _Generic((b), \
            ARITHMETIC_TYPES: op, \
            default: (printf("Ошибка: второй аргумент не арифметический тип\n"), 0) \
        ), \
        default: (printf("Ошибка: первый аргумент не арифметический тип\n"), 0) \
    )

// Пример функции, которая работает только с арифметическими типами
void concept_demo_arithmetic_op(int a, int b, const char* op_name) {
    printf("Операция %s: %d и %d\n", op_name, a, b);
}

void concept_demo_arithmetic_op_double(double a, double b, const char* op_name) {
    printf("Операция %s: %.2f и %.2f\n", op_name, a, b);
}

// Макрос для условного вызова функции в зависимости от типа
#define CALL_ARITHMETIC_OP(val1, val2, op_name) \
    _Generic((val1), \
        int: _Generic((val2), \
            int: concept_demo_arithmetic_op(val1, val2, op_name), \
            default: printf("Типы не совпадают для операции %s\n", op_name) \
        ), \
        double: _Generic((val2), \
            double: concept_demo_arithmetic_op_double(val1, val2, op_name), \
            default: printf("Типы не совпадают для операции %s\n", op_name) \
        ), \
        default: printf("Тип не поддерживается для операции %s\n", op_name) \
    )

// Структура для демонстрации "концепта" Comparable
typedef struct {
    int value;
} ComparableInt;

// Функция сравнения для Comparable
int comparable_less_than(ComparableInt a, ComparableInt b) {
    return a.value < b.value;
}

// Макрос для проверки, реализует ли тип операцию сравнения
#define IMPLEMENTS_LESS_THAN(type) _Generic((type){0}, \
    ComparableInt: 1, \
    default: 0 \
)

// Функция для демонстрации Concepts
void demonstrate_concepts() {
    printf("=== Демонстрация Concepts (аналог) ===\n\n");
    
    // Проверка различных типов на соответствие концептам
    int int_val = 42;
    double double_val = 3.14;
    char char_val = 'A';
    const char* str_val = "test";
    
    CHECK_CONCEPTS(int_val, int_result);
    CHECK_CONCEPTS(double_val, double_result);
    CHECK_CONCEPTS(char_val, char_result);
    // Для строковой константы проверим первый символ
    CHECK_CONCEPTS(*"test", str_result);
    
    printf("Результаты проверки концептов:\n");
    printf("int: arithmetic=%d, integral=%d, floating_point=%d\n", 
           int_result.satisfies_arithmetic, 
           int_result.satisfies_integral, 
           int_result.satisfies_floating_point);
           
    printf("double: arithmetic=%d, integral=%d, floating_point=%d\n", 
           double_result.satisfies_arithmetic, 
           double_result.satisfies_integral, 
           double_result.satisfies_floating_point);
           
    printf("char: arithmetic=%d, integral=%d, floating_point=%d\n", 
           char_result.satisfies_arithmetic, 
           char_result.satisfies_integral, 
           char_result.satisfies_floating_point);
    
    // Условное выполнение в зависимости от удовлетворения концептам
    printf("\nУсловное выполнение в зависимости от концептов:\n");
    
    IF_SATISFIES_ARITHMETIC(int_val, 
        printf("int - арифметический тип\n"),
        printf("int - НЕ арифметический тип\n"));
    
    IF_SATISFIES_INTEGRAL(int_val,
        printf("int - целочисленный тип\n"),
        printf("int - НЕ целочисленный тип\n"));
    
    IF_SATISFIES_FLOATING_POINT(double_val,
        printf("double - число с плавающей точкой\n"),
        printf("double - НЕ число с плавающей точкой\n"));
    
    // Демонстрация условных операций
    printf("\nДемонстрация условных операций:\n");
    CALL_ARITHMETIC_OP(10, 20, "сложение");
    CALL_ARITHMETIC_OP(5.5, 3.2, "умножение");
    
    // Проверка на реализацию операций
    printf("\nПроверка реализации операций:\n");
    ComparableInt comp1 = {10};
    ComparableInt comp2 = {20};
    
    if (IMPLEMENTS_LESS_THAN(comp1)) {
        printf("ComparableInt реализует операцию сравнения\n");
        printf("comp1 < comp2: %s\n", comparable_less_than(comp1, comp2) ? "да" : "нет");
    }
    
    // Имитация ограничений на шаблоны
    printf("\nИмитация ограничений на шаблоны (requires):\n");
    int a = 5, b = 10;
    double x = 2.5, y = 4.5;
    
    // Проверка арифметических операций
    REQUIRE_ARITHMETIC_OPERATION(
        printf("Операция для арифметических типов: %d + %d = %d\n", a, b, a+b),
        a, b
    );
    
    REQUIRE_ARITHMETIC_OPERATION(
        printf("Операция для арифметических типов: %.2f * %.2f = %.2f\n", x, y, x*y),
        x, y
    );
    
    printf("\nConcepts в C++20 позволяют:\n");
    printf("- Определять требования к типам данных\n");
    printf("- Улучшать сообщения об ошибках\n");
    printf("- Делать шаблоны более читаемыми и понятными\n");
    printf("- Создавать переиспользуемые ограничения на типы\n");
}

// Дополнительная информация о Concepts
void concepts_detailed_info() {
    printf("\n--- Подробнее о Concepts ---\n");
    printf("\nВ C++20 Concepts позволяют явно указывать требования к шаблонным параметрам:\n\n");
    
    printf("// Определение простого концепта\n");
    printf("template<typename T>\n");
    printf("concept Integral = std::is_integral_v<T>;\n\n");
    
    printf("// Использование концепта в шаблоне\n");
    printf("template<Integral T>\n");
    printf("T gcd(T a, T b) {\n");
    printf("    if (b == 0) return a;\n");
    printf("    return gcd(b, a % b);\n");
    printf("}\n\n");
    
    printf("// Альтернативный синтаксис с requires\n");
    printf("template<typename T>\n");
    printf("requires Integral<T>\n");
    printf("T gcd(T a, T b) { /* ... */ }\n\n");
    
    printf("// Или с inline requires\n");
    printf("template<typename T>\n");
    printf("T gcd(T a, T b) requires Integral<T> { /* ... */ }\n");
    
    printf("\nConcepts улучшают диагностику ошибок и делают код более самодокументированным.\n");
}

int main() {
    demonstrate_concepts();
    concepts_detailed_info();
    
    return 0;
}
