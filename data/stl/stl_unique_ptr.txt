<h2>std::unique_ptr</h2>

<p><strong>Категория:</strong> Умные указатели</p>

<h3>Описание</h3>
<p>std::unique_ptr — это умный указатель, обеспечивающий уникальное владение объектом. Он автоматически освобождает память при выходе из области видимости. Копирование запрещено, но перемещение разрешено.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++11</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::unique_ptr и их реализация на C (для демонстрации аналогов):</p>

<h4>C++</h4>
<pre>
<code>
#include &lt;memory&gt;
#include &lt;iostream&gt;

int main() {
    std::unique_ptr&lt;int&gt; p(new int(42));
    std::cout &lt;&lt; *p &lt;&lt; "\n";

    std::unique_ptr&lt;int&gt; p2 = std::move(p);
    if (!p)
        std::cout &lt;&lt; "p is null\n";

    std::cout &lt;&lt; *p2 &lt;&lt; "\n";

    p2.reset(new int(100));
    std::cout &lt;&lt; *p2 &lt;&lt; "\n";

    int* raw = p2.release();
    std::cout &lt;&lt; *raw &lt;&lt; "\n";
    delete raw;

    return 0;
}

<h4>C реализация</h4>
<pre>
<code>
#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;

typedef struct {
    void* ptr;
    void (*deleter)(void*);
} UniquePtr;

UniquePtr* unique_ptr_create(void* ptr) {
    UniquePtr* uptr = malloc(sizeof(UniquePtr));
    uptr->ptr = ptr;
    uptr->deleter = &free;
    return uptr;
}

UniquePtr* unique_ptr_create_with_deleter(void* ptr, void (*custom_deleter)(void*)) {
    UniquePtr* uptr = malloc(sizeof(UniquePtr));
    uptr->ptr = ptr;
    uptr->deleter = custom_deleter;
    return uptr;
}

void* unique_ptr_get(UniquePtr* uptr) {
    return uptr ? uptr->ptr : NULL;
}

void* unique_ptr_release(UniquePtr* uptr) {
    if (!uptr)
        return NULL;
    void* p = uptr->ptr;
    uptr->ptr = NULL;
    return p;
}

void* unique_ptr_reset(UniquePtr* uptr, void* new_ptr) {
    if (!uptr)
        return NULL;
    void* old = uptr->ptr;
    uptr->ptr = new_ptr;
    return old;
}

void unique_ptr_destroy(UniquePtr* uptr) {
    if (uptr) {
        if (uptr->ptr && uptr->deleter)
            uptr->deleter(uptr->ptr);
        free(uptr);
    }
}

int unique_ptr_is_null(UniquePtr* uptr) {
    return (!uptr || uptr->ptr == NULL);
}

void array_deleter(void* ptr) {
    free(ptr);
}

int main() {
    int* raw = malloc(sizeof(int));
    *raw = 42;

    UniquePtr* uptr = unique_ptr_create(raw);
    printf("%d\n", *(int*)unique_ptr_get(uptr));

    if (!unique_ptr_is_null(uptr))
        printf("not null\n");

    int* arr = malloc(5 * sizeof(int));
    for (int i = 0; i < 5; i++)
        arr[i] = i + 1;

    UniquePtr* arr_uptr = unique_ptr_create_with_deleter(arr, array_deleter);

    for (int i = 0; i < 5; i++)
        printf("%d ", ((int*)unique_ptr_get(arr_uptr))[i]);
    printf("\n");

    int* new_raw = malloc(sizeof(int));
    *new_raw = 100;

    void* old = unique_ptr_reset(uptr, new_raw);
    if (old) {
        printf("%d\n", *(int*)old);
        free(old);
    }

    printf("%d\n", *(int*)unique_ptr_get(uptr));

    unique_ptr_destroy(uptr);
    unique_ptr_destroy(arr_uptr);

    return 0;
}
</code>
