<h2>std::chrono</h2>

<p><strong>Категория:</strong> Временные библиотеки</p>

<h3>Описание</h3>
<p>std::chrono - это библиотека для работы со временем в C++, включая точки во времени, продолжительности и часы. Она была добавлена в C++11 и предоставляет типобезопасные и удобные средства для измерения времени, работы с таймерами и задержками.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++11</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::chrono и их реализация на C (для демонстрации аналогов):</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка POSIX и стандартной библиотеки C
 * Компиляция: gcc -o chrono_example chrono_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;time.h&gt;
#include &lt;sys/time.h&gt;
#include &lt;unistd.h&gt;

// Структура для представления продолжительности (duration)
typedef struct {
    long long count;
    long long period_numerator;   // числитель периода (например, 1 для секунд)
    long long period_denominator; // знаменатель периода (например, 1 для секунд, 1000000000 для наносекунд)
} Duration;

// Структура для представления точки во времени (time_point)
typedef struct {
    Duration duration_since_epoch;
} TimePoint;

// Структура для часов (clock)
typedef struct {
    TimePoint now;
} Clock;

// Создание продолжительности в наносекундах
Duration duration_ns(long long ns) {
    Duration d;
    d.count = ns;
    d.period_numerator = 1;
    d.period_denominator = 1000000000;  // 10^9 нс в секунде
    return d;
}

// Создание продолжительности в микросекундах
Duration duration_us(long long us) {
    Duration d;
    d.count = us;
    d.period_numerator = 1;
    d.period_denominator = 1000000;  // 10^6 мкс в секунде
    return d;
}

// Создание продолжительности в миллисекундах
Duration duration_ms(long long ms) {
    Duration d;
    d.count = ms;
    d.period_numerator = 1;
    d.period_denominator = 1000;  // 10^3 мс в секунде
    return d;
}

// Создание продолжительности в секундах
Duration duration_s(long long s) {
    Duration d;
    d.count = s;
    d.period_numerator = 1;
    d.period_denominator = 1;  // 1 секунда
    return d;
}

// Преобразование продолжительности в наносекунды
long long duration_to_ns(Duration* d) {
    if (!d) return 0;
    
    // Сначала вычисляем в базовых единицах (наносекундах)
    return d->count * d->period_numerator * 1000000000LL / d->period_denominator;
}

// Преобразование продолжительности в секунды
long long duration_to_s(Duration* d) {
    if (!d) return 0;
    
    return d->count * d->period_numerator / d->period_denominator;
}

// Сложение продолжительностей
Duration duration_add(Duration* d1, Duration* d2) {
    if (!d1 || !d2) {
        Duration zero = {0, 1, 1};
        return zero;
    }
    
    // Преобразуем обе продолжительности к общей базе (наносекунды)
    long long total_ns = duration_to_ns(d1) + duration_to_ns(d2);
    
    Duration result;
    result.count = total_ns;
    result.period_numerator = 1;
    result.period_denominator = 1000000000;
    return result;
}

// Получение текущего времени
TimePoint time_point_now() {
    struct timespec ts;
    clock_gettime(CLOCK_REALTIME, &ts);
    
    TimePoint tp;
    tp.duration_since_epoch = duration_ns(ts.tv_sec * 1000000000LL + ts.tv_nsec);
    return tp;
}

// Получение разницы между двумя точками во времени
Duration time_point_diff(TimePoint* tp1, TimePoint* tp2) {
    if (!tp1 || !tp2) {
        Duration zero = {0, 1, 1};
        return zero;
    }
    
    long long diff_ns = duration_to_ns(&tp2->duration_since_epoch) - 
                        duration_to_ns(&tp1->duration_since_epoch);
    
    Duration diff;
    diff.count = diff_ns;
    diff.period_numerator = 1;
    diff.period_denominator = 1000000000;
    return diff;
}

// Добавление продолжительности к точке во времени
TimePoint time_point_add(TimePoint* tp, Duration* d) {
    if (!tp || !d) {
        TimePoint zero = {{0, 1, 1}};
        return zero;
    }
    
    TimePoint result = *tp;
    long long base_ns = duration_to_ns(&result.duration_since_epoch);
    long long add_ns = duration_to_ns(d);
    
    result.duration_since_epoch = duration_ns(base_ns + add_ns);
    return result;
}

// Сравнение точек во времени
int time_point_compare(TimePoint* tp1, TimePoint* tp2) {
    if (!tp1 || !tp2) return 0;
    
    long long ns1 = duration_to_ns(&tp1->duration_since_epoch);
    long long ns2 = duration_to_ns(&tp2->duration_since_epoch);
    
    if (ns1 < ns2) return -1;
    else if (ns1 > ns2) return 1;
    else return 0;
}

// Печать продолжительности
void duration_print(Duration* d, const char* label) {
    if (!d) return;
    
    printf("%s: %lld.%09lld секунд (%lld нс)\n", 
           label,
           duration_to_s(d),
           duration_to_ns(d) % 1000000000LL,
           duration_to_ns(d));
}

// Печать точки во времени
void time_point_print(TimePoint* tp, const char* label) {
    if (!tp) return;
    
    long long total_ns = duration_to_ns(&tp->duration_since_epoch);
    long long secs = total_ns / 1000000000LL;
    long long nsecs = total_ns % 1000000000LL;
    
    printf("%s: %lld.%09lld секунд с эпохи\n", label, secs, nsecs);
}

// Пример использования
int main() {
    printf("=== Демонстрация библиотеки времени (аналог std::chrono) ===\n\n");
    
    // Получаем текущее время
    TimePoint start = time_point_now();
    time_point_print(&start, "Время начала");
    
    // Ждем немного
    printf("Ждем 1.5 секунды...\n");
    usleep(1500000); // 1.5 секунды
    
    TimePoint end = time_point_now();
    time_point_print(&end, "Время окончания");
    
    // Вычисляем разницу
    Duration elapsed = time_point_diff(&start, &end);
    duration_print(&elapsed, "Прошедшее время");
    
    // Работа с разными единицами времени
    Duration sec = duration_s(5);
    Duration milli = duration_ms(500);
    Duration micro = duration_us(250);
    Duration nano = duration_ns(100);
    
    duration_print(&sec, "5 секунд");
    duration_print(&milli, "500 миллисекунд");
    duration_print(&micro, "250 микросекунд");
    duration_print(&nano, "100 наносекунд");
    
    // Сложение продолжительностей
    Duration sum1 = duration_add(&sec, &milli);
    Duration sum2 = duration_add(&sum1, &micro);
    Duration total = duration_add(&sum2, &nano);
    
    duration_print(&total, "Суммарная продолжительность");
    
    // Добавление продолжительности к точке во времени
    TimePoint future = time_point_add(&end, &total);
    time_point_print(&future, "Будущее время");
    
    // Сравнение точек во времени
    int cmp_result = time_point_compare(&start, &end);
    printf("\nСравнение start и end: ");
    if (cmp_result < 0) printf("start < end\n");
    else if (cmp_result > 0) printf("start > end\n");
    else printf("start == end\n");
    
    return 0;
}
