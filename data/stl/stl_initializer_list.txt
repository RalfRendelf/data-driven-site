<h2>std::initializer_list</h2>

<p><strong>Категория:</strong> Контейнеры</p>

<h3>Описание</h3>
<p>std::initializer_list - это класс шаблона, который предоставляет способ передачи списка инициализации в функцию или конструктор. Введен в C++11 для поддержки uniform initialization (унифицированной инициализации).</p>

<h3>Добавлен в стандарт:</h3>
<p>C++11</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::initializer_list:</p>

<pre>
<code>
// Пример использования std::initializer_list на C++

#include &lt;iostream&gt;
#include &lt;initializer_list&gt;
#include &lt;vector&gt;
#include &lt;numeric&gt;

class MyContainer {
private:
    std::vector&lt;int&gt; data;
    
public:
    // Конструктор, принимающий initializer_list
    MyContainer(std::initializer_list&lt;int&gt; init_list) {
        std::cout << "Creating container with initializer list: ";
        for (auto it = init_list.begin(); it != init_list.end(); ++it) {
            std::cout << *it << " ";
            data.push_back(*it);
        }
        std::cout << std::endl;
    }
    
    // Метод для добавления элементов из initializer_list
    void add_elements(std::initializer_list&lt;int&gt; new_elements) {
        std::cout << "Adding elements: ";
        for (int elem : new_elements) {
            std::cout << elem << " ";
            data.push_back(elem);
        }
        std::cout << std::endl;
    }
    
    // Метод для вычисления суммы элементов
    int sum() const {
        return std::accumulate(data.begin(), data.end(), 0);
    }
    
    // Метод для печати элементов
    void print() const {
        std::cout << "Container contents: ";
        for (int elem : data) {
            std::cout << elem << " ";
        }
        std::cout << std::endl;
    }
    
    size_t size() const { return data.size(); }
};

// Функция, принимающая initializer_list
int sum_list(std::initializer_list&lt;int&gt; il) {
    return std::accumulate(il.begin(), il.end(), 0);
}

int main() {
    // Создание контейнера с использованием initializer_list
    MyContainer container{1, 2, 3, 4, 5};
    container.print();
    
    std::cout << "Sum of elements: " << container.sum() << std::endl;
    
    // Добавление элементов с использованием initializer_list
    container.add_elements({10, 20, 30});
    container.print();
    
    std::cout << "Sum after adding: " << container.sum() << std::endl;
    
    // Использование функции с initializer_list
    int total = sum_list({1, 2, 3, 4, 5});
    std::cout << "Sum of list {1, 2, 3, 4, 5}: " << total << std::endl;
    
    // Использование в векторе (C++11 и выше)
    std::vector&lt;int&gt; vec{10, 20, 30, 40, 50};
    std::cout << "Vector from initializer_list: ";
    for (int n : vec) {
        std::cout << n << " ";
    }
    std::cout << std::endl;
    
    return 0;
}
</code>

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o initializer_list_example initializer_list_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;stdarg.h&gt;

// Структура для имитации initializer_list
typedef struct {
    int* data;
    size_t size;
} IntList;

// Функция для создания IntList из переменного числа аргументов
IntList create_int_list(size_t count, ...) {
    IntList list;
    list.size = count;
    list.data = malloc(count * sizeof(int));
    
    va_list args;
    va_start(args, count);
    
    for (size_t i = 0; i < count; i++) {
        list.data[i] = va_arg(args, int);
    }
    
    va_end(args);
    
    return list;
}

// Структура для контейнера
typedef struct {
    int* data;
    size_t size;
    size_t capacity;
} MyContainer;

// Функция для инициализации контейнера с помощью списка инициализации
MyContainer* create_container_from_list(IntList init_list) {
    MyContainer* container = malloc(sizeof(MyContainer));
    container->size = init_list.size;
    container->capacity = init_list.size;
    container->data = malloc(container->capacity * sizeof(int));
    
    printf("Creating container with initializer list: ");
    for (size_t i = 0; i < init_list.size; i++) {
        container->data[i] = init_list.data[i];
        printf("%d ", init_list.data[i]);
    }
    printf("\n");
    
    return container;
}

// Функция для добавления элементов из списка
void add_elements(MyContainer* container, IntList new_elements) {
    // Расширяем массив при необходимости
    if (container->size + new_elements.size > container->capacity) {
        container->capacity = container->size + new_elements.size;
        container->data = realloc(container->data, container->capacity * sizeof(int));
    }
    
    printf("Adding elements: ");
    for (size_t i = 0; i < new_elements.size; i++) {
        container->data[container->size + i] = new_elements.data[i];
        printf("%d ", new_elements.data[i]);
    }
    container->size += new_elements.size;
    printf("\n");
}

// Функция для вычисления суммы элементов
int sum_container(MyContainer* container) {
    int sum = 0;
    for (size_t i = 0; i < container->size; i++) {
        sum += container->data[i];
    }
    return sum;
}

// Функция для печати элементов
void print_container(MyContainer* container) {
    printf("Container contents: ");
    for (size_t i = 0; i < container->size; i++) {
        printf("%d ", container->data[i]);
    }
    printf("\n");
}

// Функция для освобождения памяти
void destroy_container(MyContainer* container) {
    if (container) {
        if (container->data) {
            free(container->data);
        }
        free(container);
    }
}

// Функция для вычисления суммы из списка
int sum_list(IntList list) {
    int sum = 0;
    for (size_t i = 0; i < list.size; i++) {
        sum += list.data[i];
    }
    return sum;
}

// Пример использования
int main() {
    // Создаем список инициализации
    IntList init_list = create_int_list(5, 1, 2, 3, 4, 5);
    
    // Создаем контейнер с использованием списка инициализации
    MyContainer* container = create_container_from_list(init_list);
    print_container(container);
    
    printf("Sum of elements: %d\n", sum_container(container));
    
    // Создаем новый список для добавления
    IntList new_elements = create_int_list(3, 10, 20, 30);
    
    // Добавляем элементы
    add_elements(container, new_elements);
    print_container(container);
    
    printf("Sum after adding: %d\n", sum_container(container));
    
    // Использование функции с "списком инициализации"
    IntList calc_list = create_int_list(5, 1, 2, 3, 4, 5);
    int total = sum_list(calc_list);
    printf("Sum of list {1, 2, 3, 4, 5}: %d\n", total);
    
    // Освобождаем память
    free(init_list.data);
    free(new_elements.data);
    free(calc_list.data);
    destroy_container(container);
    
    return 0;
}
