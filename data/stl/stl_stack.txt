<h2>std::stack</h2>

<p><strong>Категория:</strong> Контейнеры</p>

<h3>Описание</h3>
<p>std::stack - это контейнер адаптер, реализующий структуру данных LIFO (Last In, First Out). Элементы добавляются и удаляются с одного конца, называемого вершиной стека. По умолчанию использует std::deque в качестве базового контейнера.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++98</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::stack:</p>

<pre>
<code>
// Пример использования std::stack на C++

#include &lt;iostream&gt;
#include &lt;stack&gt;
#include &lt;vector&gt;

int main() {
    // Создание стека
    std::stack&lt;int&gt; s;
    
    // Добавление элементов в стек
    s.push(1);
    s.push(2);
    s.push(3);
    s.push(4);
    
    std::cout &lt;&lt; "Размер стека: " &lt;&lt; s.size() &lt;&lt; std::endl;
    
    // Извлечение элементов из стека
    std::cout &lt;&lt; "Элементы стека: ";
    while (!s.empty()) {
        std::cout &lt;&lt; s.top() &lt;&lt; " ";  // Просмотр верхнего элемента
        s.pop();                      // Удаление верхнего элемента
    }
    std::cout &lt;&lt; std::endl;
    
    // Проверка на пустоту
    std::cout &lt;&lt; "Стек пуст: " &lt;&lt; (s.empty() ? "да" : "нет") &lt;&lt; std::endl;
    
    // Добавление элементов
    for (int i = 1; i &lt;= 5; ++i) {
        s.push(i * 10);
    }
    
    // Просмотр верхнего элемента
    std::cout &lt;&lt; "Верхний элемент: " &lt;&lt; s.top() &lt;&lt; std::endl;
    
    // Размер стека
    std::cout &lt;&lt; "Размер стека: " &lt;&lt; s.size() &lt;&lt; std::endl;
    
    return 0;
}
</code>

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o stack_example stack_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;

#define INITIAL_CAPACITY 8

// Структура для стека
typedef struct {
    int* data;
    size_t top;
    size_t capacity;
} Stack;

// Инициализация стека
Stack* stack_create() {
    Stack* s = malloc(sizeof(Stack));
    s->data = malloc(INITIAL_CAPACITY * sizeof(int));
    s->top = 0;
    s->capacity = INITIAL_CAPACITY;
    return s;
}

// Уничтожение стека
void stack_destroy(Stack* s) {
    if (!s) return;
    if (s->data) free(s->data);
    free(s);
}

// Проверка на пустоту
int stack_empty(Stack* s) {
    return s ? (s->top == 0) : 1;
}

// Получение размера
size_t stack_size(Stack* s) {
    return s ? s->top : 0;
}

// Добавление элемента в стек
void stack_push(Stack* s, int value) {
    if (!s) return;
    
    // Увеличение размера, если необходимо
    if (s->top >= s->capacity) {
        s->capacity *= 2;
        int* new_data = realloc(s->data, s->capacity * sizeof(int));
        if (!new_data) return; // ошибка выделения памяти
        s->data = new_data;
    }
    
    s->data[s->top] = value;
    s->top++;
}

// Удаление верхнего элемента
void stack_pop(Stack* s) {
    if (!s || s->top == 0) return;
    
    s->top--;
}

// Получение верхнего элемента
int stack_top(Stack* s) {
    if (!s || s->top == 0) return 0; // или другое значение по умолчанию
    return s->data[s->top - 1];
}

// Печать стека
void stack_print(Stack* s, const char* label) {
    printf("%s: [", label);
    if (!stack_empty(s)) {
        for (size_t i = 0; i &lt; s->top; i++) {
            printf("%d", s->data[i]);
            if (i &lt; s->top - 1) printf(", ");
        }
    }
    printf("] (top: %zu, capacity: %zu)\n", s->top, s->capacity);
}

// Пример использования
int main() {
    Stack* s = stack_create();
    
    // Добавление элементов в стек
    stack_push(s, 1);
    stack_push(s, 2);
    stack_push(s, 3);
    stack_push(s, 4);
    
    stack_print(s, "Стек после добавления 1, 2, 3, 4");
    printf("Размер стека: %zu\n", stack_size(s));
    
    // Извлечение элементов из стека
    printf("Элементы стека: ");
    while (!stack_empty(s)) {
        printf("%d ", stack_top(s)); // Просмотр верхнего элемента
        stack_pop(s);               // Удаление верхнего элемента
    }
    printf("\n");
    
    // Проверка на пустоту
    printf("Стек пуст: %s\n", stack_empty(s) ? "да" : "нет");
    
    // Добавление новых элементов
    for (int i = 1; i &lt;= 5; ++i) {
        stack_push(s, i * 10);
    }
    
    stack_print(s, "Стек после добавления 10, 20, 30, 40, 50");
    
    // Просмотр верхнего элемента
    printf("Верхний элемент: %d\n", stack_top(s));
    
    // Размер стека
    printf("Размер стека: %zu\n", stack_size(s));
    
    // Уничтожение стека
    stack_destroy(s);
    
    return 0;
}
</code>
