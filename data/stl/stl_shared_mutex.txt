<h2>std::shared_mutex</h2>

<p><strong>Категория:</strong> Многопоточность</p>

<h3>Описание</h3>
<p>std::shared_mutex - это примитив синхронизации, который позволяет нескольким потокам одновременно читать разделяемые данные (в режиме shared lock), но только одному потоку писать в эти данные (в режиме exclusive lock). Это полезно для ситуаций, когда чтение происходит чаще, чем запись.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++17</p>

<h3>Пример использования на C++:</h3>
<p>В этом примере демонстрируется использование std::shared_mutex в классе SharedData:</p>

<pre>
<code>
#include &lt;iostream&gt;
#include &lt;thread&gt;
#include &lt;shared_mutex&gt;
#include &lt;vector&gt;
#include &lt;chrono&gt;

class SharedData {
public:
    mutable std::shared_mutex mutex;
    std::vector&lt;int&gt; data;
    
    SharedData() : data({1, 2, 3, 4, 5}) {}
    
    // Метод для чтения данных (множество потоков могут читать одновременно)
    std::vector&lt;int&gt; read_data() const {
        std::shared_lock&lt;std::shared_mutex&gt; lock(mutex);
        std::cout &lt;&lt; "Чтение данных потоком " &lt;&lt; std::this_thread::get_id() &lt;&lt; std::endl;
        
        // Имитация работы при чтении
        std::this_thread::sleep_for(std::chrono::milliseconds(100));
        
        return data;
    }
    
    // Метод для записи данных (только один поток может писать)
    void write_data(const std::vector&lt;int&gt;& new_data) {
        std::unique_lock&lt;std::shared_mutex&gt; lock(mutex);
        std::cout &lt;&lt; "Запись данных потоком " &lt;&lt; std::this_thread::get_id() &lt;&lt; std::endl;
        
        // Изменение данных
        data = new_data;
        
        // Имитация работы при записи
        std::this_thread::sleep_for(std::chrono::milliseconds(200));
        
        std::cout &lt;&lt; "Запись данных завершена потоком " &lt;&lt; std::this_thread::get_id() &lt;&lt; std::endl;
    }
};

SharedData shared_data;

void reader_function(int thread_id) {
    for (int i = 0; i &lt; 3; i++) {
        auto data = shared_data.read_data();
        std::cout &lt;&lt; "Reader " &lt;&lt; thread_id &lt;&lt; " read: ";
        for (int val : data) {
            std::cout &lt;&lt; val &lt;&lt; " ";
        }
        std::cout &lt;&lt; std::endl;
        std::this_thread::sleep_for(std::chrono::milliseconds(200));
    }
}

void writer_function(int thread_id) {
    for (int i = 0; i &lt; 2; i++) {
        std::vector&lt;int&gt; new_data = {(thread_id + 1) * 10, (thread_id + 1) * 20, (thread_id + 1) * 30};
        shared_data.write_data(new_data);
        std::this_thread::sleep_for(std::chrono::milliseconds(300));
    }
}

int main() {
    std::vector&lt;std::thread&gt; threads;
    
    // Создание потоков-читателей
    for (int i = 0; i &lt; 3; i++) {
        threads.emplace_back(reader_function, i);
    }
    
    // Создание потоков-писателей
    for (int i = 0; i &lt; 2; i++) {
        threads.emplace_back(writer_function, i);
    }
    
    // Ожидание завершения всех потоков
    for (auto&amp; t : threads) {
        t.join();
    }
    
    std::cout &lt;&lt; "Программа завершена" &lt;&lt; std::endl;
    
    return 0;
}

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C с использованием pthread rwlock:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка POSIX threads
 * Компиляция: gcc -o shared_mutex_example shared_mutex_example.c -lpthread
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;pthread.h&gt;
#include &lt;unistd.h&gt;
#include &lt;time.h&gt;

// Структура для хранения информации о shared_mutex
typedef struct {
    pthread_rwlock_t rwlock;
} SharedMutex;

// Инициализация shared_mutex
SharedMutex* shared_mutex_create() {
    SharedMutex* sm = malloc(sizeof(SharedMutex));
    pthread_rwlock_init(&sm->rwlock, NULL);
    return sm;
}

// Уничтожение shared_mutex
void shared_mutex_destroy(SharedMutex* sm) {
    if (sm != NULL) {
        pthread_rwlock_destroy(&sm->rwlock);
        free(sm);
    }
}

// Установка shared lock (для чтения)
void shared_mutex_lock_shared(SharedMutex* sm) {
    if (sm != NULL) {
        pthread_rwlock_rdlock(&sm->rwlock);
    }
}

// Освобождение shared lock
void shared_mutex_unlock_shared(SharedMutex* sm) {
    if (sm != NULL) {
        pthread_rwlock_unlock(&sm->rwlock);
    }
}

// Установка exclusive lock (для записи)
void shared_mutex_lock(SharedMutex* sm) {
    if (sm != NULL) {
        pthread_rwlock_wrlock(&sm->rwlock);
    }
}

// Освобождение exclusive lock
void shared_mutex_unlock(SharedMutex* sm) {
    if (sm != NULL) {
        pthread_rwlock_unlock(&sm->rwlock);
    }
}

// Попытка установки shared lock (неблокирующая)
int shared_mutex_try_lock_shared(SharedMutex* sm) {
    if (sm != NULL) {
        int result = pthread_rwlock_tryrdlock(&sm->rwlock);
        return (result == 0) ? 1 : 0;  // 1 - успех, 0 - неудача
    }
    return 0;
}

// Попытка установки exclusive lock (неблокирующая)
int shared_mutex_try_lock(SharedMutex* sm) {
    if (sm != NULL) {
        int result = pthread_rwlock_trywrlock(&sm->rwlock);
        return (result == 0) ? 1 : 0;  // 1 - успех, 0 - неудача
    }
    return 0;
}

// Структура для разделяемых данных
typedef struct {
    int* data;
    size_t size;
    SharedMutex* mutex;
} SharedData;

// Функция для чтения данных (множество потоков могут читать одновременно)
void* reader_function(void* arg) {
    SharedData* shared_data = (SharedData*)arg;
    int thread_id = *(int*)arg;
    
    for (int i = 0; i &lt; 3; i++) {
        // Установка shared lock для чтения
        shared_mutex_lock_shared(shared_data->mutex);
        
        printf("Чтение: Поток %d читает данные: ", thread_id);
        for (size_t j = 0; j &lt; shared_data->size; j++) {
            printf("%d ", shared_data->data[j]);
        }
        printf("\n");
        
        // Имитация работы при чтении
        usleep(100000); // 0.1 секунды
        
        // Освобождение shared lock
        shared_mutex_unlock_shared(shared_data->mutex);
        
        // Пауза между чтениями
        usleep(200000); // 0.2 секунды
    }
    
    return NULL;
}

// Функция для записи данных (только один поток может писать)
void* writer_function(void* arg) {
    SharedData* shared_data = (SharedData*)arg;
    int thread_id = *(int*)((int*)arg + 1); // второй элемент - ID писателя
    
    for (int i = 0; i &lt; 2; i++) {
        // Установка exclusive lock для записи
        shared_mutex_lock(shared_data->mutex);
        
        printf("Запись: Поток %d изменяет данные\n", thread_id);
        
        // Изменение данных
        for (size_t j = 0; j &lt; shared_data->size; j++) {
            shared_data->data[j] = (thread_id + 1) * (j + 1) * 10;
        }
        
        // Имитация работы при записи
        usleep(200000); // 0.2 секунды
        
        printf("Запись: Поток %d завершил изменение данных\n", thread_id);
        
        // Освобождение exclusive lock
        shared_mutex_unlock(shared_data->mutex);
        
        // Пауза между записями
        usleep(300000); // 0.3 секунды
    }
    
    return NULL;
}

// Пример использования
int main() {
    // Инициализация разделяемых данных
    SharedData shared_data;
    shared_data.size = 5;
    shared_data.data = malloc(shared_data.size * sizeof(int));
    shared_data.mutex = shared_mutex_create();
    
    // Инициализация начальных данных
    for (size_t i = 0; i &lt; shared_data.size; i++) {
        shared_data.data[i] = i + 1;
    }
    
    // Создание потоков
    pthread_t readers[3];
    pthread_t writers[2];
    int reader_ids[3] = {1, 2, 3};
    int writer_ids[2] = {1, 2};
    
    // Запуск потоков-читателей
    for (int i = 0; i &lt; 3; i++) {
        pthread_create(&readers[i], NULL, reader_function, &reader_ids[i]);
    }
    
    // Запуск потоков-писателей
    for (int i = 0; i &lt; 2; i++) {
        pthread_create(&writers[i], NULL, writer_function, &writer_ids[i]);
    }
    
    // Ожидание завершения всех потоков
    for (int i = 0; i &lt; 3; i++) {
        pthread_join(readers[i], NULL);
    }
    
    for (int i = 0; i &lt; 2; i++) {
        pthread_join(writers[i], NULL);
    }
    
    printf("Финальные данные: ");
    for (size_t i = 0; i &lt; shared_data.size; i++) {
        printf("%d ", shared_data.data[i]);
    }
    printf("\n");
    
    // Освобождение ресурсов
    free(shared_data.data);
    shared_mutex_destroy(shared_data.mutex);
    
    printf("Программа завершена\n");
    
    return 0;
}
