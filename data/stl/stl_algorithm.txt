<h2>std::algorithm</h2>

<p><strong>Категория:</strong> Алгоритмы</p>

<h3>Описание</h3>
<p>std::algorithm - это библиотека алгоритмов, входящая в состав стандартной библиотеки C++. Она предоставляет набор функций для выполнения различных операций с элементами контейнеров, таких как поиск, сортировка, манипуляции с элементами и другие.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++98 (часть алгоритмов), остальные добавлены в последующих версиях</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::algorithm:</p>

<pre>
<code>
// Пример использования std::algorithm на C++

#include &lt;iostream&gt;
#include &lt;vector&gt;
#include &lt;algorithm&gt;
#include &lt;numeric&gt;

int main() {
    std::vector&lt;int&gt; vec = {5, 2, 8, 1, 9, 3, 7, 4, 6};
    
    std::cout << "Исходный вектор: ";
    for (const auto&amp; element : vec) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    // Сортировка
    std::sort(vec.begin(), vec.end());
    std::cout << "Отсортированный вектор: ";
    for (const auto&amp; element : vec) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    // Поиск
    auto it = std::find(vec.begin(), vec.end(), 7);
    if (it != vec.end()) {
        std::cout << "Найден элемент 7 на позиции: " << (it - vec.begin()) << std::endl;
    }
    
    // Максимальный и минимальный элементы
    auto minmax = std::minmax_element(vec.begin(), vec.end());
    std::cout << "Минимальный элемент: " << *minmax.first << std::endl;
    std::cout << "Максимальный элемент: " << *minmax.second << std::endl;
    
    // Подсчет элементов, удовлетворяющих условию
    int count = std::count_if(vec.begin(), vec.end(), [](int x) { return x > 5; });
    std::cout << "Количество элементов > 5: " << count << std::endl;
    
    // Заполнение значениями
    std::fill(vec.begin(), vec.end(), 10);
    std::cout << "После fill(10): ";
    for (const auto&amp; element : vec) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    // Обратный порядок
    std::reverse(vec.begin(), vec.end());
    std::cout << "В обратном порядке: ";
    for (const auto&amp; element : vec) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    // Перемешивание
    std::random_shuffle(vec.begin(), vec.end());  // deprecated, лучше использовать std::shuffle
    std::cout << "После перемешивания: ";
    for (const auto&amp; element : vec) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    // Суммирование
    int sum = std::accumulate(vec.begin(), vec.end(), 0);
    std::cout << "Сумма элементов: " << sum << std::endl;
    
    // Копирование
    std::vector&lt;int&gt; vec2(vec.size());
    std::copy(vec.begin(), vec.end(), vec2.begin());
    std::cout << "Скопированный вектор: ";
    for (const auto&amp; element : vec2) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    return 0;
}

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o algorithm_example algorithm_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;time.h&gt;

// Функция сравнения для qsort
int compare_ints(const void* a, const void* b) {
    int int_a = *(int*)a;
    int int_b = *(int*)b;

    if (int_a == int_b) return 0;
    else if (int_a < int_b) return -1;
    else return 1;
}

// Функция сортировки (аналог std::sort)
void sort_array(int* arr, size_t size) {
    qsort(arr, size, sizeof(int), compare_ints);
}

// Функция поиска (аналог std::find)
int* find_element(int* arr, size_t size, int target) {
    for (size_t i = 0; i < size; i++) {
        if (arr[i] == target) {
            return &arr[i];
        }
    }
    return NULL;
}

// Функция поиска минимума и максимума (аналог std::minmax_element)
void minmax_element(int* arr, size_t size, int** min, int** max) {
    if (size == 0) {
        *min = NULL;
        *max = NULL;
        return;
    }
    
    *min = &arr[0];
    *max = &arr[0];
    
    for (size_t i = 1; i < size; i++) {
        if (arr[i] < **min) {
            *min = &arr[i];
        }
        if (arr[i] > **max) {
            *max = &arr[i];
        }
    }
}

// Функция подсчета элементов по условию (аналог std::count_if)
size_t count_if(int* arr, size_t size, int (*predicate)(int)) {
    size_t count = 0;
    for (size_t i = 0; i < size; i++) {
        if (predicate(arr[i])) {
            count++;
        }
    }
    return count;
}

// Функция заполнения (аналог std::fill)
void fill_array(int* arr, size_t size, int value) {
    for (size_t i = 0; i < size; i++) {
        arr[i] = value;
    }
}

// Функция реверса (аналог std::reverse)
void reverse_array(int* arr, size_t size) {
    for (size_t i = 0; i < size / 2; i++) {
        int temp = arr[i];
        arr[i] = arr[size - 1 - i];
        arr[size - 1 - i] = temp;
    }
}

// Функция перемешивания (аналог std::shuffle)
void shuffle_array(int* arr, size_t size) {
    for (size_t i = size - 1; i > 0; i--) {
        size_t j = rand() % (i + 1);
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
}

// Функция суммирования (аналог std::accumulate)
int accumulate_array(int* arr, size_t size, int initial_value) {
    int sum = initial_value;
    for (size_t i = 0; i < size; i++) {
        sum += arr[i];
    }
    return sum;
}

// Функция копирования (аналог std::copy)
void copy_array(int* src, int* dest, size_t size) {
    for (size_t i = 0; i < size; i++) {
        dest[i] = src[i];
    }
}

// Предикат для подсчета элементов > 5
int greater_than_5(int x) {
    return x > 5 ? 1 : 0;
}

// Печать массива
void print_array(int* arr, size_t size, const char* label) {
    printf("%s: [", label);
    for (size_t i = 0; i < size; i++) {
        printf("%d", arr[i]);
        if (i < size - 1) printf(", ");
    }
    printf("]\n");
}

// Пример использования
int main() {
    srand(time(NULL)); // инициализация генератора случайных чисел
    
    int arr[] = {5, 2, 8, 1, 9, 3, 7, 4, 6};
    size_t size = sizeof(arr) / sizeof(arr[0]);
    
    print_array(arr, size, "Исходный массив");
    
    // Сортировка
    sort_array(arr, size);
    print_array(arr, size, "Отсортированный массив");
    
    // Поиск
    int* found = find_element(arr, size, 7);
    if (found != NULL) {
        printf("Найден элемент 7 на позиции: %ld\n", found - arr);
    }
    
    // Максимальный и минимальный элементы
    int* min, *max;
    minmax_element(arr, size, &min, &max);
    printf("Минимальный элемент: %d\n", *min);
    printf("Максимальный элемент: %d\n", *max);
    
    // Подсчет элементов, удовлетворяющих условию
    size_t count = count_if(arr, size, greater_than_5);
    printf("Количество элементов > 5: %zu\n", count);
    
    // Заполнение значениями
    fill_array(arr, size, 10);
    print_array(arr, size, "После fill(10)");
    
    // Обратный порядок
    reverse_array(arr, size);
    print_array(arr, size, "В обратном порядке");
    
    // Перемешивание
    shuffle_array(arr, size);
    print_array(arr, size, "После перемешивания");
    
    // Суммирование
    int sum = accumulate_array(arr, size, 0);
    printf("Сумма элементов: %d\n", sum);
    
    // Копирование
    int* arr2 = malloc(size * sizeof(int));
    copy_array(arr, arr2, size);
    print_array(arr2, size, "Скопированный массив");
    
    // Освобождение памяти
    free(arr2);
    
    return 0;
}
