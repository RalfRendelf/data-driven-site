<h2>std::variant</h2>

<p><strong>Категория:</strong> Контейнеры</p>

<h3>Описание</h3>
<p>std::variant - это тип, представляющий собой безопасную сумму (discriminated union), который может содержать значение одного из нескольких возможных типов. В отличие от union в C, variant контролирует, какой из типов в данный момент активен, и правильно обрабатывает конструирование и разрушение значений.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++17</p>

<h3>Основные методы:</h3>
<ul>
<li>std::get - получить значение определенного типа или по индексу</li>
<li>std::get_if - безопасное получение указателя на значение</li>
<li>std::holds_alternative - проверка, содержит ли variant значение указанного типа</li>
<li>std::visit - применение функции ко всему значению variant</li>
</ul>

<h3>Пример использования:</h3>
<p>Вот простой пример использования std::variant:</p>

<pre>
<code>
#include &lt;variant&gt;
#include &lt;iostream&gt;

using VarType = std::variant&lt;int, float, std::string&gt;;

int main() {
    VarType var = 42;  // содержит int
    std::cout &lt;&lt; std::get&lt;int&gt;(var) &lt;&lt; std::endl;
    
    var = std::string("hello");  // теперь содержит строку
    std::cout &lt;&lt; std::get&lt;std::string&gt;(var) &lt;&lt; std::endl;
    
    return 0;
}
</code>
</pre>

<h3>Аналог в мире C:</h3>
<p>В C для реализации подобной функциональности используется объединение (union) с дополнительным полем для отслеживания типа:</p>

<pre>
<code>
#include &lt;stdio.h&gt;

typedef enum {
    TYPE_INT,
    TYPE_FLOAT,
    TYPE_STRING
} var_type_t;

typedef struct {
    var_type_t type;
    union {
        int int_val;
        float float_val;
        char* string_val;
    } data;
} variant_t;

int main() {
    variant_t var;
    var.type = TYPE_INT;
    var.data.int_val = 42;
    
    if (var.type == TYPE_INT) {
        printf("Value: %d\n", var.data.int_val);
    }
    
    return 0;
}
</code>
</pre>
<h2>std::variant</h2>

<p><strong>Категория:</strong> Контейнеры</p>

<h3>Описание</h3>
<p>std::variant - это тип, представляющий собой безопасную сумму (discriminated union), который может содержать значение одного из нескольких возможных типов. В отличие от union в C, variant контролирует, какой из типов в данный момент активен, и правильно обрабатывает конструирование и разрушение значений.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++17</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::variant:</p>

<pre>
<code>
// Пример использования std::variant на C++

#include &lt;iostream&gt;
#include &lt;variant&gt;
#include &lt;string&gt;

// Visitor для обработки variant
struct Visitor {
    void operator()(int i) { std::cout << "Integer: " << i << std::endl; }
    void operator()(double d) { std::cout << "Double: " << d << std::endl; }
    void operator()(const std::string& s) { std::cout << "String: " << s << std::endl; }
};

int main() {
    // Создание variant
    std::variant&lt;int, double, std::string&gt; var;
    
    // Присваивание значений
    var = 42;
    std::cout << "Index: " << var.index() << std::endl;
    
    var = 3.14;
    std::cout << "Index: " << var.index() << std::endl;
    
    var = std::string("Hello, Variant!");
    std::cout << "Index: " << var.index() << std::endl;
    
    // Использование std::get для получения значения
    try {
        std::string& str = std::get&lt;std::string&gt;(var);
        std::cout << "Got string: " << str << std::endl;
    } catch (const std::bad_variant_access& e) {
        std::cout << "Exception: " << e.what() << std::endl;
    }
    
    // Использование std::get_if для безопасного получения значения
    if (auto* str_ptr = std::get_if&lt;std::string&gt;(&var)) {
        std::cout << "Got string via get_if: " << *str_ptr << std::endl;
    }
    
    // Использование visitor для обработки variant
    std::visit(Visitor{}, var);
    
    // Создание variant с разными значениями
    std::variant&lt;int, double, std::string&gt; var2 = 100;
    std::variant&lt;int, double, std::string&gt; var3 = 2.718;
    
    std::visit(Visitor{}, var2);
    std::visit(Visitor{}, var3);
    
    return 0;
}
</code>
</pre>

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o variant_example variant_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;string.h&gt;

// Перечисление для отслеживания активного типа
typedef enum {
    VARIANT_INT,
    VARIANT_DOUBLE,
    VARIANT_STRING
} VariantType;

// Структура для строки (для безопасного управления памятью)
typedef struct {
    char* data;
    size_t length;
} VariantString;

// Union для хранения возможных значений
typedef union {
    int int_value;
    double double_value;
    VariantString string_value;
} VariantData;

// Основная структура variant
typedef struct {
    VariantType type;
    VariantData data;
} Variant;

// Функции для создания variant
Variant variant_create_int(int value) {
    Variant v;
    v.type = VARIANT_INT;
    v.data.int_value = value;
    return v;
}

Variant variant_create_double(double value) {
    Variant v;
    v.type = VARIANT_DOUBLE;
    v.data.double_value = value;
    return v;
}

Variant variant_create_string(const char* str) {
    Variant v;
    v.type = VARIANT_STRING;
    v.data.string_value.length = strlen(str);
    v.data.string_value.data = malloc(v.data.string_value.length + 1);
    strcpy(v.data.string_value.data, str);
    return v;
}

// Функция для освобождения памяти (если variant содержит строку)
void variant_destroy(Variant* v) {
    if (v && v->type == VARIANT_STRING) {
        if (v->data.string_value.data) {
            free(v->data.string_value.data);
            v->data.string_value.data = NULL;
        }
    }
}

// Функции для получения значений
int variant_get_int(Variant* v) {
    if (v && v->type == VARIANT_INT) {
        return v->data.int_value;
    }
    return 0; // или вызвать ошибку
}

double variant_get_double(Variant* v) {
    if (v && v->type == VARIANT_DOUBLE) {
        return v->data.double_value;
    }
    return 0.0; // или вызвать ошибку
}

char* variant_get_string(Variant* v) {
    if (v && v->type == VARIANT_STRING) {
        return v->data.string_value.data;
    }
    return NULL; // или вызвать ошибку
}

// Функция для проверки типа
VariantType variant_get_type(Variant* v) {
    if (v) {
        return v->type;
    }
    return VARIANT_INT; // значение по умолчанию
}

// Функция для безопасного получения типа
int variant_is_int(Variant* v) {
    return v && v->type == VARIANT_INT;
}

int variant_is_double(Variant* v) {
    return v && v->type == VARIANT_DOUBLE;
}

int variant_is_string(Variant* v) {
    return v && v->type == VARIANT_STRING;
}

// Visitor функция для обработки variant
void variant_visit(Variant* v) {
    switch (v->type) {
        case VARIANT_INT:
            printf("Integer: %d\n", v->data.int_value);
            break;
        case VARIANT_DOUBLE:
            printf("Double: %f\n", v->data.double_value);
            break;
        case VARIANT_STRING:
            printf("String: %s\n", v->data.string_value.data);
            break;
        default:
            printf("Unknown type\n");
            break;
    }
}

// Пример использования
int main() {
    // Создание variant
    Variant var = variant_create_int(42);
    
    printf("Type index: %d\n", variant_get_type(&var));
    variant_visit(&var);
    
    // Уничтожаем текущий вариант и создаем новый
    variant_destroy(&var);
    var = variant_create_double(3.14);
    
    printf("Type index: %d\n", variant_get_type(&var));
    variant_visit(&var);
    
    // Уничтожаем текущий вариант и создаем строковый
    variant_destroy(&var);
    var = variant_create_string("Hello, Variant!");
    
    printf("Type index: %d\n", variant_get_type(&var));
    variant_visit(&var);
    
    // Проверка типа и безопасное извлечение
    if (variant_is_string(&var)) {
        printf("Got string: %s\n", variant_get_string(&var));
    }
    
    // Создание других вариантов
    Variant var2 = variant_create_int(100);
    Variant var3 = variant_create_double(2.718);
    
    variant_visit(&var2);
    variant_visit(&var3);
    
    // Освобождение памяти
    variant_destroy(&var);
    variant_destroy(&var2); // хотя var2 не содержит строку, безопасно вызвать
    variant_destroy(&var3); // хотя var3 не содержит строку, безопасно вызвать
    
    return 0;
}
</code>
</pre>
