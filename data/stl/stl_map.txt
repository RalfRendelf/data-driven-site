<h2>std::map</h2>

<p><strong>Категория:</strong> Контейнеры</p>

<h3>Описание</h3>
<p>std::map - это ассоциативный контейнер, содержащий пары ключ-значение, в которых ключи упорядочены по возрастанию. Он реализован как сбалансированное бинарное дерево поиска (обычно красно-черное дерево), что обеспечивает логарифмическую сложность основных операций.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++98</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::map:</p>

<pre>
<code>
// Пример использования std::map на C++

#include &lt;iostream&gt;
#include &lt;map&gt;
#include &lt;string&gt;

int main() {
    // Создание map
    std::map&lt;int, std::string&gt; myMap;
    
    // Вставка элементов
    myMap[1] = "one";
    myMap[2] = "two";
    myMap[3] = "three";
    
    // Использование emplace
    myMap.emplace(4, "four");
    
    // Использование insert
    myMap.insert(std::make_pair(5, "five"));
    
    std::cout << "Размер map: " << myMap.size() << std::endl;
    
    // Использование итераторов
    std::cout << "Элементы map через итераторы: " << std::endl;
    for (auto it = myMap.begin(); it != myMap.end(); ++it) {
        std::cout << it->first << " => " << it->second << std::endl;
    }
    
    // Использование range-based for loop
    std::cout << "Элементы map через range-based for: " << std::endl;
    for (const auto&amp; pair : myMap) {
        std::cout << pair.first << " => " << pair.second << std::endl;
    }
    
    // Доступ к элементам
    std::cout << "Значение по ключу 3: " << myMap.at(3) << std::endl;
    
    // Проверка наличия ключа
    if (myMap.count(2) > 0) {
        std::cout << "Ключ 2 существует в map" << std::endl;
    }
    
    // Использование find
    auto it = myMap.find(4);
    if (it != myMap.end()) {
        std::cout << "Найден элемент: " << it->first << " => " << it->second << std::endl;
    }
    
    // Удаление элемента
    myMap.erase(3);
    std::cout << "После удаления ключа 3:" << std::endl;
    for (const auto&amp; pair : myMap) {
        std::cout << pair.first << " => " << pair.second << std::endl;
    }
    
    // lower_bound и upper_bound
    auto lb = myMap.lower_bound(2);  // первый элемент >= 2
    auto ub = myMap.upper_bound(3);  // первый элемент > 3
    std::cout << "lower_bound(2): " << lb->first << " => " << lb->second << std::endl;
    std::cout << "upper_bound(3): " << ub->first << " => " << ub->second << std::endl;
    
    // Очистка map
    myMap.clear();
    std::cout << "После очистки размер: " << myMap.size() << std::endl;
    
    return 0;
}

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o map_example map_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;string.h&gt;

// Структура для пары ключ-значение
typedef struct {
    int key;
    char* value;
} KeyValuePair;

// Структура для узла дерева
typedef struct TreeNode {
    KeyValuePair pair;
    struct TreeNode* left;
    struct TreeNode* right;
    struct TreeNode* parent;
    int color; // 0 = черный, 1 = красный
} TreeNode;

// Структура для map
typedef struct {
    TreeNode* root;
    TreeNode* nil; // специальный узел для представления NULL
    size_t size;
} Map;

// Инициализация узла nil
TreeNode* create_nil_node() {
    TreeNode* nil = malloc(sizeof(TreeNode));
    nil->color = 0; // черный
    nil->left = NULL;
    nil->right = NULL;
    nil->parent = NULL;
    nil->pair.key = 0;
    nil->pair.value = NULL;
    return nil;
}

// Инициализация map
Map* map_create() {
    Map* map = malloc(sizeof(Map));
    map->nil = create_nil_node();
    map->root = map->nil;
    map->size = 0;
    return map;
}

// Уничтожение узла и его потомков
void destroy_tree_nodes(TreeNode* node, TreeNode* nil) {
    if (node != nil) {
        destroy_tree_nodes(node->left, nil);
        destroy_tree_nodes(node->right, nil);
        if (node->pair.value) {
            free(node->pair.value);
        }
        free(node);
    }
}

// Уничтожение map
void map_destroy(Map* map) {
    if (!map) return;
    
    destroy_tree_nodes(map->root, map->nil);
    free(map->nil);
    free(map);
}

// Создание нового узла
TreeNode* create_node(Map* map, int key, const char* value) {
    TreeNode* node = malloc(sizeof(TreeNode));
    node->pair.key = key;
    node->pair.value = malloc(strlen(value) + 1);
    strcpy(node->pair.value, value);
    node->left = map->nil;
    node->right = map->nil;
    node->parent = NULL;
    node->color = 1; // новый узел всегда красный
    return node;
}

// Левый поворот (для балансировки красно-черного дерева)
void rotate_left(Map* map, TreeNode* x) {
    TreeNode* y = x->right;
    x->right = y->left;
    
    if (y->left != map->nil) {
        y->left->parent = x;
    }
    
    y->parent = x->parent;
    
    if (x->parent == NULL) {
        map->root = y;
    } else if (x == x->parent->left) {
        x->parent->left = y;
    } else {
        x->parent->right = y;
    }
    
    y->left = x;
    x->parent = y;
}

// Правый поворот (для балансировки красно-черного дерева)
void rotate_right(Map* map, TreeNode* y) {
    TreeNode* x = y->left;
    y->left = x->right;
    
    if (x->right != map->nil) {
        x->right->parent = y;
    }
    
    x->parent = y->parent;
    
    if (y->parent == NULL) {
        map->root = x;
    } else if (y == y->parent->right) {
        y->parent->right = x;
    } else {
        y->parent->left = x;
    }
    
    x->right = y;
    y->parent = x;
}

// Восстановление свойств красно-черного дерева после вставки
void fix_after_insert(Map* map, TreeNode* z) {
    while (z->parent && z->parent->color == 1) {
        if (z->parent == z->parent->parent->left) {
            TreeNode* y = z->parent->parent->right;
            if (y->color == 1) {
                z->parent->color = 0;
                y->color = 0;
                z->parent->parent->color = 1;
                z = z->parent->parent;
            } else {
                if (z == z->parent->right) {
                    z = z->parent;
                    rotate_left(map, z);
                }
                z->parent->color = 0;
                z->parent->parent->color = 1;
                rotate_right(map, z->parent->parent);
            }
        } else {
            TreeNode* y = z->parent->parent->left;
            if (y->color == 1) {
                z->parent->color = 0;
                y->color = 0;
                z->parent->parent->color = 1;
                z = z->parent->parent;
            } else {
                if (z == z->parent->left) {
                    z = z->parent;
                    rotate_right(map, z);
                }
                z->parent->color = 0;
                z->parent->parent->color = 1;
                rotate_left(map, z->parent->parent);
            }
        }
    }
    map->root->color = 0;
}

// Вставка элемента
void map_put(Map* map, int key, const char* value) {
    TreeNode* z = create_node(map, key, value);
    
    TreeNode* y = NULL;
    TreeNode* x = map->root;
    
    while (x != map->nil) {
        y = x;
        if (z->pair.key < x->pair.key) {
            x = x->left;
        } else if (z->pair.key > x->pair.key) {
            x = x->right;
        } else {
            // Ключ уже существует, обновляем значение
            free(x->pair.value);
            x->pair.value = malloc(strlen(value) + 1);
            strcpy(x->pair.value, value);
            free(z);
            return;
        }
    }
    
    z->parent = y;
    
    if (y == NULL) {
        map->root = z;
    } else if (z->pair.key < y->pair.key) {
        y->left = z;
    } else {
        y->right = z;
    }
    
    map->size++;
    fix_after_insert(map, z);
}

// Поиск узла по ключу
TreeNode* tree_search(TreeNode* node, TreeNode* nil, int key) {
    while (node != nil && key != node->pair.key) {
        if (key < node->pair.key) {
            node = node->left;
        } else {
            node = node->right;
        }
    }
    return node;
}

// Проверка наличия ключа
int map_contains(Map* map, int key) {
    if (!map) return 0;
    TreeNode* result = tree_search(map->root, map->nil, key);
    return result != map->nil;
}

// Получение значения по ключу
char* map_get(Map* map, int key) {
    if (!map) return NULL;
    TreeNode* result = tree_search(map->root, map->nil, key);
    if (result != map->nil) {
        return result->pair.value;
    }
    return NULL;
}

// Удаление элемента
void map_erase(Map* map, int key) {
    // Для упрощения реализации удалим узел без полной балансировки дерева
    // В реальной реализации std::map потребуется полная реализация удаления 
    // с сохранением свойств красно-черного дерева
    
    // Найдем узел для удаления
    TreeNode* node = tree_search(map->root, map->nil, key);
    if (node == map->nil) {
        return; // элемент не найден
    }
    
    // Удаление листа или узла с одним потомком будет проще
    if (node->left == map->nil || node->right == map->nil) {
        TreeNode* child = (node->left != map->nil) ? node->left : node->right;
        
        if (node->parent == NULL) {
            map->root = child;
        } else if (node == node->parent->left) {
            node->parent->left = child;
        } else {
            node->parent->right = child;
        }
        
        if (child != map->nil) {
            child->parent = node->parent;
        }
        
        free(node->pair.value);
        free(node);
        map->size--;
    }
    // Для узла с двумя потомками нужно найти преемника
    else {
        // Упрощенная реализация: в реальной реализации нужно использовать преемника
        // и выполнить полную балансировку
        printf("Удаление узла с двумя потомками не реализовано в этом примере\n");
    }
}

// Вспомогательная функция для inorder обхода дерева
void inorder_traversal(TreeNode* node, TreeNode* nil, void (*visit)(KeyValuePair*)) {
    if (node != nil) {
        inorder_traversal(node->left, nil, visit);
        visit(&node->pair);
        inorder_traversal(node->right, nil, visit);
    }
}

// Вспомогательная функция для печати пары
void print_pair(KeyValuePair* pair) {
    printf("%d => %s\n", pair->key, pair->value);
}

// Печать map
void map_print(Map* map, const char* label) {
    printf("%s:\n", label);
    inorder_traversal(map->root, map->nil, print_pair);
    printf("(size: %zu)\n", map->size);
}

// Получение размера
size_t map_size(Map* map) {
    return map ? map->size : 0;
}

// Проверка на пустоту
int map_empty(Map* map) {
    return map ? (map->size == 0) : 1;
}

// Очистка map
void map_clear(Map* map) {
    if (!map) return;
    
    destroy_tree_nodes(map->root, map->nil);
    map->root = map->nil;
    map->size = 0;
}

// Пример использования
int main() {
    Map* myMap = map_create();
    
    // Вставка элементов
    map_put(myMap, 1, "one");
    map_put(myMap, 2, "two");
    map_put(myMap, 3, "three");
    map_put(myMap, 4, "four");
    map_put(myMap, 5, "five");
    
    map_print(myMap, "Исходный map");
    printf("Размер map: %zu\n", map_size(myMap));
    
    // Доступ к элементам
    char* value = map_get(myMap, 3);
    if (value) {
        printf("Значение по ключу 3: %s\n", value);
    }
    
    // Проверка наличия ключа
    if (map_contains(myMap, 2)) {
        printf("Ключ 2 существует в map\n");
    }
    
    // Удаление элемента
    map_erase(myMap, 3);
    map_print(myMap, "После удаления ключа 3");
    
    // Очистка map
    map_clear(myMap);
    printf("После очистки размер: %zu\n", map_size(myMap));
    
    // Уничтожение map
    map_destroy(myMap);
    
    return 0;
}
