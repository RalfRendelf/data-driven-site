<h2>std::string / std::string_view</h2>

<p><strong>Категория:</strong> Контейнеры</p>

<h3>Описание</h3>
<p>std::string - это класс, представляющий собой контейнер для хранения и манипуляции строками текста. Он предоставляет удобный и безопасный способ работы с текстовыми данными, автоматически управляя памятью. std::string_view - это легковесный класс, появившийся в C++17, который предоставляет доступ к строке без владения ею, позволяя избежать ненужных копирований.</p>

<h3>Добавлен в стандарт:</h3>
<p>std::string - с самого начала C++, std::string_view - C++17</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::string/std::string_view:</p>

<pre>
<code>
// Пример использования std::string/std::string_view на C++

#include &lt;iostream&gt;
#include &lt;string&gt;
#include &lt;string_view&gt;

int main() {
    // Создание строк
    std::string str1 = "Hello";
    std::string str2 = "World";
    
    std::cout << "str1: " << str1 << std::endl;
    std::cout << "str2: " << str2 << std::endl;
    
    // Конкатенация
    std::string concatenated = str1 + ", " + str2 + "!";
    std::cout << "Concatenated: " << concatenated << std::endl;
    
    // Доступ к символам
    std::cout << "First character of str1: " << str1[0] << std::endl;
    std::cout << "Last character of str1: " << str1.back() << std::endl;
    
    // Размер строки
    std::cout << "Length of concatenated: " << concatenated.length() << std::endl;
    
    // Поиск подстроки
    size_t pos = concatenated.find("World");
    if (pos != std::string::npos) {
        std::cout << "Found 'World' at position: " << pos << std::endl;
    }
    
    // substr
    std::string sub = concatenated.substr(0, 5); // "Hello"
    std::cout << "Substring: '" << sub << "'" << std::endl;
    
    // Сравнение
    if (str1 < str2) {
        std::cout << str1 << " comes before " << str2 << " lexicographically" << std::endl;
    }
    
    // Работа с string_view
    std::string_view sv = concatenated;
    std::cout << "String view: " << sv << std::endl;
    std::cout << "First part of view: " << sv.substr(0, 5) << std::endl;
    
    // Изменение строки
    str1.append(" C++");
    std::cout << "After appending: " << str1 << std::endl;
    
    return 0;
}

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o string_example string_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;string.h&gt;
#include &lt;assert.h&gt;

// Структура для хранения строки (аналог std::string)
typedef struct {
    char* data;
    size_t size;
    size_t capacity;
} String;

// Структура для представления строки без владения (аналог std::string_view)
typedef struct {
    const char* data;
    size_t size;
} StringView;

// Инициализация строки с начальным размером
String* string_create(const char* init_str) {
    String* str = malloc(sizeof(String));
    if (init_str == NULL) {
        str->size = 0;
        str->capacity = 1;
        str->data = malloc(str->capacity);
        str->data[0] = '\0';
    } else {
        str->size = strlen(init_str);
        str->capacity = str->size + 1;
        str->data = malloc(str->capacity);
        strcpy(str->data, init_str);
    }
    return str;
}

// Уничтожение строки
void string_destroy(String* str) {
    if (str != NULL) {
        if (str->data != NULL) {
            free(str->data);
        }
        free(str);
    }
}

// Создание string_view из строки
StringView string_view_create(const char* str, size_t len) {
    StringView sv;
    sv.data = str;
    sv.size = len > 0 ? len : (str ? strlen(str) : 0);
    return sv;
}

// Получение длины строки
size_t string_length(String* str) {
    return str ? str->size : 0;
}

// Получение длины string_view
size_t string_view_length(StringView* sv) {
    return sv ? sv->size : 0;
}

// Доступ к символу по индексу
char string_at(String* str, size_t index) {
    if (str && index < str->size) {
        return str->data[index];
    }
    return '\0';  // возвращаем null-символ, если индекс вне диапазона
}

// Доступ к символу в string_view
char string_view_at(StringView* sv, size_t index) {
    if (sv && index < sv->size) {
        return sv->data[index];
    }
    return '\0';
}

// Добавление символа в конец строки
void string_push_back(String* str, char ch) {
    if (str) {
        if (str->size + 1 >= str->capacity) {
            // Увеличиваем емкость
            str->capacity *= 2;
            str->data = realloc(str->data, str->capacity);
        }
        
        str->data[str->size] = ch;
        str->size++;
        str->data[str->size] = '\0';  // null-терминатор
    }
}

// Конкатенация строк
String* string_concat(String* str1, String* str2) {
    if (!str1 || !str2) return NULL;
    
    String* result = malloc(sizeof(String));
    result->size = str1->size + str2->size;
    result->capacity = result->size + 1;
    result->data = malloc(result->capacity);
    
    strcpy(result->data, str1->data);
    strcat(result->data, str2->data);
    
    return result;
}

// Сравнение строк
int string_compare(String* str1, String* str2) {
    if (!str1 || !str2) return -1;
    return strcmp(str1->data, str2->data);
}

// Сравнение string_view со строкой
int string_view_compare(StringView* sv, const char* str) {
    if (!sv || !str) return -1;
    
    size_t str_len = strlen(str);
    size_t min_len = sv->size < str_len ? sv->size : str_len;
    
    int cmp = strncmp(sv->data, str, min_len);
    if (cmp == 0) {
        if (sv->size < str_len) return -1;
        else if (sv->size > str_len) return 1;
    }
    
    return cmp;
}

// Поиск подстроки в строке
int string_find(String* str, const char* substr) {
    if (!str || !substr) return -1;
    
    char* pos = strstr(str->data, substr);
    if (pos) {
        return pos - str->data;  // возвращаем индекс
    }
    return -1;  // не найдено
}

// Создание подстроки
String* string_substr(String* str, size_t start, size_t length) {
    if (!str || start >= str->size) return NULL;
    
    size_t len = (start + length > str->size) ? str->size - start : length;
    
    String* result = malloc(sizeof(String));
    result->data = malloc(len + 1);
    strncpy(result->data, str->data + start, len);
    result->data[len] = '\0';
    result->size = len;
    result->capacity = len + 1;
    
    return result;
}

// Преобразование string_view в строку
String* string_from_view(StringView* sv) {
    if (!sv) return NULL;
    
    String* str = malloc(sizeof(String));
    str->size = sv->size;
    str->capacity = sv->size + 1;
    str->data = malloc(str->capacity);
    strncpy(str->data, sv->data, sv->size);
    str->data[sv->size] = '\0';
    
    return str;
}

// Добавление строки к существующей
void string_append(String* str, const char* append_str) {
    if (!str || !append_str) return;
    
    size_t append_len = strlen(append_str);
    if (str->size + append_len >= str->capacity) {
        str->capacity = str->size + append_len + 1;
        str->data = realloc(str->data, str->capacity);
    }
    
    strcat(str->data, append_str);
    str->size += append_len;
}

// Печать строки
void string_print(String* str, const char* label) {
    printf("%s: \"%s\" (length: %zu, capacity: %zu)\n", 
           label, str->data, string_length(str), str->capacity);
}

// Пример использования
int main() {
    // Создание строк
    String* str1 = string_create("Hello");
    String* str2 = string_create("World");
    
    string_print(str1, "str1");
    string_print(str2, "str2");
    
    // Конкатенация
    String* concatenated = string_concat(str1, str2);
    string_push_back(concatenated, '!');
    string_insert(concatenated, 5, ", "); // вставляем ", " в позицию 5
    string_print(concatenated, "Concatenated");
    
    // Доступ к символам
    printf("First character of str1: %c\n", string_at(str1, 0));
    printf("Last character of str1: %c\n", string_at(str1, string_length(str1)-1));
    
    // Размер строки
    printf("Length of concatenated: %zu\n", string_length(concatenated));
    
    // Поиск подстроки
    int pos = string_find(concatenated, "World");
    if (pos != -1) {
        printf("Found 'World' at position: %d\n", pos);
    }
    
    // substr
    String* sub = string_substr(concatenated, 0, 5); // "Hello"
    string_print(sub, "Substring");
    
    // Сравнение
    if (string_compare(str1, str2) < 0) {
        printf("str1 comes before str2 lexicographically\n");
    }
    
    // Создание string_view
    StringView sv = string_view_create(concatenated->data, string_length(concatenated));
    printf("String view: \"%.*s\"\n", (int)string_view_length(&sv), sv.data);
    
    // Подстрока из string_view
    StringView sub_sv = {sv.data, 5}; // первые 5 символов
    printf("First part of view: \"%.*s\"\n", (int)string_view_length(&sub_sv), sub_sv.data);
    
    // Изменение строки
    string_append(str1, " C++");
    string_print(str1, "After appending");
    
    // Освобождение памяти
    string_destroy(str1);
    string_destroy(str2);
    string_destroy(concatenated);
    string_destroy(sub);
    
    return 0;
}

// Вспомогательная функция для вставки строки в указанную позицию
void string_insert(String* str, size_t pos, const char* insert_str) {
    if (!str || !insert_str || pos > str->size) return;
    
    size_t insert_len = strlen(insert_str);
    size_t new_size = str->size + insert_len;
    
    if (new_size >= str->capacity) {
        str->capacity = new_size + 1;
        str->data = realloc(str->data, str->capacity);
    }
    
    // Сдвигаем хвост строки вправо
    memmove(str->data + pos + insert_len, str->data + pos, str->size - pos + 1);
    // Копируем вставляемую строку
    memcpy(str->data + pos, insert_str, insert_len);
    
    str->size = new_size;
}
