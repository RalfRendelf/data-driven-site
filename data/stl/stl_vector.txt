<h2>std::vector</h2>

<p><strong>Категория:</strong> Контейнеры</p>

<h3>Описание</h3>
<p>std::vector - это контейнер, который представляет собой динамический массив. Он управляет массивом элементов, который может изменять свой размер. В отличие от обычных массивов, вектор может автоматически увеличивать свою емкость при добавлении новых элементов.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++98</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::vector:</p>

<pre>
<code>
// Пример использования std::vector на C++

#include &lt;iostream&gt;
#include &lt;vector&gt;
#include &lt;algorithm&gt;

int main() {
    // Создание вектора
    std::vector&lt;int&gt; vec;
    
    // Добавление элементов
    vec.push_back(10);
    vec.push_back(20);
    vec.push_back(30);
    
    std::cout << "Размер вектора: " << vec.size() << std::endl;
    
    // Доступ к элементам
    for (size_t i = 0; i < vec.size(); ++i) {
        std::cout << "vec[" << i << "] = " << vec[i] << std::endl;
    }
    
    // Использование итераторов
    std::cout << "Элементы вектора через итераторы: ";
    for (auto it = vec.begin(); it != vec.end(); ++it) {
        std::cout << *it << " ";
    }
    std::cout << std::endl;
    
    // Использование range-based for loop
    std::cout << "Элементы вектора через range-based for: ";
    for (const auto&amp; element : vec) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    // Вставка элемента в середину
    vec.insert(vec.begin() + 1, 15);
    std::cout << "После вставки 15 на позицию 1: ";
    for (const auto&amp; element : vec) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    // Удаление элемента
    vec.erase(vec.begin());
    std::cout << "После удаления первого элемента: ";
    for (const auto&amp; element : vec) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    // Изменение размера
    vec.resize(10, 0); // Увеличить до 10 элементов, заполнив 0
    std::cout << "После resize(10, 0): ";
    for (const auto&amp; element : vec) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    // Очистка вектора
    vec.clear();
    std::cout << "После очистки размер: " << vec.size() << std::endl;
    
    return 0;
}

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o vector_example vector_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;

// Структура для вектора
typedef struct {
    int* data;
    size_t size;
    size_t capacity;
} Vector;

// Инициализация вектора
Vector* vector_create() {
    Vector* vec = malloc(sizeof(Vector));
    vec->data = NULL;
    vec->size = 0;
    vec->capacity = 0;
    return vec;
}

// Уничтожение вектора
void vector_destroy(Vector* vec) {
    if (vec) {
        if (vec->data) {
            free(vec->data);
        }
        free(vec);
    }
}

// Увеличение емкости вектора
void vector_resize_if_needed(Vector* vec) {
    if (vec->size >= vec->capacity) {
        size_t new_capacity = vec->capacity == 0 ? 1 : vec->capacity * 2;
        vec->data = realloc(vec->data, new_capacity * sizeof(int));
        vec->capacity = new_capacity;
    }
}

// Добавление элемента в конец
void vector_push_back(Vector* vec, int value) {
    vector_resize_if_needed(vec);
    vec->data[vec->size] = value;
    vec->size++;
}

// Получение размера
size_t vector_size(Vector* vec) {
    return vec ? vec->size : 0;
}

// Получение емкости
size_t vector_capacity(Vector* vec) {
    return vec ? vec->capacity : 0;
}

// Получение элемента по индексу
int vector_at(Vector* vec, size_t index) {
    if (vec && index < vec->size) {
        return vec->data[index];
    }
    return 0; // значение по умолчанию
}

// Установка элемента по индексу
void vector_set(Vector* vec, size_t index, int value) {
    if (vec && index < vec->size) {
        vec->data[index] = value;
    }
}

// Вставка элемента по индексу
void vector_insert(Vector* vec, size_t index, int value) {
    if (!vec || index > vec->size) return;
    
    vector_resize_if_needed(vec);
    
    // Сдвигаем элементы вправо
    for (size_t i = vec->size; i > index; i--) {
        vec->data[i] = vec->data[i-1];
    }
    
    vec->data[index] = value;
    vec->size++;
}

// Удаление элемента по индексу
void vector_erase(Vector* vec, size_t index) {
    if (!vec || index >= vec->size) return;
    
    // Сдвигаем элементы влево
    for (size_t i = index; i < vec->size - 1; i++) {
        vec->data[i] = vec->data[i+1];
    }
    
    vec->size--;
}

// Изменение размера
void vector_resize(Vector* vec, size_t new_size, int fill_value) {
    if (!vec) return;
    
    if (new_size > vec->capacity) {
        vec->data = realloc(vec->data, new_size * sizeof(int));
        vec->capacity = new_size;
    }
    
    // Заполняем новые элементы значением
    for (size_t i = vec->size; i < new_size; i++) {
        vec->data[i] = fill_value;
    }
    
    vec->size = new_size;
}

// Очистка вектора
void vector_clear(Vector* vec) {
    if (vec) {
        vec->size = 0;
    }
}

// Печать вектора
void vector_print(Vector* vec, const char* label) {
    printf("%s: ", label);
    for (size_t i = 0; i < vector_size(vec); i++) {
        printf("%d ", vector_at(vec, i));
    }
    printf("(size: %zu, capacity: %zu)\n", vector_size(vec), vector_capacity(vec));
}

// Пример использования
int main() {
    Vector* vec = vector_create();
    
    // Добавление элементов
    vector_push_back(vec, 10);
    vector_push_back(vec, 20);
    vector_push_back(vec, 30);
    
    vector_print(vec, "После добавления 10, 20, 30");
    
    // Доступ к элементам
    printf("Размер вектора: %zu\n", vector_size(vec));
    
    // Вставка элемента в середину
    vector_insert(vec, 1, 15);
    vector_print(vec, "После вставки 15 на позицию 1");
    
    // Удаление элемента
    vector_erase(vec, 0);
    vector_print(vec, "После удаления первого элемента");
    
    // Изменение размера
    vector_resize(vec, 10, 0);
    vector_print(vec, "После resize(10, 0)");
    
    // Очистка вектора
    vector_clear(vec);
    printf("После очистки размер: %zu\n", vector_size(vec));
    
    // Уничтожение вектора
    vector_destroy(vec);
    
    return 0;
}
