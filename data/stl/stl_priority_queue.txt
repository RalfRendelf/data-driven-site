<h2>std::priority_queue</h2>

<p><strong>Категория:</strong> Контейнеры</p>

<h3>Описание</h3>
<p>std::priority_queue - это контейнер-адаптер, реализующий структуру данных очереди с приоритетом. Элементы извлекаются из очереди в порядке убывания приоритета (максимальный элемент извлекается первым). По умолчанию реализована как max-heap на основе std::vector.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++98</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::priority_queue:</p>

<pre>
<code>
// Пример использования std::priority_queue на C++

#include &lt;iostream&gt;
#include &lt;queue&gt;
#include &lt;vector&gt;

int main() {
    // Создание очереди с приоритетом
    std::priority_queue&lt;int&gt; pq;
    
    // Добавление элементов в очередь
    pq.push(3);
    pq.push(1);
    pq.push(4);
    pq.push(2);
    pq.push(5);
    
    std::cout << "Размер очереди: " << pq.size() << std::endl;
    
    // Извлечение элементов из очереди (в порядке убывания)
    std::cout << "Элементы очереди (по приоритету): ";
    while (!pq.empty()) {
        std::cout << pq.top() << " ";  // Просмотр максимального элемента
        pq.pop();                      // Удаление максимального элемента
    }
    std::cout << std::endl;
    
    // Проверка на пустоту
    std::cout << "Очередь пуста: " << (pq.empty() ? "да" : "нет") << std::endl;
    
    // Создание очереди с приоритетом с обратным порядком (min-heap)
    std::priority_queue&lt;int, std::vector&lt;int&gt;, std::greater&lt;int&gt;&gt; min_pq;
    min_pq.push(3);
    min_pq.push(1);
    min_pq.push(4);
    min_pq.push(2);
    
    std::cout << "Min-heap элементы: ";
    while (!min_pq.empty()) {
        std::cout << min_pq.top() << " ";
        min_pq.pop();
    }
    std::cout << std::endl;
    
    return 0;
}

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o priority_queue_example priority_queue_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;

#define INITIAL_CAPACITY 8

// Структура для очереди с приоритетом (max-heap)
typedef struct {
    int* data;
    size_t size;
    size_t capacity;
} PriorityQueue;

// Инициализация очереди с приоритетом
PriorityQueue* priority_queue_create() {
    PriorityQueue* pq = malloc(sizeof(PriorityQueue));
    pq->data = malloc(INITIAL_CAPACITY * sizeof(int));
    pq->size = 0;
    pq->capacity = INITIAL_CAPACITY;
    return pq;
}

// Уничтожение очереди с приоритетом
void priority_queue_destroy(PriorityQueue* pq) {
    if (!pq) return;
    if (pq->data) free(pq->data);
    free(pq);
}

// Проверка на пустоту
int priority_queue_empty(PriorityQueue* pq) {
    return pq ? (pq->size == 0) : 1;
}

// Получение размера
size_t priority_queue_size(PriorityQueue* pq) {
    return pq ? pq->size : 0;
}

// Вспомогательная функция для просеивания вверх
void heapify_up(PriorityQueue* pq, size_t index) {
    if (index && pq->data[(index - 1) / 2] < pq->data[index]) {
        // Обмен с родительским узлом
        int temp = pq->data[index];
        pq->data[index] = pq->data[(index - 1) / 2];
        pq->data[(index - 1) / 2] = temp;
        
        // Рекурсивно просеиваем вверх
        heapify_up(pq, (index - 1) / 2);
    }
}

// Вспомогательная функция для просеивания вниз
void heapify_down(PriorityQueue* pq, size_t index) {
    size_t largest = index;
    size_t left_child = 2 * index + 1;
    size_t right_child = 2 * index + 2;
    
    // Проверка, является ли левый потомок большим
    if (left_child < pq->size && pq->data[left_child] > pq->data[largest]) {
        largest = left_child;
    }
    
    // Проверка, является ли правый потомок большим
    if (right_child < pq->size && pq->data[right_child] > pq->data[largest]) {
        largest = right_child;
    }
    
    // Если наибольший элемент не является текущим, меняем и продолжаем просеивание
    if (largest != index) {
        int temp = pq->data[index];
        pq->data[index] = pq->data[largest];
        pq->data[largest] = temp;
        
        heapify_down(pq, largest);
    }
}

// Добавление элемента в очередь
void priority_queue_push(PriorityQueue* pq, int value) {
    if (!pq) return;
    
    // Увеличение размера, если необходимо
    if (pq->size >= pq->capacity) {
        pq->capacity *= 2;
        int* new_data = realloc(pq->data, pq->capacity * sizeof(int));
        if (!new_data) return; // ошибка выделения памяти
        pq->data = new_data;
    }
    
    // Добавление элемента в конец
    pq->data[pq->size] = value;
    pq->size++;
    
    // Восстановление свойства кучи
    heapify_up(pq, pq->size - 1);
}

// Удаление максимального элемента
void priority_queue_pop(PriorityQueue* pq) {
    if (!pq || pq->size == 0) return;
    
    // Заменяем корень последним элементом
    pq->data[0] = pq->data[pq->size - 1];
    pq->size--;
    
    // Восстановление свойства кучи
    if (pq->size > 1) {
        heapify_down(pq, 0);
    }
}

// Получение максимального элемента
int priority_queue_top(PriorityQueue* pq) {
    if (!pq || pq->size == 0) return 0; // или другое значение по умолчанию
    return pq->data[0];
}

// Печать очереди с приоритетом (для отладки)
void priority_queue_print(PriorityQueue* pq, const char* label) {
    printf("%s: [", label);
    for (size_t i = 0; i < pq->size; i++) {
        printf("%d", pq->data[i]);
        if (i < pq->size - 1) printf(", ");
    }
    printf("] (size: %zu, capacity: %zu)\n", pq->size, pq->capacity);
}

// Пример использования
int main() {
    PriorityQueue* pq = priority_queue_create();
    
    // Добавление элементов в очередь
    priority_queue_push(pq, 3);
    priority_queue_push(pq, 1);
    priority_queue_push(pq, 4);
    priority_queue_push(pq, 2);
    priority_queue_push(pq, 5);
    
    priority_queue_print(pq, "Очередь после добавления 3, 1, 4, 2, 5");
    printf("Размер очереди: %zu\n", priority_queue_size(pq));
    
    // Извлечение элементов из очереди (в порядке убывания)
    std::cout << "Элементы очереди (по приоритету): ";
    while (!priority_queue_empty(pq)) {
        printf("%d ", priority_queue_top(pq)); // Просмотр максимального элемента
        priority_queue_pop(pq);               // Удаление максимального элемента
    }
    printf("\n");
    
    // Проверка на пустоту
    printf("Очередь пуста: %s\n", priority_queue_empty(pq) ? "да" : "нет");
    
    // Уничтожение очереди
    priority_queue_destroy(pq);
    
    return 0;
}
