<h2>C++ в микроконтроллерах</h2>

<p><strong>Категория:</strong> Программирование</p>

<h3>Описание</h3>
<p>Использование C++ в разработке для микроконтроллеров позволяет применять преимущества объектно-ориентированного программирования при сохранении низкоуровневого контроля над оборудованием. C++ предоставляет мощные средства абстракции без дополнительных накладных расходов при правильном использовании.</p>

<h3>C++ в микроконтроллерах:</h3>
<p>Использование C++ для программирования микроконтроллеров имеет ряд преимуществ по сравнению с C:</p>

<ul>
<li>Абстракция периферии через классы</li>
<li>RAII (Resource Acquisition Is Initialization) для управления ресурсами</li>
<li>Шаблоны для генерации эффективного кода во время компиляции</li>
<li>Инкапсуляция и наследование для лучшей организации кода</li>
<li>Строгая типизация для предотвращения ошибок</li>
</ul>

<h4>Пример использования C++ для инициализации GPIO:</h4>

<pre>
<code>
// C++ пример
class GpioPin {
public:
    enum class Direction { Input, Output };
    enum class Level { Low, High };
    enum class Port { A, B, C, D, E, F };

    GpioPin(Port port, uint8_t pin, Direction dir) : 
        port_(port), pin_(pin), direction_(dir) {
        initialize();
    }

    void set(Level level) {
        auto base = getBaseAddress(port_);
        if (level == Level::High) {
            *reinterpret_cast<volatile uint32_t*>(base + 0x18) = (1U << pin_);
        } else {
            *reinterpret_cast<volatile uint32_t*>(base + 0x28) = (1U << pin_);
        }
    }

    Level get() const {
        auto base = getBaseAddress(port_);
        return (*reinterpret_cast<volatile uint32_t*>(base + 0x10) & (1U << pin_)) 
               ? Level::High : Level::Low;
    }

private:
    void initialize() {
        // Включение тактирования порта
        auto clk_reg = reinterpret_cast<volatile uint32_t*>(0x40023830);
        *clk_reg |= (1U << static_cast<uint8_t>(port_));
        
        // Настройка пина
        auto base = getBaseAddress(port_);
        auto moder = reinterpret_cast<volatile uint32_t*>(base);
        *moder &= ~(0x3 << (pin_ * 2));  // Очистка битов
        *moder |= (direction_ == Direction::Output ? 0x1 : 0x0) << (pin_ * 2);
    }

    uintptr_t getBaseAddress(Port port) {
        return 0x40020000 + (static_cast<uint8_t>(port) * 0x400);
    }

    Port port_;
    uint8_t pin_;
    Direction direction_;
};

// Использование
GpioPin led(GpioPin::Port::A, 5, GpioPin::Direction::Output);
led.set(GpioPin::Level::High);
</code>
</pre>

typedef enum {
    GPIO_DIR_INPUT,
    GPIO_DIR_OUTPUT
} gpio_direction_t;

typedef enum {
    GPIO_PORT_A,
    GPIO_PORT_B,
    GPIO_PORT_C,
    GPIO_PORT_D,
    GPIO_PORT_E,
    GPIO_PORT_F
} gpio_port_t;

typedef enum {
    GPIO_LEVEL_LOW,
    GPIO_LEVEL_HIGH
} gpio_level_t;

typedef struct {
    gpio_port_t port;
    uint8_t pin;
    gpio_direction_t direction;
} gpio_pin_t;

void gpio_init(gpio_pin_t* pin_obj, gpio_port_t port, uint8_t pin_num, gpio_direction_t dir) {
    pin_obj->port = port;
    pin_obj->pin = pin_num;
    pin_obj->direction = dir;
    
    // Включение тактирования порта
    volatile uint32_t* clk_reg = (volatile uint32_t*)0x40023830;
    *clk_reg |= (1U << port);
    
    // Настройка пина
    uintptr_t base = 0x40020000 + (port * 0x400);
    volatile uint32_t* moder = (volatile uint32_t*)base;
    *moder &= ~(0x3 << (pin_num * 2));  // Очистка битов
    *moder |= (dir == GPIO_DIR_OUTPUT ? 0x1 : 0x0) << (pin_num * 2);
}

void gpio_set(gpio_pin_t* pin_obj, gpio_level_t level) {
    uintptr_t base = 0x40020000 + (pin_obj->port * 0x400);
    if (level == GPIO_LEVEL_HIGH) {
        *(volatile uint32_t*)(base + 0x18) = (1U << pin_obj->pin);
    } else {
        *(volatile uint32_t*)(base + 0x28) = (1U << pin_obj->pin);
    }
}

gpio_level_t gpio_get(gpio_pin_t* pin_obj) {
    uintptr_t base = 0x40020000 + (pin_obj->port * 0x400);
    return (*(volatile uint32_t*)(base + 0x10) & (1U << pin_obj->pin)) 
           ? GPIO_LEVEL_HIGH : GPIO_LEVEL_LOW;
}

<h3>new/delete и аллокаторы:</h3>
<p>В embedded системах использование динамической памяти ограничено из-за отсутствия операционной системы и необходимости детерминированного поведения. Однако, при правильном подходе, new/delete могут быть использованы.</p>

<h4>Пример реализации кастомного аллокатора:</h4>

<pre>
<code>
// C++ пример
#include <cstddef>
#include <cstdint>

class PoolAllocator {
private:
    static constexpr size_t POOL_SIZE = 4096;
    alignas(std::max_align_t) static uint8_t pool_[POOL_SIZE];
    static size_t offset_;

public:
    static void* allocate(size_t size) {
        if (offset_ + size > POOL_SIZE) {
            return nullptr;  // Пул исчерпан
        }
        
        void* ptr = &pool_[offset_];
        offset_ += size;
        return ptr;
    }
    
    static void deallocate(void* ptr, size_t size) {
        // Простая реализация: не освобождаем память до сброса
        // В реальных системах может использоваться более сложная логика
    }
    
    static void reset() {
        offset_ = 0;
    }
};

size_t PoolAllocator::offset_ = 0;
uint8_t PoolAllocator::pool_[PoolAllocator::POOL_SIZE];

// Переопределение new/delete для класса
class Sensor {
public:
    void* operator new(size_t size) {
        return PoolAllocator::allocate(size);
    }
    
    void operator delete(void* ptr, size_t size) {
        PoolAllocator::deallocate(ptr, size);
    }
    
    Sensor(int id) : id_(id) {}
    
private:
    int id_;
};
</code>
</pre>

<h3>Исключения (и почему их отключают):</h3>
<p>Исключения в C++ имеют значительные накладные расходы, которые неприемлемы в embedded системах:</p>

<ul>
<li>Увеличивают размер исполняемого кода</li>
<li>Добавляют неопределенность во время выполнения</li>
<li>Требуют дополнительной поддержки со стороны компилятора</li>
<li>Могут вызвать непредсказуемую задержку при обработке</li>
</ul>

<p>Поэтому в embedded средах обычно отключают исключения с помощью флага компилятора -fno-exceptions и используют коды возврата.</p>

<h4>Пример обработки ошибок без исключений:</h4>

<pre>
<code>
// C++ пример без исключений

    ErrorHardwareFault

class UartDriver {

<h3>RAII в embedded:</h3>
<p>RAII (Resource Acquisition Is Initialization) - это ключевой паттерн C++, который особенно важен в embedded системах для корректного управления ресурсами.</p>

<h4>Пример RAII для управления GPIO:</h4>

<pre>
<code>
// C++ пример
class GpioLock {
public:
    GpioLock() {
        // Блокировка GPIO периферии
        enableClock();
    }
    
    ~GpioLock() {
        // Снятие блокировки GPIO периферии
        disableClock();
    }
    
private:
    void enableClock() {
        // Включение тактирования GPIO
        *reinterpret_cast<volatile uint32_t*>(0x40023830) |= (1U << 0); // GPIOA
    }
    
    void disableClock() {
        // Выключение тактирования GPIO
        *reinterpret_cast<volatile uint32_t*>(0x40023830) &= ~(1U << 0);
    }
};

// Использование RAII
void configureGpio() {
    GpioLock lock;  // Тактирование включается
    
    // Работа с GPIO
    GpioPin led(GpioPin::Port::A, 5, GpioPin::Direction::Output);
    led.set(GpioPin::Level::High);
    
} // Тактирование автоматически выключается при выходе из функции
</code>
</pre>

<h3>constexpr и шаблоны для оптимизации:</h3>
<p>Шаблоны и constexpr позволяют выполнять вычисления на этапе компиляции, что особенно ценно в embedded системах, где каждая инструкция имеет значение.</p>

<h4>Пример использования constexpr и шаблонов:</h4>

<pre>
<code>
// C++ пример
template<int PinNumber>
class DigitalOut {
public:
    constexpr DigitalOut() {
        // Настройка пина во время компиляции
        initialize();
    }
    
    void set(bool value) {
        if (value) {
            setHigh();
        } else {
            setLow();
        }
    }
    
private:
    constexpr void initialize() {
        // Установка модер
        auto moder = reinterpret_cast<volatile uint32_t*>(PORT_BASE + 0x00);
        *moder &= ~(0x3 << (PinNumber * 2));
        *moder |= (0x1 << (PinNumber * 2));
    }
    
    void setHigh() {
        auto bsrr = reinterpret_cast<volatile uint32_t*>(PORT_BASE + 0x18);
        *bsrr = (1U << PinNumber);
    }
    
    void setLow() {
        auto bsrr = reinterpret_cast<volatile uint32_t*>(PORT_BASE + 0x28);
        *bsrr = (1U << PinNumber);
    }
    
    static constexpr uintptr_t PORT_BASE = 0x40020000;
};

// Создание вывода для конкретного пина
DigitalOut<5> led;  // Пин 5, известный на этапе компиляции
led.set(true);
</code>
</pre>

<h3>Zero-cost abstractions:</h3>
<p>Одна из ключевых концепций C++ в embedded - это zero-cost abstractions, когда абстракции не добавляют дополнительных накладных расходов во время выполнения.</p>

<h4>Пример zero-cost абстракции:</h4>

<pre>
<code>
// C++ пример
template<typename Peripheral>
class RegisterWrapper {
public:
    template<uint32_t Offset>
    static volatile uint32_t& reg() {
        return *reinterpret_cast<volatile uint32_t*>(Peripheral::BASE_ADDR + Offset);
    }
    
    template<uint32_t Offset, typename FlagType>
    static void set_flag(FlagType flag) {
        reg<Offset>() |= static_cast<uint32_t>(flag);
    }
    
    template<uint32_t Offset, typename FlagType>
    static void clear_flag(FlagType flag) {
        reg<Offset>() &= ~static_cast<uint32_t>(flag);
    }
};

struct GpioA {
    static constexpr uintptr_t BASE_ADDR = 0x40020000;
};

// Использование
using GpioARegs = RegisterWrapper<GpioA>;
GpioARegs::set_flag<0x00>(1U << 10);  // Установка бита в MODER для пина 5
</code>
</pre>
