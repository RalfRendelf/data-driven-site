<h2>Bare-metal программирование</h2>

<p><strong>Категория:</strong> Программирование</p>

<h3>Описание</h3>
<p>Bare-metal программирование означает разработку программного обеспечения, которое работает непосредственно на физическом оборудовании без операционной системы или абстрактных библиотек, таких как HAL. Это позволяет получить полный контроль над оборудованием и достичь максимальной производительности и минимального потребления ресурсов.</p>

<h3>Работа с регистрами напрямую:</h3>
<p>В bare-metal программировании мы взаимодействуем с регистрами периферии напрямую, используя адреса, указанные в технической документации микроконтроллера.</p>

<h4>Пример работы с GPIO через прямой доступ к регистрам:</h4>

<pre>
<code>
#include "stm32f4xx.h"

// Макросы для работы с GPIO
#define GPIOA_ENABLE_CLOCK()   (RCC->AHB1ENR |= RCC_AHB1ENR_GPIOAEN)
#define GPIOA_SET_PIN_5()      (GPIOA->BSRR = GPIO_BSRR_BS_5)
#define GPIOA_CLEAR_PIN_5()    (GPIOA->BSRR = GPIO_BSRR_BR_5)
#define GPIOA_TOGGLE_PIN_5()   (GPIOA->ODR ^= GPIO_ODR_OD5)

// Настройка GPIOA pin 5 как выход
void setup_gpio_a_pin5(void) {
    // Включить тактирование GPIOA
    GPIOA_ENABLE_CLOCK();
    
    // Установить режим: 00 - Input, 01 - Output, 10 - Alternate, 11 - Analog
    GPIOA->MODER &= ~GPIO_MODER_MODER5;  // Очистить биты для пина 5
    GPIOA->MODER |= GPIO_MODER_MODER5_0; // Установить в 01 - Output mode
    
    // Установить тип вывода: 0 - Push-pull, 1 - Open-drain
    GPIOA->OTYPER &= ~GPIO_OTYPER_OT_5;  // Push-pull
    
    // Установить скорость вывода: 00 - Low, 01 - Medium, 10 - High, 11 - Very high
    GPIOA->OSPEEDR |= GPIO_OSPEEDER_OSPEEDR5; // Установить в 11 - Very high speed
    
    // Установить подтягивающий резистор: 00 - No pull, 01 - Pull-up, 10 - Pull-down
    GPIOA->PUPDR &= ~GPIO_PUPDR_PUPDR5;  // Установить в 00 - No pull-up, no pull-down
}

// Функция мигания светодиодом
void blink_led(int times) {
    for(int i = 0; i < times; i++) {
        GPIOA_SET_PIN_5();    // Включить светодиод
        delay_ms(500);        // Задержка 500мс
        GPIOA_CLEAR_PIN_5();  // Выключить светодиод
        delay_ms(500);        // Задержка 500мс
    }
}

// Простая задержка (в реальных приложениях лучше использовать таймер)
void delay_ms(int ms) {
    // Приблизительная задержка - зависит от частоты процессора
    volatile int i, j;
    for(i = 0; i < ms; i++) {
        for(j = 0; j < 16800; j++); // Приблизительно 1ms при 168MHz
    }
}
</code>
</pre>

<h4>Пример работы с UART через прямой доступ к регистрам:</h4>

<pre>
<code>
#include "stm32f4xx.h"

// Настройка UART2 через прямую работу с регистрами
void uart2_setup(void) {
    // Включить тактирование UART2 и GPIOA
    RCC->AHB1ENR |= RCC_AHB1ENR_GPIOAEN;  // Тактирование GPIOA
    RCC->APB1ENR |= RCC_APB1ENR_USART2EN; // Тактирование UART2
    
    // Настроить GPIOA pins 2 и 3 как альтернативные функции для UART
    // PA2 - TX, PA3 - RX
    GPIOA->MODER &= ~(GPIO_MODER_MODER2 | GPIO_MODER_MODER3); // Очистить биты
    GPIOA->MODER |= (GPIO_MODER_MODER2_1 | GPIO_MODER_MODER3_1); // Альтернативная функция
    
    GPIOA->AFR[0] |= (7 << (2 * 4)) | (7 << (3 * 4)); // AF7 для UART2
    
    // Настроить UART
    // Установить скорость передачи данных (baud rate)
    // Установим OVER8=0 (oversampling by 16)
    USART2->CR1 &= ~USART_CR1_OVER8; // OVER8 = 0
    
    // Пример расчета для 115200 baud при 16MHz
    // Упрощенный расчет: 16000000 / 16 / 115200 = 8.68
    uint32_t baud_rate = 115200;
    uint32_t pclk1 = 16000000; // Предполагаем 16MHz
    uint32_t usart_div = (pclk1 * 25) / (4 * baud_rate);
    uint32_t mantissa = usart_div / 100;
    uint32_t fraction = ((usart_div - (mantissa * 100)) * 16 + 50) / 100;
    
    USART2->BRR = (mantissa << 4) | (fraction & 0x0F);
    
    // Настроить параметры передачи: 8 бит, 1 стоп-бит, без проверки четности
    USART2->CR1 &= ~(USART_CR1_M | USART_CR1_PCE); // 8 бит, без проверки четности
    USART2->CR2 &= ~USART_CR2_STOP; // 1 стоп-бит
    
    // Включить UART
    USART2->CR1 |= USART_CR1_UE; // Enable USART
    USART2->CR1 |= (USART_CR1_RE | USART_CR1_TE); // Enable receiver and transmitter
}

// Отправка байта через UART
void uart2_send_byte(uint8_t byte) {
    // Ждать, пока буфер передачи не освободится
    while (!(USART2->SR & USART_SR_TXE));
    
    // Отправить байт
    USART2->DR = byte;
}

// Отправка строки через UART
void uart2_send_string(const char* str) {
    while (*str) {
        uart2_send_byte(*str++);
    }
}
</code>
</pre>

<h3>Bit-banding:</h3>
<p>Bit-banding - это возможность адресовать отдельные биты в регистре как отдельные 32-битные слова в специальной области памяти. Это позволяет атомарно изменять отдельные биты без операций чтения-модификации-записи.</p>

<h4>Пример использования bit-banding:</h4>

<pre>
<code>
#include "stm32f4xx.h"

// Макросы для bit-banding
#define BIT_BAND_SRAM_REF   0x20000000
#define BIT_BAND_SRAM_BASE  0x22000000
#define BIT_BAND_PERI_REF   0x40000000
#define BIT_BAND_PERI_BASE  0x42000000

// Макрос для получения адреса bit-band для периферии
#define PERIPH_BIT_BAND_ADDR(addr, bit) \
    (BIT_BAND_PERI_BASE + ((addr) - BIT_BAND_PERI_REF) * 32 + (bit) * 4)

// Пример: изменение бита ODR5 регистра GPIOA
#define GPIOA_ODR5_BB_ADDR  PERIPH_BIT_BAND_ADDR(&(GPIOA->ODR), 5)

// Атомарное изменение бита через bit-banding
void set_gpioa_pin5_bb(int state) {
    // Прямой доступ к одному биту регистра ODR через bit-banding
    volatile uint32_t* bb_addr = (volatile uint32_t*)GPIOA_ODR5_BB_ADDR;
    
    if(state) {
        *bb_addr = 1; // Установить бит (включить светодиод)
    } else {
        *bb_addr = 0; // Очистить бит (выключить светодиод)
    }
}
</code>
</pre>

<h3>Memory-mapped I/O:</h3>
<p>Memory-mapped I/O - это метод, при котором регистры периферийных устройств отображаются в адресное пространство памяти. Это позволяет обращаться к периферии так же, как к обычной памяти.</p>

<h4>Пример использования memory-mapped I/O:</h4>

<pre>
<code>
#include "stm32f4xx.h"

// Адреса регистров таймера TIM2
#define TIM2_BASE     (APB1PERIPH_BASE + 0x0000)  // Базовый адрес TIM2
#define TIM2_CR1      *(volatile uint32_t*)(TIM2_BASE + 0x00)  // Control register 1
#define TIM2_CNT      *(volatile uint32_t*)(TIM2_BASE + 0x24)  // Counter register
#define TIM2_PSC      *(volatile uint32_t*)(TIM2_BASE + 0x28)  // Prescaler register
#define TIM2_ARR      *(volatile uint32_t*)(TIM2_BASE + 0x2C)  // Auto-reload register

// Настройка таймера TIM2 для генерации прерываний
void timer2_setup(void) {
    // Включить тактирование TIM2
    RCC->APB1ENR |= RCC_APB1ENR_TIM2EN;
    
    // Настроить prescaler для получения 1 кГц (при 168 МГц)
    // 168000000 / (8400 * 20000) = 1 Гц
    TIM2_PSC = 8399;  // 8400 - 1 для делителя
    TIM2_ARR = 19999; // Автоматический перезапуск после 20000 шагов
    
    // Включить прерывание по обновлению
    TIM2->DIER |= TIM_DIER_UIE;
    
    // Включить таймер
    TIM2_CR1 |= TIM_CR1_CEN;
    
    // Включить прерывание в NVIC
    NVIC_EnableIRQ(TIM2_IRQn);
    NVIC_SetPriority(TIM2_IRQn, 0);
}

// Обработчик прерывания от TIM2
void TIM2_IRQHandler(void) {
    if(TIM2->SR & TIM_SR_UIF) {  // Проверить флаг прерывания
        TIM2->SR &= ~TIM_SR_UIF; // Очистить флаг прерывания
        
        // Переключить светодиод
        GPIOA->ODR ^= GPIO_ODR_OD5;
    }
}
</code>
</pre>

<h3>Минимальный проект без библиотек:</h3>
<p>Минимальный bare-metal проект состоит из вектора прерываний, обработчика сброса и начального стека. Ниже приведен пример минимального проекта на ассемблере и C для STM32F4.</p>

<h4>Файл startup_stm32f407xx.s:</h4>

<pre>
<code>
.syntax unified
.thumb
.text

/* Вектор прерываний */
.section .isr_vector, "a"
.word _estack
.word Reset_Handler
.word NMI_Handler
.word HardFault_Handler
.word MemManage_Handler
.word BusFault_Handler
.word UsageFault_Handler
.word 0
.word 0
.word 0
.word 0
.word SVC_Handler
.word DebugMon_Handler
.word 0
.word PendSV_Handler
.word SysTick_Handler

/* Определение обработчиков прерываний */
.weak NMI_Handler
.thumb_set NMI_Handler, Default_Handler

.weak HardFault_Handler
.thumb_set HardFault_Handler, Default_Handler

.weak MemManage_Handler
.thumb_set MemManage_Handler, Default_Handler

.weak BusFault_Handler
.thumb_set BusFault_Handler, Default_Handler

.weak UsageFault_Handler
.thumb_set UsageFault_Handler, Default_Handler

.weak SVC_Handler
.thumb_set SVC_Handler, Default_Handler

.weak DebugMon_Handler
.thumb_set DebugMon_Handler, Default_Handler

.weak PendSV_Handler
.thumb_set PendSV_Handler, Default_Handler

.weak SysTick_Handler
.thumb_set SysTick_Handler, Default_Handler

Default_Handler:
    b Default_Handler

/* Обработчик сброса */
.type Reset_Handler, %function
Reset_Handler:
    /* Копирование секции .data из Flash в RAM */
    ldr r1, =_sdata       /* Начало данных в RAM */
    ldr r2, =_edata       /* Конец данных в RAM */
    ldr r0, =_sidata      /* Начало данных в Flash */
    
CopyDataLoop:
    cmp r1, r2
    itt lt
    ldrlt r3, [r0], #4
    strlt r3, [r1], #4
    blt CopyDataLoop
    
    /* Заполнение .bss нулями */
    ldr r1, =_sbss
    ldr r2, =_ebss
    movs r3, #0
    
ZeroBSSLoop:
    cmp r1, r2
    itt lt
    strlt r3, [r1], #4
    blt ZeroBSSLoop
    
    /* Вызов функции main */
    bl main
    /* Если main возвращает управление, зациклиться здесь */
Hang:
    b Hang

.size Reset_Handler, .-Reset_Handler
</code>
</pre>

<h4>Файл main.c минимального проекта:</h4>

<pre>
<code>
#include <stdint.h>

// Определение адресов регистров GPIOA
#define RCC_BASE        0x40023800
#define RCC_AHB1ENR     (*(volatile uint32_t*)(RCC_BASE + 0x30))
#define GPIOA_BASE      0x40020000
#define GPIOA_MODER     (*(volatile uint32_t*)(GPIOA_BASE + 0x00))
#define GPIOA_ODR       (*(volatile uint32_t*)(GPIOA_BASE + 0x14))
#define GPIOA_BSRR      (*(volatile uint32_t*)(GPIOA_BASE + 0x18))

// Определения битов
#define RCC_AHB1ENR_GPIOAEN    (1 << 0)
#define GPIO_MODER_MODER5      (0x3 << 10)  // Биты 11:10 для пина 5
#define GPIO_MODER_MODER5_0    (0x1 << 10)  // Бит 10
#define GPIO_MODER_MODER5_1    (0x1 << 11)  // Бит 11
#define GPIO_ODR_OD5           (0x1 << 5)   // Бит 5
#define GPIO_BSRR_BS_5         (0x1 << 5)   // Установить пин 5
#define GPIO_BSRR_BR_5         (0x1 << 21)  // Сбросить пин 5

// Функция задержки
void delay(volatile uint32_t count) {
    while(count--);
}

// Основная функция
int main(void) {
    // Включить тактирование GPIOA
    RCC_AHB1ENR |= RCC_AHB1ENR_GPIOAEN;
    
    // Настроить пин PA5 как выход (для светодиода)
    GPIOA_MODER &= ~GPIO_MODER_MODER5;      // Очистить биты для пина 5
    GPIOA_MODER |= GPIO_MODER_MODER5_0;     // Установить в 01 - Output mode
    
    // Бесконечный цикл мигания светодиодом
    while(1) {
        GPIOA_BSRR = GPIO_BSRR_BS_5;  // Установить пин (включить светодиод)
        delay(1680000);               // Задержка
        
        GPIOA_BSRR = GPIO_BSRR_BR_5;  // Сбросить пин (выключить светодиод)
        delay(1680000);               // Задержка
    }
    
    return 0;
}
</code>
</pre>

<h3>Преимущества bare-metal программирования:</h3>
<ul>
<li>Максимальная производительность и минимальные накладные расходы</li>
<li>Полный контроль над оборудованием</li>
<li>Минимальный размер кода</li>
<li>Предсказуемое поведение и временные характеристики</li>
<li>Минимальное потребление памяти</li>
</ul>

<h3>Недостатки bare-metal программирования:</h3>
<ul>
<li>Более высокая сложность разработки</li>
<li>Необходимость глубокого понимания архитектуры микроконтроллера</li>
<li>Более высокий риск ошибок при неправильной работе с регистрами</li>
<li>Меньшая переносимость кода между различными моделями микроконтроллеров</li>
<li>Более длительное время разработки</li>
</ul>
