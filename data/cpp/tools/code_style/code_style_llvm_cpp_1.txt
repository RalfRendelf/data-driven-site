bool isNumberValid(int InputNumber) {
    // 2 пробела для отступов
    if (InputNumber < 0) {
        return false;
    }
    
    // Переменные именуются в CamelCase
    int CalculatedValue = InputNumber * 2;
    
    return CalculatedValue > 10;
}
