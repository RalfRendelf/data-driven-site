bool isValidNumber(int input_number) {
    // 2 пробела для отступов вместо табуляции
    if (input_number < 0) {
        return false;
    }
    
    // Переменные именуются в lower_underscore
    int calculated_value = input_number * 2;
    
    return calculated_value > 10;
}
