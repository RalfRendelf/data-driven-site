cmake_minimum_required(VERSION 3.20)

project(MyProject LANGUAGES CXX)

add_library(my_lib
    src/lib.cpp
    include/lib.hpp
)

target_include_directories(my_lib
    PUBLIC
        ${PROJECT_SOURCE_DIR}/include
)

add_executable(my_app src/main.cpp)

target_link_libraries(my_app
    PRIVATE
        my_lib
)
