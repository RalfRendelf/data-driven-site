// Пример многопоточной программы:
#include <iostream>
#include <thread>
#include <vector>

void worker(int id) {
    std::cout << "Worker " << id << " is running" << std::endl;
}

int main() {
    std::vector<std::thread> threads;
    for(int i = 0; i < 3; i++) {
        threads.emplace_back(worker, i);
    }
    
    for(auto& t : threads) {
        t.join();
    }
    
    return 0;
}
