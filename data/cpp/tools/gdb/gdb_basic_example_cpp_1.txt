// Пример программы для отладки:
#include <iostream>

int main() {
    int x = 10;
    int y = 20;
    int sum = x + y;
    
    std::cout << "Sum is: " << sum << std::endl;
    
    for(int i = 0; i < 5; i++) {
        std::cout << "Iteration: " << i << std::endl;
    }
    
    return 0;
}

// Компиляция с отладочной информацией:
// g++ -g -o program program.cpp

// Запуск GDB:
// gdb ./program
// (gdb) break main
// (gdb) run
// (gdb) print x
// (gdb) print y
// (gdb) continue
