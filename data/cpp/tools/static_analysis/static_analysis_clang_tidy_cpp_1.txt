// Пример кода с проблемами:
#include <iostream>

int* dangerous_function() {
    int local_var = 42;
    return &local_var;  // Ошибка: возврат адреса локальной переменной
}

int main() {
    int* ptr = dangerous_function();
    std::cout << *ptr << std::endl;  // Неопределенное поведение
    return 0;
}
