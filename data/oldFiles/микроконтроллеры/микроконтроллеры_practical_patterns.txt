<h2>Практические шаблоны</h2>

<p><strong>Категория:</strong> Программирование</p>

<h3>Описание</h3>
<p>Практические шаблоны представляют собой проверенные решения для часто встречающихся задач в разработке встраиваемых систем. Эти шаблоны помогают ускорить разработку, улучшить структуру кода и повысить его надежность и повторное использование.</p>

<h3>Драйверы периферии (UART/SPI/I2C):</h3>
<p>Драйверы периферии обеспечивают абстракцию для взаимодействия с аппаратными интерфейсами. Они должны быть независимыми от используемых библиотек (HAL, LL или bare-metal) и иметь унифицированный интерфейс.</p>

<h3>Очереди сообщений:</h3>
<p>Очереди сообщений позволяют организовать асинхронное взаимодействие между компонентами системы. Особенно полезны в многозадачных системах.</p>

<h4>Пример реализации очереди сообщений:</h4>

<pre>
<code>
#ifndef MESSAGE_QUEUE_H
#define MESSAGE_QUEUE_H

#include <stdint.h>
#include <stdbool.h>
#include <string.h>

#define MSG_QUEUE_MAX_MSG_SIZE 256
#define MSG_QUEUE_MAX_MESSAGES 32

// Типы сообщений
typedef enum {
    MSG_TYPE_SENSOR_DATA = 0x01,
    MSG_TYPE_COMMAND = 0x02,
    MSG_TYPE_EVENT = 0x03,
    MSG_TYPE_RESPONSE = 0x04
} msg_type_t;

// Структура сообщения
typedef struct {
    msg_type_t type;
    uint16_t length;
    uint8_t data[MSG_QUEUE_MAX_MSG_SIZE];
} message_t;

// Структура очереди сообщений
typedef struct {
    message_t messages[MSG_QUEUE_MAX_MESSAGES];
    uint8_t head;
    uint8_t tail;
    uint8_t count;
    bool full;
} msg_queue_t;

// Инициализация очереди
void msg_queue_init(msg_queue_t* queue);

// Поместить сообщение в очередь
bool msg_queue_enqueue(msg_queue_t* queue, const message_t* msg);

// Получить сообщение из очереди
bool msg_queue_dequeue(msg_queue_t* queue, message_t* msg);

// Проверить, пуста ли очередь
bool msg_queue_is_empty(const msg_queue_t* queue);

// Проверить, полна ли очередь
bool msg_queue_is_full(const msg_queue_t* queue);

// Реализация
void msg_queue_init(msg_queue_t* queue) {
    if (!queue) return;
    
    memset(queue, 0, sizeof(msg_queue_t));
}

bool msg_queue_enqueue(msg_queue_t* queue, const message_t* msg) {
    if (!queue || !msg || queue->full) {
        return false;
    }
    
    // Копируем сообщение
    queue->messages[queue->head] = *msg;
    
    // Обновляем указатели
    queue->head = (queue->head + 1) % MSG_QUEUE_MAX_MESSAGES;
    queue->count++;
    
    if (queue->count == MSG_QUEUE_MAX_MESSAGES) {
        queue->full = true;
    }
    
    return true;
}

bool msg_queue_dequeue(msg_queue_t* queue, message_t* msg) {
    if (!queue || !msg || msg_queue_is_empty(queue)) {
        return false;
    }
    
    // Копируем сообщение
    *msg = queue->messages[queue->tail];
    
    // Обновляем указатели
    queue->tail = (queue->tail + 1) % MSG_QUEUE_MAX_MESSAGES;
    queue->count--;
    queue->full = false;
    
    return true;
}

bool msg_queue_is_empty(const msg_queue_t* queue) {
    return (queue && queue->count == 0);
}

bool msg_queue_is_full(const msg_queue_t* queue) {
    return (queue && queue->full);
}

#endif
</code>

<h3>State machine (FSM):</h3>
<p>Конечный автомат (FSM - Finite State Machine) - это важный паттерн для управления состоянием системы, особенно в задачах, где поведение системы зависит от текущего состояния и входных сигналов.</p>

<h4>Пример реализации FSM для управления соединением:</h4>

<pre>
<code>
#ifndef CONNECTION_FSM_H
#define CONNECTION_FSM_H

#include <stdint.h>
#include <stdbool.h>

// Состояния автомата соединения
typedef enum {
    CONN_STATE_DISCONNECTED = 0,
    CONN_STATE_CONNECTING,
    CONN_STATE_CONNECTED,
    CONN_STATE_DISCONNECTING,
    CONN_STATE_ERROR
} conn_state_t;

// События автомата соединения
typedef enum {
    CONN_EVENT_CONNECT_REQ = 0,
    CONN_EVENT_DISCONNECT_REQ,
    CONN_EVENT_CONNECT_SUCCESS,
    CONN_EVENT_CONNECT_FAILED,
    CONN_EVENT_DISCONNECT_SUCCESS,
    CONN_EVENT_ERROR_OCCURRED
} conn_event_t;

// Данные состояния
typedef struct {
    conn_state_t current_state;
    uint32_t connection_attempts;
    uint32_t last_error_code;
} conn_fsm_data_t;

// Функция перехода состояния
typedef conn_state_t (*transition_func_t)(conn_fsm_data_t* fsm, conn_event_t event);

// Инициализация FSM
void conn_fsm_init(conn_fsm_data_t* fsm);

// Обработка события
void conn_fsm_process_event(conn_fsm_data_t* fsm, conn_event_t event);

// Получение текущего состояния
conn_state_t conn_fsm_get_state(const conn_fsm_data_t* fsm);

// Реализация
void conn_fsm_init(conn_fsm_data_t* fsm) {
    if (!fsm) return;
    
    fsm->current_state = CONN_STATE_DISCONNECTED;
    fsm->connection_attempts = 0;
    fsm->last_error_code = 0;
}

conn_state_t conn_fsm_get_state(const conn_fsm_data_t* fsm) {
    return fsm ? fsm->current_state : CONN_STATE_ERROR;
}

void conn_fsm_process_event(conn_fsm_data_t* fsm, conn_event_t event) {
    if (!fsm) return;
    
    conn_state_t old_state = fsm->current_state;
    
    // Обработка события в зависимости от текущего состояния
    switch (fsm->current_state) {
        case CONN_STATE_DISCONNECTED:
            if (event == CONN_EVENT_CONNECT_REQ) {
                fsm->current_state = CONN_STATE_CONNECTING;
                fsm->connection_attempts++;
            }
            break;
            
        case CONN_STATE_CONNECTING:
            switch (event) {
                case CONN_EVENT_CONNECT_SUCCESS:
                    fsm->current_state = CONN_STATE_CONNECTED;
                    fsm->connection_attempts = 0;
                    break;
                    
                case CONN_EVENT_CONNECT_FAILED:
                case CONN_EVENT_ERROR_OCCURRED:
                    fsm->current_state = CONN_STATE_DISCONNECTED;
                    break;
                    
                default:
                    break;
            }
            break;
            
        case CONN_STATE_CONNECTED:
            if (event == CONN_EVENT_DISCONNECT_REQ) {
                fsm->current_state = CONN_STATE_DISCONNECTING;
            } else if (event == CONN_EVENT_ERROR_OCCURRED) {
                fsm->current_state = CONN_STATE_ERROR;
            }
            break;
            
        case CONN_STATE_DISCONNECTING:
            if (event == CONN_EVENT_DISCONNECT_SUCCESS) {
                fsm->current_state = CONN_STATE_DISCONNECTED;
            } else if (event == CONN_EVENT_ERROR_OCCURRED) {
                fsm->current_state = CONN_STATE_ERROR;
            }
            break;
            
        case CONN_STATE_ERROR:
            if (event == CONN_EVENT_CONNECT_REQ) {
                fsm->current_state = CONN_STATE_CONNECTING;
                fsm->connection_attempts++;
            }
            break;
    }
    
    // Вызов колбэков при изменении состояния
    if (old_state != fsm->current_state) {
        // Здесь можно вызвать callback-функции для обработки перехода
        // on_state_changed(old_state, fsm->current_state);
    }
}

<h3>Ring buffer:</h3>
<p>Кольцевой буфер (ring buffer) - эффективная структура данных для буферизации потоковых данных, особенно полезен при работе с последовательными интерфейсами.</p>

<h4>Пример реализации ring buffer:</h4>

<pre>
<code>
#ifndef RING_BUFFER_H
#define RING_BUFFER_H

#include <stdint.h>
#include <stdbool.h>

// Структура кольцевого буфера
typedef struct {
    uint8_t* buffer;
    uint16_t size;
    uint16_t head;
    uint16_t tail;
    uint16_t count;
    bool full;
} ring_buffer_t;

// Инициализация кольцевого буфера
void ring_buffer_init(ring_buffer_t* rb, uint8_t* buffer, uint16_t size);

// Поместить байт в буфер
bool ring_buffer_put(ring_buffer_t* rb, uint8_t data);

// Получить байт из буфера
bool ring_buffer_get(ring_buffer_t* rb, uint8_t* data);

// Количество доступных байтов
uint16_t ring_buffer_available(const ring_buffer_t* rb);

// Количество свободного места
uint16_t ring_buffer_free_space(const ring_buffer_t* rb);

// Очистка буфера
void ring_buffer_clear(ring_buffer_t* rb);

// Реализация
void ring_buffer_init(ring_buffer_t* rb, uint8_t* buffer, uint16_t size) {
    if (!rb || !buffer || !size) return;
    
    rb->buffer = buffer;
    rb->size = size;
    rb->head = 0;
    rb->tail = 0;
    rb->count = 0;
    rb->full = false;
}

bool ring_buffer_put(ring_buffer_t* rb, uint8_t data) {
    if (!rb || rb->full) return false;
    
    rb->buffer[rb->head] = data;
    rb->head = (rb->head + 1) % rb->size;
    rb->count++;
    
    if (rb->count == rb->size) {
        rb->full = true;
    }
    
    return true;
}

bool ring_buffer_get(ring_buffer_t* rb, uint8_t* data) {
    if (!rb || !data || rb->count == 0) return false;
    
    *data = rb->buffer[rb->tail];
    rb->tail = (rb->tail + 1) % rb->size;
    rb->count--;
    rb->full = false;
    
    return true;
}

uint16_t ring_buffer_available(const ring_buffer_t* rb) {
    return rb ? rb->count : 0;
}

uint16_t ring_buffer_free_space(const ring_buffer_t* rb) {
    return rb ? (rb->size - rb->count) : 0;
}

void ring_buffer_clear(ring_buffer_t* rb) {
    if (!rb) return;
    
    rb->head = 0;
    rb->tail = 0;
    rb->count = 0;
    rb->full = false;
}

    // Проверка CRC и установка флага завершения обновления
    return 0;
}

void bootloader_jump_to_app(void) {
    // Чтение адреса вектора сброса из пользовательского приложения
    uint32_t* app_stack = (uint32_t*)APPLICATION_START_ADDR;
    uint32_t* app_reset = (uint32_t*)(APPLICATION_START_ADDR + 4);
    
    // Установка MSP
    __asm volatile (
        "msr msp, %0\n"
        :
        : "r" (*app_stack)
    );
    
    // Переход к пользовательскому приложению
    asm volatile (
        "bx %0\n"
        :
        : "r" (*app_reset)
    );
}

// Простая реализация CRC32
uint32_t bootloader_calc_crc32(uint8_t* data, uint32_t size) {
    uint32_t crc = 0xFFFFFFFF;
    const uint32_t polynomial = 0xEDB88320;
    
    for (uint32_t i = 0; i < size; i++) {
        crc ^= data[i];
        for (int j = 0; j < 8; j++) {
            crc = (crc >> 1) ^ (crc & 1 ? polynomial : 0);
        }
    }
    
    return crc ^ 0xFFFFFFFF;
}
