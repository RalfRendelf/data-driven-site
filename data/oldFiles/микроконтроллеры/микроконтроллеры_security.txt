<h2>Безопасность</h2>

<p><strong>Категория:</strong> Оптимизация</p>

<h3>Описание</h3>
<p>Безопасность встраиваемых систем становится все более важной по мере увеличения числа подключенных IoT-устройств. Защита данных, аутентификация, защита от несанкционированного доступа и обеспечение целостности прошивки - ключевые аспекты современных встраиваемых систем.</p>

<h3>CRC / Hash:</h3>
<p>CRC (Cyclic Redundancy Check) и хеш-функции используются для проверки целостности данных. CRC - это быстрый алгоритм для обнаружения случайных ошибок, а хеш-функции обеспечивают криптографическую проверку целостности.</p>

// Вычисление CRC32 для буфера данных
uint32_t calculate_crc32(const uint8_t *data, size_t length) {
    uint32_t crc = 0xFFFFFFFF;
    
    for(size_t i = 0; i < length; i++) {
        uint8_t table_index = (uint8_t)((crc ^ data[i]) & 0xFF);
        crc = (crc >> 8) ^ crc32_table[table_index];
    }
    
    return crc ^ 0xFFFFFFFF;
}

// Проверка целостности прошивки
uint8_t verify_firmware_integrity(const uint8_t* firmware, size_t fw_size, uint32_t stored_checksum) {
    uint32_t calculated_checksum = calculate_crc32(firmware, fw_size);
    return (calculated_checksum == stored_checksum);
}

// Основные функции SHA-256
#define ROTR(n,x) (((x) >> (n)) | ((x) << (32 - (n))))
#define CH(x,y,z) (((x) & (y)) ^ (~(x) & (z)))
#define MAJ(x,y,z) (((x) & (y)) ^ ((x) & (z)) ^ ((y) & (z)))
#define EP0(x) (ROTR(2,x) ^ ROTR(13,x) ^ ROTR(22,x))
#define EP1(x) (ROTR(6,x) ^ ROTR(11,x) ^ ROTR(25,x))
#define SIG0(x) (ROTR(7,x) ^ ROTR(18,x) ^ ((x) >> 3))
#define SIG1(x) (ROTR(17,x) ^ ROTR(19,x) ^ ((x) >> 10))

void sha256_transform(uint32_t state[8], const uint8_t data[64]) {
    uint32_t a, b, c, d, e, f, g, h, i, j, t1, t2, m[64];
    
    for (i = 0, j = 0; i < 16; ++i, j += 4)
        m[i] = (data[j] << 24) | (data[j + 1] << 16) | (data[j + 2] << 8) | (data[j + 3]);
    for (; i < 64; ++i)
        m[i] = SIG1(m[i - 2]) + m[i - 7] + SIG0(m[i - 15]) + m[i - 16];
    
    a = state[0]; b = state[1]; c = state[2]; d = state[3];
    e = state[4]; f = state[5]; g = state[6]; h = state[7];
    
    for (i = 0; i < 64; ++i) {
        t1 = h + EP1(e) + CH(e,f,g) + k[i] + m[i];
        t2 = EP0(a) + MAJ(a,b,c);
        h = g; g = f; f = e; e = d + t1;
        d = c; c = b; b = a; a = t1 + t2;
    }
    
    state[0] += a; state[1] += b; state[2] += c; state[3] += d;
    state[4] += e; state[5] += f; state[6] += g; state[7] += h;
}

void sha256(const uint8_t *data, size_t len, uint8_t hash[32]) {
    uint32_t state[8] = {0x6a09e667, 0xbb67ae85, 0x3c6ef372, 0xa54ff53a, 0x510e527f, 0x9b05688c, 0x1f83d9ab, 0x5be0cd19};
    uint8_t block[64];
    size_t offset = 0;
    
    // Обработка полных 64-байтовых блоков
    for (; len >= 64; len -= 64, data += 64) {
        sha256_transform(state, data);
    }
    
    // Обработка остатка
    offset = len & 63;
    memcpy(block, data, offset);
    
    // Добавление бита 1 после данных
    block[offset++] = 0x80;
    
    // Заполнение нулями до 56 байтов (если не помещается длина)
    if (offset > 56) {
        memset(block + offset, 0, 64 - offset);
        sha256_transform(state, block);
        memset(block, 0, 56);
    } else {
        memset(block + offset, 0, 56 - offset);
    }
    
    // Добавление длины в битах
    uint64_t total_len = (uint64_t)(len + (data - (uint8_t*)data)) * 8;
    for (int i = 0; i < 8; i++) {
        block[56 + i] = (uint8_t)(total_len >> (56 - 8*i));
    }
    
    sha256_transform(state, block);
    
    // Преобразование результата в байты
    for (int i = 0; i < 8; i++) {
        hash[i*4] = (state[i] >> 24) & 0xff;
        hash[i*4 + 1] = (state[i] >> 16) & 0xff;
        hash[i*4 + 2] = (state[i] >> 8) & 0xff;
        hash[i*4 + 3] = state[i] & 0xff;
    }
}

// Галуаево поле умножение (GF(2^8))
uint8_t gmul(uint8_t a, uint8_t b) {
    uint8_t p = 0;
    uint8_t counter;
    uint8_t carry;
    
    for(counter = 0; counter < 8; counter++) {
        if(b & 1) {
            p ^= a;
        }
        
        carry = a & 0x80;
        a <<= 1;
        if(carry) {
            a ^= 0x1b; // 0x1b = irreducible polynomial
        }
        b >>= 1;
    }
    
    return p;
}

// Подготовка раундового ключа (упрощенная реализация)
void aes_key_expansion(const uint8_t* key, uint8_t w[][4][4], int Nb, int Nk, int Nr) {
    int i, j, r, c;
    uint8_t temp[4];
    
    // Копирование ключа в первые слова
    for(i = 0; i < Nk; i++) {
        for(j = 0; j < 4; j++) {
            w[i][j][0] = key[i * 4 + j];
        }
    }
    
    // Вычисление остальных слов
    for(i = Nk; i < Nb * (Nr + 1); i++) {
        for(j = 0; j < 4; j++) {
            temp[j] = w[i-1][j][0];
        }
        
        if(i % Nk == 0) {
            // RotWord
            uint8_t k = temp[0];
            temp[0] = temp[1];
            temp[1] = temp[2];
            temp[2] = temp[3];
            temp[3] = k;
            
            // SubWord
            for(j = 0; j < 4; j++) {
                temp[j] = sbox[temp[j]];
            }
            
            // Rcon
            temp[0] ^= 1 << ((i/Nk)-1);
        } else if(Nk > 6 && (i % Nk == 4)) {
            // SubWord для 256-битных ключей
            for(j = 0; j < 4; j++) {
                temp[j] = sbox[temp[j]];
            }
        }
        
        for(j = 0; j < 4; j++) {
            w[i][j][0] = w[i-Nk][j][0] ^ temp[j];
        }
    }
}

// AES шифрование одного блока
void aes_encrypt_block(const uint8_t* input, uint8_t* output, const uint8_t* key, int key_size) {
    int Nr, Nk, Nb = 4; // 128-битный блок
    
    if(key_size == 128) {
        Nk = 4; Nr = 10;
    } else if(key_size == 192) {
        Nk = 6; Nr = 12;
    } else { // 256 bit
        Nk = 8; Nr = 14;
    }
    
    // Раундовые ключи
    uint8_t w[Nb * (Nr + 1)][4][4];
    
    // Расширение ключа
    aes_key_expansion(key, w, Nb, Nk, Nr);
    
    // Копирование входного блока
    uint8_t state[4][4];
    for(int r = 0; r < 4; r++) {
        for(int c = 0; c < 4; c++) {
            state[r][c] = input[c * 4 + r];
        }
    }
    
    // Добавление первого ключа
    for(int r = 0; r < 4; r++) {
        for(int c = 0; c < 4; c++) {
            state[r][c] ^= w[r][c][0];
        }
    }
    
    // Основные раунды
    for(int round = 1; round <= Nr; round++) {
        // SubBytes
        for(int r = 0; r < 4; r++) {
            for(int c = 0; c < 4; c++) {
                state[r][c] = sbox[state[r][c]];
            }
        }
        
        // ShiftRows
        uint8_t temp_state[4][4];
        for(int c = 0; c < 4; c++) {
            for(int r = 0; r < 4; r++) {
                temp_state[r][c] = state[r][(c + r) % 4];
            }
        }
        for(int r = 0; r < 4; r++) {
            for(int c = 0; c < 4; c++) {
                state[r][c] = temp_state[r][c];
            }
        }
        
        // MixColumns (кроме последнего раунда)
        if(round < Nr) {
            for(int c = 0; c < 4; c++) {
                for(int r = 0; r < 4; r++) {
                    temp_state[r][c] = 
                        gmul(mix_column_matrix[r][0], state[0][c]) ^
                        gmul(mix_column_matrix[r][1], state[1][c]) ^
                        gmul(mix_column_matrix[r][2], state[2][c]) ^
                        gmul(mix_column_matrix[r][3], state[3][c]);
                }
            }
            for(int r = 0; r < 4; r++) {
                for(int c = 0; c < 4; c++) {
                    state[r][c] = temp_state[r][c];
                }
            }
        }
        
        // AddRoundKey
        for(int r = 0; r < 4; r++) {
            for(int c = 0; c < 4; c++) {
                state[r][c] ^= w[round * 4 + r][c][0];
            }
        }
    }
    
    // Копирование результата
    for(int r = 0; r < 4; r++) {
        for(int c = 0; c < 4; c++) {
            output[c * 4 + r] = state[r][c];
        }
    }
}

<h3>Secure Boot:</h3>
<p>Безопасная загрузка (Secure Boot) - это механизм, который проверяет подлинность и целостность прошивки перед её выполнением. Это предотвращает запуск несанкционированного кода.</p>

<h4>Пример реализации Secure Boot:</h4>

<pre>
<code>
#include "stm32f4xx_hal.h"
#include <stdint.h>
#include <string.h>

#define BOOT_HEADER_SIZE 256
#define SIGNATURE_OFFSET (BOOT_HEADER_SIZE - 64)  // Подпись находится в конце заголовка
#define HASH_OFFSET (BOOT_HEADER_SIZE - 32)       // Хеш находится перед подписью

// Структура заголовка прошивки
typedef struct {
    uint32_t magic;           // Магическое число для идентификации
    uint32_t version;         // Версия прошивки
    uint32_t size;            // Размер прошивки в байтах
    uint32_t timestamp;       // Время создания
    uint8_t reserved[184];    // Зарезервировано для будущего использования
    uint8_t hash[32];         // SHA-256 хеш прошивки
    uint8_t signature[32];    // ECDSA подпись
} boot_header_t;

// Проверка заголовка прошивки
uint8_t validate_boot_header(const boot_header_t* header) {
    // Проверка магического числа
    if (header->magic != 0xDEADBEEF) {
        return 0; // Неверное магическое число
    }
    
    // Проверка размера
    if (header->size > 0x100000) { // Максимум 1MB
        return 0; // Слишком большой размер
    }
    
    return 1; // Заголовок действителен
}

// Проверка целостности прошивки
uint8_t verify_firmware_integrity(const uint8_t* firmware, uint32_t size) {
    const boot_header_t* header = (const boot_header_t*)firmware;
    
    // Проверка заголовка
    if (!validate_boot_header(header)) {
        return 0;
    }
    
    // Вычисление хеша прошивки (без заголовка)
    uint8_t computed_hash[32];
    sha256(firmware + BOOT_HEADER_SIZE, header->size - BOOT_HEADER_SIZE, computed_hash);
    
    // Сравнение с хешем из заголовка
    if (memcmp(computed_hash, header->hash, 32) != 0) {
        return 0; // Хеши не совпадают
    }
    
    // Проверка подписи (упрощенная проверка)
    if (!verify_ecdsa_signature(firmware + BOOT_HEADER_SIZE, header->size - BOOT_HEADER_SIZE, 
                               header->signature, get_public_key())) {
        return 0; // Подпись недействительна
    }
    
    return 1; // Прошивка действительна
}

// Основная функция безопасной загрузки
void secure_boot_process(void) {
    const uint8_t* firmware = (const uint8_t*)0x08000000; // Адрес начала прошивки
    
    if (verify_firmware_integrity(firmware, get_firmware_size())) {
        // Успешная проверка, разрешаем загрузку
        jump_to_application(0x08000000 + BOOT_HEADER_SIZE);
    } else {
        // Ошибка проверки, переход в режим восстановления
        enter_recovery_mode();
    }
}

// Функция перехода к приложению
void jump_to_application(uint32_t app_addr) {
    // Получение адреса вектора сброса
    uint32_t* app_stack = (uint32_t*)app_addr;
    uint32_t* app_reset = (uint32_t*)(app_addr + 4);
    
    // Установка стека
    __set_MSP(*app_stack);
    
    // Переход к приложению
    asm volatile (
        "mov r0, %0 \n"
        "mov r1, %1 \n"
        "mov r2, %2 \n"
        "mov r3, %3 \n"
        "bx %4 \n"
        :
        : "r" (app_reset), "r" (app_stack), "r" (app_addr), "r" (0), "r" (*app_reset)
        : "r0", "r1", "r2", "r3"
    );
}

// Функция входа в режим восстановления
void enter_recovery_mode(void) {
    // Включить светодиод аварийного режима
    HAL_GPIO_WritePin(GPIOC, GPIO_PIN_13, GPIO_PIN_SET);
    
    // Инициализировать UART для восстановления
    init_recovery_uart();
    
    // Ожидать команд восстановления
    while(1) {
        process_recovery_commands();
        HAL_Delay(10);
    }
}

</pre>

<h3>TrustZone (Cortex‑M33):</h3>
<p>TrustZone для ARMv8-M (включая Cortex-M33) предоставляет аппаратную изоляцию между доверенным и недоверенным кодом. Это позволяет создавать безопасные системы с защитой критических функций.</p>

<h4>Пример использования TrustZone:</h4>

<pre>
<code>
#include "stm32l5xx_hal.h"  // Пример для STM32L5 с Cortex-M33
#include "arm_cmse.h"
#include <stdint.h>

// Определение безопасной функции
__attribute__((cmse_nonsecure_entry))
void secure_function(uint32_t arg1, uint32_t arg2) {
    // Эта функция может быть вызвана из небезопасного мира
    // Она выполняется в безопасном мире
    
    // Обработка аргументов
    if (arg1 == 0x12345678) {
        // Выполнить безопасную операцию
        secure_operation();
    }
    
    // Обновить небезопасные переменные только если безопасно
    if (arg2 == 0xABCDEF00) {
        update_secure_counter();
    }
}

// Безопасная функция для обновления счетчика
void update_secure_counter(void) {
    static volatile uint32_t secure_counter = 0;
    secure_counter++;
}

// Получение защищенного счетчика
__attribute__((cmse_nonsecure_entry))
uint32_t get_secure_counter(void) {
    return secure_counter;

// Инициализация TrustZone
void tz_init(void) {
    // Настройка Secure Attribute Unit (SAU)
    // SAU позволяет указать какие области памяти считаются безопасными
    
    // Включить SAU
    TZ_SAU_Enable();
    
    // Настроить области памяти
    // Область 0: безопасная область кода
    SAU->RNR = 0;  // Выбрать регион 0
    SAU->RBAR = 0x0C000000;  // Начало области
    SAU->RLAR = 0x0C0FFFFF;  // Конец области | SAU_RLAR_ENABLE_Msk;  // Включить регион
    
    // Область 1: безопасная область данных
    SAU->RNR = 1;  // Выбрать регион 1
    SAU->RBAR = 0x20000000;  // Начало области
    SAU->RLAR = 0x20001FFF | SAU_RLAR_ENABLE_Msk;  // Конец области + включить
    
    // Обнулить остальные области
    for(int i = 2; i < 8; i++) {
        SAU->RNR = i;
        SAU->RBAR = 0;
        SAU->RLAR = 0;
    }
    
    // Настроить Secure Fault Handler
    SCB->SHCSR |= SCB_SHCSR_SECUREFAULTENA_Msk;
    
    // Настроить прерывания как безопасные
    NVIC_SetTargetState(EXTI0_IRQn, 1);  // Установить EXTI0 как безопасное прерывание
}

typedef void (*ns_func_ptr)(uint32_t, uint32_t) __attribute__((cmse_nonsecure_call));

void call_secure_function(uint32_t arg1, uint32_t arg2) {
    ns_func_ptr func_ptr = (ns_func_ptr)secure_function_addr;
    
    // Проверить, что указатель находится в небезопасной области
    if (cmse_is_nsfptr(func_ptr)) {
        // Вызвать безопасную функцию
        func_ptr(arg1, arg2);

}

// Инициализация небезопасного мира
void init_nonsecure_world(void) {
    // Отключить все периферийные устройства от безопасного мира
    // Настройка MPU для изоляции
    configure_mpu_for_ns_world();
    
    // Загрузка и запуск небезопасного приложения
    load_and_run_ns_app();
}

// Настройка MPU для небезопасного мира
void configure_mpu_for_ns_world(void) {
    // Отключить MPU на время настройки
    MPU->CTRL = 0;
    
    // Настроить регион для небезопасного SRAM
    MPU->RNR = 0;
    MPU->RBAR = 0x20020000;  // Начало небезопасного SRAM
    MPU->RLAR = 0x2003FFFF | MPU_RLAR_EN_Msk;  // Конец + включить регион
    MPU->RASR = MPU_RASR_ENABLE_Msk | 
                MPU_RASR_ATTR_IDX(0) | 
                MPU_RASR_SIZE_128KB | 
                MPU_RASR_AP_RW_RW_Msk |
                MPU_RASR_XN_Msk;
    
    // Включить MPU
    MPU->CTRL = MPU_CTRL_ENABLE_Msk | MPU_CTRL_PRIVDEFENA_Msk;
}

<h3>Рекомендации по безопасности:</h3>
<ul>
<li>Использовать аппаратные функции безопасности, если доступны (AES, RNG, TrustZone)</li>
<li>Регулярно обновлять прошивку для устранения уязвимостей</li>
<li>Шифровать конфиденциальные данные как в памяти, так и при передаче</li>
<li>Реализовывать защиту от физических атак (обнаружение вскрытия, защита от сброса питания)</li>
<li>Проверять целостность кода и данных при запуске и во время выполнения</li>
<li>Использовать безопасные каналы связи (TLS, DTLS)</li>
<li>Реализовывать минимально необходимые привилегии для компонентов</li>
</ul>
