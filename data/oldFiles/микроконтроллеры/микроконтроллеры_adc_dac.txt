<h2>ADC/DAC (Аналого-цифровой и цифро-аналоговый преобразователи)</h2>

<p><strong>Категория:</strong> Периферия</p>

<h3>Описание</h3>
<p>Аналого-цифровой преобразователь (ADC) и цифро-аналоговый преобразователь (DAC) являются важными компонентами микроконтроллера, обеспечивающими интерфейс между аналоговыми сигналами из окружающей среды и цифровой обработкой внутри микроконтроллера. ADC преобразует аналоговое напряжение в цифровое значение, а DAC делает обратное преобразование.</p>

<h3>Основные характеристики:</h3>
<ul>
<li>Разрядность (обычно 8, 10, 12, 16 бит)</li>
<li>Количество каналов для многоканальных АЦП</li>
<li>Время преобразования</li>
<li>Диапазон входного напряжения</li>
<li>Режимы работы (одиночный, непрерывный, сканирование)</li>
</ul>

<h3>Пример использования ADC:</h3>

<pre>
<code>
#include "stm32f4xx_hal.h"

ADC_HandleTypeDef hadc1;

void ADC1_Init() {
    ADC_ChannelConfTypeDef sConfig = {0};

    // Включить тактирование ADC1
    __HAL_RCC_ADC1_CLK_ENABLE();
    __HAL_RCC_GPIOA_CLK_ENABLE();

    // Настроить GPIO пин PA0 как аналоговый вход
    GPIO_InitTypeDef gpio_init = {0};
    gpio_init.Pin = GPIO_PIN_0;
    gpio_init.Mode = GPIO_MODE_ANALOG;
    gpio_init.Pull = GPIO_NOPULL;
    HAL_GPIO_Init(GPIOA, &gpio_init);

    // Настроить параметры ADC
    hadc1.Instance = ADC1;
    hadc1.Init.ClockPrescaler = ADC_CLOCK_SYNC_PCLK_DIV4;
    hadc1.Init.Resolution = ADC_RESOLUTION_12B;
    hadc1.Init.ScanConvMode = DISABLE;
    hadc1.Init.ContinuousConvMode = ENABLE;  // Непрерывное преобразование
    hadc1.Init.DiscontinuousConvMode = DISABLE;
    hadc1.Init.ExternalTrigConvEdge = ADC_EXTERNALTRIGCONVEDGE_NONE;
    hadc1.Init.ExternalTrigConv = ADC_SOFTWARE_START;
    hadc1.Init.DataAlign = ADC_DATAALIGN_RIGHT;
    hadc1.Init.NbrOfConversion = 1;
    hadc1.Init.DMAContinuousRequests = DISABLE;
    hadc1.Init.EOCSelection = ADC_EOC_SINGLE_CONV;

    if (HAL_ADC_Init(&hadc1) != HAL_OK) {
        Error_Handler();
    }

    // Настроить канал для преобразования
    sConfig.Channel = ADC_CHANNEL_0;  // PA0
    sConfig.Rank = 1;
    sConfig.SamplingTime = ADC_SAMPLETIME_480CYCLES;  // Максимальное время для лучшей точности
    if (HAL_ADC_ConfigChannel(&hadc1, &sConfig) != HAL_OK) {
        Error_Handler();
    }
}

uint32_t ADC_ReadValue() {
    uint32_t adc_value = 0;
    
    HAL_ADC_Start(&hadc1);
    
    if (HAL_ADC_PollForConversion(&hadc1, 100) == HAL_OK) {
        adc_value = HAL_ADC_GetValue(&hadc1);
    }
    
    HAL_ADC_Stop(&hadc1);
    
    return adc_value;
}

float ADC_ValueToVoltage(uint32_t adc_value) {
    // Для 12-битного АЦП и опорного напряжения 3.3В
    return ((float)adc_value * 3.3f) / 4096.0f;
}

int main(void) {
    HAL_Init();
    SystemClock_Config();
    ADC1_Init();

    while(1) {
        uint32_t adc_raw = ADC_ReadValue();
        float voltage = ADC_ValueToVoltage(adc_raw);
        
        // Использование полученного значения для каких-либо целей
        // Например, отправка по UART:
        char buffer[50];
        sprintf(buffer, "ADC: %lu, Voltage: %.2fV\r\n", adc_raw, voltage);
        // UART_SendString(buffer);  // Предполагается наличие UART
        
        HAL_Delay(100);
    }
}
</code>

#include "stm32f4xx_hal.h"

ADC_HandleTypeDef hadc1;
DMA_HandleTypeDef hdma_adc1;

#define ADC_BUFFER_SIZE 100
uint16_t adc_buffer[ADC_BUFFER_SIZE];

void ADC_DMA_Init() {
    ADC_ChannelConfTypeDef sConfig = {0};

    // Включить тактирование
    __HAL_RCC_ADC1_CLK_ENABLE();
    __HAL_RCC_DMA2_CLK_ENABLE();
    __HAL_RCC_GPIOA_CLK_ENABLE();

    // Настроить GPIO
    GPIO_InitTypeDef gpio_init = {0};
    gpio_init.Pin = GPIO_PIN_0;
    gpio_init.Mode = GPIO_MODE_ANALOG;
    gpio_init.Pull = GPIO_NOPULL;
    HAL_GPIO_Init(GPIOA, &gpio_init);

    // Настроить DMA для ADC
    hdma_adc1.Instance = DMA2_Stream0;
    hdma_adc1.Init.Channel = DMA_CHANNEL_0;
    hdma_adc1.Init.Direction = DMA_PERIPH_TO_MEMORY;
    hdma_adc1.Init.PeriphInc = DMA_PINC_DISABLE;
    hdma_adc1.Init.MemInc = DMA_MINC_ENABLE;
    hdma_adc1.Init.PeriphDataAlignment = DMA_PDATAALIGN_HALFWORD;
    hdma_adc1.Init.MemDataAlignment = DMA_MDATAALIGN_HALFWORD;
    hdma_adc1.Init.Mode = DMA_CIRCULAR;
    hdma_adc1.Init.Priority = DMA_PRIORITY_MEDIUM;
    hdma_adc1.Init.FIFOMode = DMA_FIFOMODE_DISABLE;
    if (HAL_DMA_Init(&hdma_adc1) != HAL_OK) {
        Error_Handler();
    }

    __HAL_LINKDMA(&hadc1, DMA_Handle, hdma_adc1);

    // Настроить ADC
    hadc1.Instance = ADC1;
    hadc1.Init.ClockPrescaler = ADC_CLOCK_SYNC_PCLK_DIV4;
    hadc1.Init.Resolution = ADC_RESOLUTION_12B;
    hadc1.Init.ScanConvMode = DISABLE;
    hadc1.Init.ContinuousConvMode = ENABLE;
    hadc1.Init.DiscontinuousConvMode = DISABLE;
    hadc1.Init.ExternalTrigConvEdge = ADC_EXTERNALTRIGCONVEDGE_NONE;
    hadc1.Init.ExternalTrigConv = ADC_SOFTWARE_START;
    hadc1.Init.DataAlign = ADC_DATAALIGN_RIGHT;
    hadc1.Init.NbrOfConversion = 1;
    hadc1.Init.DMAContinuousRequests = ENABLE;
    hadc1.Init.EOCSelection = ADC_EOC_SINGLE_CONV;

    if (HAL_ADC_Init(&hadc1) != HAL_OK) {
        Error_Handler();
    }

    // Настроить канал
    sConfig.Channel = ADC_CHANNEL_0;
    sConfig.Rank = 1;
    sConfig.SamplingTime = ADC_SAMPLETIME_480CYCLES;
    if (HAL_ADC_ConfigChannel(&hadc1, &sConfig) != HAL_OK) {
        Error_Handler();
    }

    // Запустить ADC с DMA
    HAL_ADC_Start_DMA(&hadc1, (uint32_t*)adc_buffer, ADC_BUFFER_SIZE);
}

int main(void) {
    HAL_Init();
    SystemClock_Config();
    ADC_DMA_Init();

    while(1) {
        // Обработка данных из буфера
        for(int i = 0; i < ADC_BUFFER_SIZE; i++) {
            float voltage = ((float)adc_buffer[i] * 3.3f) / 4096.0f;
            // Обработать значение напряжения
        }
        
        HAL_Delay(1000);
    }
}

<h3>Пример использования DAC:</h3>

<pre>
<code>
#include "stm32f4xx_hal.h"

DAC_HandleTypeDef hdac;

void DAC_Init() {
    DAC_ChannelConfTypeDef sConfig = {0};

    // Включить тактирование DAC
    __HAL_RCC_DAC_CLK_ENABLE();
    __HAL_RCC_GPIOA_CLK_ENABLE();

    // Настроить GPIO пин PA4 как выход DAC
    GPIO_InitTypeDef gpio_init = {0};
    gpio_init.Pin = GPIO_PIN_4;
    gpio_init.Mode = GPIO_MODE_ANALOG;
    gpio_init.Pull = GPIO_NOPULL;
    HAL_GPIO_Init(GPIOA, &gpio_init);

    // Настроить DAC
    hdac.Instance = DAC;
    if (HAL_DAC_Init(&hdac) != HAL_OK) {
        Error_Handler();
    }

    // Настроить канал DAC
    sConfig.DAC_Trigger = DAC_TRIGGER_NONE;
    sConfig.DAC_OutputBuffer = DAC_OUTPUTBUFFER_ENABLE;
    if (HAL_DAC_ConfigChannel(&hdac, &sConfig, DAC_CHANNEL_1) != HAL_OK) {
        Error_Handler();
    }

    // Включить канал DAC
    HAL_DAC_Enable(&hdac, DAC_CHANNEL_1);
}

void DAC_SetValue(uint32_t value) {
    HAL_DAC_SetValue(&hdac, DAC_CHANNEL_1, DAC_ALIGN_12B_R, value);
}

// Генерация синусоидального сигнала
const uint16_t sine_wave[32] = {
    2047, 2447, 2831, 3185, 3498, 3755, 3947, 4063,
    4095, 4038, 3901, 3682, 3390, 3038, 2640, 2215,
    1784, 1369, 987, 653, 379, 177, 52, 8,
    40, 155, 350, 620, 959, 1357, 1800, 2274
};

void Generate_SineWave() {
    for(int i = 0; i < 32; i++) {
        HAL_DAC_SetValue(&hdac, DAC_CHANNEL_1, DAC_ALIGN_12B_R, sine_wave[i]);
        HAL_Delay(1);  // Частота сигнала зависит от задержки
    }
}

int main(void) {
    HAL_Init();
    SystemClock_Config();
    DAC_Init();

    while(1) {
        // Пример: изменение выходного напряжения по синусоиде
        Generate_SineWave();
    }
}

<h3>Регистры АЦП:</h3>
<ul>
<li>ADCON - регистр управления</li>
<li>ADCH - регистр результата преобразования</li>
<li>ADCSRA - регистр управления и состояния A</li>
<li>ADMUX - регистр выбора мультиплексора</li>
<li>ADC_DR - регистр данных</li>
<li>ADC_SQR - регистр последовательности каналов</li>
</ul>

<h3>Преимущества использования ADC/DAC:</h3>
<ul>
<li>Возможность работы с аналоговыми датчиками (температура, давление, свет и т.д.)</li>
<li>Преобразование сигналов для обработки в цифровой форме</li>
<li>Воспроизведение аналоговых сигналов</li>
<li>Высокая точность измерений</li>
</ul>

<h3>Рекомендации по работе с ADC/DAC:</h3>
<ul>
<li>Использование фильтров на аналоговых входах для уменьшения шума</li>
<li>Правильный выбор опорного напряжения для достижения максимальной точности</li>
<li>Проверка ограничений по частоте дискретизации</li>
<li>Учет времени установления сигнала при переключении каналов</li>
</ul>
