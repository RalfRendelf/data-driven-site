<h2>GPIO (General Purpose Input/Output)</h2>

<p><strong>Категория:</strong> Периферия</p>

<h3>Описание</h3>
<p>GPIO (General Purpose Input/Output) - это универсальные выводы микроконтроллера, которые могут быть настроены как входы или выходы и использоваться для цифрового взаимодействия с внешними устройствами. GPIO позволяет управлять светодиодами, кнопками, датчиками и другими цифровыми компонентами.</p>

<h3>Основные характеристики:</h3>
<ul>
<li>Возможность настройки каждого пина как вход или выход</li>
<li>Поддержка различных уровней напряжения (обычно 3.3В или 5В)</li>
<li>Встроенные подтягивающие резисторы (pull-up/pull-down)</li>
<li>Возможность работы в режиме с открытым коллектором</li>
<li>Поддержка прерываний по изменению состояния</li>
</ul>

<h3>Пример программы на C для работы с GPIO на STM32:</h3>

<pre>
<code>
#include "stm32f4xx_hal.h"

// Объявление GPIO структуры
GPIO_InitTypeDef GPIO_InitStruct = {0};

void GPIO_Init() {
    // Включить тактирование GPIOA
    __HAL_RCC_GPIOA_CLK_ENABLE();
    
    // Настройка PA5 (обычно подключен к светодиоду на Discovery платах) как выход
    GPIO_InitStruct.Pin = GPIO_PIN_5;
    GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
    GPIO_InitStruct.Pull = GPIO_NOPULL;
    GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
    HAL_GPIO_Init(GPIOA, &GPIO_InitStruct);
    
    // Настройка PA0 (обычно подключен к кнопке) как вход
    GPIO_InitStruct.Pin = GPIO_PIN_0;
    GPIO_InitStruct.Mode = GPIO_MODE_INPUT;
    GPIO_InitStruct.Pull = GPIO_PULLUP;
    HAL_GPIO_Init(GPIOA, &GPIO_InitStruct);
}

void LED_Toggle() {
    HAL_GPIO_TogglePin(GPIOA, GPIO_PIN_5);
}

uint8_t Button_Read() {
    return HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_0);
}

int main(void) {
    // Инициализация HAL
    HAL_Init();
    
    // Настройка системного тактирования
    SystemClock_Config();
    
    // Инициализация GPIO
    GPIO_Init();
    
    while(1) {
        // Если кнопка нажата (PA0 = LOW), включить светодиод
        if(Button_Read() == GPIO_PIN_RESET) {
            HAL_GPIO_WritePin(GPIOA, GPIO_PIN_5, GPIO_PIN_SET);
        } else {
            HAL_GPIO_WritePin(GPIOA, GPIO_PIN_5, GPIO_PIN_RESET);
        }
    }
}

<h3>Регистры GPIO:</h3>
<ul>
<li>MODER - регистр режима (вход/выход/альтернативная функция)</li>
<li>OTYPER - регистр типа выхода (push-pull/open-drain)</li>
<li>OSPEEDR - регистр скорости вывода</li>
<li>PUPDR - регистр подтягивающих резисторов</li>
<li>IDR - регистр данных на входе</li>
<li>ODR - регистр данных на выходе</li>
<li>BSRR - регистр установки/сброса битов</li>
</ul>

<h3>Аналог в мире C:</h3>
<p>GPIO в микроконтроллерах можно эмулировать в C программе, моделируя регистры и состояние пинов:</p>
<ul>
<li>Регистры - целочисленные переменные</li>
<li>Пины - биты в этих переменных</li>
<li>Чтение/запись - побитовые операции</li>
</ul>

<h3>Пример эмуляции GPIO на C:</h3>

<pre>
<code>
/***
 * Эмуляция работы с GPIO
 * Компиляция: gcc -o gpio_emulation gpio_emulation.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdint.h&gt;
#include &lt;stdbool.h&gt;

// Структура для эмуляции GPIO порта
typedef struct {
    volatile uint32_t MODER;   // Регистр режима
    volatile uint32_t OTYPER;  // Регистр типа
    volatile uint32_t OSPEEDR; // Регистр скорости
    volatile uint32_t PUPDR;   // Регистр подтяжки
    volatile uint32_t IDR;     // Регистр данных на входе
    volatile uint32_t ODR;     // Регистр данных на выходе
    volatile uint32_t BSRR;    // Регистр установки/сброса
} GpioPort;

// Инициализация GPIO порта
GpioPort* init_gpio_port() {
    GpioPort* port = (GpioPort*)malloc(sizeof(GpioPort));
    
    port->MODER = 0;
    port->OTYPER = 0;
    port->OSPEEDR = 0;
    port->PUPDR = 0;
    port->IDR = 0;
    port->ODR = 0;
    port->BSRR = 0;
    
    return port;
}

// Настройка пина как вход или выход
void set_pin_mode(GpioPort* port, uint8_t pin_number, bool is_output) {
    if (is_output) {
        port->MODER |= (1 << (pin_number * 2));   // Установка в 01 (output)
        port->MODER &= ~(1 << (pin_number * 2 + 1));
    } else {
        port->MODER &= ~(1 << (pin_number * 2));  // Установка в 00 (input)
        port->MODER &= ~(1 << (pin_number * 2 + 1));
    }
}

// Установка значения на пине
void write_pin(GpioPort* port, uint8_t pin_number, bool value) {
    uint32_t moder_pin = (port->MODER >> (pin_number * 2)) & 0x03;
    
    // Проверка, что пин настроен как выход (режим 01)
    if (moder_pin == 1) {
        if (value) {
            port->ODR |= (1 << pin_number);
        } else {
            port->ODR &= ~(1 << pin_number);
        }
    }
}

// Чтение значения с пина
bool read_pin(GpioPort* port, uint8_t pin_number) {
    uint32_t moder_pin = (port->MODER >> (pin_number * 2)) & 0x03;
    
    // Если пин настроен как вход (режим 00), читаем из IDR
    if (moder_pin == 0) {
        return (port->IDR >> pin_number) & 1;
    } 
    // Иначе если пин настроен как выход, читаем из ODR
    else if (moder_pin == 1) {
        return (port->ODR >> pin_number) & 1;
    }
    
    return false;
}

// Эмуляция функции Toggle (как в STM32 HAL)
void toggle_pin(GpioPort* port, uint8_t pin_number) {
    uint32_t moder_pin = (port->MODER >> (pin_number * 2)) & 0x03;
    
    // Проверка, что пин настроен как выход
    if (moder_pin == 1) {
        port->ODR ^= (1 << pin_number);  // XOR для инвертирования бита
    }
}

// Эмуляция функции BSRR (как в STM32)
void gpio_write(GpioPort* port, uint16_t GPIO_Pin, uint8_t Pin_State) {
    if (Pin_State != 0) {
        // Установка битов через BSRR (верхняя половина для сброса)
        port->BSRR = GPIO_Pin;
        port->ODR |= GPIO_Pin;
    } else {
        // Сброс битов через BSRR (нижняя половина для установки)
        port->BSRR = (uint32_t)GPIO_Pin << 16;
        port->ODR &= ~GPIO_Pin;
    }
}

// Печать состояния GPIO
void print_gpio_status(GpioPort* port) {
    printf("MODER: 0x%08lX, ODR: 0x%08lX, IDR: 0x%08lX\n", 
           (unsigned long)port->MODER, (unsigned long)port->ODR, (unsigned long)port->IDR);
}

int main() {
    GpioPort* gpio_port = init_gpio_port();
    
    printf("Initial GPIO status:\n");
    print_gpio_status(gpio_port);
    
    // Настройка пина 5 как выход
    set_pin_mode(gpio_port, 5, true);
    printf("\nAfter setting pin 5 as output:\n");
    print_gpio_status(gpio_port);
    
    // Настройка пина 3 как вход
    set_pin_mode(gpio_port, 3, false);
    // Имитация входного сигнала на пине 3
    gpio_port->IDR |= (1 << 3);  // Установка 3-го бита в 1
    printf("\nAfter setting pin 3 as input with high signal:\n");
    print_gpio_status(gpio_port);
    
    // Запись значения на пин 5
    write_pin(gpio_port, 5, true);
    printf("\nAfter setting pin 5 high:\n");
    print_gpio_status(gpio_port);
    
    // Чтение значения с пина 3
    bool pin3_value = read_pin(gpio_port, 3);
    printf("\nPin 3 value: %s\n", pin3_value ? "HIGH" : "LOW");
    
    // Переключение значения на пине 5
    toggle_pin(gpio_port, 5);
    printf("\nAfter toggling pin 5:\n");
    print_gpio_status(gpio_port);
    
    // Освобождение памяти
    free(gpio_port);
    
    return 0;
}
