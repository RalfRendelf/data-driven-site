<h2>C++17 (Стандарт C++17)</h2>

<p><strong>Категория:</strong> Стандарты</p>

<h3>Описание</h3>
<p>C++17 (ISO/IEC 14882:2017) — это стандарт языка программирования C++, принятый в 2017 году. Этот стандарт включает в себя множество улучшений, направленных на повышение удобства использования, производительности и безопасности кода. Основные изменения включают structured bindings, if/switch с инициализацией, std::optional, std::variant, std::any, модули файловой системы и другие.</p>

<h3>Основные особенности</h3>
<ol>
<li>Structured bindings</li>
<li>if/switch с инициализацией</li>
<li>std::optional</li>
<li>std::variant</li>
<li>std::any</li>
<li>std::filesystem</li>
<li>Константы времени компиляции (if constexpr)</li>
<li>Распаковка параметров (fold expressions)</li>
<li>Инициализация в range-based циклах</li>
<li>Разделители разрядов</li>
</ol>

<h3>Сложность</h3>
<ul>
<li>Временная сложность: N/A</li>
<li>Пространственная сложность: N/A</li>

<h3>Преимущества и недостатки</h3>
<p>Преимущества:</p>
<ul>
<li>Упрощение синтаксиса и повышение читаемости кода</li>
<li>Более безопасные альтернативы старым возможностям (std::optional)</li>
<li>Улучшенная поддержка метапрограммирования</li>
<li>Улучшенная работа с файловой системой</li>
</ul>

<p>Недостатки:</p>
<ul>
<li>Требует современного компилятора</li>
<li>Увеличение сложности языка</li>
<li>Меньше документации и примеров по сравнению с предыдущими стандартами</li>
</ul>

# Python не имеет отношения к стандартам C++, но демонстрируем схожие концепции
from typing import Optional, Union
import os
from pathlib import Path

# Structured bindings аналогично распаковке в Python
point = (10, 20)
x, y = point  # как structured bindings в C++17

# Optional типы (аналог std::optional)
def safe_divide(a: float, b: float) -> Optional[float]:
    if b == 0:
        return None
    return a / b

# Variant аналогично Union в Python
def process_value(value: Union[int, str, float]) -> str:
    if isinstance(value, int):
        return f"Integer: {value}"
    elif isinstance(value, str):
        return f"String: {value}"
    elif isinstance(value, float):
        return f"Float: {value}"

# Работа с файловой системой (аналог std::filesystem)
path = Path("./example.txt")
if path.exists():
    print(f"File size: {path.stat().st_size} bytes")

# if с инициализацией (аналог в Python - walrus operator в Python 3.8+)
if (result := safe_divide(10, 2)) is not None:
    print(f"Division result: {result}")

// C не поддерживает возможности C++17, но показываем близкие концепции
#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;sys/stat.h&gt;

// Макрос для имитации structured bindings
#define DESTRUCTURE_2(var1, var2, array) \
    int var1 = array[0]; \
    int var2 = array[1];

// Аналог optional через структуру с флагом
typedef struct {
    int value;
    int has_value;
} OptionalInt;

OptionalInt safe_divide(int a, int b) {
    OptionalInt result;
    if (b == 0) {
        result.has_value = 0;
        return result;
    }
    result.value = a / b;
    result.has_value = 1;
    return result;
}

int main() {
    // Имитация structured bindings
    int coords[] = {10, 20};
    DESTRUCTURE_2(x, y, coords);
    printf("Coordinates: (%d, %d)\n", x, y);
    
    // Имитация optional
    OptionalInt result = safe_divide(10, 2);
    if (result.has_value) {
        printf("Division result: %d\n", result.value);
    } else {
        printf("Division by zero!\n");
    }
    
    // Работа с файловой системой
    struct stat info;
    if (stat("./example.txt", &info) == 0) {
        printf("File size: %ld bytes\n", info.st_size);
    }
    
    return 0;
}

#include &lt;iostream&gt;
#include &lt;map&gt;
#include &lt;vector&gt;
#include &lt;optional&gt;
#include &lt;variant&gt;
#include &lt;any&gt;
#include &lt;filesystem&gt;
#include &lt;string&gt;

int main() {
    // Structured bindings
    std::map&lt;std::string, int&gt; m = {{"apple", 1}, {"banana", 2}};
    for (const auto&amp; [key, value] : m) {
        std::cout &lt;&lt; key &lt;&lt; ": " &lt;&lt; value &lt;&lt; std::endl;
    }
    
    // if с инициализацией
    if (auto result = 10 / 2; result > 0) {
        std::cout &lt;&lt; "Positive result: " &lt;&lt; result &lt;&lt; std::endl;
    }
    
    // switch с инициализацией
    switch (auto value = 42; value % 2) {
        case 0:
            std::cout &lt;&lt; "Even number" &lt;&lt; std::endl;
            break;
        case 1:
            std::cout &lt;&lt; "Odd number" &lt;&lt; std::endl;
            break;
    }
    
    // std::optional
    std::optional&lt;int&gt; opt = 42;
    if (opt.has_value()) {
        std::cout &lt;&lt; "Optional value: " &lt;&lt; opt.value() &lt;&lt; std::endl;
    }
    
    // std::variant
    std::variant&lt;int, std::string, double&gt; var = 42;
    std::cout &lt;&lt; "Variant holds int: " &lt;&lt; std::get&lt;int&gt;(var) &lt;&lt; std::endl;
    
    // std::any
    std::any anything = 42;
    if (anything.type() == typeid(int)) {
        std::cout &lt;&lt; "Any holds int: " &lt;&lt; std::any_cast&lt;int&gt;(anything) &lt;&lt; std::endl;
    }
    
    // if constexpr (для шаблонов)
    constexpr bool flag = true;
    if constexpr (flag) {
        std::cout &lt;&lt; "Compile-time condition is true" &lt;&lt; std::endl;
    }
    
    // Fold expressions
    auto sum = [](auto... args) {
        return (args + ... + 0);  // левая свертка
    };
    std::cout &lt;&lt; "Sum: " &lt;&lt; sum(1, 2, 3, 4, 5) &lt;&lt; std::endl;
    
    // std::filesystem
    if (std::filesystem::exists("./")) {
        std::cout &lt;&lt; "Current directory size: " &lt;&lt; 
            std::filesystem::file_size("./main.cpp") &lt;&lt; " bytes" &lt;&lt; std::endl;
    }
    
    return 0;
}

// Java также имеет схожие концепции с C++
import java.io.File;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.util.Optional;
import java.util.Map;
import java.util.HashMap;

public class Cpp17Features {
    public static void main(String[] args) {
        // Structured bindings аналогично деструктуризации в других языках
        // В Java нет прямого аналога, но можно использовать record (Java 14+)
        
        // Optional (начиная с Java 8)
        Optional&lt;Integer&gt; opt = Optional.of(42);
        if (opt.isPresent()) {
            System.out.println("Optional value: " + opt.get());
        }
        
        // Или лучше через orElse
        int value = opt.orElse(-1);
        System.out.println("Value with default: " + value);
        
        // Работа с файловой системой (аналог std::filesystem)
        Path currentDir = Paths.get(".");
        System.out.println("Current directory exists: " + currentDir.toFile().exists());
        
        File file = new File("main.java");
        if (file.exists()) {
            System.out.println("File size: " + file.length() + " bytes");
        }
        
        // Map для демонстрации итерации (аналог structured bindings)
        Map&lt;String, Integer&gt; map = new HashMap&lt;&gt;();
        map.put("one", 1);
        map.put("two", 2);
        
        for (Map.Entry&lt;String, Integer&gt; entry : map.entrySet()) {
            String key = entry.getKey();
            Integer value2 = entry.getValue();
            System.out.println(key + ": " + value2);
        }
        
        // Имитация if с инициализацией
        Integer result = safeDivide(10, 2);
        if (result != null) {
            System.out.println("Division result: " + result);
        }
    }
    
    // Метод для имитации std::optional
    public static Optional&lt;Integer&gt; safeDivideOptional(int a, int b) {
        return b != 0 ? Optional.of(a / b) : Optional.empty();
    }
    
    // Метод с возвратом null для имитации if с инициализацией
    public static Integer safeDivide(int a, int b) {
        return b != 0 ? a / b : null;
    }
}
