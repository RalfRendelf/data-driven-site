<h2>C++23 (Стандарт C++23)</h2>

<p><strong>Категория:</strong> Стандарты</p>

<h3>Описание</h3>
<p>C++23 — это стандарт языка программирования C++, официально принятый в 2023 году. Этот стандарт продолжает развитие языка, добавляя улучшения в существующие функции и вводя новые возможности, направленные на повышение удобства использования, производительности и безопасности кода. Ключевые нововведения включают std::expected, улучшенные диагностики переполнения, упрощенную работу с контейнерами и другие.</p>

<h3>Основные особенности</h3>
<ol>
<li>std::expected</li>
<li>std::flat_map и std::flat_set</li>
<li>Улучшенная диагностика переполнения</li>
<li>std::mdspan для работы с многомерными массивами</li>
<li>Улучшенные макросы __VA_OPT__</li>
<li>std::forward_like для сохранения свойств ссылок</li>
<li>std::lazy_split_view для ленивого разделения</li>
<li>std::byteswap для обмена байтами</li>
<li>Улучшенная поддержка UTF-8</li>
<li>Динамическая атрибутика</li>
</ol>

<li>Временная сложность: N/A</li>

<h3>Преимущества и недостатки</h3>
<p>Преимущества:</p>
<ul>
<li>Улучшенная обработка ошибок с помощью std::expected</li>
<li>Более эффективные контейнеры (flat_map, flat_set)</li>
<li>Улучшенная безопасность при работе с памятью</li>
<li>Упрощенная работа с многомерными массивами</li>
<li>Лучшая диагностика ошибок компиляции</li>
</ul>

<p>Недостатки:</p>
<ul>
<li>Требует современного компилятора с поддержкой C++23</li>
<li>Меньше документации и примеров по сравнению с предыдущими стандартами</li>
<li>Некоторые возможности требуют времени для освоения</li>
</ul>

from typing import Union, Optional, TypeVar
import array
import struct

# Expected аналогично Either в функциональных языках
T = TypeVar('T')
E = TypeVar('E')

class Result:
    def __init__(self, value=None, error=None):
        self.value = value
        self.error = error
    
    def is_ok(self):
        return self.error is None
    
    def unwrap(self):
        if self.is_ok():
            return self.value
        raise Exception(self.error)

def divide(a: float, b: float) -> Result:
    if b == 0:
        return Result(error="Division by zero")
    return Result(value=a/b)

# Работа с многомерными массивами (аналог mdspan)
import numpy as np
matrix = np.array([[1, 2, 3], [4, 5, 6]])
print(matrix.shape)  # (2, 3)

# Обмен байтами (имитация)
def byteswap_32(value: int) -> int:
    # Обмен байтами в 32-битном числе
    return struct.unpack("<I", struct.pack(">I", value))[0]

# Использование
result = divide(10, 2)
if result.is_ok():
    print(f"Division result: {result.unwrap()}")
else:
    print(f"Error: {result.error}")

// C не поддерживает возможности C++23, но показываем близкие концепции
#include &lt;stdio.h&gt;
#include &lt;stdint.h&gt;
#include &lt;stdbool.h&gt;
#include &lt;stdlib.h&gt;

// Структура для имитации std::expected
typedef struct {
    int value;
    bool has_error;
    char* error_message;
} ExpectedInt;

ExpectedInt divide(int a, int b) {
    ExpectedInt result;
    if (b == 0) {
        result.has_error = true;
        result.error_message = "Division by zero";
        return result;
    }
    result.value = a / b;
    result.has_error = false;
    result.error_message = NULL;
    return result;
}

// Обмен байтами (аналог std::byteswap)
uint32_t byteswap_32(uint32_t value) {
    return ((value & 0xFF) << 24) |
           (((value >> 8) & 0xFF) << 16) |
           (((value >> 16) & 0xFF) << 8) |
           ((value >> 24) & 0xFF);
}

// Работа с двумерными массивами (аналог mdspan)
typedef struct {
    int* data;
    size_t rows;
    size_t cols;
} Matrix;

Matrix create_matrix(size_t rows, size_t cols) {
    Matrix m;
    m.rows = rows;
    m.cols = cols;
    m.data = calloc(rows * cols, sizeof(int));
    return m;
}

int get_element(Matrix* m, size_t row, size_t col) {
    return m->data[row * m->cols + col];
}

void set_element(Matrix* m, size_t row, size_t col, int value) {
    m->data[row * m->cols + col] = value;
}

void free_matrix(Matrix* m) {
    free(m->data);
    m->data = NULL;
}

int main() {
    // Использование имитации std::expected
    ExpectedInt result = divide(10, 2);
    if (!result.has_error) {
        printf("Division result: %d\n", result.value);
    } else {
        printf("Error: %s\n", result.error_message);
    }
    
    // Использование имитации mdspan
    Matrix mat = create_matrix(2, 3);
    set_element(&mat, 0, 0, 42);
    printf("Matrix[0][0] = %d\n", get_element(&mat, 0, 0));
    free_matrix(&mat);
    
    // Использование byteswap
    uint32_t swapped = byteswap_32(0x12345678);
    printf("Byteswapped 0x12345678 = 0x%x\n", swapped);
    
    return 0;
}

// Java также имеет схожие концепции с C++
import java.util.Optional;
import java.util.Arrays;
import java.util.List;
import java.util.ArrayList;
import java.util.function.Function;

public class Cpp23Features {
    
    // Имитация std::expected через Optional и отдельную структуру
    static class Expected<T> {
        private final T value;
        private final String error;
        private final boolean hasError;
        
        private Expected(T value, String error, boolean hasError) {
            this.value = value;
            this.error = error;
            this.hasError = hasError;
        }
        
        public static <T> Expected<T> ok(T value) {
            return new Expected<>(value, null, false);
        }
        
        public static <T> Expected<T> error(String error) {
            return new Expected<>(null, error, true);
        }
        
        public boolean hasValue() {
            return !hasError;
        }
        
        public T getValue() {
            if (hasError) {
                throw new RuntimeException("Trying to get value from error state");
            }
            return value;
        }
        
        public String getError() {
            return error;
        }
    }
    
    // Функция деления с возвратом Expected
    public static Expected<Integer> divide(int a, int b) {
        if (b == 0) {
            return Expected.error("Division by zero");
        }
        return Expected.ok(a / b);
    }
    
    // Имитация mdspan для работы с многомерными массивами
    static class Matrix {
        private final int[][] data;
        private final int rows;
        private final int cols;
        
        public Matrix(int rows, int cols) {
            this.rows = rows;
            this.cols = cols;
            this.data = new int[rows][cols];
        }
        
        public int get(int row, int col) {
            return data[row][col];
        }
        
        public void set(int row, int col, int value) {
            data[row][col] = value;
        }
    }
    
    // Байтовый обмен (имитация std::byteswap)
    public static int byteswap32(int value) {
        return Integer.reverseBytes(value);
    }
    
    public static void main(String[] args) {
        // Использование имитации std::expected
        Expected<Integer> result = divide(10, 2);
        if (result.hasValue()) {
            System.out.println("Division result: " + result.getValue());
        } else {
            System.out.println("Error: " + result.getError());
        }
        
        // Использование имитации mdspan
        Matrix matrix = new Matrix(2, 3);
        matrix.set(0, 0, 42);
        System.out.println("Matrix[0][0] = " + matrix.get(0, 0));
        
        // Использование byteswap
        int swapped = byteswap32(0x12345678);
        System.out.printf("Byteswapped 0x%08x = 0x%08x%n", 0x12345678, swapped);
        
        // Работа со списками (аналог flat_map)
        List<String> keys = Arrays.asList("one", "two", "three");
        List<Integer> values = Arrays.asList(1, 2, 3);
        
        // Создание плоской структуры данных
        List<Object[]> flatStructure = new ArrayList<>();
        for (int i = 0; i < keys.size(); i++) {
            flatStructure.add(new Object[]{keys.get(i), values.get(i)});
        }
        
        for (Object[] pair : flatStructure) {
            System.out.println(pair[0] + ": " + pair[1]);
        }
    }
}
