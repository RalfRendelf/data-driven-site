#include <optional>
#include <iostream>

std::optional<int> divide(int a, int b) {
    if (b == 0) {
        return std::nullopt;  // возврат пустого значения
    }
    return a / b;
}

int main() {
    auto result = divide(10, 2);
    if (result.has_value()) {
        std::cout << "Result: " << result.value() << std::endl;
    }
    
    auto result2 = divide(10, 0);
    if (!result2.has_value()) {
        std::cout << "Division by zero!" << std::endl;
    }
    
    // Использование значения по умолчанию
    int value = result2.value_or(-1);
    std::cout << "Default value: " << value << std::endl;
    
    return 0;
}
<h2>std::optional</h2>

<p><strong>Категория:</strong> Контейнеры</p>

<h3>Описание</h3>
<p>std::optional - это шаблонный класс, который позволяет представить значение, которое может отсутствовать. Он предоставляет способ безопасно работать с возможными отсутствующими значениями, избегая необходимости использовать специальные значения или исключения для обозначения отсутствия результата.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++17</p>

// Пример использования std::optional на C++

#include &lt;iostream&gt;
#include &lt;optional&gt;
#include &lt;string&gt;

// Функция, возвращающая optional результат
std::optional&lt;int&gt; divide(int numerator, int denominator) {
    if (denominator == 0) {
        return std::nullopt; // Отсутствие значения
    }
    return numerator / denominator;
}

int main() {
    // Создание optional
    std::optional&lt;int&gt; opt1; // Пустой optional
    std::optional&lt;int&gt; opt2 = 42; // optional с значением
    
    std::cout << "opt1 has value: " << opt1.has_value() << std::endl;
    std::cout << "opt2 has value: " << opt2.has_value() << std::endl;
    
    // Безопасное извлечение значения
    if (opt2.has_value()) {
        std::cout << "Value of opt2: " << opt2.value() << std::endl;
    }
    
    // Использование value_or для получения значения по умолчанию
    std::cout << "opt1 or default: " << opt1.value_or(0) << std::endl;
    std::cout << "opt2 or default: " << opt2.value_or(0) << std::endl;
    
    // Пример использования функции с optional
    auto result1 = divide(10, 2);
    auto result2 = divide(10, 0);
    
    if (result1) {
        std::cout << "10 / 2 = " << *result1 << std::endl;
    }
    
    if (!result2) {
        std::cout << "Cannot divide by zero!" << std::endl;
    }
    
    // Использование emplace
    std::optional&lt;std::string&gt; str_opt;
    str_opt.emplace("Hello, optional!");
    std::cout << "String optional: " << str_opt.value() << std::endl;
    
    return 0;
}

<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o optional_example optional_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;string.h&gt;
#include &lt;stdbool.h&gt;

// Структура для optional целого числа
typedef struct {
    int value;
    bool has_value;
} OptionalInt;

// Структура для optional строки
typedef struct {
    char* value;
    bool has_value;
} OptionalStr;

// Функция создания пустого OptionalInt
OptionalInt optional_int_empty() {
    OptionalInt opt = {0, false};
    return opt;
}

// Функция создания OptionalInt со значением
OptionalInt optional_int_with_value(int value) {
    OptionalInt opt;
    opt.value = value;
    opt.has_value = true;
    return opt;
}

// Функция создания пустого OptionalStr
OptionalStr optional_str_empty() {
    OptionalStr opt;
    opt.value = NULL;
    opt.has_value = false;
    return opt;
}

// Функция создания OptionalStr со значением
OptionalStr optional_str_with_value(const char* value) {
    OptionalStr opt;
    opt.value = malloc(strlen(value) + 1);
    strcpy(opt.value, value);
    opt.has_value = true;
    return opt;
}

// Проверка, содержит ли значение
bool optional_int_has_value(OptionalInt* opt) {
    return opt ? opt->has_value : false;
}

bool optional_str_has_value(OptionalStr* opt) {
    return opt ? opt->has_value : false;
}

// Получение значения или значения по умолчанию
int optional_int_value_or(OptionalInt* opt, int default_value) {
    if (opt && opt->has_value) {
        return opt->value;
    }
    return default_value;
}

char* optional_str_value_or(OptionalStr* opt, const char* default_value) {
    if (opt && opt->has_value && opt->value) {
        return opt->value;
    }
    return (char*)default_value;
}

// Освобождение памяти для OptionalStr
void optional_str_destroy(OptionalStr* opt) {
    if (opt && opt->has_value && opt->value) {
        free(opt->value);
        opt->value = NULL;
        opt->has_value = false;
    }
}

// Функция деления, возвращающая optional результат
OptionalInt divide(int numerator, int denominator) {
    if (denominator == 0) {
        return optional_int_empty(); // Отсутствие значения
    }
    return optional_int_with_value(numerator / denominator);
}

// Пример использования
int main() {
    // Создание optional
    OptionalInt opt1 = optional_int_empty(); // Пустой optional
    OptionalInt opt2 = optional_int_with_value(42); // optional с значением
    
    printf("opt1 has value: %s\n", optional_int_has_value(&opt1) ? "true" : "false");
    printf("opt2 has value: %s\n", optional_int_has_value(&opt2) ? "true" : "false");
    
    // Безопасное извлечение значения
    if (optional_int_has_value(&opt2)) {
        printf("Value of opt2: %d\n", opt2.value);
    }
    
    // Использование value_or для получения значения по умолчанию
    printf("opt1 or default: %d\n", optional_int_value_or(&opt1, 0));
    printf("opt2 or default: %d\n", optional_int_value_or(&opt2, 0));
    
    // Пример использования функции с optional
    OptionalInt result1 = divide(10, 2);
    OptionalInt result2 = divide(10, 0);
    
    if (optional_int_has_value(&result1)) {
        printf("10 / 2 = %d\n", result1.value);
    }
    
    if (!optional_int_has_value(&result2)) {
        printf("Cannot divide by zero!\n");
    }
    
    // Работа с optional строкой
    OptionalStr str_opt = optional_str_empty();
    printf("String optional has value: %s\n", optional_str_has_value(&str_opt) ? "true" : "false");
    
    // Присваивание значения
    str_opt = optional_str_with_value("Hello, optional!");
    if (optional_str_has_value(&str_opt)) {
        printf("String optional: %s\n", str_opt.value);
    }
    
    // Использование value_or для строки
    OptionalStr empty_str = optional_str_empty();
    printf("Empty string or default: %s\n", optional_str_value_or(&empty_str, "default"));
    
    // Освобождение памяти
    optional_str_destroy(&str_opt);
    
    return 0;
}
