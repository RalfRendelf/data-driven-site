<h2>std::enable_shared_from_this</h2>

<p><strong>Категория:</strong> Умные указатели</p>

<h3>Описание</h3>
<p>std::enable_shared_from_this - это шаблонный класс, который позволяет объекту, который уже управляется std::shared_ptr, получить дополнительные std::shared_ptr к себе. Это полезно, когда объекту нужно передать себя в качестве shared_ptr в другую функцию или объект.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++11</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::enable_shared_from_this и их реализация на C (для демонстрации аналогов):</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка POSIX threads
 * Компиляция: gcc -o enable_shared_from_this_example enable_shared_from_this_example.c -lpthread
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;pthread.h&gt;
#include &lt;string.h&gt;

// Структура для отслеживания разделяемого объекта
typedef struct SharedObject SharedObject;

struct SharedObject {
    void* ptr;
    int* ref_count;
    int* weak_ref_count;
    void (*deleter)(void*);
    pthread_mutex_t mutex;
    int destroyed;
    // Ссылка на менеджер, чтобы получить доступ к shared_ptr
    void* manager;
};

// Структура для хранения shared_ptr
typedef struct {
    SharedObject* obj;
} SharedPtr;

// Структура для хранения weak_ptr
typedef struct {
    SharedObject* obj;
} WeakPtr;

// Менеджер для объекта, который хочет получить shared_ptr к себе
typedef struct {
    SharedObject* obj;
    pthread_mutex_t mutex;
} EnableSharedFromThisManager;

// Функции для работы с SharedObject
SharedObject* shared_object_create(void* ptr, void (*deleter)(void*)) {
    SharedObject* obj = malloc(sizeof(SharedObject));
    obj->ptr = ptr;
    obj->ref_count = malloc(sizeof(int));
    *(obj->ref_count) = 0;
    obj->weak_ref_count = malloc(sizeof(int));
    *(obj->weak_ref_count) = 0;
    obj->deleter = deleter ? deleter : &free;
    obj->destroyed = 0;
    pthread_mutex_init(&obj->mutex, NULL);
    obj->manager = NULL;
    return obj;
}

void shared_object_destroy(SharedObject* obj) {
    if (obj) {
        if (obj->ptr && obj->deleter && !obj->destroyed) {
            obj->deleter(obj->ptr);
            obj->destroyed = 1;
        }
        if (obj->ref_count) free(obj->ref_count);
        if (obj->weak_ref_count) free(obj->weak_ref_count);
        pthread_mutex_destroy(&obj->mutex);
        free(obj);
    }
}

// Функции для работы с shared_ptr
SharedPtr* shared_ptr_create(void* ptr, void (*deleter)(void*)) {
    SharedPtr* sptr = malloc(sizeof(SharedPtr));
    sptr->obj = shared_object_create(ptr, deleter);
    pthread_mutex_lock(&sptr->obj->mutex);
    (*(sptr->obj->ref_count))++;
    pthread_mutex_unlock(&sptr->obj->mutex);
    return sptr;
}

SharedPtr* shared_ptr_copy(SharedPtr* original) {
    if (!original || !original->obj) return NULL;
    
    SharedPtr* copy = malloc(sizeof(SharedPtr));
    copy->obj = original->obj;
    
    pthread_mutex_lock(&original->obj->mutex);
    (*(original->obj->ref_count))++;
    pthread_mutex_unlock(&original->obj->mutex);
    
    return copy;
}

void shared_ptr_destroy(SharedPtr* sptr) {
    if (!sptr || !sptr->obj) return;
    
    SharedObject* obj = sptr->obj;
    pthread_mutex_lock(&obj->mutex);
    (*(obj->ref_count))--;
    
    if (*(obj->ref_count) == 0 && *(obj->weak_ref_count) == 0) {
        // Все ссылки уничтожены, можно освободить объект
        shared_object_destroy(obj);
    } else {
        pthread_mutex_unlock(&obj->mutex);
    }
    
    free(sptr);
}

void* shared_ptr_get(SharedPtr* sptr) {
    return sptr && sptr->obj ? sptr->obj->ptr : NULL;
}

int shared_ptr_use_count(SharedPtr* sptr) {
    if (!sptr || !sptr->obj) return 0;
    
    pthread_mutex_lock(&sptr->obj->mutex);
    int count = *(sptr->obj->ref_count);
    pthread_mutex_unlock(&sptr->obj->mutex);
    return count;
}

// Функции для работы с weak_ptr
WeakPtr* weak_ptr_create(SharedPtr* sptr) {
    if (!sptr || !sptr->obj) return NULL;
    
    WeakPtr* wptr = malloc(sizeof(WeakPtr));
    wptr->obj = sptr->obj;
    
    pthread_mutex_lock(&sptr->obj->mutex);
    (*(sptr->obj->weak_ref_count))++;
    pthread_mutex_unlock(&sptr->obj->mutex);
    
    return wptr;
}

SharedPtr* weak_ptr_lock(WeakPtr* wptr) {
    if (!wptr || !wptr->obj) return NULL;
    
    pthread_mutex_lock(&wptr->obj->mutex);
    if (wptr->obj->destroyed) {
        pthread_mutex_unlock(&wptr->obj->mutex);
        return NULL;
    }
    
    SharedPtr* new_sptr = malloc(sizeof(SharedPtr));
    new_sptr->obj = wptr->obj;
    (*(wptr->obj->ref_count))++;
    pthread_mutex_unlock(&wptr->obj->mutex);
    
    return new_sptr;
}

void weak_ptr_destroy(WeakPtr* wptr) {
    if (!wptr || !wptr->obj) return;
    
    SharedObject* obj = wptr->obj;
    pthread_mutex_lock(&obj->mutex);
    (*(obj->weak_ref_count))--;
    
    if (*(obj->ref_count) == 0 && *(obj->weak_ref_count) == 0) {
        // Все ссылки уничтожены, можно освободить объект
        shared_object_destroy(obj);
    } else {
        pthread_mutex_unlock(&obj->mutex);
    }
    
    free(wptr);
}

// Инициализация менеджера для enable_shared_from_this
EnableSharedFromThisManager* enable_shared_from_this_manager_create() {
    EnableSharedFromThisManager* manager = malloc(sizeof(EnableSharedFromThisManager));
    pthread_mutex_init(&manager->mutex, NULL);
    return manager;
}

void enable_shared_from_this_manager_destroy(EnableSharedFromThisManager* manager) {
    if (manager) {
        pthread_mutex_destroy(&manager->mutex);
        free(manager);
    }
}

// Функция, позволяющая объекту получить shared_ptr к себе
SharedPtr* shared_from_this(EnableSharedFromThisManager* manager) {
    if (!manager) return NULL;
    
    // В реальной реализации std::enable_shared_from_this это работает через внутреннюю
    // структуру, но в нашей реализации мы будем использовать weak_ptr, 
    // который был установлен заранее
    pthread_mutex_lock(&manager->mutex);
    WeakPtr* wptr = (WeakPtr*)manager->obj;  // Предполагаем, что obj содержит weak_ptr
    pthread_mutex_unlock(&manager->mutex);
    
    if (wptr) {
        return weak_ptr_lock(wptr);
    }
    
    return NULL;
}

// Установка связи менеджера с объектом
void set_manager_object(EnableSharedFromThisManager* manager, SharedObject* obj) {
    pthread_mutex_lock(&manager->mutex);
    manager->obj = obj;  // Здесь мы храним weak_ptr к объекту
    pthread_mutex_unlock(&manager->mutex);
}

// Структура для демонстрации объекта, который может получить shared_ptr к себе
typedef struct {
    int id;
    char name[50];
    EnableSharedFromThisManager* manager;
} SelfSharedObject;

SelfSharedObject* self_shared_object_create(int id, const char* name) {
    SelfSharedObject* obj = malloc(sizeof(SelfSharedObject));
    obj->id = id;
    strncpy(obj->name, name, sizeof(obj->name) - 1);
    obj->name[sizeof(obj->name) - 1] = '\0';
    obj->manager = enable_shared_from_this_manager_create();
    return obj;
}

void self_shared_object_destroy(SelfSharedObject* obj) {
    if (obj) {
        if (obj->manager) {
            enable_shared_from_this_manager_destroy(obj->manager);
        }
        free(obj);
    }
}

// Функция, позволяющая объекту получить shared_ptr к себе
SharedPtr* self_shared_object_get_shared_ptr(SelfSharedObject* obj) {
    if (!obj || !obj->manager) return NULL;
    return shared_from_this(obj->manager);
}

// Пример использования
int main() {
    printf("=== Демонстрация enable_shared_from_this (аналог) ===\n\n");
    
    // Создаем объект, который может получить shared_ptr к себе
    SelfSharedObject* raw_obj = self_shared_object_create(123, "Test Object");
    
    // Создаем shared_ptr для объекта
    SharedPtr* obj_ptr = shared_ptr_create(raw_obj, (void (*)(void*))&self_shared_object_destroy);
    
    // Создаем weak_ptr и устанавливаем его в менеджере
    WeakPtr* obj_weak_ptr = weak_ptr_create(obj_ptr);
    set_manager_object(raw_obj->manager, obj_weak_ptr->obj);
    
    printf("Исходный shared_ptr имеет %d ссылок\n", shared_ptr_use_count(obj_ptr));
    
    // Теперь объект может получить shared_ptr к себе
    SharedPtr* another_ref = self_shared_object_get_shared_ptr(raw_obj);
    
    if (another_ref) {
        SelfSharedObject* accessed_obj = (SelfSharedObject*)shared_ptr_get(another_ref);
        printf("Объект получен через shared_from_this: ID=%d, Name=%s\n", 
               accessed_obj->id, accessed_obj->name);
        printf("Теперь объект имеет %d ссылок\n", shared_ptr_use_count(obj_ptr));
        
        // Используем дополнительный shared_ptr
        printf("Дополнительный shared_ptr также указывает на объект: ID=%d\n", 
               ((SelfSharedObject*)shared_ptr_get(another_ref))->id);
        
        // Уничтожаем дополнительный shared_ptr
        shared_ptr_destroy(another_ref);
    } else {
        printf("Не удалось получить shared_ptr к себе\n");
    }
    
    // Проверяем количество ссылок после уничтожения
    printf("После уничтожения дополнительного shared_ptr: %d ссылок\n", 
           shared_ptr_use_count(obj_ptr));
    
    // Создаем еще один shared_ptr через shared_from_this
    SharedPtr* third_ref = self_shared_object_get_shared_ptr(raw_obj);
    if (third_ref) {
        printf("Получен третий shared_ptr через shared_from_this\n");
        printf("Теперь объект имеет %d ссылок\n", shared_ptr_use_count(obj_ptr));
        
        // Уничтожаем третий shared_ptr
        shared_ptr_destroy(third_ref);
        printf("После уничтожения третьего shared_ptr: %d ссылок\n", 
               shared_ptr_use_count(obj_ptr));
    }
    
    // Уничтожаем исходный shared_ptr
    shared_ptr_destroy(obj_ptr);
    weak_ptr_destroy(obj_weak_ptr);
    
    printf("Объект должен быть уничтожен, так как все shared_ptr уничтожены\n");
    
    return 0;
}
