<h2>std::stop_token</h2>

<p><strong>Категория:</strong> Многопоточность</p>

<h3>Описание</h3>
<p>std::stop_token - это класс, который предоставляет способ проверки запроса остановки асинхронной операции. Он используется вместе с std::jthread и другими механизмами для координации остановки асинхронных операций. Введен в C++20.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++20</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::stop_token:</p>

<pre>
<code>
// Пример использования std::stop_token на C++

#include &lt;iostream&gt;
#include &lt;thread&gt;
#include &lt;chrono&gt;
#include &lt;stop_token&gt;

// Функция, которая будет выполняться в потоке
void worker_function(std::stop_token stoken, int id) {
    std::cout << "Thread " << id << " started" << std::endl;
    
    // Цикл с проверкой токена остановки
    while (!stoken.stop_requested()) {
        std::cout << "Thread " << id << " doing work..." << std::endl;
        
        // Проверяем токен остановки каждые 500 мс
        for (int i = 0; i < 5; ++i) {
            if (stoken.stop_requested()) {
                break;
            }
            std::this_thread::sleep_for(std::chrono::milliseconds(100));
        }
    }
    
    std::cout << "Thread " << id << " stopping..." << std::endl;
}

int main() {
    std::jthread t1(worker_function, 1);
    std::jthread t2(worker_function, 2);
    
    std::cout << "Main thread sleeping for 2 seconds..." << std::endl;
    std::this_thread::sleep_for(std::chrono::seconds(2));
    
    // Запрос остановки для первого потока
    std::cout << "Requesting stop for thread 1..." << std::endl;
    t1.request_stop();
    
    std::cout << "Waiting for thread 1 to finish..." << std::endl;
    
    // t1 будет автоматически присоединен при выходе из области видимости
    
    std::cout << "Thread 1 finished, continuing with thread 2..." << std::endl;
    std::this_thread::sleep_for(std::chrono::seconds(1));
    
    std::cout << "Program ending, thread 2 will be stopped automatically." << std::endl;
    
    return 0;
}
</code>
</pre>

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка POSIX threads
 * Компиляция: gcc -o stop_token_example stop_token_example.c -lpthread
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;pthread.h&gt;
#include &lt;unistd.h&gt;
#include &lt;stdatomic.h&gt;

// Структура, имитирующая функциональность stop_token
typedef struct {
    atomic_bool* stop_requested_flag;
} StopToken;

// Структура для передачи данных в поток
typedef struct {
    StopToken token;
    int id;
} ThreadData;

// Функция для проверки запроса остановки
int stop_requested(StopToken* token) {
    return atomic_load(token->stop_requested_flag);
}

// Функция, которая будет выполняться в потоке
void* worker_function(void* arg) {
    ThreadData* data = (ThreadData*)arg;
    StopToken* stoken = &data->token;
    
    printf("Thread %d started\n", data->id);
    
    // Цикл с проверкой токена остановки
    while (!stop_requested(stoken)) {
        printf("Thread %d doing work...\n", data->id);
        
        // Проверяем токен остановки каждые 500 мс
        for (int i = 0; i < 5; ++i) {
            if (stop_requested(stoken)) {
                break;
            }
            usleep(100000); // спим 100 мс
        }
    }
    
    printf("Thread %d stopping...\n", data->id);
    return NULL;
}

// Функция для создания и запуска потока с токеном остановки
pthread_t create_thread_with_token(int id) {
    // Создаем флаг остановки
    atomic_bool* stop_flag = malloc(sizeof(atomic_bool));
    atomic_init(stop_flag, false);
    
    // Подготавливаем данные для потока
    ThreadData* data = malloc(sizeof(ThreadData));
    data->token.stop_requested_flag = stop_flag;
    data->id = id;
    
    // Создаем поток
    pthread_t thread;
    pthread_create(&thread, NULL, worker_function, data);
    
    // Сохраняем токен в структуре pthread
    pthread_setspecific(pthread_key_create(NULL, NULL), stop_flag);
    
    return thread;
}

// Функция для установки флага остановки
void request_stop(atomic_bool* flag) {
    if (flag) {
        atomic_store(flag, true);
    }
}

// Пример использования
int main() {
    // Создаем флаги остановки для двух потоков
    atomic_bool stop_flag1, stop_flag2;
    atomic_init(&stop_flag1, false);
    atomic_init(&stop_flag2, false);
    
    // Подготавливаем данные для потоков
    ThreadData data1, data2;
    data1.token.stop_requested_flag = &stop_flag1;
    data1.id = 1;
    data2.token.stop_requested_flag = &stop_flag2;
    data2.id = 2;
    
    // Создаем потоки
    pthread_t t1, t2;
    pthread_create(&t1, NULL, worker_function, &data1);
    pthread_create(&t2, NULL, worker_function, &data2);
    
    printf("Main thread sleeping for 2 seconds...\n");
    usleep(2000000); // спим 2 секунды
    
    // Запрос остановки для первого потока
    printf("Requesting stop for thread 1...\n");
    request_stop(&stop_flag1);
    
    printf("Waiting for thread 1 to finish...\n");
    pthread_join(t1, NULL);
    
    printf("Thread 1 finished, continuing with thread 2...\n");
    usleep(1000000); // спим 1 секунду
    
    printf("Stopping thread 2...\n");
    request_stop(&stop_flag2);
    pthread_join(t2, NULL);
    
    printf("Both threads finished.\n");
    
    // Освобождаем память
    free((void*)&stop_flag1);
    free((void*)&stop_flag2);
    
    return 0;
}
