<h2>std::queue</h2>

<p><strong>Категория:</strong> Контейнеры</p>

<h3>Описание</h3>
<p>std::queue - это контейнер адаптер, реализующий структуру данных FIFO (First In, First Out). Элементы добавляются в конец очереди и извлекаются из начала очереди. По умолчанию использует std::deque в качестве базового контейнера.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++98</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::queue:</p>

<pre>
<code>
// Пример использования std::queue на C++

#include &lt;iostream&gt;
#include &lt;queue&gt;

int main() {
    // Создание очереди
    std::queue&lt;int&gt; q;
    
    // Добавление элементов в очередь
    q.push(1);
    q.push(2);
    q.push(3);
    q.push(4);
    
    std::cout &lt;&lt; "Размер очереди: " &lt;&lt; q.size() &lt;&lt; std::endl;
    
    // Извлечение элементов из очереди
    std::cout &lt;&lt; "Элементы очереди: ";
    while (!q.empty()) {
        std::cout &lt;&lt; q.front() &lt;&lt; " "; // Просмотр первого элемента
        q.pop();                       // Удаление первого элемента
    }
    std::cout &lt;&lt; std::endl;
    
    // Проверка на пустоту
    std::cout &lt;&lt; "Очередь пуста: " &lt;&lt; (q.empty() ? "да" : "нет") &lt;&lt; std::endl;
    
    // Добавление элементов
    for (int i = 1; i &lt;= 5; ++i) {
        q.push(i * 10);
    }
    
    // Просмотр первого и последнего элементов
    std::cout &lt;&lt; "Первый элемент: " &lt;&lt; q.front() &lt;&lt; std::endl;
    std::cout &lt;&lt; "Последний элемент: " &lt;&lt; q.back() &lt;&lt; std::endl;
    
    // Размер очереди
    std::cout &lt;&lt; "Размер очереди: " &lt;&lt; q.size() &lt;&lt; std::endl;
    
    return 0;
}

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o queue_example queue_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;

#define INITIAL_CAPACITY 8

// Структура для очереди
typedef struct {
    int* data;
    size_t front;
    size_t rear;
    size_t size;
    size_t capacity;
} Queue;

// Инициализация очереди
Queue* queue_create() {
    Queue* q = malloc(sizeof(Queue));
    q->data = malloc(INITIAL_CAPACITY * sizeof(int));
    q->front = 0;
    q->rear = 0;
    q->size = 0;
    q->capacity = INITIAL_CAPACITY;
    return q;
}

// Уничтожение очереди
void queue_destroy(Queue* q) {
    if (!q) return;
    if (q->data) free(q->data);
    free(q);
}

// Проверка на пустоту
int queue_empty(Queue* q) {
    return q ? (q->size == 0) : 1;
}

// Получение размера
size_t queue_size(Queue* q) {
    return q ? q->size : 0;
}

// Добавление элемента в конец очереди
void queue_push(Queue* q, int value) {
    if (!q) return;
    
    // Увеличение размера, если необходимо
    if (q->size == q->capacity) {
        q->capacity *= 2;
        int* new_data = realloc(q->data, q->capacity * sizeof(int));
        if (!new_data) return; // ошибка выделения памяти
        q->data = new_data;
        
        // Если элементы оборачиваются через конец массива
        if (q->rear < q->front) {
            // Перемещаем элементы с начала массива в конец выделенного участка
            for (size_t i = 0; i < q->rear; i++) {
                q->data[q->capacity/2 + i] = q->data[i];
            }
            q->rear = q->capacity/2 + q->rear;
        }
    }
    
    q->data[q->rear] = value;
    q->rear = (q->rear + 1) % q->capacity;
    q->size++;
}

// Удаление первого элемента
void queue_pop(Queue* q) {
    if (!q || q->size == 0) return;
    
    q->front = (q->front + 1) % q->capacity;
    q->size--;
}

// Получение первого элемента
int queue_front(Queue* q) {
    if (!q || q->size == 0) return 0; // или другое значение по умолчанию
    return q->data[q->front];
}

// Получение последнего элемента
int queue_back(Queue* q) {
    if (!q || q->size == 0) return 0; // или другое значение по умолчанию
    size_t idx = (q->rear == 0) ? q->capacity - 1 : q->rear - 1;
    return q->data[idx];
}

// Печать очереди
void queue_print(Queue* q, const char* label) {
    printf("%s: [", label);
    if (!queue_empty(q)) {
        size_t current = q->front;
        for (size_t i = 0; i < q->size; i++) {
            printf("%d", q->data[current]);
            if (i < q->size - 1) printf(", ");
            current = (current + 1) % q->capacity;
        }
    }
    printf("] (size: %zu, front: %zu, rear: %zu)\n", 
           queue_size(q), q->front, q->rear);
}

// Пример использования
int main() {
    Queue* q = queue_create();
    
    // Добавление элементов в очередь
    queue_push(q, 1);
    queue_push(q, 2);
    queue_push(q, 3);
    queue_push(q, 4);
    
    queue_print(q, "Очередь после добавления 1, 2, 3, 4");
    printf("Размер очереди: %zu\n", queue_size(q));
    
    // Извлечение элементов из очереди
    printf("Элементы очереди: ");
    while (!queue_empty(q)) {
        printf("%d ", queue_front(q)); // Просмотр первого элемента
        queue_pop(q);                  // Удаление первого элемента
    }
    printf("\n");
    
    // Проверка на пустоту
    printf("Очередь пуста: %s\n", queue_empty(q) ? "да" : "нет");
    
    // Добавление новых элементов
    for (int i = 1; i <= 5; ++i) {
        queue_push(q, i * 10);
    }
    
    queue_print(q, "Очередь после добавления 10, 20, 30, 40, 50");
    
    // Просмотр первого и последнего элементов
    printf("Первый элемент: %d\n", queue_front(q));
    printf("Последний элемент: %d\n", queue_back(q));
    
    // Размер очереди
    printf("Размер очереди: %zu\n", queue_size(q));
    
    // Уничтожение очереди
    queue_destroy(q);
    
    return 0;
}
