<h2>std::expected (C++23)</h2>

<p><strong>Категория:</strong> Контейнеры</p>

<h3>Описание</h3>
<p>std::expected - это класс, добавленный в C++23, предназначенный для представления значений, которые могут быть успешными или содержать ошибку. Он предоставляет альтернативу для обработки ошибок через исключения или специальные значения, позволяя возвращать как результат, так и причину ошибки в одном объекте.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++23</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные концепции std::expected и их реализация на C (для демонстрации аналогов):</p>

<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o expected_example expected_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;string.h&gt;

// Типы для определения состояния Expected
typedef enum {
    EXPECTED_VALUE,
    EXPECTED_ERROR
} ExpectedState;

// Коды ошибок
typedef enum {
    ERROR_NONE = 0,
    ERROR_DIVISION_BY_ZERO,
    ERROR_INVALID_ARGUMENT,
    ERROR_OUT_OF_RANGE,
    ERROR_FILE_NOT_FOUND,
    ERROR_PERMISSION_DENIED
} ErrorCode;

// Структура для хранения ошибки
typedef struct {
    ErrorCode code;
    char message[256];
} Error;

// Основная структура для Expected
typedef struct {
    ExpectedState state;
    union {
        double value;        // для успешного результата
        Error error;         // для ошибки
    } data;
} Expected;

// Создание Expected с успешным значением
Expected expected_with_value(double value) {
    Expected ex;
    ex.state = EXPECTED_VALUE;
    ex.data.value = value;
    return ex;
}

// Создание Expected с ошибкой
Expected expected_with_error(ErrorCode code, const char* message) {
    Expected ex;
    ex.state = EXPECTED_ERROR;
    ex.data.error.code = code;
    strncpy(ex.data.error.message, message, sizeof(ex.data.error.message) - 1);
    ex.data.error.message[sizeof(ex.data.error.message) - 1] = '\0';
    return ex;
}

// Проверка, содержит ли Expected значение
int expected_has_value(Expected* ex) {
    if (!ex) return 0;
    return ex->state == EXPECTED_VALUE;
}

// Проверка, содержит ли Expected ошибку
int expected_has_error(Expected* ex) {
    if (!ex) return 1;
    return ex->state == EXPECTED_ERROR;
}

// Получение значения (требуется проверка на наличие значения!)
double expected_get_value(Expected* ex) {
    if (ex && ex->state == EXPECTED_VALUE) {
        return ex->data.value;
    }
    return 0.0; // значение по умолчанию в случае ошибки
}

// Получение ошибки
Error* expected_get_error(Expected* ex) {
    if (ex && ex->state == EXPECTED_ERROR) {
        return &(ex->data.error);
    }
    return NULL;
}

// Функция для безопасного получения значения с значением по умолчанию
double expected_value_or(Expected* ex, double default_value) {
    if (expected_has_value(ex)) {
        return expected_get_value(ex);
    }
    return default_value;
}

// Функция, которая может завершиться успешно или с ошибкой (деление)
Expected safe_divide(double dividend, double divisor) {
    if (divisor == 0.0) {
        return expected_with_error(ERROR_DIVISION_BY_ZERO, "Деление на ноль");
    }
    
    return expected_with_value(dividend / divisor);
}

// Функция, которая может завершиться успешно или с ошибкой (квадратный корень)
Expected safe_sqrt(double value) {
    if (value < 0.0) {
        return expected_with_error(ERROR_INVALID_ARGUMENT, "Квадратный корень из отрицательного числа");
    }
    
    // Простая аппроксимация sqrt (для демонстрации)
    if (value == 0.0) return expected_with_value(0.0);
    
    double x = value;
    for (int i = 0; i < 20; i++) {
        x = 0.5 * (x + value / x);  // метод Ньютона
    }
    
    return expected_with_value(x);
}

// Функция, которая может завершиться успешно или с ошибкой (преобразование строки в число)
Expected string_to_double(const char* str) {
    if (!str || strlen(str) == 0) {
        return expected_with_error(ERROR_INVALID_ARGUMENT, "Пустая строка");
    }
    
    char* endptr;
    double result = strtod(str, &endptr);
    
    if (*endptr != '\0') {
        return expected_with_error(ERROR_INVALID_ARGUMENT, "Неверный формат числа");
    }
    
    return expected_with_value(result);
}

// Функция для отображения (map) значения
Expected expected_map(Expected* ex, double (*func)(double)) {
    if (expected_has_value(ex)) {
        return expected_with_value(func(expected_get_value(ex)));
    }
    
    // Если была ошибка, возвращаем ту же ошибку
    return *ex;
}

// Функция для плоского отображения (flatMap) - для цепочек операций
Expected expected_flat_map(Expected* ex, Expected (*func)(double)) {
    if (expected_has_value(ex)) {
        return func(expected_get_value(ex));
    }
    
    // Если была ошибка, возвращаем ту же ошибку
    return *ex;
}

// Функция для печати Expected
void print_expected(Expected* ex, const char* label) {
    if (expected_has_value(ex)) {
        printf("%s: SUCCESS(value=%.2f)\n", label, expected_get_value(ex));
    } else {
        Error* err = expected_get_error(ex);
        printf("%s: ERROR(code=%d, message=\"%s\")\n", label, err->code, err->message);
    }
}

// Демонстрация использования Expected
void demonstrate_expected_usage() {
    printf("=== Демонстрация std::expected (аналог) ===\n\n");
    
    // Простое использование
    Expected ex1 = safe_divide(10.0, 2.0);
    print_expected(&ex1, "10 / 2");
    
    Expected ex2 = safe_divide(10.0, 0.0);
    print_expected(&ex2, "10 / 0 (ошибка)");
    
    Expected ex3 = safe_sqrt(25.0);
    print_expected(&ex3, "sqrt(25)");
    
    Expected ex4 = safe_sqrt(-5.0);
    print_expected(&ex4, "sqrt(-5) (ошибка)");
    
    // Преобразование строки в число
    Expected ex5 = string_to_double("123.45");
    print_expected(&ex5, "string_to_double(\"123.45\")");
    
    Expected ex6 = string_to_double("invalid");
    print_expected(&ex6, "string_to_double(\"invalid\")");
    
    // Использование значения по умолчанию
    printf("\nИспользование значений по умолчанию:\n");
    double val1 = expected_value_or(&ex2, -1.0);  // деление на ноль
    printf("Значение ex2 или -1.0: %.2f\n", val1);
    
    double val2 = expected_value_or(&ex3, -1.0);  // успешный результат
    printf("Значение ex3 или -1.0: %.2f\n", val2);
    
    // Использование map
    printf("\nИспользование map (умножение на 2):\n");
    Expected mapped = expected_map(&ex1, [](double x) { return x * 2; });
    print_expected(&mapped, "mapped result");
    
    // Цепочка операций с flat_map
    printf("\nЦепочка операций с flat_map:\n");
    Expected chain_result = expected_flat_map(&ex1, 
        [](double x) { 
            Expected step1 = safe_sqrt(x); 
            if (expected_has_value(&step1)) {
                return safe_divide(expected_get_value(&step1), 2.0);
            }
            return step1;
        }
    );
    print_expected(&chain_result, "sqrt(10/2) / 2");
    
    // Пример цепочки операций
    printf("\nЦепочка операций: string -> double -> sqrt -> divide by 2\n");
    Expected chain1 = string_to_double("64");
    Expected chain2 = expected_flat_map(&chain1, safe_sqrt);
    Expected chain3 = expected_flat_map(&chain2, 
        [](double x) { return safe_divide(x, 2.0); });
    print_expected(&chain3, "final result of chain");
    
    // Пример с ошибкой в цепочке
    printf("\nЦепочка операций с ошибкой посередине:\n");
    Expected err_chain1 = string_to_double("64");
    Expected err_chain2 = expected_flat_map(&err_chain1, 
        [](double x) { return safe_divide(x, 0.0); });  // деление на ноль
    Expected err_chain3 = expected_flat_map(&err_chain2, safe_sqrt);
    print_expected(&err_chain3, "result with error in chain");
}

// Дополнительная информация о std::expected
void expected_detailed_info() {
    printf("\n--- Подробнее о std::expected (C++23) ---\n");
    printf("\nstd::expected<T, E> предоставляет следующие преимущества:\n\n");
    
    printf("// Примеры использования в C++23:\n");
    printf("#include <expected>\n\n");
    
    printf("// Объявление\n");
    printf("std::expected<int, std::error_code> result;\n\n");
    
    printf("// Создание с успешным значением\n");
    printf("auto success = std::expected<int, std::string>(42);\n\n");
    
    printf("// Создание с ошибкой\n");
    printf("auto failure = std::unexpected(std::error_code(1, std::generic_category()));\n\n");
    
    printf("// Проверка результата\n");
    printf("if (result) {\n");
    printf("    std::cout << \"Success: \" << *result << std::endl;\n");
    printf("} else {\n");
    printf("    std::cout << \"Error: \" << result.error() << std::endl;\n");
    printf("}\n\n");
    
    printf("// Использование значения по умолчанию\n");
    printf("int value = result.value_or(0);\n\n");
    
    printf("// Использование lambda для обработки ошибки\n");
    printf("auto processed = result.or_else([](auto ec) { return std::expected<int, int>{0}; });\n\n");
    
    printf("Преимущества std::expected:\n");
    printf("- Более безопасная альтернатива исключениям\n");
    printf("- Явное указание возможных ошибок в сигнатуре функции\n");
    printf("- Лучшая оптимизация компилятором по сравнению с исключениями\n");
    printf("- Поддержка цепочек операций (monadic operations)\n");
    printf("- Альтернатива возвращению ошибок через специальные значения\n");
}

// Сравнение с другими подходами
void compare_with_other_approaches() {
    printf("\n--- Сравнение с другими подходами ---\n");
    printf("\nТрадиционный подход с возвратом кода ошибки:\n");
    printf("int divide(int a, int b, int* result) {\n");
    printf("    if (b == 0) return -1; // код ошибки\n");
    printf("    *result = a / b;\n");
    printf("    return 0; // успех\n");
    printf("}\n\n");
    
    printf("С использованием указателя для ошибки:\n");
    printf("double* safe_divide_ptrs(double a, double b, char** error) {\n");
    printf("    if (b == 0.0) {\n");
    printf("        *error = \"Division by zero\";\n");
    printf("        return NULL;\n");
    printf("    }\n");
    printf("    double* result = malloc(sizeof(double));\n");
    printf("    *result = a / b;\n");
    printf("    return result;\n");
    printf("}\n\n");
    
    printf("С использованием std::expected:\n");
    printf("std::expected<double, Error> safe_divide_expected(double a, double b) {\n");
    printf("    if (b == 0.0) {\n");
    printf("        return std::unexpected(Error{DIV_BY_ZERO, \"Division by zero\"});\n");
    printf("    }\n");
    printf("    return a / b;\n");
    printf("}\n");
}

int main() {
    demonstrate_expected_usage();
    expected_detailed_info();
    compare_with_other_approaches();
    
    printf("\nstd::expected в C++23 обеспечивает современный подход к обработке ошибок,\n");
    printf("позволяя более явно выражать намерения разработчика и строить цепочки операций.\n");
    
    return 0;
}
