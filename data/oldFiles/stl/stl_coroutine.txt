<h2>std::coroutine</h2>

<p><strong>Категория:</strong> Корутины</p>

<h3>Описание</h3>
<p>std::coroutine - это набор типов и функций, добавленных в C++20 для поддержки корутин. Корутины - это особый тип функции, которая может приостанавливать свое выполнение и возобновлять его позже. Они позволяют реализовать асинхронные вычисления, генераторы и другие формы cooperative multitasking.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++20</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::coroutine:</p>

<pre>
<code>
// Пример использования std::coroutine на C++

#include &lt;iostream&gt;
#include &lt;coroutine&gt;
#include &lt;memory&gt;

// Простой пример корутинного типа
template &lt;typename T&gt;
class Generator {
public:
    struct promise_type {
        T value;
        
        Generator get_return_object() {
            return Generator(std::coroutine_handle&lt;promise_type&gt;::from_promise(*this));
        }
        
        std::suspend_always initial_suspend() { return {}; }
        std::suspend_always final_suspend() noexcept { return {}; }
        
        void return_void() {}
        
        std::suspend_always yield_value(T v) {
            value = v;
            return {};
        }
        
        void unhandled_exception() { std::terminate(); }
    };
    
    std::coroutine_handle&lt;promise_type&gt; handle;
    
    explicit Generator(std::coroutine_handle&lt;promise_type&gt; h) : handle(h) {}
    ~Generator() {
        if (handle) handle.destroy();
    }
    
    T getValue() { return handle.promise().value; }
    void moveToNext() { handle.resume(); }
    bool isFinished() { return handle.done(); }
};

// Пример функции-корутины
Generator&lt;int&gt; generateNumbers() {
    co_yield 1;
    co_yield 2;
    co_yield 3;
    co_return;
}

int main() {
    auto gen = generateNumbers();
    
    while (!gen.isFinished()) {
        std::cout << gen.getValue() << std::endl;
        gen.moveToNext();
    }
    
    return 0;
}

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o coroutine_example coroutine_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;setjmp.h&gt;

// Состояние корутины
typedef enum {
    COROUTINE_NEW,
    COROUTINE_RUNNING,
    COROUTINE_SUSPENDED,
    COROUTINE_FINISHED
} CoroutineState;

// Структура для корутины
typedef struct {
    jmp_buf env;
    CoroutineState state;
    int value;
    int *param;
    int should_yield;
} SimpleCoroutine;

// Макрос для инициализации корутины
#define COROUTINE_INIT(coro) do { \
    (coro)->state = COROUTINE_NEW; \
    (coro)->value = 0; \
    (coro)->should_yield = 0; \
} while(0)

// Макрос для определения начала корутины
#define COROUTINE_BEGIN(coro) do { \
    switch((coro)->state) { \
        case COROUTINE_NEW: goto start; \
        case COROUTINE_SUSPENDED: goto resume; \
        default: return 0; \
    } \
    start: \
    (coro)->state = COROUTINE_RUNNING; \
} while(0)

// Макрос для завершения корутины
#define COROUTINE_END(coro) do { \
    (coro)->state = COROUTINE_FINISHED; \
    return 1; \
} while(0)

// Макрос для приостанова корутины
#define COROUTINE_YIELD(coro, val) do { \
    (coro)->value = (val); \
    (coro)->state = COROUTINE_SUSPENDED; \
    return 1; \
    resume: ; \
} while(0)

// Макрос для возобновления корутины
#define COROUTINE_RESUME(coro) setjmp((coro)->env)

// Пример корутинной функции для генерации чисел
int generate_numbers(SimpleCoroutine* coro) {
    COROUTINE_BEGIN(coro);
    
    // Генерация чисел 1, 2, 3
    COROUTINE_YIELD(coro, 1);
    COROUTINE_YIELD(coro, 2);
    COROUTINE_YIELD(coro, 3);
    
    COROUTINE_END(coro);
}

// Функция для запуска корутины и получения значения
int run_coroutine(SimpleCoroutine* coro) {
    if (coro->state == COROUTINE_FINISHED) {
        return 0;
    }
    
    if (COROUTINE_RESUME(coro) == 0) {
        // Первый запуск или возобновление
        return generate_numbers(coro);
    } else {
        // Возобновление после yield
        return 1;
    }
}

// Пример использования
int main() {
    SimpleCoroutine coro;
    COROUTINE_INIT(&coro);
    
    printf("Запуск корутины для генерации чисел:\n");
    
    while (run_coroutine(&coro)) {
        if (coro.state == COROUTINE_SUSPENDED) {
            printf("Сгенерировано значение: %d\n", coro.value);
        }
    }
    
    printf("Корутина завершена\n");
    
    return 0;
}
