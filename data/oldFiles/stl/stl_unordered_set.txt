<h2>std::unordered_set</h2>

<p><strong>Категория:</strong> Контейнеры</p>

<h3>Описание</h3>
<p>std::unordered_set - это ассоциативный контейнер, который содержит уникальные объекты. Он организован как хеш-таблица, что позволяет осуществлять поиск, вставку и удаление за постоянное среднее время. В отличие от std::set, элементы не упорядочены.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++11</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::unordered_set:</p>

<pre>
<code>
// Пример использования std::unordered_set на C++

#include &lt;iostream&gt;
#include &lt;unordered_set&gt;
#include &lt;vector&gt;

int main() {
    // Создание unordered_set
    std::unordered_set&lt;int&gt; mySet = {5, 2, 8, 1, 9};
    
    std::cout << "Размер unordered_set: " << mySet.size() << std::endl;
    
    // Вставка элементов
    mySet.insert(3);
    mySet.insert(7);
    
    // Использование итераторов
    std::cout << "Элементы unordered_set: ";
    for (auto it = mySet.begin(); it != mySet.end(); ++it) {
        std::cout << *it << " ";
    }
    std::cout << std::endl;
    
    // Использование range-based for loop
    std::cout << "Элементы unordered_set через range-based for: ";
    for (const auto&amp; element : mySet) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    // Проверка наличия элемента
    if (mySet.count(5) > 0) {
        std::cout << "Элемент 5 существует в unordered_set" << std::endl;
    }
    
    // Использование find
    auto it = mySet.find(8);
    if (it != mySet.end()) {
        std::cout << "Найден элемент: " << *it << std::endl;
    }
    
    // Удаление элемента
    mySet.erase(3);
    std::cout << "После удаления элемента 3: ";
    for (const auto&amp; element : mySet) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    // Количество элементов с заданным значением (всегда 0 или 1 для unordered_set)
    std::cout << "Количество элементов со значением 5: " << mySet.count(5) << std::endl;
    
    // Очистка unordered_set
    mySet.clear();
    std::cout << "После очистки размер: " << mySet.size() << std::endl;
    
    return 0;
}

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o unordered_set_example unordered_set_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;

#define INITIAL_CAPACITY 16
#define LOAD_FACTOR_THRESHOLD 0.75

// Структура для узла хеш-таблицы
typedef struct {
    int value;
    int is_occupied;
    int is_deleted;
} HashSetNode;

// Структура для unordered_set
typedef struct {
    HashSetNode* buckets;
    size_t size;
    size_t capacity;
} UnorderedSet;

// Хеш-функция для int
size_t hash_function(int key, size_t capacity) {
    // Простая хеш-функция
    unsigned int hash = key;
    hash ^= hash << 16;
    hash ^= hash >> 8;
    return hash % capacity;
}

// Инициализация unordered_set
UnorderedSet* unordered_set_create() {
    UnorderedSet* set = malloc(sizeof(UnorderedSet));
    set->buckets = calloc(INITIAL_CAPACITY, sizeof(HashSetNode));
    set->size = 0;
    set->capacity = INITIAL_CAPACITY;
    return set;
}

// Уничтожение unordered_set
void unordered_set_destroy(UnorderedSet* set) {
    if (!set) return;
    free(set->buckets);
    free(set);
}

// Удвоение размера таблицы при превышении порога загрузки
void resize_if_needed(UnorderedSet* set) {
    if ((double)set->size / set->capacity <= LOAD_FACTOR_THRESHOLD) {
        return;
    }
    
    size_t old_capacity = set->capacity;
    HashSetNode* old_buckets = set->buckets;
    
    set->capacity *= 2;
    set->buckets = calloc(set->capacity, sizeof(HashSetNode));
    set->size = 0;
    
    // Перехеширование всех элементов
    for (size_t i = 0; i < old_capacity; i++) {
        if (old_buckets[i].is_occupied && !old_buckets[i].is_deleted) {
            unordered_set_insert(set, old_buckets[i].value);
        }
    }
    
    free(old_buckets);
}

// Вставка элемента
void unordered_set_insert(UnorderedSet* set, int value) {
    if (!set) return;
    
    resize_if_needed(set);
    
    size_t index = hash_function(value, set->capacity);
    size_t original_index = index;
    
    // Линейный probing для разрешения коллизий
    do {
        if (!set->buckets[index].is_occupied || set->buckets[index].is_deleted) {
            // Найдена пустая или удаленная ячейка
            set->buckets[index].value = value;
            set->buckets[index].is_occupied = 1;
            set->buckets[index].is_deleted = 0;
            set->size++;
            return;
        } else if (set->buckets[index].value == value) {
            // Значение уже существует, ничего не делаем
            return;
        }
        
        // Пробуем следующую ячейку
        index = (index + 1) % set->capacity;
    } while (index != original_index);
    
    // Таблица полностью заполнена (не должно произойти при правильном управлении размером)
    printf("Хеш-таблица переполнена\n");
}

// Поиск элемента
int unordered_set_find(UnorderedSet* set, int value) {
    if (!set) return 0;
    
    size_t index = hash_function(value, set->capacity);
    size_t original_index = index;
    
    // Линейный probing для поиска
    do {
        if (!set->buckets[index].is_occupied) {
            // Нашли пустую ячейку - элемент не существует
            return 0;
        } else if (set->buckets[index].value == value && 
                   set->buckets[index].is_occupied && 
                   !set->buckets[index].is_deleted) {
            // Нашли элемент
            return 1;
        }
        
        // Пробуем следующую ячейку
        index = (index + 1) % set->capacity;
    } while (index != original_index);
    
    // Таблица полностью пройдена - элемент не найден
    return 0;
}

// Проверка наличия элемента
int unordered_set_contains(UnorderedSet* set, int value) {
    return unordered_set_find(set, value);
}

// Удаление элемента
void unordered_set_erase(UnorderedSet* set, int value) {
    if (!set) return;
    
    size_t index = hash_function(value, set->capacity);
    size_t original_index = index;
    
    // Линейный probing для поиска
    do {
        if (!set->buckets[index].is_occupied) {
            // Нашли пустую ячейку - элемент не существует
            return;
        } else if (set->buckets[index].value == value && 
                   set->buckets[index].is_occupied && 
                   !set->buckets[index].is_deleted) {
            // Нашли элемент для удаления
            set->buckets[index].is_deleted = 1; // помечаем как удаленный
            set->size--;
            return;
        }
        
        // Пробуем следующую ячейку
        index = (index + 1) % set->capacity;
    } while (index != original_index);
}

// Получение размера
size_t unordered_set_size(UnorderedSet* set) {
    return set ? set->size : 0;
}

// Проверка на пустоту
int unordered_set_empty(UnorderedSet* set) {
    return set ? (set->size == 0) : 1;
}

// Очистка unordered_set
void unordered_set_clear(UnorderedSet* set) {
    if (!set) return;
    
    for (size_t i = 0; i < set->capacity; i++) {
        set->buckets[i].is_occupied = 0;
        set->buckets[i].is_deleted = 0;
    }
    set->size = 0;
}

// Печать unordered_set
void unordered_set_print(UnorderedSet* set, const char* label) {
    printf("%s: [", label);
    int first = 1;
    for (size_t i = 0; i < set->capacity; i++) {
        if (set->buckets[i].is_occupied && !set->buckets[i].is_deleted) {
            if (!first) printf(", ");
            printf("%d", set->buckets[i].value);
            first = 0;
        }
    }
    printf("] (size: %zu, capacity: %zu)\n", unordered_set_size(set), set->capacity);
}

// Пример использования
int main() {
    UnorderedSet* mySet = unordered_set_create();
    
    // Вставка начальных значений
    int initialValues[] = {5, 2, 8, 1, 9};
    for (int i = 0; i < 5; i++) {
        unordered_set_insert(mySet, initialValues[i]);
    }
    
    unordered_set_print(mySet, "Исходный unordered_set");
    printf("Размер unordered_set: %zu\n", unordered_set_size(mySet));
    
    // Вставка дополнительных элементов
    unordered_set_insert(mySet, 3);
    unordered_set_insert(mySet, 7);
    unordered_set_print(mySet, "После вставки 3 и 7");
    
    // Проверка наличия элемента
    if (unordered_set_contains(mySet, 5)) {
        printf("Элемент 5 существует в unordered_set\n");
    }
    
    // Удаление элемента
    unordered_set_erase(mySet, 3);
    unordered_set_print(mySet, "После удаления элемента 3");
    
    // Очистка unordered_set
    unordered_set_clear(mySet);
    printf("После очистки размер: %zu\n", unordered_set_size(mySet));
    
    // Уничтожение unordered_set
    unordered_set_destroy(mySet);
    
    return 0;
}
