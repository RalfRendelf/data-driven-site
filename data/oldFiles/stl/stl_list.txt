<h2>std::list</h2>

<p><strong>Категория:</strong> Контейнеры</p>

<h3>Описание</h3>
<p>std::list - это контейнер, реализующий двусвязный список. Он хранит элементы в узлах, каждый из которых содержит указатель на следующий и предыдущий узлы. Вставка, удаление и перемещение элементов эффективны, так как не требуют перемещения других элементов.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++98</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::list:</p>

<pre>
<code>
// Пример использования std::list на C++

#include &lt;iostream&gt;
#include &lt;list&gt;
#include &lt;algorithm&gt;

int main() {
    // Создание списка
    std::list&lt;int&gt; lst = {1, 2, 3, 4, 5};
    
    std::cout << "Размер списка: " << lst.size() << std::endl;
    
    // Добавление элементов
    lst.push_back(6);
    lst.push_front(0);
    
    // Использование итераторов
    std::cout << "Элементы списка через итераторы: ";
    for (auto it = lst.begin(); it != lst.end(); ++it) {
        std::cout << *it << " ";
    }
    std::cout << std::endl;
    
    // Использование range-based for loop
    std::cout << "Элементы списка через range-based for: ";
    for (const auto&amp; element : lst) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    // front и back
    std::cout << "Первый элемент: " << lst.front() << std::endl;
    std::cout << "Последний элемент: " << lst.back() << std::endl;
    
    // Вставка элемента в середину
    auto it = lst.begin();
    std::advance(it, 3); // перемещаемся к 4-му элементу
    lst.insert(it, 99);
    
    std::cout << "После вставки 99 на 4-ю позицию: ";
    for (const auto&amp; element : lst) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    // Удаление элемента
    it = lst.begin();
    std::advance(it, 3); // перемещаемся к 4-му элементу (99)
    lst.erase(it);
    
    std::cout << "После удаления 4-го элемента: ";
    for (const auto&amp; element : lst) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    // Удаление всех элементов со значением 3
    lst.remove(3);
    std::cout << "После remove(3): ";
    for (const auto&amp; element : lst) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    // Сортировка
    lst.sort();
    std::cout << "После сортировки: ";
    for (const auto&amp; element : lst) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    // Объединение списков
    std::list&lt;int&gt; lst2 = {10, 20, 30};
    lst.merge(lst2);
    std::cout << "После merge с {10, 20, 30}: ";
    for (const auto&amp; element : lst) {
        std::cout << element << " ";
    }
    std::cout << std::endl;
    
    // Очистка списка
    lst.clear();
    std::cout << "После очистки размер: " << lst.size() << std::endl;
    
    return 0;
}

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o list_example list_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;

// Структура для узла списка
typedef struct ListNode {
    int data;
    struct ListNode* next;
    struct ListNode* prev;
} ListNode;

// Структура для списка
typedef struct {
    ListNode* head;
    ListNode* tail;
    size_t size;
} List;

// Инициализация списка
List* list_create() {
    List* lst = malloc(sizeof(List));
    lst->head = NULL;
    lst->tail = NULL;
    lst->size = 0;
    return lst;
}

// Уничтожение списка
void list_destroy(List* lst) {
    if (!lst) return;
    
    ListNode* current = lst->head;
    while (current != NULL) {
        ListNode* next = current->next;
        free(current);
        current = next;
    }
    free(lst);
}

// Создание узла
ListNode* create_node(int data) {
    ListNode* node = malloc(sizeof(ListNode));
    node->data = data;
    node->next = NULL;
    node->prev = NULL;
    return node;
}

// Добавление элемента в начало
void list_push_front(List* lst, int value) {
    if (!lst) return;
    
    ListNode* new_node = create_node(value);
    
    if (lst->head == NULL) {
        lst->head = lst->tail = new_node;
    } else {
        new_node->next = lst->head;
        lst->head->prev = new_node;
        lst->head = new_node;
    }
    
    lst->size++;
}

// Добавление элемента в конец
void list_push_back(List* lst, int value) {
    if (!lst) return;
    
    ListNode* new_node = create_node(value);
    
    if (lst->tail == NULL) {
        lst->head = lst->tail = new_node;
    } else {
        lst->tail->next = new_node;
        new_node->prev = lst->tail;
        lst->tail = new_node;
    }
    
    lst->size++;
}

// Удаление элемента из начала
int list_pop_front(List* lst) {
    if (!lst || lst->head == NULL) return 0;
    
    ListNode* old_head = lst->head;
    int value = old_head->data;
    
    lst->head = lst->head->next;
    if (lst->head) {
        lst->head->prev = NULL;
    } else {
        lst->tail = NULL; // список стал пустым
    }
    
    free(old_head);
    lst->size--;
    
    return value;
}

// Удаление элемента из конца
int list_pop_back(List* lst) {
    if (!lst || lst->tail == NULL) return 0;
    
    ListNode* old_tail = lst->tail;
    int value = old_tail->data;
    
    lst->tail = lst->tail->prev;
    if (lst->tail) {
        lst->tail->next = NULL;
    } else {
        lst->head = NULL; // список стал пустым
    }
    
    free(old_tail);
    lst->size--;
    
    return value;
}

// Получение размера
size_t list_size(List* lst) {
    return lst ? lst->size : 0;
}

// Проверка на пустоту
int list_empty(List* lst) {
    return lst ? (lst->size == 0) : 1;
}

// Получение первого элемента
int list_front(List* lst) {
    if (lst && lst->head) {
        return lst->head->data;
    }
    return 0;
}

// Получение последнего элемента
int list_back(List* lst) {
    if (lst && lst->tail) {
        return lst->tail->data;
    }
    return 0;
}

// Вставка элемента по итератору (представленном как узел)
ListNode* list_insert(List* lst, ListNode* pos, int value) {
    if (!lst) return NULL;
    
    if (pos == lst->head) {
        list_push_front(lst, value);
        return lst->head;
    }
    
    ListNode* new_node = create_node(value);
    
    if (pos == NULL) { // вставка в конец
        if (lst->tail) {
            lst->tail->next = new_node;
            new_node->prev = lst->tail;
            lst->tail = new_node;
        } else { // список пуст
            lst->head = lst->tail = new_node;
        }
        lst->size++;
        return new_node;
    }
    
    // вставка перед pos
    new_node->next = pos;
    new_node->prev = pos->prev;
    pos->prev->next = new_node;
    pos->prev = new_node;
    
    lst->size++;
    return new_node;
}

// Удаление элемента по итератору (представленном как узел)
ListNode* list_erase(List* lst, ListNode* pos) {
    if (!lst || !pos) return NULL;
    
    ListNode* next_node = pos->next;
    
    if (pos->prev) {
        pos->prev->next = pos->next;
    } else {
        lst->head = pos->next; // удаляем голову
    }
    
    if (pos->next) {
        pos->next->prev = pos->prev;
    } else {
        lst->tail = pos->prev; // удаляем хвост
    }
    
    free(pos);
    lst->size--;
    
    return next_node;
}

// Удаление всех элементов с заданным значением
void list_remove(List* lst, int value) {
    if (!lst) return;
    
    ListNode* current = lst->head;
    while (current != NULL) {
        ListNode* next = current->next;
        if (current->data == value) {
            list_erase(lst, current);
        }
        current = next;
    }
}

// Печать списка
void list_print(List* lst, const char* label) {
    printf("%s: [", label);
    ListNode* current = lst->head;
    while (current != NULL) {
        printf("%d", current->data);
        if (current->next != NULL) printf(", ");
        current = current->next;
    }
    printf("] (size: %zu)\n", list_size(lst));
}

// Пример использования
int main() {
    List* lst = list_create();
    
    // Заполнение начальными значениями
    for (int i = 1; i <= 5; i++) {
        list_push_back(lst, i);
    }
    
    list_print(lst, "Исходный список");
    printf("Размер списка: %zu\n", list_size(lst));
    
    // Добавление элементов
    list_push_back(lst, 6);
    list_push_front(lst, 0);
    list_print(lst, "После добавления 6 в конец и 0 в начало");
    
    // front и back
    printf("Первый элемент: %d\n", list_front(lst));
    printf("Последний элемент: %d\n", list_back(lst));
    
    // Вставка элемента в середину
    ListNode* it = lst->head;
    for (int i = 0; i < 3; i++) { // перемещаемся к 4-му элементу
        if (it) it = it->next;
    }
    list_insert(lst, it, 99);
    list_print(lst, "После вставки 99 на 4-ю позицию");
    
    // Удаление элемента
    it = lst->head;
    for (int i = 0; i < 3; i++) { // перемещаемся к 4-му элементу (99)
        if (it) it = it->next;
    }
    list_erase(lst, it);
    list_print(lst, "После удаления 4-го элемента");
    
    // Удаление всех элементов со значением 3
    list_remove(lst, 3);
    list_print(lst, "После remove(3)");
    
    // Очистка списка
    list_clear(lst);
    printf("После очистки размер: %zu\n", list_size(lst));
    
    // Уничтожение списка
    list_destroy(lst);
    
    return 0;
}

// Вспомогательная функция для очистки списка
void list_clear(List* lst) {
    if (!lst) return;
    
    ListNode* current = lst->head;
    while (current != NULL) {
        ListNode* next = current->next;
        free(current);
        current = next;
    }
    
    lst->head = NULL;
    lst->tail = NULL;
    lst->size = 0;
}
