<h2>&lt;iterator&gt;</h2>

<p><strong>Категория:</strong> Алгоритмы</p>

<h3>Описание</h3>
<p>Заголовочный файл &lt;iterator&gt; содержит определения и алгоритмы для работы с итераторами. Итераторы обеспечивают унифицированный способ доступа к элементам контейнеров, независимо от их внутренней структуры.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++98 (часть функций), остальные добавлены в последующих версиях</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы из &lt;iterator&gt;:</p>

<pre>
<code>
// Пример использования &lt;iterator&gt; на C++

#include &lt;iostream&gt;
#include &lt;iterator&gt;
#include &lt;vector&gt;
#include &lt;list&gt;
#include &lt;algorithm&gt;

int main() {
    std::vector&lt;int&gt; vec = {1, 2, 3, 4, 5};
    
    // std::begin и std::end - универсальные функции для получения итераторов
    std::cout << "Vector elements: ";
    for (auto it = std::begin(vec); it != std::end(vec); ++it) {
        std::cout << *it << " ";
    }
    std::cout << std::endl;
    
    // std::advance - перемещение итератора на определенное количество позиций
    auto it = std::begin(vec);
    std::advance(it, 2);
    std::cout << "Element at position 2: " << *it << std::endl;
    
    // std::distance - вычисление расстояния между двумя итераторами
    auto dist = std::distance(std::begin(vec), it);
    std::cout << "Distance from begin to position 2: " << dist << std::endl;
    
    // std::next - получение итератора, смещенного на определенное количество позиций
    auto next_it = std::next(std::begin(vec), 3);
    std::cout << "Element at position 3: " << *next_it << std::endl;
    
    // std::prev - получение итератора, смещенного назад на определенное количество позиций
    auto prev_it = std::prev(next_it, 1);
    std::cout << "Previous element: " << *prev_it << std::endl;
    
    // std::iterator_traits - получение информации о типе итератора
    typedef std::iterator_traits&lt;std::vector&lt;int&gt;::iterator&gt; traits;
    std::cout << "Iterator value type: " << typeid(traits::value_type).name() << std::endl;
    
    // std::back_inserter - создание итератора вставки в конец контейнера
    std::vector&lt;int&gt; result;
    std::copy(vec.begin(), vec.end(), std::back_inserter(result));
    std::cout << "Copied vector: ";
    for (int n : result) {
        std::cout << n << " ";
    }
    std::cout << std::endl;
    
    // std::front_inserter - создание итератора вставки в начало контейнера
    std::list&lt;int&gt; lst;
    std::copy(vec.begin(), vec.end(), std::front_inserter(lst));
    std::cout << "Front inserted list: ";
    for (int n : lst) {
        std::cout << n << " ";
    }
    std::cout << std::endl;
    
    // std::inserter - создание итератора вставки в произвольную позицию
    std::set&lt;int&gt; ordered_set;
    std::copy(vec.begin(), vec.end(), std::inserter(ordered_set, ordered_set.begin()));
    std::cout << "Set with inserter: ";
    for (int n : ordered_set) {
        std::cout << n << " ";
    }
    std::cout << std::endl;
    
    return 0;
}

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o iterator_example iterator_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;

// Структура для имитации итератора
typedef struct {
    int* ptr;
    int* start;
    int* end;
} IntIterator;

// Функция для создания итератора
IntIterator make_iterator(int* array, size_t size) {
    IntIterator it;
    it.ptr = array;
    it.start = array;
    it.end = array + size;
    return it;
}

// Функция для получения значения
int iterator_get(IntIterator* it) {
    if (it->ptr >= it->start && it->ptr < it->end) {
        return *(it->ptr);
    }
    return 0; // значение по умолчанию
}

// Функция для инкремента итератора
void iterator_increment(IntIterator* it) {
    if (it->ptr < it->end) {
        it->ptr++;
    }
}

// Функция для декремента итератора
void iterator_decrement(IntIterator* it) {
    if (it->ptr > it->start) {
        it->ptr--;
    }
}

// Функция для проверки на достижение конца
int iterator_equals_end(IntIterator* it) {
    return it->ptr >= it->end;
}

// Функция для проверки на достижение начала
int iterator_equals_begin(IntIterator* it) {
    return it->ptr <= it->start;
}

// Функция для перемещения итератора на определенное количество позиций
void iterator_advance(IntIterator* it, int offset) {
    if (it->ptr + offset >= it->start && it->ptr + offset <= it->end) {
        it->ptr += offset;
    }
}

// Функция для вычисления расстояния между итераторами
int iterator_distance(IntIterator* start, IntIterator* end) {
    return end->ptr - start->ptr;
}

// Функция для получения следующего итератора
IntIterator iterator_next(IntIterator it, int offset) {
    IntIterator next = it;
    iterator_advance(&next, offset);
    return next;
}

// Функция для получения предыдущего итератора
IntIterator iterator_prev(IntIterator it, int offset) {
    IntIterator prev = it;
    iterator_advance(&prev, -offset);
    return prev;
}

// Функция для печати массива с использованием итератора
void print_with_iterator(int* array, size_t size, const char* label) {
    printf("%s: ", label);
    IntIterator it = make_iterator(array, size);
    
    while (!iterator_equals_end(&it)) {
        printf("%d ", iterator_get(&it));
        iterator_increment(&it);
    }
    printf("\n");
}

// Функция для копирования элементов с использованием итератора
void copy_with_iterator(int* source, int* dest, size_t size) {
    IntIterator src_it = make_iterator(source, size);
    IntIterator dest_it = make_iterator(dest, size);
    
    while (!iterator_equals_end(&src_it)) {
        *(dest_it.ptr) = iterator_get(&src_it);
        iterator_increment(&src_it);
        iterator_increment(&dest_it);
    }
}

// Пример использования
int main() {
    int numbers[] = {1, 2, 3, 4, 5};
    size_t size = sizeof(numbers) / sizeof(numbers[0]);
    
    // Печать массива с использованием итератора
    print_with_iterator(numbers, size, "Original array");
    
    // Перемещение итератора на 2 позиции
    IntIterator it = make_iterator(numbers, size);
    iterator_advance(&it, 2);
    printf("Element at position 2: %d\n", iterator_get(&it));
    
    // Вычисление расстояния от начала до позиции
    IntIterator begin_it = make_iterator(numbers, size);
    int dist = iterator_distance(&begin_it, &it);
    printf("Distance from begin to position 2: %d\n", dist);
    
    // Получение следующего итератора
    IntIterator next_it = iterator_next(begin_it, 3);
    printf("Element at position 3: %d\n", iterator_get(&next_it));
    
    // Получение предыдущего итератора
    IntIterator prev_it = iterator_prev(next_it, 1);
    printf("Previous element: %d\n", iterator_get(&prev_it));
    
    // Копирование с использованием итераторов
    int copied[size];
    copy_with_iterator(numbers, copied, size);
    print_with_iterator(copied, size, "Copied array");
    
    return 0;
}
