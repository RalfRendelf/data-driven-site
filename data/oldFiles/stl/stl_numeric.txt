<h2>&lt;numeric&gt;</h2>

<p><strong>Категория:</strong> Алгоритмы</p>

<h3>Описание</h3>
<p>Заголовочный файл &lt;numeric&gt; содержит коллекцию функций для выполнения различных операций над числовыми последовательностями. Он включает в себя алгоритмы для вычисления сумм, произведений, разностей и других математических операций.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++98 (часть функций), остальные добавлены в последующих версиях</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы из &lt;numeric&gt;:</p>

<pre>
<code>
// Пример использования &lt;numeric&gt; на C++

#include &lt;iostream&gt;
#include &lt;numeric&gt;
#include &lt;vector&gt;
#include &lt;functional&gt;

int main() {
    std::vector&lt;int&gt; numbers = {1, 2, 3, 4, 5};
    
    // std::accumulate - вычисляет сумму элементов
    int sum = std::accumulate(numbers.begin(), numbers.end(), 0);
    std::cout << "Sum: " << sum << std::endl;
    
    // std::accumulate с пользовательской операцией
    int product = std::accumulate(numbers.begin(), numbers.end(), 1, 
                                 std::multiplies&lt;int&gt;());
    std::cout << "Product: " << product << std::endl;
    
    // std::inner_product - вычисляет скалярное произведение
    std::vector&lt;int&gt; v1 = {1, 2, 3};
    std::vector&lt;int&gt; v2 = {4, 5, 6};
    int dot_product = std::inner_product(v1.begin(), v1.end(), v2.begin(), 0);
    std::cout << "Dot product: " << dot_product << std::endl;
    
    // std::partial_sum - вычисляет частичные суммы
    std::vector&lt;int&gt; partial_sums(numbers.size());
    std::partial_sum(numbers.begin(), numbers.end(), partial_sums.begin());
    std::cout << "Partial sums: ";
    for (int n : partial_sums) {
        std::cout << n << " ";
    }
    std::cout << std::endl;
    
    // std::adjacent_difference - вычисляет разности соседних элементов
    std::vector&lt;int&gt; adj_diff(numbers.size());
    std::adjacent_difference(numbers.begin(), numbers.end(), adj_diff.begin());
    std::cout << "Adjacent differences: ";
    for (int n : adj_diff) {
        std::cout << n << " ";
    }
    std::cout << std::endl;
    
    // std::gcd - наибольший общий делитель (C++17)
#ifdef __cpp_lib_gcd_lcm
    int gcd_result = std::gcd(12, 18);
    std::cout << "GCD(12, 18): " << gcd_result << std::endl;
    
    // std::lcm - наименьшее общее кратное (C++17)
    int lcm_result = std::lcm(12, 18);
    std::cout << "LCM(12, 18): " << lcm_result << std::endl;
#endif
    
    return 0;
}

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o numeric_example numeric_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;

// Функция для вычисления суммы элементов массива
int accumulate(int* arr, size_t size, int initial) {
    int result = initial;
    for (size_t i = 0; i < size; i++) {
        result += arr[i];
    }
    return result;
}

// Функция для вычисления произведения элементов массива
int accumulate_product(int* arr, size_t size, int initial) {
    int result = initial;
    for (size_t i = 0; i < size; i++) {
        result *= arr[i];
    }
    return result;
}

// Функция для вычисления скалярного произведения
int inner_product(int* arr1, int* arr2, size_t size, int initial) {
    int result = initial;
    for (size_t i = 0; i < size; i++) {
        result += arr1[i] * arr2[i];
    }
    return result;
}

// Функция для вычисления частичных сумм
void partial_sum(int* input, int* output, size_t size) {
    if (size == 0) return;
    
    output[0] = input[0];
    for (size_t i = 1; i < size; i++) {
        output[i] = output[i-1] + input[i];
    }
}

// Функция для вычисления разностей соседних элементов
void adjacent_difference(int* input, int* output, size_t size) {
    if (size == 0) return;
    
    output[0] = input[0];
    for (size_t i = 1; i < size; i++) {
        output[i] = input[i] - input[i-1];
    }
}

// Функция для вычисления НОД (алгоритм Евклида)
int gcd(int a, int b) {
    while (b != 0) {
        int temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}

// Функция для вычисления НОК
int lcm(int a, int b) {
    return (a * b) / gcd(a, b);
}

// Функция для печати массива
void print_array(int* arr, size_t size, const char* label) {
    printf("%s: ", label);
    for (size_t i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
}

// Пример использования
int main() {
    int numbers[] = {1, 2, 3, 4, 5};
    size_t size = sizeof(numbers) / sizeof(numbers[0]);
    
    // Вычисление суммы
    int sum = accumulate(numbers, size, 0);
    printf("Sum: %d\n", sum);
    
    // Вычисление произведения
    int product = accumulate_product(numbers, size, 1);
    printf("Product: %d\n", product);
    
    // Вычисление скалярного произведения
    int v1[] = {1, 2, 3};
    int v2[] = {4, 5, 6};
    size_t vec_size = sizeof(v1) / sizeof(v1[0]);
    int dot_product = inner_product(v1, v2, vec_size, 0);
    printf("Dot product: %d\n", dot_product);
    
    // Вычисление частичных сумм
    int partial_sums[size];
    partial_sum(numbers, partial_sums, size);
    print_array(partial_sums, size, "Partial sums");
    
    // Вычисление разностей соседних элементов
    int adj_diff[size];
    adjacent_difference(numbers, adj_diff, size);
    print_array(adj_diff, size, "Adjacent differences");
    
    // Вычисление НОД и НОК
    int gcd_result = gcd(12, 18);
    int lcm_result = lcm(12, 18);
    printf("GCD(12, 18): %d\n", gcd_result);
    printf("LCM(12, 18): %d\n", lcm_result);
    
    return 0;
}
