<h2>Exceptions (Обработка исключений)</h2>

<p><strong>Категория:</strong> Язык</p>

<h3>Описание</h3>
<p>Exceptions (исключения) - это механизм в C++, позволяющий обрабатывать ошибки во время выполнения программы. Он состоит из трех ключевых элементов: try, catch и throw. Исключения позволяют отделить код, обнаруживающий ошибку, от кода, обрабатывающего эту ошибку, и обеспечивают более чистую структуру программы по сравнению с традиционными способами обработки ошибок.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++98</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные концепции Exceptions и их реализация на C (для демонстрации аналогов):</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o exceptions_example exceptions_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;string.h&gt;
#include &lt;setjmp.h&gt;

// Структура для представления исключения
typedef struct {
    char message[256];
    int error_code;
    char type[50];
} Exception;

// Глобальная переменная для хранения текущего исключения
Exception current_exception;

// Структура для хранения точки восстановления (аналог try/catch)
typedef struct {
    jmp_buf jump_buffer;
    int caught;
    Exception caught_exception;
} ExceptionHandler;

// Макрос для инициализации обработчика исключений
ExceptionHandler* exception_handler_create() {
    ExceptionHandler* handler = malloc(sizeof(ExceptionHandler));
    handler->caught = 0;
    return handler;
}

// Макрос для установки точки восстановления (TRY)
#define TRY(handler) \
    if (setjmp(handler->jump_buffer) == 0)

// Макрос для выбрасывания исключения (THROW)
#define THROW(type, msg, code) \
    do { \
        strncpy(current_exception.message, msg, sizeof(current_exception.message) - 1); \
        current_exception.error_code = code; \
        strncpy(current_exception.type, type, sizeof(current_exception.type) - 1); \
        printf("Выброшено исключение: %s [%s] (код: %d)\n", \
               current_exception.message, current_exception.type, current_exception.error_code); \
        longjmp(current_exception_jmp_buf, 1); \
    } while(0)

// Глобальная переменная для longjmp
static jmp_buf current_exception_jmp_buf;

// Функция для установки точки восстановления
int setup_exception_handling(ExceptionHandler* handler) {
    return setjmp(handler->jump_buffer);
}

// Макрос для обработки исключений (CATCH)
#define CATCH(handler, exception_type) \
    else if (handler->caught && strcmp(handler->caught_exception.type, exception_type) == 0)

// Макрос для обработки любого исключения (CATCH_ALL)
#define CATCH_ALL(handler) \
    else if (handler->caught)

// Функция для выбрасывания исключения
void throw_exception(const char* type, const char* message, int error_code) {
    strncpy(current_exception.message, message, sizeof(current_exception.message) - 1);
    current_exception.message[sizeof(current_exception.message) - 1] = '\0';
    current_exception.error_code = error_code;
    strncpy(current_exception.type, type, sizeof(current_exception.type) - 1);
    current_exception.type[sizeof(current_exception.type) - 1] = '\0';
    
    printf("Выброшено исключение: %s [%s] (код: %d)\n", 
           current_exception.message, current_exception.type, current_exception.error_code);
    
    longjmp(current_exception_jmp_buf, 1);
}

// Специфичные типы исключений
void throw_invalid_argument(const char* message) {
    throw_exception("InvalidArgumentException", message, 100);
}

void throw_out_of_range(const char* message) {
    throw_exception("OutOfRangeException", message, 101);
}

void throw_runtime_error(const char* message) {
    throw_exception("RuntimeError", message, 102);
}

// Функция, которая может выбросить исключение
int divide_numbers(int numerator, int denominator) {
    if (denominator == 0) {
        throw_runtime_error("Деление на ноль невозможно");
    }
    
    if (numerator < 0 || denominator < 0) {
        throw_invalid_argument("Числа должны быть положительными");
    }
    
    return numerator / denominator;
}

// Функция для доступа к массиву с проверкой границ
int get_array_element(int* arr, int size, int index) {
    if (index < 0 || index >= size) {
        throw_out_of_range("Индекс вне диапазона");
    }
    
    return arr[index];
}

// Функция, демонстрирующая вложенные обработчики исключений
int complex_operation(int a, int b) {
    ExceptionHandler* inner_handler = exception_handler_create();
    
    if (setup_exception_handling(inner_handler) == 0) {
        // Код внутри try блока
        printf("Выполнение сложной операции с %d и %d\n", a, b);
        
        if (a > 100) {
            throw_invalid_argument("Параметр a слишком большой");
        }
        
        int result = divide_numbers(a, b);
        printf("Результат деления: %d\n", result);
        
        // Создаем искусственный массив для демонстрации
        int test_array[] = {10, 20, 30, 40, 50};
        int element = get_array_element(test_array, 5, 2);
        printf("Элемент массива: %d\n", element);
        
        free(inner_handler);
        return result;
    } else {
        // Код внутри catch блока
        printf("Обработка исключения во вложенном обработчике\n");
        free(inner_handler);
        // Перебрасываем исключение наверх
        longjmp(current_exception_jmp_buf, 1);
    }
}

// Пример использования
int main() {
    printf("=== Демонстрация обработки исключений (аналог) ===\n\n");
    
    ExceptionHandler* handler = exception_handler_create();
    
    if (setjmp(current_exception_jmp_buf) == 0) {
        // TRY блок
        printf("Попытка выполнения кода с возможными исключениями...\n");
        
        // Устанавливаем точку восстановления
        if (setup_exception_handling(handler) == 0) {
            // Выполняем операции, которые могут выбросить исключение
            int result1 = divide_numbers(20, 4);
            printf("Результат 1: %d\n", result1);
            
            // Вызов функции, которая может выбросить исключение
            int result2 = complex_operation(50, 5);
            printf("Результат 2: %d\n", result2);
            
            // Этот вызов должен вызвать исключение
            int result3 = divide_numbers(10, 0);
            printf("Результат 3: %d\n", result3);  // Эта строка не будет выполнена
            
        } else {
            // Не должно сработать, так как исключение будет перехвачено внутри complex_operation
        }
    } else {
        // CATCH блок - сюда попадаем при выбрасывании исключения
        printf("Обработка исключения в основном блоке:\n");
        printf("  Тип: %s\n", current_exception.type);
        printf("  Сообщение: %s\n", current_exception.message);
        printf("  Код ошибки: %d\n", current_exception.error_code);
    }
    
    // Еще один пример с разными типами исключений
    printf("\n--- Демонстрация разных типов исключений ---\n");
    
    // Пример с делением на ноль
    if (setjmp(current_exception_jmp_buf) == 0) {
        printf("Попытка деления на ноль...\n");
        int result = divide_numbers(10, 0);
        printf("Результат: %d\n", result);
    } else {
        printf("Исключение перехвачено: %s - %s\n", 
               current_exception.type, current_exception.message);
    }
    
    // Пример с неверным индексом
    if (setjmp(current_exception_jmp_buf) == 0) {
        printf("\nПопытка доступа к элементу за пределами массива...\n");
        int test_array[] = {1, 2, 3};
        int element = get_array_element(test_array, 3, 5);
        printf("Элемент: %d\n", element);
    } else {
        printf("Исключение перехвачено: %s - %s\n", 
               current_exception.type, current_exception.message);
    }
    
    // Пример с недопустимым аргументом
    if (setjmp(current_exception_jmp_buf) == 0) {
        printf("\nПопытка выполнения операции с отрицательными числами...\n");
        int result = divide_numbers(-5, 3);
        printf("Результат: %d\n", result);
    } else {
        printf("Исключение перехвачено: %s - %s\n", 
               current_exception.type, current_exception.message);
    }
    
    free(handler);
    
    printf("\nДемонстрация обработки исключений завершена\n");
    printf("Обработка исключений позволяет централизованно обрабатывать ошибки\n");
    printf("и отделять логику обработки ошибок от основной логики программы.\n");
    
    return 0;
}
