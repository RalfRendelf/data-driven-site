<h2>std::timed_mutex</h2>

<p><strong>Категория:</strong> Многопоточность</p>

<h3>Описание</h3>
<p>std::timed_mutex - это примитив синхронизации, который позволяет захватывать мьютекс с возможностью указания таймаута. В отличие от обычного std::mutex, timed_mutex предоставляет методы для захвата с таймаутом, что предотвращает бесконечное ожидание в случае дедлоков или зависания других потоков.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++11</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::timed_mutex и их реализация на C (для демонстрации аналогов):</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка POSIX threads
 * Компиляция: gcc -o timed_mutex_example timed_mutex_example.c -lpthread
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;pthread.h&gt;
#include &lt;unistd.h&gt;
#include &lt;time.h&gt;
#include &lt;sys/time.h&gt;

// Структура для хранения информации о таймированном мьютексе
typedef struct {
    pthread_mutex_t mutex;
    pthread_t owner;
    int is_locked;
} TimedMutex;

// Инициализация таймированного мьютекса
TimedMutex* timed_mutex_create() {
    TimedMutex* tm = malloc(sizeof(TimedMutex));
    pthread_mutex_init(&tm->mutex, NULL);
    tm->owner = 0;
    tm->is_locked = 0;
    return tm;
}

// Блокировка таймированного мьютекса (аналог lock())
void timed_mutex_lock(TimedMutex* tm) {
    if (tm != NULL) {
        pthread_mutex_lock(&tm->mutex);
        tm->owner = pthread_self();
        tm->is_locked = 1;
    }
}

// Освобождение таймированного мьютекса (аналог unlock())
void timed_mutex_unlock(TimedMutex* tm) {
    if (tm != NULL) {
        tm->is_locked = 0;
        tm->owner = 0;
        pthread_mutex_unlock(&tm->mutex);
    }
}

// Попытка заблокировать с таймаутом (аналог try_lock_for())
int timed_mutex_try_lock_for(TimedMutex* tm, int seconds) {
    if (tm != NULL) {
        struct timespec timeout;
        struct timeval now;
        
        gettimeofday(&now, NULL);
        timeout.tv_sec = now.tv_sec + seconds;
        timeout.tv_nsec = now.tv_usec * 1000;
        
        int result = pthread_mutex_timedlock(&tm->mutex, &timeout);
        if (result == 0) {
            tm->owner = pthread_self();
            tm->is_locked = 1;
            return 1;  // Успешно заблокирован
        }
    }
    return 0;  // Не удалось заблокировать
}

// Попытка заблокировать с абсолютным таймаутом (аналог try_lock_until())
int timed_mutex_try_lock_until(TimedMutex* tm, int seconds_from_now) {
    if (tm != NULL) {
        struct timespec abs_timeout;
        struct timeval now;
        
        gettimeofday(&now, NULL);
        abs_timeout.tv_sec = now.tv_sec + seconds_from_now;
        abs_timeout.tv_nsec = now.tv_usec * 1000;
        
        int result = pthread_mutex_timedlock(&tm->mutex, &abs_timeout);
        if (result == 0) {
            tm->owner = pthread_self();
            tm->is_locked = 1;
            return 1;  // Успешно заблокирован
        }
    }
    return 0;  // Не удалось заблокировать
}

// Попытка немедленной блокировки (аналог try_lock())
int timed_mutex_try_lock(TimedMutex* tm) {
    if (tm != NULL) {
        int result = pthread_mutex_trylock(&tm->mutex);
        if (result == 0) {
            tm->owner = pthread_self();
            tm->is_locked = 1;
            return 1;  // Успешно заблокирован
        }
    }
    return 0;  // Не удалось заблокировать
}

// Проверка, заблокирован ли мьютекс
int timed_mutex_is_locked(TimedMutex* tm) {
    if (tm != NULL) {
        return tm->is_locked;
    }
    return 0;
}

// Уничтожение таймированного мьютекса
void timed_mutex_destroy(TimedMutex* tm) {
    if (tm != NULL) {
        pthread_mutex_destroy(&tm->mutex);
        free(tm);
    }
}

// Общие данные для примера
typedef struct {
    int counter;
    TimedMutex* mutex;
    int delay_ms;  // задержка в миллисекундах
} SharedData;

// Функция для выполнения в потоке
void* worker_function(void* arg) {
    SharedData* data = (SharedData*)arg;
    pthread_t current_thread = pthread_self();
    
    printf("Поток %lu пытается захватить мьютекс с таймаутом...\n", current_thread);
    
    // Попытка захватить мьютекс с таймаутом 2 секунды
    if (timed_mutex_try_lock_for(data->mutex, 2)) {
        printf("Поток %lu успешно захватил мьютекс\n", current_thread);
        
        // Имитация работы с разделяемыми данными
        data->counter++;
        printf("Поток %lu увеличил счетчик до %d\n", current_thread, data->counter);
        
        // Задержка для демонстрации таймаута в других потоках
        usleep(data->delay_ms * 1000);
        
        timed_mutex_unlock(data->mutex);
        printf("Поток %lu освободил мьютекс\n", current_thread);
    } else {
        printf("Поток %lu не смог захватить мьютекс за отведенное время\n", current_thread);
    }
    
    return NULL;
}

// Пример использования
int main() {
    SharedData data;
    data.counter = 0;
    data.mutex = timed_mutex_create();
    data.delay_ms = 1500;  // 1.5 секунды
    
    pthread_t thread1, thread2, thread3;
    
    // Создание потоков с разными задержками
    data.delay_ms = 2500;  // Первый поток будет держать мьютекс 2.5 секунды
    pthread_create(&thread1, NULL, worker_function, &data);
    
    // Небольшая задержка перед запуском следующих потоков
    usleep(100000); // 0.1 секунды
    
    data.delay_ms = 1000;  // Другие потоки будут держать мьютекс 1 секунду
    pthread_create(&thread2, NULL, worker_function, &data);
    pthread_create(&thread3, NULL, worker_function, &data);
    
    // Ожидание завершения потоков
    pthread_join(thread1, NULL);
    pthread_join(thread2, NULL);
    pthread_join(thread3, NULL);
    
    printf("Финальное значение счетчика: %d\n", data.counter);
    
    timed_mutex_destroy(data.mutex);
    
    return 0;
}
