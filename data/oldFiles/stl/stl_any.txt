<h2>std::any</h2>

<p><strong>Категория:</strong> Контейнеры</p>

<h3>Описание</h3>
<p>std::any - это тип, который может хранить значение любого типа, который является CopyConstructible. Он был добавлен в C++17 и позволяет безопасно хранить и извлекать значения различных типов в едином контейнере. Это полезно для ситуаций, когда тип данных не известен на этапе компиляции.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++17</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::any и их реализация на C (для демонстрации аналогов):</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o any_example any_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;string.h&gt;
#include &lt;stdint.h&gt;

#define ANY_TYPE_INT 1
#define ANY_TYPE_DOUBLE 2
#define ANY_TYPE_STRING 3
#define ANY_TYPE_CUSTOM 4

// Структура для представления типа данных
typedef struct {
    int type_id;
    void* data;
    size_t size;
    void (*destructor)(void*);
    void* (*copy_constructor)(const void*);
} AnyValue;

// Структура для представления std::any
typedef struct {
    AnyValue value;
    int has_value;
} Any;

// Функция для удаления строк
void string_destructor(void* str) {
    if (str) {
        free(str);
    }
}

// Функция для копирования строк
void* string_copy_constructor(const void* str) {
    if (str) {
        const char* s = (const char*)str;
        char* new_str = malloc(strlen(s) + 1);
        strcpy(new_str, s);
        return new_str;
    }
    return NULL;
}

// Инициализация Any с целым числом
Any any_from_int(int value) {
    Any any;
    any.value.type_id = ANY_TYPE_INT;
    any.value.data = malloc(sizeof(int));
    memcpy(any.value.data, &value, sizeof(int));
    any.value.size = sizeof(int));
    any.value.destructor = free;
    any.value.copy_constructor = NULL;
    any.has_value = 1;
    return any;
}

// Инициализация Any с числом с плавающей точкой
Any any_from_double(double value) {
    Any any;
    any.value.type_id = ANY_TYPE_DOUBLE;
    any.value.data = malloc(sizeof(double));
    memcpy(any.value.data, &value, sizeof(double));
    any.value.size = sizeof(double));
    any.value.destructor = free;
    any.value.copy_constructor = NULL;
    any.has_value = 1;
    return any;
}

// Инициализация Any со строкой
Any any_from_string(const char* str) {
    Any any;
    any.value.type_id = ANY_TYPE_STRING;
    any.value.data = malloc(strlen(str) + 1);
    strcpy((char*)any.value.data, str);
    any.value.size = strlen(str) + 1;
    any.value.destructor = string_destructor;
    any.value.copy_constructor = string_copy_constructor;
    any.has_value = 1;
    return any;
}

// Инициализация Any с произвольным типом
Any any_from_custom(void* data, size_t size, 
                    void (*destructor)(void*), 
                    void* (*copy_constructor)(const void*)) {
    Any any;
    any.value.type_id = ANY_TYPE_CUSTOM;
    any.value.data = malloc(size);
    memcpy(any.value.data, data, size);
    any.value.size = size;
    any.value.destructor = destructor;
    any.value.copy_constructor = copy_constructor;
    any.has_value = 1;
    return any;
}

// Освобождение ресурсов Any
void any_reset(Any* any) {
    if (any && any->has_value && any->value.data && any->value.destructor) {
        any->value.destructor(any->value.data);
        any->value.data = NULL;
        any->has_value = 0;
    }
}

// Проверка, содержит ли Any значение
int any_has_value(Any* any) {
    return any ? any->has_value : 0;
}

// Получение значения типа int из Any
int* any_cast_int(Any* any) {
    if (!any || !any->has_value || any->value.type_id != ANY_TYPE_INT) {
        return NULL;
    }
    return (int*)any->value.data;
}

// Получение значения типа double из Any
double* any_cast_double(Any* any) {
    if (!any || !any->has_value || any->value.type_id != ANY_TYPE_DOUBLE) {
        return NULL;
    }
    return (double*)any->value.data;
}

// Получение значения типа строка из Any
char* any_cast_string(Any* any) {
    if (!any || !any->has_value || any->value.type_id != ANY_TYPE_STRING) {
        return NULL;
    }
    return (char*)any->value.data;
}

// Копирование Any
Any any_copy(Any* any) {
    if (!any || !any->has_value) {
        Any empty_any = {0};
        return empty_any;
    }
    
    Any new_any;
    new_any.has_value = any->has_value;
    new_any.value.type_id = any->value.type_id;
    new_any.value.size = any->value.size;
    new_any.value.destructor = any->value.destructor;
    new_any.value.copy_constructor = any->value.copy_constructor;
    
    if (any->value.copy_constructor) {
        new_any.value.data = any->value.copy_constructor(any->value.data);
    } else {
        new_any.value.data = malloc(any->value.size);
        memcpy(new_any.value.data, any->value.data, any->value.size);
    }
    
    return new_any;
}

// Структура для демонстрации хранения пользовательского типа
typedef struct {
    int id;
    char name[50];
} CustomStruct;

void custom_struct_destructor(void* data) {
    if (data) {
        free(data);
    }
}

void* custom_struct_copy_constructor(const void* data) {
    if (data) {
        CustomStruct* cs = malloc(sizeof(CustomStruct));
        memcpy(cs, data, sizeof(CustomStruct));
        return cs;
    }
    return NULL;
}

// Печать значения Any
void any_print(Any* any) {
    if (!any_has_value(any)) {
        printf("[empty]\n");
        return;
    }
    
    switch (any->value.type_id) {
        case ANY_TYPE_INT:
            printf("int: %d\n", *(int*)any->value.data);
            break;
        case ANY_TYPE_DOUBLE:
            printf("double: %.2f\n", *(double*)any->value.data);
            break;
        case ANY_TYPE_STRING:
            printf("string: %s\n", (char*)any->value.data);
            break;
        case ANY_TYPE_CUSTOM:
            printf("custom type\n");
            break;
        default:
            printf("unknown type\n");
            break;
    }
}

// Пример использования
int main() {
    printf("=== Демонстрация std::any (аналог) ===\n\n");
    
    // Создание Any с разными типами данных
    Any int_any = any_from_int(42);
    Any double_any = any_from_double(3.14159);
    Any string_any = any_from_string("Hello, Any!");
    
    printf("Any с целым числом: ");
    any_print(&int_any);
    
    printf("Any с числом с плавающей точкой: ");
    any_print(&double_any);
    
    printf("Any со строкой: ");
    any_print(&string_any);
    
    // Извлечение значений
    int* int_val = any_cast_int(&int_any);
    if (int_val) {
        printf("Извлеченное целое число: %d\n", *int_val);
    }
    
    double* double_val = any_cast_double(&double_any);
    if (double_val) {
        printf("Извлеченное число с плавающей точкой: %.2f\n", *double_val);
    }
    
    char* string_val = any_cast_string(&string_any);
    if (string_val) {
        printf("Извлеченная строка: %s\n", string_val);
    }
    
    // Проверка наличия значения
    printf("int_any имеет значение: %s\n", any_has_value(&int_any) ? "да" : "нет");
    printf("Пустой Any имеет значение: %s\n", any_has_value(NULL) ? "да" : "нет");
    
    // Создание Any с пользовательским типом
    CustomStruct cs = {123, "Test Struct"};
    Any custom_any = any_from_custom(&cs, sizeof(CustomStruct), 
                                     custom_struct_destructor, 
                                     custom_struct_copy_constructor);
    
    printf("\nAny с пользовательской структурой: ");
    any_print(&custom_any);
    
    // Копирование Any
    Any copied_any = any_copy(&string_any);
    printf("Скопированный Any со строкой: ");
    any_print(&copied_any);
    
    // Очистка значений
    any_reset(&int_any);
    printf("\nПосле очистки int_any: ");
    any_print(&int_any);
    
    any_reset(&string_any);
    any_reset(&double_any);
    any_reset(&custom_any);
    any_reset(&copied_any);
    
    printf("Все ресурсы освобождены\n");
    
    return 0;
}
