<h2>std::weak_ptr</h2>

<p><strong>Категория:</strong> Умные указатели</p>

<h3>Описание</h3>
<p>std::weak_ptr — это умный указатель, который хранит «слабую» ссылку на объект, управляемый std::shared_ptr. weak_ptr не увеличивает счётчик ссылок и используется для предотвращения циклических зависимостей. Чтобы получить доступ к объекту, weak_ptr нужно преобразовать в shared_ptr с помощью lock().</p>

<h3>Добавлен в стандарт:</h3>
<p>C++11</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::weak_ptr и их реализация на C (для демонстрации аналогов):</p>

<h4>C++</h4>
<pre>
<code>
#include &lt;memory&gt;
#include &lt;iostream&gt;

int main() {
    std::shared_ptr&lt;int&gt; sp = std::make_shared&lt;int&gt;(42);
    std::weak_ptr&lt;int&gt; wp = sp;

    std::cout &lt;&lt; "use_count: " &lt;&lt; wp.use_count() &lt;&lt; "\n";

    if (auto locked = wp.lock()) {
        std::cout &lt;&lt; "Value: " &lt;&lt; *locked &lt;&lt; "\n";
    }

    sp.reset();

    if (wp.expired()) {
        std::cout &lt;&lt; "expired\n";
    }

    return 0;
}
</code>
</pre>

<h4>C реализация</h4>
<pre>
<code>
#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;pthread.h&gt;

typedef struct {
    void* ptr;
    int* ref_count;
    int* weak_ref_count;
    void (*deleter)(void*);
    pthread_mutex_t mutex;
    int destroyed;
} SharedObject;

typedef struct {
    SharedObject* obj;
} SharedPtr;

typedef struct {
    SharedObject* obj;
} WeakPtr;

SharedPtr* shared_ptr_create(void* ptr) {
    SharedPtr* sp = malloc(sizeof(SharedPtr));
    sp->obj = malloc(sizeof(SharedObject));
    sp->obj->ptr = ptr;
    sp->obj->ref_count = malloc(sizeof(int));
    *(sp->obj->ref_count) = 1;
    sp->obj->weak_ref_count = malloc(sizeof(int));
    *(sp->obj->weak_ref_count) = 0;
    sp->obj->deleter = &free;
    sp->obj->destroyed = 0;
    pthread_mutex_init(&sp->obj->mutex, NULL);
    return sp;
}

WeakPtr* weak_ptr_create(SharedPtr* sp) {
    if (!sp || !sp->obj)
        return NULL;

    WeakPtr* wp = malloc(sizeof(WeakPtr));
    wp->obj = sp->obj;

    pthread_mutex_lock(&sp->obj->mutex);
    (*(sp->obj->weak_ref_count))++;
    pthread_mutex_unlock(&sp->obj->mutex);

    return wp;
}

SharedPtr* weak_ptr_lock(WeakPtr* wp) {
    if (!wp || !wp->obj)
        return NULL;

    pthread_mutex_lock(&wp->obj->mutex);
    if (wp->obj->destroyed) {
        pthread_mutex_unlock(&wp->obj->mutex);
        return NULL;
    }

    SharedPtr* sp = malloc(sizeof(SharedPtr));
    sp->obj = wp->obj;
    (*(wp->obj->ref_count))++;
    pthread_mutex_unlock(&wp->obj->mutex);

    return sp;
}

int weak_ptr_expired(WeakPtr* wp) {
    if (!wp || !wp->obj)
        return 1;

    pthread_mutex_lock(&wp->obj->mutex);
    int expired = wp->obj->destroyed;
    pthread_mutex_unlock(&wp->obj->mutex);

    return expired;
}

int weak_ptr_use_count(WeakPtr* wp) {
    if (wp && wp->obj) {
        pthread_mutex_lock(&wp->obj->mutex);
        int count = *(wp->obj->ref_count);
        pthread_mutex_unlock(&wp->obj->mutex);
        return count;
    }
    return 0;
}

int weak_ptr_weak_count(WeakPtr* wp) {
    if (wp && wp->obj) {
        pthread_mutex_lock(&wp->obj->mutex);
        int count = *(wp->obj->weak_ref_count);
        pthread_mutex_unlock(&wp->obj->mutex);
        return count;
    }
    return 0;
}

void shared_ptr_destroy(SharedPtr* sp) {
    if (sp && sp->obj) {
        pthread_mutex_lock(&sp->obj->mutex);
        (*(sp->obj->ref_count))--;

        if (*(sp->obj->ref_count) == 0) {
            if (sp->obj->ptr && sp->obj->deleter) {
                sp->obj->deleter(sp->obj->ptr);
                sp->obj->ptr = NULL;
            }
            sp->obj->destroyed = 1;
        }

        if (*(sp->obj->ref_count) == 0 && *(sp->obj->weak_ref_count) == 0) {
            free(sp->obj->ref_count);
            free(sp->obj->weak_ref_count);
            pthread_mutex_unlock(&sp->obj->mutex);
            pthread_mutex_destroy(&sp->obj->mutex);
            free(sp->obj);
        } else {
            pthread_mutex_unlock(&sp->obj->mutex);
        }
    }
    free(sp);
}

void weak_ptr_destroy(WeakPtr* wp) {
    if (wp && wp->obj) {
        pthread_mutex_lock(&wp->obj->mutex);
        (*(wp->obj->weak_ref_count))--;

        if (wp->obj->destroyed &&
            *(wp->obj->ref_count) == 0 &&
            *(wp->obj->weak_ref_count) == 0) {

            free(wp->obj->ref_count);
            free(wp->obj->weak_ref_count);
            pthread_mutex_unlock(&wp->obj->mutex);
            pthread_mutex_destroy(&wp->obj->mutex);
            free(wp->obj);
        } else {
            pthread_mutex_unlock(&wp->obj->mutex);
        }
    }
    free(wp);
}

int main() {
    int* raw = malloc(sizeof(int));
    *raw = 42;

    SharedPtr* sp = shared_ptr_create(raw);
    WeakPtr* wp = weak_ptr_create(sp);

    printf("shared count: %d\n", weak_ptr_use_count(wp));
    printf("weak count: %d\n", weak_ptr_weak_count(wp));

    SharedPtr* locked = weak_ptr_lock(wp);
    if (locked) {
        printf("locked value: %d\n", *(int*)locked->obj->ptr);
        shared_ptr_destroy(locked);
    }

    shared_ptr_destroy(sp);

    printf("expired: %s\n", weak_ptr_expired(wp) ? "true" : "false");

    weak_ptr_destroy(wp);

    return 0;
}
