<h2>Move Semantics (Семантика перемещения)</h2>

<p><strong>Категория:</strong> Язык</p>

<h3>Описание</h3>
<p>Move semantics - это возможность эффективно перемещать ресурсы из одного объекта в другой вместо их копирования. В C++11 была введена возможность перемещения объектов, которая позволяет "украсть" ресурсы у временных объектов, тем самым избегая дорогостоящих операций копирования. Это особенно важно для объектов, владеющих динамически выделенной памятью.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++11</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные концепции Move semantics и их реализация на C (для демонстрации аналогов):</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o move_semantics_example move_semantics_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;string.h&gt;

// Структура, имитирующая класс с владением ресурсами (аналог std::string)
typedef struct {
    char* data;
    size_t size;
    size_t capacity;
} MoveString;

// Инициализация MoveString
MoveString* move_string_create(const char* str) {
    MoveString* ms = malloc(sizeof(MoveString));
    if (!str) {
        ms->data = NULL;
        ms->size = 0;
        ms->capacity = 0;
        return ms;
    }
    
    ms->size = strlen(str);
    ms->capacity = ms->size + 1;
    ms->data = malloc(ms->capacity);
    strcpy(ms->data, str);
    
    printf("Создана строка: %s (адрес: %p)\n", ms->data, (void*)ms->data);
    return ms;
}

// Копирующий конструктор (аналог copy constructor)
MoveString* move_string_copy(const MoveString* original) {
    if (!original) return NULL;
    
    MoveString* copy = malloc(sizeof(MoveString));
    copy->size = original->size;
    copy->capacity = original->capacity;
    
    if (original->data) {
        copy->data = malloc(copy->capacity);
        strcpy(copy->data, original->data);
        printf("Скопирована строка: %s (адрес: %p)\n", copy->data, (void*)copy->data);
    } else {
        copy->data = NULL;
    }
    
    return copy;
}

// Перемещающий конструктор (аналог move constructor)
MoveString* move_string_move(MoveString* source) {
    if (!source) return NULL;
    
    MoveString* moved = malloc(sizeof(MoveString));
    // Передаем владение ресурсами
    moved->data = source->data;
    moved->size = source->size;
    moved->capacity = source->capacity;
    
    // Обнуляем исходный объект
    source->data = NULL;
    source->size = 0;
    source->capacity = 0;
    
    printf("Выполнено перемещение строки: %s (адрес: %p)\n", 
           moved->data ? moved->data : "(null)", (void*)moved->data);
    
    return moved;
}

// Оператор присваивания с копированием (copy assignment)
void move_string_copy_assign(MoveString* dest, const MoveString* source) {
    if (!dest || !source) return;
    
    // Освобождаем старые ресурсы
    if (dest->data) {
        printf("Удаляем старую строку: %s\n", dest->data);
        free(dest->data);
    }
    
    // Копируем данные
    dest->size = source->size;
    dest->capacity = source->capacity;
    dest->data = malloc(dest->capacity);
    strcpy(dest->data, source->data);
    
    printf("Выполнено копирование присваивания: %s\n", dest->data);
}

// Оператор присваивания с перемещением (move assignment)
void move_string_move_assign(MoveString* dest, MoveString* source) {
    if (!dest || !source) return;
    
    // Освобождаем старые ресурсы
    if (dest->data) {
        printf("Удаляем старую строку: %s\n", dest->data);
        free(dest->data);
    }
    
    // Передаем владение ресурсами
    dest->data = source->data;
    dest->size = source->size;
    dest->capacity = source->capacity;
    
    // Обнуляем исходный объект
    source->data = NULL;
    source->size = 0;
    source->capacity = 0;
    
    printf("Выполнено перемещение присваивания: %s\n", dest->data ? dest->data : "(null)");
}

// Уничтожение MoveString
void move_string_destroy(MoveString* ms) {
    if (ms) {
        if (ms->data) {
            printf("Уничтожение строки: %s\n", ms->data);
            free(ms->data);
        }
        free(ms);
    }
}

// Функция, возвращающая MoveString (демонстрирует RVO/NRVO и move при необходимости)
MoveString* create_temporary_string(const char* prefix, int id) {
    char buffer[256];
    snprintf(buffer, sizeof(buffer), "%s_%d", prefix, id);
    
    MoveString* temp = move_string_create(buffer);
    return temp;  // Может быть применен RVO, или move возвращаемого значения
}

// Функция, принимающая MoveString и возвращающая его (демонстрирует перемещение)
MoveString* process_string(MoveString* input) {
    if (!input) return NULL;
    
    printf("Обработка строки: %s\n", input->data);
    
    // Возвращаем владение строкой (перемещение)
    MoveString* result = malloc(sizeof(MoveString));
    result->data = input->data;
    result->size = input->size;
    result->capacity = input->capacity;
    
    // Обнуляем исходный объект
    input->data = NULL;
    input->size = 0;
    input->capacity = 0;
    
    return result;
}

// Печать информации о строке
void move_string_print(MoveString* ms, const char* label) {
    if (!ms) {
        printf("%s: NULL\n", label);
        return;
    }
    printf("%s: %s (size=%zu, capacity=%zu, data_addr=%p)\n", 
           label, 
           ms->data ? ms->data : "(null)",
           ms->size,
           ms->capacity,
           (void*)ms->data);
}

// Пример использования
int main() {
    printf("=== Демонстрация Move Semantics (аналог) ===\n\n");
    
    // Создание строк
    printf("1. Создание строк:\n");
    MoveString* str1 = move_string_create("Hello World");
    move_string_print(str1, "str1");
    
    MoveString* str2 = move_string_create("Another String");
    move_string_print(str2, "str2");
    
    // Копирование
    printf("\n2. Копирование строки:\n");
    MoveString* str_copy = move_string_copy(str1);
    move_string_print(str_copy, "str_copy");
    
    // Перемещение
    printf("\n3. Перемещение строки:\n");
    MoveString* str_moved = move_string_move(str2);
    move_string_print(str_moved, "str_moved");
    move_string_print(str2, "str2 после перемещения (должна быть пустой)");
    
    // Оператор присваивания с копированием
    printf("\n4. Копирование присваивания:\n");
    MoveString* str_assign_copy = move_string_create("Initial Value");
    move_string_print(str_assign_copy, "str_assign_copy до");
    move_string_copy_assign(str_assign_copy, str1);
    move_string_print(str_assign_copy, "str_assign_copy после");
    
    // Оператор присваивания с перемещением
    printf("\n5. Перемещение присваивания:\n");
    MoveString* str_assign_move = move_string_create("Initial Value 2");
    move_string_print(str_assign_move, "str_assign_move до");
    MoveString* temp_str = move_string_create("Temp Value");
    move_string_move_assign(str_assign_move, temp_str);
    move_string_print(str_assign_move, "str_assign_move после");
    move_string_print(temp_str, "temp_str после перемещения (должна быть пустой)");
    
    // Демонстрация возврата временного объекта
    printf("\n6. Возврат временного объекта (может использовать move):\n");
    MoveString* returned_str = create_temporary_string("Temporary", 123);
    move_string_print(returned_str, "returned_str");
    
    // Демонстрация перемещения при передаче между функциями
    printf("\n7. Перемещение при передаче между функциями:\n");
    MoveString* processed = process_string(returned_str);
    move_string_print(processed, "processed строка");
    move_string_print(returned_str, "returned_str после обработки (должна быть пустой)");
    
    // Очистка
    printf("\n8. Очистка ресурсов:\n");
    move_string_destroy(str1);
    move_string_destroy(str_copy);
    move_string_destroy(str_moved);
    move_string_destroy(str_assign_copy);
    move_string_destroy(str_assign_move);
    move_string_destroy(processed);
    move_string_destroy(str2);  // Уничтожение уже перемещенного объекта (должно быть безопасно)
    
    printf("\nДемонстрация Move Semantics завершена\n");
    printf("Move semantics позволяет эффективно передавать владение ресурсами,\n");
    printf("избегая дорогостоящих операций копирования.\n");
    
    return 0;
}
