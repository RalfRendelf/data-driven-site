<h2>SFINAE (Substitution Failure Is Not An Error)</h2>

<p><strong>Категория:</strong> Язык</p>

<h3>Описание</h3>
<p>SFINAE (Substitution Failure Is Not An Error) — это принцип в системе шаблонов C++, согласно которому ошибки, возникающие при подстановке конкретных типов вместо параметров шаблона, не считаются фатальными ошибками компиляции. Вместо этого, шаблон просто исключается из рассмотрения при разрешении перегрузок. Это позволяет создавать более гибкие и мощные системы шаблонов.</p>

<h3>Добавлен в стандарт:</h3>
<p>Концепция была сформулирована после C++98, но стала важной частью шаблонной системы начиная с C++03</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные концепции SFINAE и их реализация на C (для демонстрации аналогов):</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка стандартной библиотеки C
 * Компиляция: gcc -o sfinae_example sfinae_example.c
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;string.h&gt;

// Макрос для определения, является ли тип целочисленным (аналог std::is_integral)
#define IS_INTEGRAL(type) _Generic((type){0}, \
    char: 1, signed char: 1, unsigned char: 1, \
    short: 1, unsigned short: 1, \
    int: 1, unsigned int: 1, \
    long: 1, unsigned long: 1, \
    long long: 1, unsigned long long: 1, \
    default: 0 \
)

// Макрос для определения, является ли тип числом с плавающей точкой
#define IS_FLOATING_POINT(type) _Generic((type){0}, \
    float: 1, double: 1, long double: 1, \
    default: 0 \
)

// Макрос для определения, является ли тип арифметическим
#define IS_ARITHMETIC(type) (IS_INTEGRAL(type) || IS_FLOATING_POINT(type))

// Структура для хранения информации о типе
typedef struct {
    char type_name[50];
    int is_integral;
    int is_floating_point;
    int is_arithmetic;
} TypeInfo;

// Макрос для получения информации о типе
#define GET_TYPE_INFO(type, var_name) \
    TypeInfo var_name = {0}; \
    strncpy(var_name.type_name, #type, sizeof(var_name.type_name) - 1); \
    var_name.type_name[sizeof(var_name.type_name) - 1] = '\0'; \
    var_name.is_integral = IS_INTEGRAL(type); \
    var_name.is_floating_point = IS_FLOATING_POINT(type); \
    var_name.is_arithmetic = IS_ARITHMETIC(type);

// Функция для проверки, можно ли выполнить арифметическую операцию (аналог SFINAE)
int can_perform_arithmetic(void* value, int type_check) {
    return type_check; // Возвращает результат проверки типа
}

// Макрос для условного выполнения в зависимости от типа (аналог enable_if)
#define CONDITIONAL_EXECUTE(expr, integral_action, floating_action, other_action) \
    _Generic((expr), \
        char: integral_action, signed char: integral_action, unsigned char: integral_action, \
        short: integral_action, unsigned short: integral_action, \
        int: integral_action, unsigned int: integral_action, \
        long: integral_action, unsigned long: integral_action, \
        long long: integral_action, unsigned long long: integral_action, \
        float: floating_action, double: floating_action, long double: floating_action, \
        default: other_action \
    )

// Функция, имитирующая перегрузку с SFINAE для арифметических типов
void process_value_arithmetic(int value) {
    printf("Обработка целочисленного значения: %d\n", value);
}

void process_value_arithmetic_double(double value) {
    printf("Обработка вещественного значения: %.2f\n", value);
}

// Макрос для выбора подходящей функции в зависимости от типа
#define PROCESS_VALUE_SF_SAFE(val) \
    CONDITIONAL_EXECUTE((val), \
        process_value_arithmetic((int)(val)), \
        process_value_arithmetic_double((double)(val)), \
        printf("Тип не поддерживается для арифметических операций\n") \
    )

// Структура для демонстрации проверки наличия метода (условно)
typedef struct {
    int has_method_process;
    int has_method_compute;
} TypeCapabilities;

// Макрос для проверки возможностей типа
#define HAS_METHOD_PROCESS(type) _Generic((type){0}, \
    int: 1, float: 1, double: 1, \
    default: 0 \
)

#define HAS_METHOD_COMPUTE(type) _Generic((type){0}, \
    int: 1, double: 1, \
    default: 0 \
)

// Функция, демонстрирующая SFINAE для разных типов
void sfinae_demo() {
    printf("=== Демонстрация SFINAE ===\n\n");
    
    // Проверка типов
    GET_TYPE_INFO(42, int_info);
    GET_TYPE_INFO(3.14, double_info);
    GET_TYPE_INFO('A', char_info);
    GET_TYPE_INFO("string", string_info);
    
    printf("Информация о типах:\n");
    printf("int: is_integral=%d, is_floating_point=%d, is_arithmetic=%d\n", 
           int_info.is_integral, int_info.is_floating_point, int_info.is_arithmetic);
    printf("double: is_integral=%d, is_floating_point=%d, is_arithmetic=%d\n", 
           double_info.is_integral, double_info.is_floating_point, double_info.is_arithmetic);
    printf("char: is_integral=%d, is_floating_point=%d, is_arithmetic=%d\n", 
           char_info.is_integral, char_info.is_floating_point, char_info.is_arithmetic);
    printf("string: is_integral=%d, is_floating_point=%d, is_arithmetic=%d\n", 
           string_info.is_integral, string_info.is_floating_point, string_info.is_arithmetic);
    
    // Демонстрация условного выполнения
    printf("\nУсловное выполнение в зависимости от типа:\n");
    PROCESS_VALUE_SF_SAFE(42);
    PROCESS_VALUE_SF_SAFE(3.14159);
    PROCESS_VALUE_SF_SAFE('X');
    
    // Проверка возможностей типов
    printf("\nПроверка возможностей типов:\n");
    int sample_int = 10;
    double sample_double = 20.5;
    char sample_char = 'Z';
    
    printf("int имеет метод process: %s\n", HAS_METHOD_PROCESS(sample_int) ? "да" : "нет");
    printf("double имеет метод process: %s\n", HAS_METHOD_PROCESS(sample_double) ? "да" : "нет");
    printf("char имеет метод process: %s\n", HAS_METHOD_PROCESS(sample_char) ? "да" : "нет");
    
    printf("int имеет метод compute: %s\n", HAS_METHOD_COMPUTE(sample_int) ? "да" : "нет");
    printf("double имеет метод compute: %s\n", HAS_METHOD_COMPUTE(sample_double) ? "да" : "нет");
    printf("char имеет метод compute: %s\n", HAS_METHOD_COMPUTE(sample_char) ? "да" : "нет");
    
    // Имитация SFINAE для выбора функции
    printf("\nИмитация SFINAE для выбора подходящей функции:\n");
    int int_val = 100;
    double double_val = 200.5;
    char char_val = 'C';
    
    // Проверка, можно ли выполнить арифметическую операцию
    if (can_perform_arithmetic(&int_val, IS_ARITHMETIC(int_val))) {
        printf("Для int: можно выполнить арифметическую операцию\n");
    }
    
    if (can_perform_arithmetic(&double_val, IS_ARITHMETIC(double_val))) {
        printf("Для double: можно выполнить арифметическую операцию\n");
    }
    
    if (can_perform_arithmetic(&char_val, IS_ARITHMETIC(char_val))) {
        printf("Для char: можно выполнить арифметическую операцию\n");
    }
    
    // Попробуем с неарифметическим типом
    char* str_val = "test";
    if (can_perform_arithmetic(str_val, IS_ARITHMETIC(*str_val))) {
        printf("Для строки: можно выполнить арифметическую операцию\n");
    } else {
        printf("Для строки: нельзя выполнить арифметическую операцию\n");
    }
    
    printf("\nSFINAE позволяет выбирать подходящие функции в зависимости от свойств типов\n");
    printf("без возникновения ошибок компиляции при несовместимых типах.\n");
}

// Пример использования SFINAE для ограничения шаблонов
void demonstrate_sfinae_principle() {
    printf("\n--- Демонстрация принципа SFINAE ---\n");
    
    // В C мы не можем использовать настоящие шаблоны, но можем имитировать
    // ситуацию, когда одна перегрузка "отбрасывается", если не подходит
    
    printf("В C++ SFINAE позволяет определить несколько шаблонных функций,\n");
    printf("и компилятор автоматически выбирает подходящую, исключая\n");
    printf("несовместимые варианты из рассмотрения, а не вызывая ошибку.\n\n");
    
    printf("Например, если мы определяем функцию только для арифметических типов:\n");
    printf("template<typename T>\n");
    printf("typename std::enable_if<std::is_arithmetic<T>::value, void>::type\n");
    printf("process(T value) { /* ... */ }\n\n");
    
    printf("То при попытке вызвать process(string) компилятор не выдаст ошибку,\n");
    printf("а просто исключит этот вариант из рассмотрения.\n");
}

int main() {
    sfinae_demo();
    demonstrate_sfinae_principle();
    
    return 0;
}
