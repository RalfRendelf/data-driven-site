<h2>std::scoped_lock</h2>

<p><strong>Категория:</strong> Многопоточность</p>

<h3>Описание</h3>
<p>std::scoped_lock - это RAII-обертка для автоматического управления блокировками. Он обеспечивает безопасный способ захвата и освобождения одной или нескольких блокировок. Введен в C++17 как улучшенная версия std::lock_guard, с поддержкой deadlock-безопасного захвата нескольких мьютексов.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++17</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::scoped_lock:</p>

<pre>
<code>
// Пример использования std::scoped_lock на C++

#include &lt;iostream&gt;
#include &lt;mutex&gt;
#include &lt;thread&gt;
#include &lt;vector&gt;

class BankAccount {
private:
    int balance;
    std::mutex mtx;
    
public:
    BankAccount(int initial_balance) : balance(initial_balance) {}
    
    int get_balance() const { return balance; }
    
    // Функция перевода денег между аккаунтами
    static void transfer(BankAccount& from, BankAccount& to, int amount) {
        // Используем scoped_lock для deadlock-безопасного захвата мьютексов
        std::scoped_lock lock(from.mtx, to.mtx);
        
        // После захвата блокировок можем безопасно работать с обеими учетными записями
        if (from.balance >= amount) {
            from.balance -= amount;
            to.balance += amount;
            
            std::cout << "Transferred " << amount << " from account with balance " 
                      << from.balance + amount << " to account with balance " 
                      << to.balance - amount << std::endl;
        } else {
            std::cout << "Insufficient funds for transfer of " << amount 
                      << " from account with balance " << from.balance << std::endl;
        }
    }
};

int main() {
    BankAccount acc1(100);
    BankAccount acc2(50);
    
    std::cout << "Initial balances: Acc1=" << acc1.get_balance() 
              << ", Acc2=" << acc2.get_balance() << std::endl;
    
    // Создаем несколько потоков для выполнения переводов
    std::vector&lt;std::thread&gt; threads;
    
    // Поток 1: перевод с первого аккаунта на второй
    threads.emplace_back([](BankAccount& from, BankAccount& to, int amount) {
        BankAccount::transfer(from, to, amount);
    }, std::ref(acc1), std::ref(acc2), 30);
    
    // Поток 2: перевод со второго аккаунта на первый
    threads.emplace_back([](BankAccount& from, BankAccount& to, int amount) {
        BankAccount::transfer(from, to, amount);
    }, std::ref(acc2), std::ref(acc1), 20);
    
    // Ожидаем завершения всех потоков
    for (auto&amp; t : threads) {
        t.join();
    }
    
    std::cout << "Final balances: Acc1=" << acc1.get_balance() 
              << ", Acc2=" << acc2.get_balance() << std::endl;
    
    return 0;
}

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка POSIX threads
 * Компиляция: gcc -o scoped_lock_example scoped_lock_example.c -lpthread
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;pthread.h&gt;
#include &lt;unistd.h&gt;

// Структура для имитации RAII-обертки для мьютекса
typedef struct {
    pthread_mutex_t* mutex;
    int locked;
} MutexWrapper;

// Структура для имитации scoped_lock
typedef struct {
    MutexWrapper* wrappers;
    int count;
} ScopedLock;

// Структура банковского счета
typedef struct {
    int balance;
    pthread_mutex_t mutex;
} BankAccount;

// Функция для инициализации банковского счета
BankAccount* create_account(int initial_balance) {
    BankAccount* acc = malloc(sizeof(BankAccount));
    acc->balance = initial_balance;
    pthread_mutex_init(&acc->mutex, NULL);
    return acc;
}

// Функция для освобождения банковского счета
void destroy_account(BankAccount* acc) {
    if (acc) {
        pthread_mutex_destroy(&acc->mutex);
        free(acc);
    }
}

// Функция для получения баланса
int get_balance(BankAccount* acc) {
    if (!acc) return 0;
    
    int balance;
    pthread_mutex_lock(&acc->mutex);
    balance = acc->balance;
    pthread_mutex_unlock(&acc->mutex);
    
    return balance;
}

// Deadlock-безопасная функция захвата нескольких мьютексов
void acquire_locks(pthread_mutex_t* m1, pthread_mutex_t* m2) {
    while(1) {
        // Пытаемся захватить первый мьютекс
        if (pthread_mutex_trylock(m1) == 0) {
            // Успешно захватили первый, теперь пробуем захватить второй
            if (pthread_mutex_trylock(m2) == 0) {
                // Оба мьютекса успешно захвачены
                break;
            } else {
                // Не удалось захватить второй, освобождаем первый и повторяем
                pthread_mutex_unlock(m1);
                usleep(1000); // Небольшая задержка перед повторной попыткой
            }
        } else {
            // Не удалось захватить первый, делаем паузу и пробуем снова
            usleep(1000);
        }
    }
}

// Функция перевода денег между аккаунтами
void transfer(BankAccount* from, BankAccount* to, int amount) {
    if (!from || !to) return;
    
    // Deadlock-безопасный захват мьютексов
    acquire_locks(&from->mutex, &to->mutex);
    
    // После захвата блокировок можем безопасно работать с обеими учетными записями
    if (from->balance >= amount) {
        from->balance -= amount;
        to->balance += amount;
        
        printf("Transferred %d from account with balance %d to account with balance %d\n", 
               amount, from->balance + amount, to->balance - amount);
    } else {
        printf("Insufficient funds for transfer of %d from account with balance %d\n", 
               amount, from->balance);
    }
    
    // Освобождаем мьютексы
    pthread_mutex_unlock(&from->mutex);
    pthread_mutex_unlock(&to->mutex);
}

// Структура для передачи данных в поток
typedef struct {
    BankAccount* from;
    BankAccount* to;
    int amount;
} TransferData;

// Функция потока для выполнения перевода
void* thread_transfer(void* arg) {
    TransferData* data = (TransferData*)arg;
    transfer(data->from, data->to, data->amount);
    free(data);
    return NULL;
}

// Пример использования
int main() {
    BankAccount* acc1 = create_account(100);
    BankAccount* acc2 = create_account(50);
    
    printf("Initial balances: Acc1=%d, Acc2=%d\n", 
           get_balance(acc1), get_balance(acc2));
    
    // Создаем несколько потоков для выполнения переводов
    pthread_t t1, t2;
    
    // Подготовка данных для первого потока
    TransferData* data1 = malloc(sizeof(TransferData));
    data1->from = acc1;
    data1->to = acc2;
    data1->amount = 30;
    
    // Подготовка данных для второго потока
    TransferData* data2 = malloc(sizeof(TransferData));
    data2->from = acc2;
    data2->to = acc1;
    data2->amount = 20;
    
    // Запуск потоков
    pthread_create(&t1, NULL, thread_transfer, data1);
    pthread_create(&t2, NULL, thread_transfer, data2);
    
    // Ожидаем завершения потоков
    pthread_join(t1, NULL);
    pthread_join(t2, NULL);
    
    printf("Final balances: Acc1=%d, Acc2=%d\n", 
           get_balance(acc1), get_balance(acc2));
    
    // Освобождаем ресурсы
    destroy_account(acc1);
    destroy_account(acc2);
    
    return 0;
}
