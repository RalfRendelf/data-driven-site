<h2>std::jthread</h2>

<p><strong>Категория:</strong> Многопоточность</p>

<h3>Описание</h3>
<p>std::jthread - это класс, представляющий поток исполнения, который автоматически присоединяется при уничтожении объекта, если не был отсоединен ранее. Также включает в себя поддержку механизма остановки через std::stop_token. Введен в C++20 как улучшенная версия std::thread.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++20</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::jthread:</p>

<pre>
<code>
// Пример использования std::jthread на C++

#include &lt;iostream&gt;
#include &lt;thread&gt;
#include &lt;chrono&gt;
#include &lt;vector&gt;

// Функция, которая будет выполняться в потоке
void worker_function(std::stop_token stoken, int id) {
    std::cout << "Thread " << id << " started" << std::endl;
    
    for (int i = 0; i < 10; ++i) {
        // Проверка токена остановки
        if (stoken.stop_requested()) {
            std::cout << "Thread " << id << " received stop request" << std::endl;
            return;
        }
        
        std::cout << "Thread " << id << " doing work: " << i << std::endl;
        std::this_thread::sleep_for(std::chrono::milliseconds(500));
    }
    
    std::cout << "Thread " << id << " completed normally" << std::endl;
}

int main() {
    std::vector&lt;std::jthread&gt; threads;
    
    // Создание нескольких потоков
    for (int i = 0; i < 3; ++i) {
        threads.emplace_back(worker_function, i);
    }
    
    std::cout << "Main thread sleeping..." << std::endl;
    std::this_thread::sleep_for(std::chrono::milliseconds(1500));
    
    // Запрос остановки всех потоков
    std::cout << "Requesting stop for all threads..." << std::endl;
    for (auto&amp; t : threads) {
        t.request_stop();
    }
    
    std::cout << "Threads will join automatically when jthread destructors are called" << std::endl;
    
    return 0;
}

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка POSIX threads
 * Компиляция: gcc -o jthread_example jthread_example.c -lpthread
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;pthread.h&gt;
#include &lt;unistd.h&gt;
#include &lt;signal.h&gt;
#include &lt;stdatomic.h&gt;

// Структура, имитирующая функциональность jthread
typedef struct {
    pthread_t thread;
    atomic_bool stop_requested;
    int id;
} JThread;

// Функция, которая будет выполняться в потоке
void* worker_function(void* arg) {
    JThread* jt = (JThread*)arg;
    printf("Thread %d started\n", jt->id);
    
    for (int i = 0; i < 10; ++i) {
        // Проверка флага остановки
        if (atomic_load(&jt->stop_requested)) {
            printf("Thread %d received stop request\n", jt->id);
            return NULL;
        }
        
        printf("Thread %d doing work: %d\n", jt->id, i);
        usleep(500000); // спим 500 мс
    }
    
    printf("Thread %d completed normally\n", jt->id);
    return NULL;
}

// Функция для создания jthread
JThread* create_jthread(void* (*func)(void*), int id) {
    JThread* jt = malloc(sizeof(JThread));
    jt->id = id;
    atomic_store(&jt->stop_requested, false);
    
    // Создание потока
    pthread_create(&jt->thread, NULL, func, jt);
    
    return jt;
}

// Функция для запроса остановки
void request_stop(JThread* jt) {
    if (jt) {
        atomic_store(&jt->stop_requested, true);
    }
}

// Функция для ожидания завершения потока (имитирует автоматическое присоединение)
void join_jthread(JThread* jt) {
    if (jt) {
        pthread_join(jt->thread, NULL);
    }
}

// Функция для уничтожения jthread
void destroy_jthread(JThread* jt) {
    if (jt) {
        // Автоматическое присоединение при уничтожении
        if (jt->thread) {
            pthread_join(jt->thread, NULL);
        }
        free(jt);
    }
}

// Пример использования
int main() {
    #define NUM_THREADS 3
    JThread* threads[NUM_THREADS];
    
    // Создание нескольких потоков
    for (int i = 0; i < NUM_THREADS; ++i) {
        threads[i] = create_jthread(worker_function, i);
    }
    
    printf("Main thread sleeping...\n");
    usleep(1500000); // спим 1.5 с
    
    // Запрос остановки всех потоков
    printf("Requesting stop for all threads...\n");
    for (int i = 0; i < NUM_THREADS; ++i) {
        request_stop(threads[i]);
    }
    
    printf("Joining threads...\n");
    for (int i = 0; i < NUM_THREADS; ++i) {
        join_jthread(threads[i]);
    }
    
    printf("Destroying threads...\n");
    for (int i = 0; i < NUM_THREADS; ++i) {
        destroy_jthread(threads[i]);
    }
    
    return 0;
}
