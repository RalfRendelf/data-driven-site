<h2>std::future/std::promise</h2>

<p><strong>Категория:</strong> Многопоточность</p>

<h3>Описание</h3>
<p>std::future и std::promise - это механизм для асинхронной передачи значений между потоками. std::promise используется для установки значения, а std::future - для получения этого значения. Когда поток устанавливает значение через promise, другой поток может получить его через future.</p>

<h3>Добавлен в стандарт:</h3>
<p>C++11</p>

<h3>Основные методы:</h3>
<p>В этом разделе представлены основные методы std::future/std::promise:</p>

<h3>Реализация на C++:</h3>
<pre>
<code>
#include &lt;iostream&gt;
#include &lt;future&gt;
#include &lt;thread&gt;
#include &lt;chrono&gt;

void setter(std::promise&lt;int&gt; prom) {
    std::cout << "Производитель вычисляет значение..." << std::endl;
    
    // Имитация вычислений
    std::this_thread::sleep_for(std::chrono::seconds(2));
    
    // Установка результата
    prom.set_value(42);
    std::cout << "Производитель установил значение: 42" << std::endl;
}

void getter(std::future&lt;int&gt; fut) {
    std::cout << "Потребитель ожидает значение..." << std::endl;
    
    // Блокировка до получения значения
    int value = fut.get();
    std::cout << "Потребитель получил значение: " << value << std::endl;
}

int main() {
    // Создание пары future/promise
    std::promise&lt;int&gt; prom;
    std::future&lt;int&gt; fut = prom.get_future();
    
    // Создание потоков
    std::thread t1(setter, std::move(prom));
    std::thread t2(getter, std::move(fut));
    
    // Ожидание завершения потоков
    t1.join();
    t2.join();
    
    std::cout << "Программа завершена" << std::endl;
    
    return 0;
}
</code>
</pre>

<h3>Реализация на C:</h3>
<p>Для сравнения, вот как можно реализовать аналогичную функциональность на C с использованием pthread и общих данных:</p>

<pre>
<code>
/***
 * Для компиляции необходима поддержка POSIX threads
 * Компиляция: gcc -o future_promise_example future_promise_example.c -lpthread
 ***/

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;pthread.h&gt;
#include &lt;unistd.h&gt;
#include &lt;stdbool.h&gt;

// Структура для хранения общего состояния
typedef struct {
    pthread_mutex_t mutex;
    pthread_cond_t cond;
    int* value;
    bool ready;
    bool has_exception;
    void* exception;
} FuturePromiseState;

// Структура для future
typedef struct {
    FuturePromiseState* state;
} Future;

// Структура для promise
typedef struct {
    FuturePromiseState* state;
} Promise;

// Инициализация общего состояния
FuturePromiseState* state_create() {
    FuturePromiseState* state = malloc(sizeof(FuturePromiseState));
    pthread_mutex_init(&state->mutex, NULL);
    pthread_cond_init(&state->cond, NULL);
    state->value = NULL;
    state->ready = false;
    state->has_exception = false;
    state->exception = NULL;
    return state;
}

// Уничтожение состояния
void state_destroy(FuturePromiseState* state) {
    if (state != NULL) {
        pthread_mutex_destroy(&state->mutex);
        pthread_cond_destroy(&state->cond);
        free(state);
    }
}

// Создание пары future/promise
void create_future_promise_pair(Future* future, Promise* promise) {
    FuturePromiseState* state = state_create();
    
    future->state = state;
    promise->state = state;
}

// Установка значения через promise
void promise_set_value(Promise* promise, int value) {
    FuturePromiseState* state = promise->state;
    
    pthread_mutex_lock(&state->mutex);
    
    if (!state->ready) {
        state->value = malloc(sizeof(int));
        *(state->value) = value;
        state->ready = true;
        pthread_cond_broadcast(&state->cond);
    }
    
    pthread_mutex_unlock(&state->mutex);
}

// Установка исключения через promise
void promise_set_exception(Promise* promise, void* exception) {
    FuturePromiseState* state = promise->state;
    
    pthread_mutex_lock(&state->mutex);
    
    if (!state->ready) {
        state->exception = exception;
        state->has_exception = true;
        state->ready = true;
        pthread_cond_broadcast(&state->cond);
    }
    
    pthread_mutex_unlock(&state->mutex);
}

// Получение значения через future
int future_get(Future* future) {
    FuturePromiseState* state = future->state;
    
    pthread_mutex_lock(&state->mutex);
    
    // Ждать, пока значение не будет готово
    while (!state->ready) {
        pthread_cond_wait(&state->cond, &state->mutex);
    }
    
    int result = 0;
    
    if (state->has_exception) {
        // Обработка исключения
        printf("Произошло исключение при выполнении\n");
    } else {
        result = *(state->value);
    }
    
    pthread_mutex_unlock(&state->mutex);
    
    return result;
}

// Проверка, готово ли значение
bool future_is_ready(Future* future) {
    FuturePromiseState* state = future->state;
    
    pthread_mutex_lock(&state->mutex);
    bool ready = state->ready;
    pthread_mutex_unlock(&state->mutex);
    
    return ready;
}

// Функция производителя
void* setter_function(void* arg) {
    Promise* promise = (Promise*)arg;
    
    printf("Производитель вычисляет значение...\n");
    
    // Имитация вычислений
    sleep(2);
    
    // Установка результата
    promise_set_value(promise, 42);
    printf("Производитель установил значение: 42\n");
    
    return NULL;
}

// Функция потребителя
void* getter_function(void* arg) {
    Future* future = (Future*)arg;
    
    printf("Потребитель ожидает значение...\n");
    
    // Блокировка до получения значения
    int value = future_get(future);
    printf("Потребитель получил значение: %d\n", value);
    
    return NULL;
}

// Пример использования
int main() {
    Future future;
    Promise promise;
    
    // Создание пары future/promise
    create_future_promise_pair(&future, &promise);
    
    pthread_t setter_thread, getter_thread;
    
    // Создание потоков
    pthread_create(&setter_thread, NULL, setter_function, &promise);
    pthread_create(&getter_thread, NULL, getter_function, &future);
    
    // Ожидание завершения потоков
    pthread_join(setter_thread, NULL);
    pthread_join(getter_thread, NULL);
    
    printf("Программа завершена\n");
    
    // Освобождение ресурсов
    if (future.state->value) {
        free(future.state->value);
    }
    state_destroy(future.state);  // состояние разделяется между future и promise
    
    return 0;
}
