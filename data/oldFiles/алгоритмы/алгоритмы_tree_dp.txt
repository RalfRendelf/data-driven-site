<h2>Tree DP (Динамическое программирование на деревьях)</h2>

<p><strong>Категория:</strong> Динамическое программирование</p>

<h3>Описание</h3>
<p>Динамическое программирование на деревьях - это метод решения задач, в которых основной структурой данных является дерево. В отличие от обычного DP, состояния связаны с вершинами дерева, и переходы между состояниями происходят по рёбрам дерева.</p>

<h3>Принцип работы</h3>
<ol>
<li>Определить состояние DP как некоторую характеристику поддерева</li>
<li>Вычислить состояние для листьев (база рекурсии)</li>
<li>Вычислить состояние для внутренней вершины на основе состояний её потомков</li>
<li>Объединить результаты поддеревьев для получения ответа</li>
</ol>

<h3>Сложность алгоритма</h3>
<ul>
<li>Временная сложность: O(n) для большинства задач</li>
<li>Пространственная сложность: O(n) для хранения дерева и DP-таблицы</li>
</ul>

<h3>Преимущества и недостатки</h3>
<p>Преимущества:</p>
<ul>
<li>Эффективное решение задач на деревьях</li>
<li>Естественное представление рекурсивной структуры задачи</li>
<li>Может быть расширено для онлайн-запросов</li>
</ul>

<p>Недостатки:</p>
<ul>
<li>Требует понимания структуры дерева</li>
<li>Сложнее для визуализации и отладки</li>
</ul>

<h3>Пример на C++: Максимальный независимый набор в дереве</h3>
<pre>
<code>
#include &lt;iostream&gt;
#include &lt;vector&gt;
#include &lt;algorithm&gt;

class TreeDP {
public:
    // Функция для вычисления максимального независимого набора
    static std::pair&lt;int, int&gt; maxIndependentSet(const std::vector&lt;std::vector&lt;int&gt;&gt;&amp; tree, int u, int parent) {
        // dp[u][0] - максимальный размер, если u не включён
        // dp[u][1] - максимальный размер, если u включён
        std::pair&lt;int, int&gt; dp_u = {0, 1};  // {not_include, include}
        
        for (int v : tree[u]) {
            if (v != parent) {
                auto dp_v = maxIndependentSet(tree, v, u);
                
                // Если мы не включаем u, то можем выбрать максимум из включения/не включения v
                int not_include_u = dp_u.first + std::max(dp_v.first, dp_v.second);
                
                // Если мы включаем u, то v не может быть включён
                int include_u = dp_u.second + dp_v.first;
                
                dp_u = {not_include_u, include_u};
            }
        }
        
        return dp_u;
    }
    
    // Функция для вычисления диаметра дерева
    static int diameter(const std::vector&lt;std::vector&lt;int&gt;&gt;&amp; tree) {
        int n = tree.size();
        std::vector&lt;int&gt; height(n, 0);
        int diameter = 0;
        
        // DFS для вычисления высоты поддеревьев и диаметра
        dfs_diameter(tree, 0, -1, height, diameter);
        
        return diameter;
    }
    
private:
    static void dfs_diameter(const std::vector&lt;std::vector&lt;int&gt;&gt;&amp; tree, int u, int parent, 
                            std::vector&lt;int&gt;&amp; height, int&amp; diameter) {
        int maxHeight1 = 0, maxHeight2 = 0;
        
        for (int v : tree[u]) {
            if (v != parent) {
                dfs_diameter(tree, v, u, height, diameter);
                
                if (height[v] > maxHeight1) {
                    maxHeight2 = maxHeight1;
                    maxHeight1 = height[v];
                } else if (height[v] > maxHeight2) {
                    maxHeight2 = height[v];
                }
            }
        }
        
        height[u] = maxHeight1 + 1;
        diameter = std::max(diameter, maxHeight1 + maxHeight2);
    }
};

int main() {
    // Пример дерева: 0-1-2-3, 1-4
    std::vector&lt;std::vector&lt;int&gt;&gt; tree = {
        {1},           // 0
        {0, 2, 4},    // 1
        {1, 3},       // 2
        {2},          // 3
        {1}           // 4
    };
    
    auto result = TreeDP::maxIndependentSet(tree, 0, -1);
    std::cout &lt;&lt; "Max independent set size: " &lt;&lt; std::max(result.first, result.second) &lt;&lt; std::endl;
    
    std::cout &lt;&lt; "Tree diameter: " &lt;&lt; TreeDP::diameter(tree) &lt;&lt; std::endl;
    
    return 0;
}
</code>

<h3>Пример на C: Подсчет количества потомков для каждой вершины</h3>
<pre>
<code>
#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;

typedef struct {
    int *data;
    int size;
    int capacity;
} List;

List* createList(int capacity) {
    List* list = malloc(sizeof(List));
    list->data = malloc(capacity * sizeof(int));
    list->size = 0;
    list->capacity = capacity;
    return list;
}

void addToList(List* list, int value) {
    if (list->size < list->capacity) {
        list->data[list->size++] = value;
    }
}

void freeList(List* list) {
    free(list->data);
    free(list);
}

// DFS для вычисления количества потомков
int countDescendants(List** tree, int u, int parent, int* descendants) {
    int count = 0;
    
    for (int i = 0; i < tree[u]->size; i++) {
        int v = tree[u]->data[i];
        if (v != parent) {
            int subtreeCount = countDescendants(tree, v, u, descendants);
            count += subtreeCount + 1;  // +1 для самой вершины v
        }
    }
    
    descendants[u] = count;
    return count;
}

int main() {
    int n = 5;
    
    // Создаем дерево: 0-1-2-3, 1-4
    List** tree = malloc(n * sizeof(List*));
    for (int i = 0; i < n; i++) {
        tree[i] = createList(n);
    }
    
    // Добавляем ребра
    addToList(tree[0], 1);
    addToList(tree[1], 0);
    addToList(tree[1], 2);
    addToList(tree[1], 4);
    addToList(tree[2], 1);
    addToList(tree[2], 3);
    addToList(tree[3], 2);
    addToList(tree[4], 1);
    
    int* descendants = malloc(n * sizeof(int));
    countDescendants(tree, 0, -1, descendants);
    
    printf("Number of descendants for each node:\n");
    for (int i = 0; i < n; i++) {
        printf("Node %d: %d\n", i, descendants[i]);
    }
    
    // Освобождаем память
    for (int i = 0; i < n; i++) {
        freeList(tree[i]);
    }
    free(tree);
    free(descendants);
    
    return 0;
}
