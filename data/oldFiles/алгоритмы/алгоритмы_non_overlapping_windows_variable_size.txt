<h2>Non-overlapping Windows: Variable Size (Непересекающиеся окна: переменного размера)</h2>

<p><strong>Категория:</strong> Массивы</p>

<h3>Описание</h3>
<p>Алгоритм скользящего окна переменного размера используется для решения задач, где размер подмассива/подстроки должен меняться в зависимости от условия. Этот подход часто используется для нахождения минимального/максимального подмассива, удовлетворяющего определенному условию.</p>

<h3>Принцип работы</h3>
<ol>
<li>Использовать два указателя (left и right) для определения границ окна</li>
<li>Расширять окно, перемещая правый указатель, пока не будет выполнено условие</li>
<li>Когда условие выполнено, пытаться минимизировать окно, перемещая левый указатель</li>
<li>Отслеживать оптимальный результат на каждом шаге</li>
</ol>

<h3>Сложность алгоритма</h3>
<ul>
<li>Временная сложность: O(n), где n - длина массива (каждый элемент посещается максимум дважды)</li>
<li>Пространственная сложность: O(1) - только для хранения указателей и результатов</li>
</ul>

<h3>Преимущества и недостатки</h3>
<p>Преимущества:</p>
<ul>
<li>Эффективное решение задач с подмассивами переменной длины</li>
<li>Оптимальная временная сложность</li>
<li>Минимальное использование дополнительной памяти</li>
</ul>

<p>Недостатки:</p>
<ul>
<li>Требует специфических условий задачи</li>
<li>Сложнее для понимания и реализации</li>
</ul>

<h3>Пример на C++: Нахождение минимальной длины подмассива с суммой >= target</h3>
<pre>
<code>
#include &lt;iostream&gt;
#include &lt;vector&gt;
#include &lt;climits&gt;
#include &lt;algorithm&gt;

int minSubArrayLen(int target, std::vector&lt;int&gt;&amp; nums) {
    int n = nums.size();
    int left = 0, right = 0;
    int sum = 0;
    int minLength = INT_MAX;
    
    while (right < n) {
        sum += nums[right];  // Расширяем окно
        
        // Сокращаем окно слева, пока сумма >= target
        while (sum >= target) {
            minLength = std::min(minLength, right - left + 1);
            sum -= nums[left];
            left++;
        }
        
        right++;  // Расширяем окно справа
    }
    
    return minLength == INT_MAX ? 0 : minLength;
}

// Найти самую длинную подстроку без повторяющихся символов
int lengthOfLongestSubstring(std::string s) {
    std::vector&lt;int&gt; chars(128); // ASCII символы
    
    int left = 0;
    int right = 0;
    int maxWindowSize = 0;
    
    while (right < s.length()) {
        chars[s[right]]++;  // Увеличиваем счетчик для текущего символа
        
        // Если символ повторяется, сдвигаем левую границу
        while (chars[s[right]] > 1) {
            chars[s[left]]--;
            left++;
        }
        
        maxWindowSize = std::max(maxWindowSize, right - left + 1);
        right++;
    }
    
    return maxWindowSize;
}

int main() {
    std::vector&lt;int&gt; nums = {2, 3, 1, 2, 4, 3};
    int target = 7;
    
    int result = minSubArrayLen(target, nums);
    std::cout << "Minimal length of subarray with sum >= " << target << ": " << result << std::endl;
    
    std::string str = "abcabcbb";
    int longest = lengthOfLongestSubstring(str);
    std::cout << "Length of longest substring without repeating characters: " << longest << std::endl;
    
    return 0;
}

<h3>Пример на C: Найти подмассив с заданной суммой</h3>
<pre>
<code>
#include &lt;stdio.h&gt;

int findSubarrayWithSum(int arr[], int size, int targetSum, int *start, int *end) {
    int currentSum = arr[0];
    *start = 0;
    *end = 0;
    
    for (int i = 1; i <= size; i++) {
        // Если текущая сумма равна целевой, возвращаем успех
        if (currentSum == targetSum) {
            *end = i - 1;
            return 1;  // Успех
        }
        
        // Если текущая сумма меньше целевой и не достигли конца, расширяем окно
        if (currentSum < targetSum && i < size) {
            currentSum += arr[i];
        }
        
        // Если текущая сумма больше целевой или достигли конца, сжимаем окно
        while (currentSum > targetSum && *start < i - 1) {
            currentSum -= arr[*start];
            (*start)++;
            
            // Проверяем снова после сжатия
            if (currentSum == targetSum) {
                *end = i - 1;
                return 1;  // Успех
            }
        }
    }
    
    return 0;  // Не найдено
}

int main() {
    int arr[] = {1, 4, 2, 6, 3, 5, 8, 7};
    int size = sizeof(arr) / sizeof(arr[0]);
    int targetSum = 12;
    
    int start, end;
    if (findSubarrayWithSum(arr, size, targetSum, &start, &end)) {
        printf("Subarray with sum %d found from index %d to %d: ", targetSum, start, end);
        for (int i = start; i <= end; i++) {
            printf("%d ", arr[i]);
        }
        printf("\n");
    } else {
        printf("No subarray with sum %d found\n", targetSum);
    }
    
    return 0;
}
