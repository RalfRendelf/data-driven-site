<h2>MVC (Model-View-Controller)</h2>

<p><strong>Категория:</strong> Model-View-Controller</p>

<h3>Описание</h3>
<p>MVC (Model-View-Controller) - это архитектурный паттерн, широко используемый при разработке пользовательских интерфейсов. Он разделяет приложение на три взаимосвязанных компонента: модель (model), представление (view) и контроллер (controller).</p>

<h3>Компоненты</h3>
<ol>
<li><strong>Model (Модель)</strong> - представляет собой структуру данных и бизнес-логику приложения</li>
<li><strong>View (Представление)</strong> - отвечает за отображение данных модели пользователю</li>
<li><strong>Controller (Контроллер)</strong> - принимает входные данные пользователя и управляет взаимодействием между Model и View</li>
</ol>

<h3>Преимущества и недостатки</h3>
<p>Преимущества:</p>
<ul>
<li>Разделение ответственности между компонентами</li>
<li>Упрощение разработки и тестирования</li>
<li>Возможность многократного использования кода</li>
<li>Улучшенное управление сложностью приложения</li>
</ul>

<p>Недостатки:</p>
<ul>
<li>Усложнение архитектуры для простых приложений</li>
<li>Повышенная сложность из-за разделения на компоненты</li>
<li>Потенциально более медленная производительность</li>
</ul>

#ifndef USERMODEL_H
#define USERMODEL_H

#include &lt;QAbstractListModel&gt;
#include &lt;QVector&gt;
#include "user.h"

class UserModel : public QAbstractListModel
{
    Q_OBJECT

public:
    enum UserRoles {
        IdRole = Qt::UserRole + 1,
        NameRole,
        EmailRole
    };

    explicit UserModel(QObject *parent = nullptr);
    
    int rowCount(const QModelIndex &parent = QModelIndex()) const override;
    QVariant data(const QModelIndex &index, int role = Qt::DisplayRole) const override;
    QHash&lt;int, QByteArray&gt; roleNames() const override;

    void addUser(const User& user);

private:
    QVector&lt;User&gt; users;
};

#endif // USERMODEL_H

// usermodel.cpp
#include "usermodel.h"

UserModel::UserModel(QObject *parent)
    : QAbstractListModel(parent)
{
    // Добавляем тестовых пользователей
    addUser(User(1, "John Doe", "john@example.com"));
    addUser(User(2, "Jane Smith", "jane@example.com"));
}

int UserModel::rowCount(const QModelIndex &) const
{
    return users.size();
}

QVariant UserModel::data(const QModelIndex &index, int role) const
{
    if (index.row() < 0 || index.row() >= users.size())
        return QVariant();

    const User &user = users[index.row()];
    
    switch (role) {
    case IdRole:
        return user.getId();
    case NameRole:
        return QString::fromStdString(user.getName());
    case EmailRole:
        return QString::fromStdString(user.getEmail());
    default:
        return QVariant();
    }
}

QHash&lt;int, QByteArray&gt; UserModel::roleNames() const
{
    QHash&lt;int, QByteArray&gt; roles;
    roles[IdRole] = "id";
    roles[NameRole] = "name";
    roles[EmailRole] = "email";
    return roles;
}

void UserModel::addUser(const User& user)
{
    beginInsertRows(QModelIndex(), users.size(), users.size());
    users.append(user);
    endInsertRows();
}

// Model
package com.example.model;

public class User {
    private int id;
    private String name;
    private String email;
    
    public User(int id, String name, String email) {
        this.id = id;
        this.name = name;
        this.email = email;
    }
    
    // Getters and setters
    public int getId() { return id; }
    public void setId(int id) { this.id = id; }
    public String getName() { return name; }
    public void setName(String name) { this.name = name; }
    public String getEmail() { return email; }
    public void setEmail(String email) { this.email = email; }
}

// Controller
package com.example.controller;

import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.GetMapping;
import com.example.model.User;
import java.util.ArrayList;
import java.util.List;

@Controller
public class UserController {
    
    @GetMapping("/users")
    public String getUsers(Model model) {
        List&lt;User&gt; users = getUsersFromService(); // Предположим, это метод сервиса
        model.addAttribute("users", users);
        return "users"; // имя представления
    }
    
    private List&lt;User&gt; getUsersFromService() {
        List&lt;User&gt; users = new ArrayList&lt;&gt;();
        users.add(new User(1, "John Doe", "john@example.com"));
        users.add(new User(2, "Jane Smith", "jane@example.com"));
        return users;
    }
}
